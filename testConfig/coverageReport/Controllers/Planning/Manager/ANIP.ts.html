  function getPlanObject() {
    const ret: PlanObject = new PlanObject();
    ret.StaticGUID = $("#StaticGUID").val();
    ret.Title = $("#Title").val();
    ret.JurisdictionName = $("#JurisdictionName").val();
    ret.OrganizationName = $("#OrganizationName").val();
    ret.PlanDate = $("#PlanDate")
      .data("kendoDateTimePicker")
      .value();
    ret.PlanDateCreated = $("#PlanDateCreated").val();
    ret.PlanVersion = $("#PlanVersion").val();
    ret.IdentificationData = $("#IdentificationData").val();
    ret.DataOnDevelopment = $("#DataOnDevelopment").val();
    ret.MailingListAndProcedure = $("#MailingListAndProcedure").val();
    ret.Versioning = $("#Versioning").val();
    ret.Safety = $("#Safety").val();
    ret.Missions = $("#Missions").val();
    ret.OrgAndOp = $("#OrgAndOp").val();
    ret.Contacts = $("#Contacts").val();
    ret.Logistical = $("#Logistical").val();
    ret.Risk = $("#Risk").val();
    ret.Status = $("#Status")
      .data("kendoDropDownList")
      .value();
    return ret;
  }

    });
  }

    EMPlanStaticGUID: string;

  function saveCardWithoutAttachment() {
    const card: EMPlanCard = new EMPlanCard();
    card.EMPlanStaticGUID = $("#StaticGUID").val();
    card.InitialEMPlanGUID = $("#EMPlanGUID").val();
    card.Title = $("#txtTitle").val();
    card.Description = $("#txtDescription").val();
    card.EMPlanCardGUID = Guid.newGuid();
    if (card.Title.length > 0) {
      const strTmp = JSON.stringify(card);
      $Ajax.ajaxAntiForgery({
        url: addCardURL,
        type: "POST",
        data: {
          modelString: strTmp,
        },
        success: function(response) {
          if (response.success === false) {
            $.notify({ message: response.Message }, { type: "danger" });
          } else {
            onUploadSuccess(null);
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }
