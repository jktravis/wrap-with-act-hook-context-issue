  export function saveEMPlanType(saveURL) {
    const txten = $("#txtenEMPlanType").val();
    const txtfr = $("#txtfrEMPlanType").val();
    const txtde = $("#txtdeEMPlanType").val();
    const txtnl = $("#txtnlEMPlanType").val();
    const txtSel = $("#drpEMPlanType")
      .data("kendoDropDownList")
      .value();
    if (txten.length > 0) {
      $Ajax.ajaxAntiForgery({
        url: saveURL,
        type: "POST",
        data: {
          enText: txten,
          deText: txtde,
          nlText: txtnl,
          frText: txtfr,
          SelectedEMPlanType: txtSel,
        },
        success: function(response) {
          if (response.success === false) {
            $("#txtValSummEMPlanType").text(response.Message);
          } else {
            reloadEMPlanTypeDD();
            clearEMPlanTypeValues();
            $("#EMPlanTypeModal").modal("hide");
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }
  export function deleteEMPlanType(delURL) {
    const txtSel = $("#drpEMPlanType")
      .data("kendoDropDownList")
      .value();
    $Ajax.ajaxAntiForgery({
      url: delURL,
      type: "POST",
      data: {
        SelectedGUID: txtSel,
      },
      success: function(response) {
        if (response.success === false) {
          $.notify({ message: response.Message }, { type: "danger" });
        } else {
          reloadEMPlanTypeDD();
        }
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }
