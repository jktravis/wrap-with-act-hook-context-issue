  function returnItemsNotInArray(newarray, oldarray) {
    const ret = [];
    let i = 0;
    $.each(newarray, function(index, value) {
      if ($.inArray(value, oldarray) === -1) {
        ret[i++] = value;
      }
    });
    return ret;
  }

    if ($("#divCards")) {

  function droptargetOnDrop(e) {
    $(".dragging").removeClass("dragging");
    const newEl = $(e.draggable.currentTarget).clone();
    const dragid = newEl.attr("id");
    const dt = e.dropTarget;
    Main.droptargetOnDragLeave(e);
    const angDT = angular.element(dt)[0];
    const dropTool = angDT.getAttribute("data-toolName");
    const extraInfo = (<HTMLInputElement>document.getElementById("MediumLogRoomDropdown")).value;
    if ($("#wasDragged").length > 0) {
      $("#wasDragged").val("true");
    }
    (<IDashboardScope>angular.element("#dashControllerID").scope()).createEntry(dragid, dropTool, extraInfo);
  }

  export var cbraModule = angular.module("cobraapp", ["ngAnimate", "ngSanitize", "kendo.directives"], function(
    $httpProvider,

    // from http://victorblog.com/2012/12/20/make-angularjs-http-service-behave-like-jquery-ajax/
    // use x-www-form-urlencoded Content-Type
    $httpProvider.defaults.headers.post["Content-Type"] = "application/x-www-form-urlencoded;charset=utf-8";
    // override $http service's default transformRequest
    $httpProvider.defaults.transformRequest = [
      function(data) {
        const param = function(obj) {
          let query = "";
          let name, value, fullSubName, subName, subValue, innerObj, i;
          for (name in obj) {
            if (obj.hasOwnProperty(name)) {
              value = obj[name];
              if (value instanceof Array) {
                for (i = 0; i < value.length; ++i) {
                  subValue = value[i];
                  fullSubName = name + "[" + i + "]";
                  innerObj = {};
                  innerObj[fullSubName] = subValue;
                  query += param(innerObj) + "&";
                }
              } else if (value instanceof Object) {
                for (subName in value) {
                  if (value.hasOwnProperty(subName)) {
                    subValue = value[subName];
                    fullSubName = name + "[" + subName + "]";
                    innerObj = {};
                    innerObj[fullSubName] = subValue;
                    query += param(innerObj) + "&";
                  }
                }
              } else if (value !== undefined && value !== null) {
                query += encodeURIComponent(name) + "=" + encodeURIComponent(value) + "&";
              }
            }
          }
          return query.length ? query.substr(0, query.length - 1) : query;
        };
        return angular.isObject(data) && String(data) !== "[object File]" ? param(data) : data;
      },
    ];
  });

        },
      });

    }
    function getCurOpPeriod() {
      const deferred = $q.defer();
      $.ajax({
        url: hostURL + curOpPeriodURL + projGUID,
        type: "GET",
        success: function(response) {
          deferred.resolve(response);
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
      return deferred.promise;
    }
    function refresh() {
      const deferred = $q.defer();
      $q.all([
        getCurOpPeriod().then(function(data) {
          if (data === "") {
            _curOpPeriod = null;
          } else {
            _curOpPeriod = data;
          }
        }),
        getAllOpPeriods()
          .then(function(data) {
            _allOpPeriods = data;
          })
          .then(function() {
            deferred.resolve();
          }),
      ]);
      return deferred.promise;
    }
    return {
      init: function(hstURL, projectGUID) {
        hostURL = hstURL;
        projGUID = projectGUID;
        return refresh();
      },
      refresh: refresh,
      allOpPeriods: function() {
        return _allOpPeriods;
      },
      curOpPeriod: function() {
        return _curOpPeriod;
      },
    };
  });
