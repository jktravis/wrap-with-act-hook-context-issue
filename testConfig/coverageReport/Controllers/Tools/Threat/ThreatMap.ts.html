  function buildMapData() {
    const baselonLat: [number, number] = [-95.7129, 37.0902]; //just whereever.  i need an initial spot to move from later on.  this is the middle of america
    const baselonLatInWebMercator = fromLonLat(baselonLat);
 
    mapView = new View({
      center: baselonLatInWebMercator,
      zoom: 10,
    });
 
    mapView.on("propertychange", function(e) {
      switch (e.key) {
        case "resolution":
          $("#zoomvalue").text(e.oldValue);
          break;
      }
    });
 
    mapObj = new Map({
      target: "map",
      layers: [
        new TileLayer({
          source: getMapDataSource(_globalPrompts.DefaultBaseMap),
        }),
      ],
      view: mapView,
    });
 
    const mousePositionControl = new MousePosition({
      coordinateFormat: createStringXY(4),
      projection: "EPSG:4326",
      // comment the following two lines to have the mouse position
      // be placed within the map.
      className: "custom-mouse-position",
      target: document.getElementById("mouse-position"),
      undefinedHTML: "&nbsp;",
    });
    mapObj.addControl(mousePositionControl);
  }

  function LoadThreatEntries() {
    const styleFunction = function(feature) {
      return [
        new Style({
          stroke: new Stroke({
            color: "black",
            width: 1,
          }),
          fill: new Fill({
            color: colorFunction(feature),
          }),
          image: new Circle({
            radius: 5,
            fill: new Fill({
              color: colorFunction(feature),
            }),
          }),
        }),
      ];
    };
 
    function colorFunction(feature) {
      const defaultColor = "rgba(52,58,64,0.9)";
      const severityVal = feature.get("severity");
      if (severityVal !== undefined && severityVal.length > 0) {
        switch (severityVal.toLowerCase()) {
          case "significant":
            return "rgba(220,53,69,0.9)";
          case "high":
            return "rgba(255,193,7,0.9)";
          case "medium":
            return "rgba(40,167,69,0.9)";
          case "low":
            return "rgba(23,162,184,0.9)";
          default:
            return defaultColor; //black
        }
      }
      return defaultColor;
    }
 
    const sinceDate = $("#sinceDate")
      .data("kendoDateTimePicker")
      .value();
    const sigChecked = (<any>document.getElementById("chksignificant")).checked;
    const highChecked = (<any>document.getElementById("chkHigh")).checked;
    const medChecked = (<any>document.getElementById("chkMedium")).checked;
    const lowChecked = (<any>document.getElementById("chkLow")).checked;
    const unkChecked = (<any>document.getElementById("chkUnknown")).checked;
    const geoJsonUrl =
      _globalPrompts.ThreatEntryGeoJSONURL +
      _globalPrompts.ThreatRegistrationGUID +
      "&sinceDate=" +
      kendo.toString(sinceDate, "G") +
      "&includeSignificant=" +
      sigChecked +
      "&includeHigh=" +
      highChecked +
      "&includeMedium=" +
      medChecked +
      "&includeLow=" +
      lowChecked +
      "&includeUnknown=" +
      unkChecked;
    threatEntrySource = new SourceVector({
      format: new GeoJSON(),
      url: geoJsonUrl,
    });
 
    threatEntryLayer = new Vector({
      source: threatEntrySource,
      style: styleFunction,
    });
    Main.blockit($("#entryListWrapper"));
    const listenerKey = threatEntrySource.on("change", function(e) {
      if (threatEntrySource.getState() === "ready") {
        let featurefound = false;
        const features = threatEntrySource.getFeatures();
        if (features !== undefined && features.length > 0) {
          if (_globalPrompts.ThreatEntryGUID.length > 0) {
            featurefound = PanAndSelectFeature(features, _globalPrompts.ThreatEntryGUID);
            _globalPrompts.ThreatEntryGUID = ""; //so they don't get panned again on change date
          }
          if (!featurefound) {
            ZoomToFeatExtent();
            //clear feature details
            showFeatureDetails(null);
          }
        }
        Main.unblockit($("#entryListWrapper"));
        LoadFeatureGrid(features);
      }
    });
    mapObj.addLayer(threatEntryLayer);
  }
 
  export function ZoomToFeatExtent() {
    if (threatEntrySource !== null && threatEntrySource !== undefined) {
      const featExtent = threatEntrySource.getExtent();
      mapView.fit(featExtent, { duration: 1000 });
    }
  }
 
  function PanAndSelectFeature(features, threatEntryGUID): boolean {
    let featurefound = false;
    for (let i = 0; i < features.length && !featurefound; i++) {
      if (features[i].get("ThreatEntryGUID") === threatEntryGUID) {
        featurefound = true;
        const geom = features[i].getGeometry();
        const coords = geom.getCoordinates();
        if (coords.length === 1) {
          //polygon, not a point
          mapView.fit(geom.getExtent(), { duration: 1000 });
          //mapView.animate({ center: geom.getCenter(), duration: 1000 });
        } else {
          mapView.animate({ center: coords, duration: 1000 });
        }
        //show feature details
        showFeatureDetails(features[i]);
      }
    }
    return featurefound;
  }
 
  let featureGridSource;
  function LoadFeatureGrid(features) {
    const entries = [];
    for (let i = 0; i < features.length; i++) {
      let goodTime = features[i].get("TimeOfEvent");
      const lastHyphen = goodTime.lastIndexOf("-");
      if (lastHyphen > 10) {
        goodTime = goodTime.substring(0, lastHyphen);
      }
      const lastPlus = goodTime.lastIndexOf("+");
      if (lastPlus > 10) {
        goodTime = goodTime.substring(0, lastPlus);
      }
      let unixTime = features[i].get("time");
      if (unixTime === undefined) {
        unixTime = new Date(goodTime).valueOf();
      }
 
      entries.push({
        title: features[i].get("title"),
        backgroundColor: features[i].get("backgroundColor"),
        TimeOfEvent: kendo.toString(goodTime, "G"),
        time: unixTime,
        ThreatEntryGUID: features[i].get("ThreatEntryGUID"),
      });
    }
 
    featureGridSource = new kendo.data.DataSource({
      data: entries,
      sort: {
        field: "TimeOfEvent",
        dir: "desc",
      },
      schema: {
        model: {
          id: "ThreatEntryGUID",
          fields: {
            ThreatEntryGUID: { editable: false },
            time: { editable: false, type: "number" },
            TimeOfEvent: { editable: false, type: "datetime" },
            backgroundColor: { editable: false },
            title: { editable: false },
          },
        },
      },
    });
 
    $("#entryListView").kendoListView({
      dataSource: featureGridSource,
      selectable: "single",
      change: onListSelect,
      template: kendo.template($("#template").html()),
      altTemplate: kendo.template($("#alttemplate").html()),
    });
  }
 
  function onListSelect() {
    const data = featureGridSource.view(),
      selectedthreatEntry = $.map(this.select(), function(item) {
        return data[$(item).index()].ThreatEntryGUID;
      });
    if (selectedthreatEntry.length > 0) {
      const features = threatEntrySource.getFeatures();
      PanAndSelectFeature(features, selectedthreatEntry[0]);
    }
  }
 
  function showFeatureDetails(feature) {
    if (feature !== null && feature !== undefined) {
      const featureDetails = [];
      let featType = "";
      if (feature.get("type") !== undefined) {
        featType = feature.get("type").toUpperCase();
      } else if (feature.get("msgType") !== undefined) {
        featType = feature.get("msgType").toUpperCase();
      }
      featureDetails.push("<div>");
      featureDetails.push(
        "<div class='row'>",
        "<div class='col-auto'>",
        "<h5 class='badge badge-" + feature.get("backgroundColor") + "'>",
        featType,
        "</h5>",
        "</div>",
        "<div class='col'></div>",
        "<div class='col-auto'>",
        "<small>",
        "<time data-format='lll' datetime='" + feature.get("time") + "'>",
        Main.dateFormatter(feature.get("TimeOfEvent")),
        "</time>",
        "</small>",
        "</div>",
        "</div>",
      );
      featureDetails.push("<h4>", feature.get("title"), "</h4>");
 
      const summary = feature.get("summary");
      if (summary !== undefined && summary.length > 0) {
        featureDetails.push("<p>", "<strong>", summary, "</strong>", "</p>");
      }
 
      const url = feature.get("url");
      //const dataUrl = feature.get('');
      if (url !== undefined && url.length > 0) {
        featureDetails.push(
          "<div class='row'>",
          "<div class='col'></div>",
          "<div class='col-auto'>",
          "<a href='" +
            feature.get("url") +
            "' class='btn btn-link' title='" +
            _globalPrompts.MoreDetailsString +
            "' target='_blank'>",
          _globalPrompts.MoreDetailsString,
          "</a>",
          "</div>",
          "</div>",
        );
      }
      featureDetails.push(
        "<div class='btn-group'>",
        "<a href='#' role='button' onclick=\"ThreatMap.CreateIncident('" +
          feature.get("ThreatEntryGUID") +
          "')\" class='btn btn-warning' title='Create Incident'>",
        "Create Incident",
        "</a>",
        "<a href='" + feature.get("url") + "' class='btn btn-warning disabled' title='Alert'>",
        "Alert",
        "</a>",
        "</div>",
      );
      featureDetails.push("</div>");
      $("#properties").html(featureDetails.join(""));
    } else {
      $("#properties").html("");
    }
  }
 
  function LoadThreatRegistrationArea() {
    $Ajax.ajaxAntiForgery({
      url: _globalPrompts.RegistrationAreaUrl,
      type: "GET",
      data: {
        threatRegistrationGUID: _globalPrompts.ThreatRegistrationGUID,
      },
      success: function(response) {
        if (response.length > 0) {
          loadWKTIntoMap(response);
        }
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }

  function loadWKTIntoMap(wkt) {
    const format = new WKT();
    const feature = format.readFeature(wkt, {
      dataProjection: "EPSG:4326",
      featureProjection: "EPSG:3857", //what is this about?  Something WKT specific?
    });
    registrationSource = new SourceVector({
      features: [feature],
    });
    const vector = new Vector({
      source: registrationSource,
      style: new Style({
        fill: new Fill({
          color: "rgba(243,180,21,0.3)",
        }),
        stroke: new Stroke({
          color: "rgba(9,7,0,0.7)",
        }),
      }),
      //maxResolution: 5000,
      minResolution: 5000,
    });
    mapObj.addLayer(vector);
  }
 
  function WireUpClickEvents() {
    mapObj.on("singleclick", function(evt) {
      let featureFound = null;
      this.forEachFeatureAtPixel(evt.pixel, function(feature, layer) {
        const featureProperties = feature.getProperties();
        if (featureProperties.title !== undefined && featureProperties.title.length > 0) {
          featureFound = feature;
        }
      });
      if (featureFound !== null) {
        //select the feature in listview, and scroll to it
        const threatEntryGUID = featureFound.get("ThreatEntryGUID");
        if (threatEntryGUID !== undefined) {
          const listviewObj = $("#entryListView");
          listviewObj.prop({ scrollTop: 0 }); //reset.  otherwise math.
          const listView = listviewObj.data("kendoListView");
          const listViewItem = listView.element.children(
            "[data-uid='" + listView.dataSource.get(threatEntryGUID).uid + "']",
          );
          listView.select(listViewItem);
          listviewObj.animate({ scrollTop: listView.select().position().top }, 400);
        }
      }
    });
  }
 
  export function CreateIncident(entryGUID) {
    $Ajax.ajaxAntiForgery({
      url: _globalPrompts.CreateIncidentFromThreatURL,
      data: {
        ThreatEntryGUID: entryGUID,
      },
      type: "POST",
      success: function(response) {
        if (response.Message !== undefined && response.Message.length > 0) {
          $.notify({ message: response.Message }, { type: "danger" });
        } else {
          //do something?  redirect?
          $.notify({ message: "Incident Created" }, { type: "info" });
        }
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }
