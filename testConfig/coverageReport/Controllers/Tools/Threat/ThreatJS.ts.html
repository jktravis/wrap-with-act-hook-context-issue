  function buildMapData() {
    const baselonLat: [number, number] = [-95.7129, 37.0902]; //just whereever.  i need an initial spot to move from later on.  this is the middle of america
    const baselonLatInWebMercator = fromLonLat(baselonLat);
 
    mapView = new View({
      center: baselonLatInWebMercator,
      zoom: 5,
    });
 
    mapObj = new Map({
      target: "map",
      layers: [
        new TileLayer({
          source: new OSM(),
        }),
      ],
      view: mapView,
    });
 
    const mousePositionControl = new MousePosition({
      coordinateFormat: createStringXY(4),
      projection: "EPSG:4326",
      // comment the following two lines to have the mouse position
      // be placed within the map.
      className: "custom-mouse-position",
      target: document.getElementById("mouse-position"),
      undefinedHTML: "&nbsp;",
    });
    mapObj.addControl(mousePositionControl);
  }
 
  export function BufferSizeChanged() {
    loadOrgAreaInf();
    ShowFacArea();
  }
  function loadOrgAreaInf() {
    if ($("#chkIncludeAllOrg").is(":visible") === false || $("#chkIncludeAllOrg").is(":checked")) {
      const buffer = $("#bufferSize")
        .data("kendoDropDownList")
        .value();
      $Ajax.ajaxAntiForgery({
        url: _globalprmp.getOrgAreaUrl,
        type: "GET",
        data: {
          bufferSize: buffer,
        },
        success: function(response) {
          if (response.success === false) {
            $.notify({ message: response.Message }, { type: "danger" });
          } else {
            $("#orgArea").text(response);
            loadWKTIntoMap(response);
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }
 
  export function ShowFacArea() {
    if ($("#chkIncludeAllOrg").is(":checked")) {
      return;
    }
    const facGUIDs = $("#regFacDiv").data("faclist");
    const facGUIDsAsString = JSON.stringify(facGUIDs);
    const buffer = $("#bufferSize")
      .data("kendoDropDownList")
      .value();
    if (facGUIDsAsString.length > 0) {
      $Ajax.ajaxAntiForgery({
        url: _globalprmp.getFacAreaUrl,
        type: "GET",
        data: {
          facGUIDs: facGUIDsAsString,
          bufferSize: buffer,
        },
        success: function(response) {
          if (response.success === false) {
            $.notify({ message: response.Message }, { type: "danger" });
          } else {
            $("#facArea").text(response);
            loadWKTIntoMap(response);
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    } else {
      $("#facArea").text("");
      loadWKTIntoMap();
    }
  }
 
  function loadWKTIntoMap(wkt?) {
    if (vectorLayer) {
      mapObj.removeLayer(vectorLayer);
    }
    if (wkt && mapView) {
      const format = new WKT();
      const feature = format.readFeature(wkt, {
        dataProjection: "EPSG:4326",
        featureProjection: "EPSG:3857", //what is this about?  Something WKT specific?
      });
      const registrationSource = new SourceVector({
        features: [feature],
      });
      vectorLayer = new Vector({
        source: registrationSource,
        style: new Style({
          fill: new Fill({
            color: "rgba(52,58,64,0.9)",
            //color:'rgba(52,58,64,0.9)'
          }),
          stroke: new Stroke({
            color: "rgba(52,58,64,0.9)",
          }),
        }),
      });
      mapObj.addLayer(vectorLayer);
      const geom = feature.getGeometry();
      const geomType = geom.getType();
      if (geomType === "Point") {
        mapView.animate({
          center: geom.getExtent(),
          zoom: defaultZoom,
          duration: defaultAnimation,
        });
      } else {
        mapView.fit(geom.getExtent(), { duration: defaultAnimation });
      }
    }
  }
  let vectorLayer;

  const defaultZoom = 12;

  export function saveReg() {
    const regName = $("#txtName").val();
    const regDesc = $("#txtDesc").val();
    const regGUID = $("#txtRegGUID").val();
    let orgWide: boolean;
    if ($("#chkIncludeAllOrg").is(":checked")) {
      orgWide = true;
    } else {
      orgWide = false;
    }
    const facGUIDs = $("#regFacDiv").data("faclist");
    //const facGUIDsAsString = JSON.stringify(facGUIDs);
    const usrGUIDs = $("#notifiedUsersDiv").data("userlist");
    //const usrGUIDsAsString = JSON.stringify(usrGUIDs);
    const watched = $("#watchedThreats")
      .data("kendoMultiSelect")
      .value();
    //const watched = JSON.stringify($("#watchedThreats").data("kendoMultiSelect").value());
    const buffer = $("#bufferSize")
      .data("kendoDropDownList")
      .value();
    if (regName.length > 0) {
      $Ajax.ajaxAntiForgery({
        url: _globalprmp.regSaveUrl,
        type: "POST",
        data: {
          RegistrationName: regName,
          RegistrationDescription: regDesc,
          OrgWide: orgWide,
          FacilityGUIDs: facGUIDs,
          NotifiedUserGUIDs: usrGUIDs,
          watchedThreats: watched,
          bufferSize: buffer,
          ThreatRegistrationGUID: regGUID,
        },
        success: function(response) {
          if (response.success === false) {
            $.notify({ message: response.Message }, { type: "danger" });
          } else {
            refreshgrid();
            $("#collapseExample").collapse("hide");
            $("#cancelBtn").hide();
            $("#divmapcontainer").hide();
            clearValues();
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    } else {
      $.notify({ message: "Must enter a valid registration name" }, { type: "danger" });
    }
  }
 
  export function editRow(regGUID, regGetURL) {
    $Ajax.ajaxAntiForgery({
      url: regGetURL,
      type: "GET",
      data: {
        threatRegistrationGUID: regGUID,
      },
      success: function(response) {
        if (response === null) {
          $.notify({ message: "Registration not found" }, { type: "danger" });
        } else {
          $("#txtName").val(response.RegistrationName);
          $("#txtDesc").val(response.RegistrationDescription);
          $("#txtRegGUID").val(response.ThreatRegistrationGUID);
          if (response.FacilityGUIDs !== null && response.FacilityGUIDs !== undefined) {
            $("#regFacDiv").data("faclist", response.FacilityGUIDs);
            $("#chkIncludeAllOrg").prop("checked", false);
          } else {
            $("#chkIncludeAllOrg").prop("checked", true);
          }
          $("#bufferSize")
            .data("kendoDropDownList")
            .value(response.BufferSize);
          $("#collapseExample").collapse("show");
          $("#cancelBtn").show();
          $("#divmapcontainer").show();
        }
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }
  export function deleteRow(regGUID, regDelURL) {
    if (regGUID.length > 0) {
      $Ajax.ajaxAntiForgery({
        url: regDelURL,
        type: "POST",
        data: {
          ThreatRegistrationGUID: regGUID,
        },
        success: function(response) {
          if (response.success === false) {
            $.notify({ message: response.Message }, { type: "danger" });
          } else {
            refreshgrid();
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    } else {
      $.notify({ message: "Must enter a valid registration GUID" }, { type: "danger" });
    }
  }
