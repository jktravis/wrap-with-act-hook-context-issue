import "../../../../node_modules/ol/ol.css";
 
import Map from "ol/Map";
import GeoJSON from "ol/format/GeoJSON";
//import Tile from 'ol/layer/Tile';
//import TileImage from 'ol/source/TileImage';
import TileLayer from "ol/layer/Tile";
//import XYZ from 'ol/source/xyz';
 
//import Overlay from 'ol/overlay';
import OSM from "ol/source/OSM";
 
import SourceVector from "ol/source/Vector";
import Vector from "ol/layer/Vector.js";
import WKT from "ol/format/WKT";
 
import { Fill, Stroke, Style, Icon, Circle, Text } from "ol/style.js";
 
import { createStringXY } from "ol/coordinate.js";
 
import { fromLonLat, transform } from "ol/proj.js";
 
import MousePosition from "ol/Control/MousePosition.js";
import View from "ol/View";
export namespace ThreatDummyData {
  interface IGlobalPrompt {
    //ThreatRegistrations: any;
    ThreatRegistrations: string;
    RegistrationAreaUrl: string;
    SendFakeThreatURL: string;
  }
 
  let _globalPrompts: IGlobalPrompt;
 
  let mapObj;
  let mapView;
 
  export function init(globalPrompts) {
    _globalPrompts = globalPrompts;
 
    buildMapData();
    WireUpClickEvents();
    if (_globalPrompts.ThreatRegistrations !== undefined && _globalPrompts.ThreatRegistrations.length > 0) {
      LoadThreatRegistrationAreas(JSON.parse(_globalPrompts.ThreatRegistrations));
    }
  }
 
  function buildMapData() {
    const baselonLat: [number, number] = [-95.7129, 37.0902]; //just whereever.  i need an initial spot to move from later on.  this is the middle of america
    const baselonLatInWebMercator = fromLonLat(baselonLat);
 
    mapView = new View({
      center: baselonLatInWebMercator,
      zoom: 5,
    });
 
    mapObj = new Map({
      target: "map",
      layers: [
        new TileLayer({
          source: new OSM(),
        }),
      ],
      view: mapView,
    });
 
    const mousePositionControl = new MousePosition({
      coordinateFormat: createStringXY(4),
      projection: "EPSG:4326",
      // comment the following two lines to have the mouse position
      // be placed within the map.
      className: "custom-mouse-position",
      target: document.getElementById("mouse-position"),
      undefinedHTML: "&nbsp;",
    });
    mapObj.addControl(mousePositionControl);
  }
 
  interface IThreatRegistration {
    ThreatRegistrationGUID: string;
    RegistrationName: string;
    RegistrationDescription: string;
  }
 
  function LoadThreatRegistrationAreas(registrations: IThreatRegistration[]) {
    for (const reg of registrations) {
      LoadThreatRegistrationArea(reg);
    }
  }
 
  function LoadThreatRegistrationArea(reg: IThreatRegistration) {
    $Ajax.ajaxAntiForgery({
      url: _globalPrompts.RegistrationAreaUrl,
      type: "GET",
      data: {
        threatRegistrationGUID: reg.ThreatRegistrationGUID,
      },
      success: function(response) {
        if (response.length > 0) {
          loadWKTIntoMap(response, reg);
        }
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }
 
  const selectColorFunc = (colorNum, colors) => {
    if (colors < 1) {
      colors === 1;
    } // defaults to one color - avoid divide by zero
    return "hsl(" + ((colorNum * (360 / colors)) % 360) + ",100%,50%,.5)";
  };
 
  function loadWKTIntoMap(wkt, reg: IThreatRegistration) {
    const format = new WKT();
    const feature = format.readFeature(wkt, {
      dataProjection: "EPSG:4326",
      featureProjection: "EPSG:3857", //what is this about?  Something WKT specific?
    });
    const registrationSource = new SourceVector({
      features: [feature],
    });
    const vector = new Vector({
      source: registrationSource,
      style: new Style({
        fill: new Fill({
          color: selectColorFunc(Math.floor(Math.random() * 100), 40),
          //color:'rgba(52,58,64,0.9)'
        }),
        stroke: new Stroke({
          color: "rgba(52,58,64,0.9)",
        }),
        text: new Text({
          font: "12px Calibri,sans-serif",
          fill: new Fill({
            color: "#000",
          }),
          text: reg.RegistrationName + " - " + reg.RegistrationDescription,
        }),
      }),
      //maxResolution: 5000,
      //minResolution: 5000
    });
    mapObj.addLayer(vector);
  }
 
  function WireUpClickEvents() {
    mapObj.on("singleclick", function(evt) {
      const trans = transform([evt.coordinate[0], evt.coordinate[1]], "EPSG:3857", "EPSG:4326");
      $("#txtLat").val(trans[1]);
      $("#txtLong").val(trans[0]);
    });
  }
 
  export function SendFakeThreat() {
    const data = {
      threatName: $("#txtTitle").val(),
      threatDescription: $("#txtDesc").val(),
      latitude: $("#txtLat").val(),
      longitude: $("#txtLong").val(),
      severity: "unknown",
    };
 
    if (data.threatName.length > 0 && data.latitude.length > 0 && data.longitude.length > 0) {
      $Ajax.ajaxAntiForgery({
        url: _globalPrompts.SendFakeThreatURL,
        type: "POST",
        contentType: "application/json",
        data: JSON.stringify(data),
        success: function(response) {
          if (response.success === true) {
            $.notify({ message: "data sent" }, { type: "info" });
          } else {
            $.notify({ message: response }, { type: "danger" });
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }
}
