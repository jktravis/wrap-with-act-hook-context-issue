  export function catchanged() {
    const drp = $("#StructureCategory").data("kendoDropDownList");
    $("#StructureCategoryString").val(drp.text());
  }
  export function typechanged() {
    const drp = $("#StructureType").data("kendoDropDownList");

  export function filterStructure() {
    return {
      DamageStructureCategoryGUID: $("#StructureCategory").val(),
    };
  }
  export function LoadFacData(facilityGUID: string) {
    $("#FacilityGUID").val(facilityGUID);
    $Ajax.ajaxAntiForgery({
      url: _globalPrompts.getFacilityURL,
      type: "GET",
      data: {
        FacilityGUID: facilityGUID,
      },
      success: function(response) {
        if (response != null) {
          $("#Street1").val(response.Street1);
          $("#Street2").val(response.Street2);
          $("#City").val(response.City);
          $("#Country")
            .data("kendoDropDownList")
            .value(response.Country);
          $("#State_Province")
            .data("kendoComboBox")
            .text(response.State_Province);
          $("#Zip_PostalCode").val(response.Zip_PostalCode);
          if ($("#County_Region")) {
            $("#County_Region").val(response.County_Region);
          }
          $("#Lat")
            .data("kendoNumericTextBox")
            .value(response.Lat);
          $("#Lon")
            .data("kendoNumericTextBox")
            .value(response.Lon);
          $("#What3Words").val(response.What3Words);
          placeMarker(true, null);
        }
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }
