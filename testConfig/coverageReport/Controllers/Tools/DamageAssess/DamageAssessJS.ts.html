  export function saveReport(baseurl) {
    let latval;
    if ($("#Lat")) {
      latval = $("#Lat").val();
    }
    let lonval;
    if ($("#Lon")) {
      lonval = $("#Lon").val();
    }
    const structureType = $("#StructureType")
      .data("kendoDropDownList")
      .value();
    const structureCategory = $("#StructureCategory")
      .data("kendoDropDownList")
      .value();
    const structureTypeString = $("#StructureType")
      .data("kendoDropDownList")
      .text();
    const structureCategoryString = $("#StructureCategory")
      .data("kendoDropDownList")
      .text();
    const damageNumber = $("#DamageNumber")
      .data("kendoNumericTextBox")
      .value();
    const ownerName = $("#OwnerName").val();
    const contactInformation = $("#ContactInformation").val();
    const damageType = $("#DamageType")
      .data("kendoDropDownList")
      .value();
    const damageAmount = $("#DamageAmount")
      .data("kendoNumericTextBox")
      .value();
    const percentDamaged = $("#PercentDamaged")
      .data("kendoNumericTextBox")
      .value();
    const propertyValue = $("#PropertyValue")
      .data("kendoNumericTextBox")
      .value();
    const insured = $("#Insured")
      .data("kendoDropDownList")
      .value();
    const damageDescription = $("#DamageDescription").val();
    const eOCNotes = $("#EOCNotes").val();
    const logguid = $("#LogEntryGUID").val();
    const statguid = $("#StaticGUID").val();
    const street1 = $("#Street1").val();
    const street2 = $("#Street2").val();
    const city = $("#City").val();
    const stCombo = $("#State_Province").data("kendoComboBox");
    const stateval = stCombo.text();
    const zip = $("#Zip_PostalCode").val();
    const county = $("#County_Region").val();
    const country = $("#Country").val();
    const linkedLogEntriesAsString = $("#LinkedLogEntriesAsString").val();
    if (structureType !== null && structureType.length > 0) {
      const url = baseurl;
 
      $Ajax.ajaxAntiForgery({
        url: url,
        type: "POST",
        data: {
          LinkedLogEntriesAsString: linkedLogEntriesAsString,
          Lat: latval,
          Lon: lonval,
          Street1: street1,
          Street2: street2,
          City: city,
          State: stateval,
          Zip: zip,
          County: county,
          Country: country,
          StructureType: structureType,
          StructureCategory: structureCategory,
          StructureTypeString: structureTypeString,
          StructureCategoryString: structureCategoryString,
          DamageNumber: damageNumber,
          OwnerName: ownerName,
          ContactInformation: contactInformation,
          DamageType: damageType,
          DamageAmount: damageAmount,
          PercentDamaged: percentDamaged,
          PropertyValue: propertyValue,
          Insured: insured,
          DamageDescription: damageDescription,
          EOCNotes: eOCNotes,
          LogEntryGUID: logguid,
          StaticGUID: statguid,
        },
        success: function(response) {
          if (response !== null) {
            Dashboard.jqueryclosemodal();
            //add refresh to notes window here if you like
            //refreshAlldamageassess(true);
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }

  function refreshAlldamageassess() {

      (<IDamageAssessScope>angular.element("#damageassessControllerID").scope()).damageassessDataSourceL.read();
    }
    //would like to figure out a way to make this repository reside in the dashboard controller somehow, so that i
    //only need to refresh this once.
    if (angular.element("#damageassessControllerIDM").length) {
      (<IDamageAssessScope>angular.element("#damageassessControllerIDM").scope()).damageassessDataSourceM.read();
    }
  }
 
  angular.module("cobraapp").factory("damageassessRepository", function() {
    return {
      getDataSourceM: function(hostURL, projectGUID, configProjectGUID) {
        return new kendo.data.DataSource({
          transport: {
            read: {
              url: hostURL + "Tools/damageassess/_GetReportListJSON/" + projectGUID,
            },
            destroy: {
              url: hostURL + "Tools/damageassess/_DeleteReport/" + projectGUID,
              type: "POST",
              dataType: "json",
            },
          },
          schema: {
            model: {
              id: "LogEntryGUID",
              fields: {
                LogEntryGUID: { editable: false },
                StructureType: { editable: false },
                DamageType: { editable: false },
                LocationString: { editable: true },
                DamageDescription: { editable: true },
                DamageAmount: { editable: true },
                DamageNumber: { editable: true },
                FormattedDateCreated: { type: "date", editable: false },
              },
            },
          },
        });
      },
      getDataSourceL: function(hostURL, projectGUID, configProjectGUID) {
        return new kendo.data.DataSource({
          transport: {
            read: {
              url: hostURL + "Tools/damageassess/_GetReportListJSON/" + projectGUID,
            },
            destroy: {

              type: "POST",
              dataType: "json",

          schema: {
            model: {
              id: "LogEntryGUID",
              fields: {

    };
