  function refreshAllattachment() {
    if (angular.element("#attachmentControllerID").length) {
      (<IAttachmentScope>angular.element("#attachmentControllerID").scope()).attachmentDataSourceL.read();
    }
    //would like to figure out a way to make this repository reside in the dashboard controller somehow, so that i
    //only need to refresh this once.
    if (angular.element("#attachmentControllerIDM").length) {
      (<IAttachmentScope>angular.element("#attachmentControllerIDM").scope()).attachmentDataSourceM.read();
    }
  }
 
  angular.module("cobraapp").factory("attachmentRepository", function() {
    return {
      getDataSourceM: function(hostURL, projectGUID, configProjectGUID) {
        return new kendo.data.DataSource({
          transport: {
            read: {
              url: hostURL + "Tools/attachment/_GetAttachmentListJSON/" + projectGUID,
            },
            update: {
              url: hostURL + "Tools/attachment/_UpdateAttachment/" + projectGUID,
              type: "POST",
              dataType: "json",
            },
          },
          schema: {
            model: {
              id: "LogEntryGUID",
              fields: {
                LogEntryGUID: { editable: false },
                FileName: { editable: false },
                FriendlyUserName: { editable: false },
                LocationString: { editable: false },
                EventDescription: { editable: true },
                FormattedDateCreated: { type: "date", editable: false },
              },
            },
          },
        });
      },
      getDataSourceL: function(hostURL, projectGUID, configProjectGUID) {
        return new kendo.data.DataSource({
          transport: {
            read: {
              url: hostURL + "Tools/attachment/_GetAttachmentListJSON/" + projectGUID,
            },
            update: {
              url: hostURL + "Tools/attachment/_UpdateAttachment/" + projectGUID,
              type: "POST",
              dataType: "json",
            },
          },
          pageSize: 10,
          schema: {
            model: {
              id: "LogEntryGUID",
              fields: {
                LogEntryGUID: { editable: false },
                FileName: { editable: false },
                FriendlyUserName: { editable: false },
                LocationString: { editable: false },
                EventDescription: { editable: true },
                FormattedDateCreated: { type: "date", editable: false },
              },
            },
          },
        });
      },
    };
  });
  angular.module("cobraapp").controller("attachmentController", function($scope, attachmentRepository) {
    $scope.loadcard = function(ocard) {
      $scope.currentCard = ocard;
    };
    $scope.isAttachmentImage = function(fileext) {
      fileext = fileext.toLowerCase();
      //".jpg", ".jpeg", ".gif", ".png", ".bmp"
      switch (fileext) {
        case ".jpg":
        case ".jpeg":
        case ".gif":
        case ".png":
        case ".bmp":
          return true;
        default:
          return false;
      }
    };
    $scope.$on("initModalJS", function() {
