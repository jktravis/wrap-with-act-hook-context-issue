  function getFeaturesAtPixel(pixel) {
    const featuresAtPoint: IFeatureDetails[] = [];
    cobramap.mapObj.forEachFeatureAtPixel(
      pixel,
      function(feature, layer) {
        const tmpFeature = cobramap.GetFeatureDetails(feature);
        if (
          tmpFeature !== undefined &&
          tmpFeature.PrimaryKey !== undefined &&
          tmpFeature.PrimaryKey.length > 0 &&
          !FeatureIsHidden(tmpFeature.PrimaryKey)
        ) {
          featuresAtPoint.push(tmpFeature);
        } else if (layer !== undefined && layer.LogEntryGUID !== undefined) {
          //this is from the KML layer, we get the data in another way.
          feature.PrimaryKey = layer.LogEntryGUID;
          feature.DataType = FeatureDataType.LogEntryGUID;
          featuresAtPoint.push(feature);
        }
      },
      {
        hitTolerance: cobramap.hitTolerance,
        layerFilter: function(layer) {
          return layer.CanSelect;
        },
      },
    );
    return featuresAtPoint;
  }

  export function ZoomToFeatExtent(propertyToSearchFor) {
    let featExtent = null;
    let zoomLevl = 0;
    if (propertyToSearchFor === null || propertyToSearchFor === undefined) {
      //i guess this is just me saying 'if nothing is specified, then just zoom to the extent of the incident entries'
      if (featEntrySource !== null && featEntrySource !== undefined) {
        featExtent = featEntrySource.getExtent();
      }
    } else {
      //this is the normal case
      if (propertyToSearchFor === "WMSLayerGUID") {
        cobramap.GetLayers("WMSLayerGUID").forEach(function(lyr) {
          if (!FeatureIsHidden(lyr.WMSLayerGUID)) {
            if (featExtent === null) {
              featExtent = lyr.WMSExtent;
            } else {
              extend(featExtent, lyr.WMSExtent);
            }
          }
        });
      } else {
        let features;
        if (
          propertyToSearchFor === "LogEntryGUID" ||
          propertyToSearchFor === "ProjectGUID" ||
          propertyToSearchFor === "RootOrganization"
        ) {
          features = featEntrySource.getFeatures();
        } else if (propertyToSearchFor === "FacilityGUID") {
          features = facSource.getFeatures();
        } else if (propertyToSearchFor === "BackgroundLayerGUID") {
          features = [];
          cobramap.GetLayers("BackgroundLayerGUID").forEach(function(lyr) {
            features = features.concat(lyr.getSource().getFeatures());
          });
        }
        zoomLevl = defaultZoomLevel;
        for (const feature of features) {
          if (feature.get(propertyToSearchFor) !== undefined) {
            if (featExtent === null) {
              featExtent = feature.getGeometry().getExtent();
            } else {
              zoomLevl = 0;
              extend(featExtent, feature.getGeometry().getExtent());
            }
          }
        }
      }
    }
    if (featExtent !== null) {
      cobramap.AnimateOrFitToExtent(featExtent, zoomLevl);
    }
  }

  function getFeaturesByPrimaryKey(itemType: FeatureDataType, primaryKey: string, secondaryKey: string): any[] {

  export function ZoomPan(fitToExtent: boolean, itemType: FeatureDataType, primaryKey: string, secondaryKey: string) {
    if (FeatureIsHidden(primaryKey) || FeatureIsHidden(secondaryKey)) {
      return;
    }
    const features = getFeaturesByPrimaryKey(itemType, primaryKey, secondaryKey);
    if (features !== undefined && features.length > 0) {
      if (features.length === 1) {
        const geom = features[0].getGeometry();
        const geomType = geom.getType();
        if (geomType === "Point") {
          //this is a point then, not a polygon
          cobramap.AnimateOrFitToExtent(geom.getExtent(), defaultZoomLevel);
          return;
        }
      }
      let featExtent = null;
      for (const feature of features) {
        if (featExtent === null) {
          featExtent = feature.getGeometry().getExtent();
        } else {
          extend(featExtent, feature.getGeometry().getExtent());
        }
      }
      var zoomLevel = defaultZoomLevel;
      if (fitToExtent) {
        zoomLevel = 0;
      }
      if (featExtent !== null) {
        cobramap.AnimateOrFitToExtent(featExtent, zoomLevel);
      }
    } else if (itemType === FeatureDataType.WMSLayerGUID) {
      cobramap.GetLayers("WMSLayerGUID").forEach(function(lyr) {
        if (lyr.WMSLayerGUID === primaryKey && !lyr.IsWFS) {
          cobramap.AnimateOrFitToExtent(lyr.WMSExtent, zoomLevel);
          return;
        }
      });
    }
  }

  function getExtentOfWMSLayers(layers: IWMSWFSLayerDTO[], isWFS: boolean) {
    if (layers !== null && layers.length > 0) {
