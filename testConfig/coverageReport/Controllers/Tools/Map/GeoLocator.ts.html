  export function GetLatLonValue(
    mapObj,
    dataString,
    localeTwo,
    localeName,
    apiKey,
    updateLatLon: (lat, lon) => void,
  ): void {
    if (dataString === undefined || dataString === null) {
      dataString = $("#Address").val();
    }
    $.ajax({
      url: getGeoLocateURL(),
      data: {
        address: dataString,
        key: apiKey,
      },
      type: "GET",
      success: function(response) {
        const lonLatPos = convertGoogleResultsToLatLon(response, updateLatLon);
        if (lonLatPos !== null && lonLatPos !== undefined) {
          mapObj.AnimateOrFitToExtent(lonLatPos, 12);
        }
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }

  function convertGoogleResultsToLatLon(response, updateLatLon: (lat, lon) => void) {
    if (response.results !== null) {
      const addressItem = response.results[0];
      if (!jQuery.isEmptyObject(addressItem)) {
        if (addressItem.geometry !== null && addressItem.geometry.location !== null) {
          const latValue = addressItem.geometry.location.lat;
          const lonValue = addressItem.geometry.location.lng;
          const addressLoc = fromLonLat([lonValue, latValue]);
          if (updateLatLon !== undefined) {
            updateLatLon(latValue, lonValue);
          }
          return addressLoc;
        }
      }
    }
  }
 
  export function GetLatLonByMGRS(latTxt, lnTxt, writeOutMGRS) {
    if ($("#MGRS")) {
      const str = $("#MGRS").val();
      const usngConv = new usng.Converter();
      const latlng = usngConv.USNGtoLL(str, true);
 
      if (
        latlng !== undefined &&
        latlng.lat !== undefined &&
        latlng.lon !== undefined &&
        !isNaN(latlng.lat) &&
        !isNaN(latlng.lon)
      ) {
        if (latTxt !== null && lnTxt !== null) {
          latTxt.value(latlng.lat);
          lnTxt.value(latlng.lon);
        } else {
          //have to manually update MGRS
          writeOutMGRS(latlng.lat, latlng.lon, null);
        }
        placeMarker(true, null);
      } else {
        if (str.length > 6) {
          //alert maybe?
          $.notify({ message: "Invalid MGRS Value" }, { type: "danger" });
        }
      }
    }
  }

    long_name: string;
    short_name: string;

  function convertGoogleResultsToAddress(response): void {
    $("#Street1").val("");
    if (response.results !== null) {
      const addressItem = response.results[0];
      if (!jQuery.isEmptyObject(addressItem)) {
        $.each(addressItem.address_components, function(index, addyComp: IGoogleAddressType) {
          if ($.inArray("street_number", addyComp.types) > -1) {
            $("#Street1")
              .val(addyComp.long_name + $("#Street1").val())
              .change();
          } else if ($.inArray("route", addyComp.types) > -1) {
            $("#Street1")
              .val($("#Street1").val() + " " + addyComp.long_name)
              .change();
          } else if ($.inArray("neighborhood", addyComp.types) > -1) {
            $("#City")
              .val(addyComp.long_name)
              .change();
          } else if ($.inArray("locality", addyComp.types) > -1) {
            $("#City")
              .val(addyComp.long_name)
              .change();
          } else if ($.inArray("administrative_area_level_1", addyComp.types) > -1) {
            setStateProvinceValue(addyComp.short_name);
          } else if ($.inArray("postal_code", addyComp.types) > -1) {
            $("#Zip_PostalCode")
              .val(addyComp.long_name)
              .change();
          } else if ($.inArray("administrative_area_level_2", addyComp.types) > -1) {
            $("#County_Region")
              .val(addyComp.long_name)
              .change();
          } else if ($.inArray("country", addyComp.types) > -1) {
            const countryDrp = $("#Country").data("kendoDropDownList");
            if (addyComp.short_name.length > 0) {
              countryDrp.value(addyComp.short_name);
            } else {
              countryDrp.value("");
            }
          }
        });
        belgiumPostProcessing();
      } else {
        BlankThisAddress();
      }
    } else {
      BlankThisAddress();
    }
  }
 
  function belgiumPostProcessing(): void {
    const countryDrp = $("#Country").data("kendoDropDownList");
    if (countryDrp.value().length > 0 && countryDrp.value() === "BE") {
      const cityVal = $("#City").val();
      const provVal = getStateProvinceValue();
      if (
        cityVal.startsWith("Brussel") ||
        cityVal.startsWith(brusselsDefaultName) ||
        provVal.startsWith("Brussel") ||
        provVal.startsWith(brusselsDefaultName)
      ) {
        setStateProvinceValue(getBelgiumFormattedProvince(brusselsDefaultName));
      } else {
        setStateProvinceValue(getBelgiumFormattedProvince($("#County_Region").val()));
      }
    }
  }

  interface ILatLon {
    Latitude: string;
    Longitude: string;
  }
  function getCurrentLatLong(): ILatLon {
    const latBox = $("#Lat");
    const lonBox = $("#Lon");
    if (latBox.length === 0 && lonBox.length === 0) {
      // no lat/lon boxes on the page, just exit function
      // probably will never hit this case
      return;
    }
    let lat = latBox.val();
    let lon = lonBox.val();
 
    const kendoLat = $("#Lat").data("kendoNumericTextBox");
    if (kendoLat !== null && kendoLat !== undefined) {
      lat = kendoLat.value(); // to get the non-localized value
    }
    const kendoLon = $("#Lon").data("kendoNumericTextBox");
    if (kendoLon !== null && kendoLon !== undefined) {
      lon = kendoLon.value(); // to get the non-localized value
    }
    return {
      Latitude: lat,
      Longitude: lon,
    };
  }
  export function setLatLongOnPage(lat, lon) {
    if (typeof angular !== "undefined" && angular.element("#mapController").scope() !== undefined) {
      (<any>angular.element("#mapController").scope()).Lat = lat;
      (<any>angular.element("#mapController").scope()).Lon = lon;
      (<any>angular.element("#mapController").scope()).$apply();
    } else {
      const latTextBox = $("#Lat").data("kendoNumericTextBox");
      if (latTextBox !== undefined) {
        latTextBox.value(lat);
      }
 
      const lonTextBox = $("#Lon").data("kendoNumericTextBox");
      if (lonTextBox !== undefined) {
        lonTextBox.value(lon);
      }
    }
  }

  function convertWhtResultsToString(response) {
    $("#What3Words").val("");
    if (response !== null) {
      if (response.status !== null) {
        if (response.status.status === 200) {
          if (response.words && response.words.length > 0) {
            $("#What3Words")
              .val(response.words)
              .change();
          } else {
            NoValueFound();
          }
        } else {
          $("div#wht_msg").html("<h5>" + response.status.message + "</h5>");
          $("div#wht_msg").show("slow");
        }
      }
    }
  }
