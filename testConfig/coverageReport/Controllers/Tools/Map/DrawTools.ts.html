  function getColor(colorToTranslate) {
    if (colorToTranslate !== undefined && colorToTranslate.length > 0) {
      if (colorToTranslate.startsWith("#")) {
        return hexToRgbA(colorToTranslate, defaultAlphaValue);
      } else {
        return colorToTranslate;
      }
    }
    return undefined;
  }
 
  function SetDrawStyles(strokeColor, fillColor, strokeWidth, iconURL) {
    const defaultColor = hexToRgbA(defaultHexColor, defaultAlphaValue);
    if (strokeColor === undefined || strokeColor.length <= 0) {
      _strokeColor = defaultColor;
    } else {
      _strokeColor = getColor(strokeColor);
    }
    if (strokeWidth === undefined || strokeWidth.length <= 0) {
      _strokeWidth = 1;
    } else {
      _strokeWidth = strokeWidth;
    }
    if (fillColor === undefined || fillColor.length <= 0) {
      _fillColor = defaultColor;
    } else {
      _fillColor = getColor(fillColor);
    }
    if (iconURL === undefined || iconURL.length <= 0) {
      //do nothing, leave undefined
    } else {
      _iconURL = iconURL;
    }
 
    if (IsDrawing()) {
      for (const feat of drawSource.getFeatures()) {
        feat.setStyle(getDrawStyle());
      }
    }
  }

  function addInteraction(interactionType) {
    draw = new Draw({
      source: drawSource,
      type: interactionType,
    });
    let listener;
    pointerMoveKey = mapObj.on("pointermove", pointerMoveHandler);
    mouseOutKey = mapObj.getViewport().addEventListener("mouseout", function() {
      if (helpTooltipElement !== undefined) {
        helpTooltipElement.classList.add("hidden");
      }
    });
    draw.on(
      "drawstart",
      function(evt) {
        sketch = evt.feature;
        let tooltipCoord = evt.coordinate;
 
        listener = sketch.getGeometry().on("change", function(evtGeom) {
          let output;
          const geom = evtGeom.target;
          if (geom instanceof Polygon) {
            output = formatArea(geom);
            tooltipCoord = geom.getInteriorPoint().getCoordinates();
          } else if (geom instanceof CircleGeom) {
            const polyGeom = fromCircle(geom); //this could be a bad idea.  intensive?
            output = formatLength(polyGeom);
            tooltipCoord = polyGeom.getLastCoordinate();
          } else if (geom instanceof LineString) {
            output = formatLength(geom);
            tooltipCoord = geom.getLastCoordinate();
          }
          if (
            output !== undefined &&
            output.length > 0 &&
            measureTooltipElement !== undefined &&
            measureTooltipElement !== null
          ) {
            measureTooltipElement.innerHTML = output;
            measureTooltip.setPosition(tooltipCoord);
          }
        });
      },
      this,
    );
    draw.on(
      "drawend",
      function(e) {
        measureTooltipElement.className = "tooltip tooltip-static badge badge-warning";
        measureTooltip.setOffset([0, -7]);

        unByKey(listener);

          featGeom = fromCircle(e.feature.getGeometry());
        } else {
          featGeom = e.feature.getGeometry();
        }

  function formatLength(line) {
    const length = getLength(line);
    let output;
    if (Main.UseMetric()) {
      if (length > 100) {
        output = Math.round((length / 1000) * 100) / 100 + " " + "km";
      } else {
        output = Math.round(length * 100) / 100 + " " + "m";
      }
    } else {
      //in feet, do we switch to miles?
      const feet = length * 3.28084;
      if (feet > 5280) {
        //switch to miles
        const miles = feet / 5280;
        output = Math.round(miles * 100) / 100 + " " + "mi";
      } else {
        output = Math.round(feet * 100) / 100 + " " + "ft";
      }
    }
    return output;
  }
 
  function formatArea(polygon) {
    const area = getArea(polygon);
    let output;
    if (Main.UseMetric()) {
      if (area > 10000) {
        output = Math.round((area / 1000000) * 100) / 100 + " " + "km<sup>2</sup>";
      } else {
        output = Math.round(area * 100) / 100 + " " + "m<sup>2</sup>";
      }
    } else {
      const feetSq = area * 10.7639;
      if (feetSq > 107639.1) {
        const miSq = feetSq * 3.587e-8;
        output = Math.round(miSq * 100) / 100 + " " + "mi<sup>2</sup>";
      } else {
        output = Math.round(feetSq * 100) / 100 + " " + "ft<sup>2</sup>";
      }
    }
    return output;
  }
