function getFeaturesAtPixel(pixel) {
  const featuresAtPoint: IFeatureDetails[] = [];
  cobramap.mapObj.forEachFeatureAtPixel(
    pixel,
    function(feature, layer) {
      const tmpFeature = cobramap.GetFeatureDetails(feature);
      if (
        tmpFeature !== undefined &&
        tmpFeature.PrimaryKey !== undefined &&
        tmpFeature.PrimaryKey.length > 0 &&
        !FeatureIsHidden(tmpFeature.PrimaryKey)
      ) {
        featuresAtPoint.push(tmpFeature);
      } else if (layer !== undefined && layer.LogEntryGUID !== undefined) {
        //this is from the KML layer, we get the data in another way.
        feature.PrimaryKey = layer.LogEntryGUID;
        feature.DataType = FeatureDataType.LogEntryGUID;
        featuresAtPoint.push(feature);
      }
    },
    {
      hitTolerance: cobramap.hitTolerance,
      layerFilter: function(layer) {
        return layer.CanSelect;
      },
    },
  );
  return featuresAtPoint;
}
 
function processFoundFeatures(featuresFound: IFeatureDetails[], clickCoords) {
  if (featuresFound.length > 0) {
    //this is kinda overkill.  i'd like to just use featuresfound, but KML features get lost that way.
    let allFeatures = [];
    for (const featureDet of featuresFound) {
      if (featureDet !== undefined) {
        const features = getFeaturesByPrimaryKey(featureDet.DataType, featureDet.PrimaryKey, featureDet.SecondaryKey);
        if (features !== undefined && features.length > 0) {
          allFeatures = allFeatures.concat(features);
        }
      }
    }
    showFeatureDetails(allFeatures, clickCoords);
  } else {
    //give a second chance.  WFS/WMS don't show up as features
    let wmsDetFound: boolean = false;
    if (wfswmsGridSource !== undefined && wfswmsGridSource.total() > 0) {
      const viewResolution = cobramap.GetViewResolution();
 
      cobramap.GetLayers("WMSLayerGUID").forEach(function(lyr) {
        if (!FeatureIsHidden(lyr.WMSLayerGUID)) {
          const src = lyr.getSource();
          const con = containsXY(lyr.WMSExtent, clickCoords[0], clickCoords[1]);
          if (con) {
            if (!lyr.IsWFS) {
              const url = src.getGetFeatureInfoUrl(clickCoords, viewResolution, "EPSG:3857", {
                INFO_FORMAT: "text/html",
              });
              if (url) {
                const parsedURI = new URI(url);
                wmsDetFound = true;
                document.getElementById("properties").innerHTML =
                  '<iframe seamless style="width:100%" src="' + parsedURI.toString() + '"></iframe>';
                $("#detailCard").show();
              }
            } else {
              const layerSource = lyr.getSource();
              const wfsFeatures = layerSource.getFeaturesAtCoordinate([clickCoords[0], clickCoords[1]]);
              if (wfsFeatures !== undefined && wfsFeatures.length > 0) {
                const featureDetails = cobramap.GetFeatureAsArrayOfString(wfsFeatures, false, clickCoords);
                $("#properties").html(featureDetails.join(""));
                $("#detailCard").show();
                wmsDetFound = true;
              }
            }
          }
        }
      });
    }
    if (!wmsDetFound) {
      showFeatureDetails(null, null);
    }
  }
}

      }),
    );
    LoadOrgDetails(
      features.filter(function(item) {
        return item.get("RootOrganization") !== undefined;

  }
}

export function ZoomToFeatExtent(propertyToSearchFor) {
  let featExtent = null;
  let zoomLevl = 0;
  if (propertyToSearchFor === null || propertyToSearchFor === undefined) {
    //i guess this is just me saying 'if nothing is specified, then just zoom to the extent of the incident entries'
    if (featEntrySource !== null && featEntrySource !== undefined) {
      featExtent = featEntrySource.getExtent();
    }
  } else {
    //this is the normal case
    if (propertyToSearchFor === "WMSLayerGUID") {
      cobramap.GetLayers("WMSLayerGUID").forEach(function(lyr) {
        if (!FeatureIsHidden(lyr.WMSLayerGUID)) {
          if (featExtent === null) {
            featExtent = lyr.WMSExtent;
          } else {
            extend(featExtent, lyr.WMSExtent);
          }
        }
      });
    } else {
      let features;
      if (
        propertyToSearchFor === "LogEntryGUID" ||
        propertyToSearchFor === "ProjectGUID" ||
        propertyToSearchFor === "RootOrganization"
      ) {
        features = featEntrySource.getFeatures();
      } else if (propertyToSearchFor === "FacilityGUID") {
        features = facSource.getFeatures();
      } else if (propertyToSearchFor === "BackgroundLayerGUID") {
        features = [];
        cobramap.GetLayers("BackgroundLayerGUID").forEach(function(lyr) {
          features = features.concat(lyr.getSource().getFeatures());
        });
      }
      zoomLevl = defaultZoomLevel;
      for (const feature of features) {
        if (feature.get(propertyToSearchFor) !== undefined) {
          if (featExtent === null) {
            featExtent = feature.getGeometry().getExtent();
          } else {
            zoomLevl = 0;
            extend(featExtent, feature.getGeometry().getExtent());
          }
        }
      }
    }
  }
  if (featExtent !== null) {
    cobramap.AnimateOrFitToExtent(featExtent, zoomLevl);
  }
}

export function GetBestKMLDescriptor(kmlEntry) {
  if (kmlEntry !== undefined) {
    if (kmlEntry.name !== undefined) {
      return kmlEntry.name;
    }
    const props = Object.getOwnPropertyNames(kmlEntry);
    if (props !== undefined) {
      for (const prop in props) {
        if (props.hasOwnProperty(prop)) {
          if (prop !== "id" && props[prop].length > 0) {
            return props[prop];
          }
        }
      }
    }
  }
}
 
function getKMLFeaturesMinimized(kmlFeatures) {
  const ret = new kendo.data.ObservableArray([]);
  if (kmlFeatures != null && kmlFeatures !== undefined) {
    for (const kmlFeature of kmlFeatures) {
      if (kmlFeature.id_ === undefined) {
        kmlFeature.setId(Guid.newGuid().toString(undefined));
      }
      const props = kmlFeature.getProperties();
      if (props !== undefined) {
        const translatedFeature = new kendo.data.ObservableObject();
        for (const prop in props) {
          if (props.hasOwnProperty(prop)) {
            if (
              prop !== "bbox" &&
              prop !== "geometry" &&
              prop !== "styleUrl" &&
              prop !== "marker-symbol" &&
              props[prop].length > 0
            ) {
              const val = props[prop];
              translatedFeature.set(prop, val);
            }
          }
        }
        translatedFeature.set("id", kmlFeature.id_);
        ret.push(translatedFeature);
      }
    }
  }
  return ret;
}

      } else if (!makeHidden && FeatureIsHidden(lyr[propertyToSearchFor])) {
        hiddenFeatures.splice(index, 1);
      }
      if (propertyToSearchFor === "WMSLayerGUID") {
        swapShowHideButtonInListView(lyr[propertyToSearchFor], makeHidden);
      }
      lyr.setVisible(!makeHidden);
    });
    spinObject(showHideBtn, true);
  }
}
 
function getFeaturesByPrimaryKey(itemType: FeatureDataType, primaryKey: string, secondaryKey: string): any[] {
  if (primaryKey !== undefined || secondaryKey !== undefined) {
    let retFeatures: any[] = [];
    if (itemType === FeatureDataType.WMSLayerGUID) {
      cobramap.GetLayers("WMSLayerGUID").forEach(function(lyr) {
        if (lyr.WMSLayerGUID === primaryKey && lyr.IsWFS) {
          retFeatures = lyr.getSource().getFeatures();
        }
      });
    } else if (itemType === FeatureDataType.FacilityGUID) {
      retFeatures = facSource.getFeatures().filter(function(item) {
        const featKey = item.get("FacilityGUID");
        return featKey !== undefined && featKey === primaryKey;
      });
    } else if (itemType === FeatureDataType.BackgroundLayerGUID) {
      if (secondaryKey !== undefined && secondaryKey !== null) {
        cobramap.GetLayers("BackgroundLayerGUID").forEach(function(lyr) {
          if (lyr.BackgroundLayerGUID === secondaryKey) {
            retFeatures = lyr
              .getSource()
              .getFeatures()
              .filter(function(item) {
                const featKey = item.get("BackgroundFeatureGUID");
                return featKey !== undefined && featKey === primaryKey;
              });
          }
        });
      } else {
        cobramap.GetLayers("BackgroundLayerGUID").forEach(function(lyr) {
          if (lyr.BackgroundLayerGUID === primaryKey) {
            retFeatures = lyr.getSource().getFeatures();
          }
        });
      }
    } else if (itemType === FeatureDataType.LogEntryGUID) {
      if (secondaryKey !== undefined) {
        //this items is a sub item underneath the logentries.  probably from kml.  need to find a clean way to handle these generally
        //https://gis.stackexchange.com/questions/298797/find-feature-by-property-from-multiple-layers
        cobramap.GetLayers("LogEntryGUID").forEach(function(lyr) {
          if (lyr.LogEntryGUID === secondaryKey) {
            //const tmpFeatures =lyr.getSource().getFeatures();
            retFeatures = lyr
              .getSource()
              .getFeatures()
              .filter(function(item) {
                const featKey = item.id_;
                return featKey !== undefined && featKey === primaryKey;
              });
          }
        });
      } else {
        const features = featEntrySource.getFeatures();
        for (const feature of features) {
          if (feature.get("LogEntryGUID") === primaryKey) {
            const kmlURL = feature.get("KMLUrl");
            if (kmlURL !== undefined && kmlURL.length > 0) {
              let kmlFeatures = null;
              cobramap.GetLayers("LogEntryGUID").forEach(function(lyr) {
                if (primaryKey === lyr.LogEntryGUID) {
                  kmlFeatures = lyr.getSource().getFeatures();
                  return;
                }
              });
              if (kmlFeatures !== null) {
                setLogEntryFeaturePropertiesOnKMLFeature(feature, kmlFeatures);
                return kmlFeatures;
              }
            }
            retFeatures.push(feature);
          }
        }
      }
    }
    return retFeatures;
  }
}
 
function setLogEntryFeaturePropertiesOnKMLFeature(logEntryFeature, kmlFeatures) {
  if (logEntryFeature !== undefined && kmlFeatures !== undefined && kmlFeatures.length > 0) {
    const props = logEntryFeature.getProperties();
    if (props !== undefined) {
      for (const kmlFeature of kmlFeatures) {
        for (const prop in props) {
          if (props.hasOwnProperty(prop)) {
            if (prop !== "geometry" && props[prop].length > 0) {
              //TODO:here i should get to see if a value already exists, and it's a popular value.  like description?
              kmlFeature.set(prop, props[prop]);
            }
          }
        }
      }
    }
  }
}
 
export function ZoomPan(fitToExtent: boolean, itemType: FeatureDataType, primaryKey: string, secondaryKey: string) {
  if (FeatureIsHidden(primaryKey) || FeatureIsHidden(secondaryKey)) {
    return;
  }
  const features = getFeaturesByPrimaryKey(itemType, primaryKey, secondaryKey);
  if (features !== undefined && features.length > 0) {
    if (features.length === 1) {
      const geom = features[0].getGeometry();
      const geomType = geom.getType();
      if (geomType === "Point") {
        //this is a point then, not a polygon
        cobramap.AnimateOrFitToExtent(geom.getExtent(), defaultZoomLevel);
        return;
      }
    }
 
    let featExtent = null;
    for (const feature of features) {
      if (featExtent === null) {
        featExtent = feature.getGeometry().getExtent();
      } else {
        extend(featExtent, feature.getGeometry().getExtent());
      }
    }
 
    var zoomLevel = defaultZoomLevel;
    if (fitToExtent) {
      zoomLevel = 0;
    }
    if (featExtent !== null) {
      cobramap.AnimateOrFitToExtent(featExtent, zoomLevel);
    }
  } else if (itemType === FeatureDataType.WMSLayerGUID) {
    cobramap.GetLayers("WMSLayerGUID").forEach(function(lyr) {
      if (lyr.WMSLayerGUID === primaryKey && !lyr.IsWFS) {
        cobramap.AnimateOrFitToExtent(lyr.WMSExtent, zoomLevel);
        return;
      }
    });
  }
}

function loadBackgroundLayersIntoMap(response: IBackgroundLayerDTO[]) {
  const loadedLayers: IBackgroundLayerDTO[] = [];
  blockIt($("#backListWrapper"));
  for (const entry of response) {
    //TODO: here i should check if the layer already exists, and remove it?
    if (entry.Features !== undefined && entry.Features.features !== undefined && entry.Features.features.length > 0) {
      const bckSource = new VectorSource({
        features: new GeoJSON({ featureProjection: "EPSG:3857" }).readFeatures(entry.Features),
      });
      const bckVector = new VectorLayer({
        source: bckSource,
        style: getStyleForFeature,
        maxResolution: cobramap.GetResolutionForZoom(parseInt(entry.MinZoomLevel, 10)),
        minResolution: cobramap.GetResolutionForZoom(parseInt(entry.MaxZoomLevel, 10)),
      });
 
      (<any>bckVector).BackgroundLayerGUID = entry.BackgroundLayerGUID;
      (<any>bckVector).BackgroundLayerName = entry.BackgroundLayerName;
      (<any>bckVector).BackExtent = bckSource.getExtent();
      bckVector.setVisible(!FeatureIsHidden(entry.BackgroundLayerGUID));
      //bckVector.setZIndex(99);
      cobramap.AddLayer(bckVector);
 
      loadedLayers.push(entry);
    }
  }
  unblockit($("#backListWrapper"));
  LoadBackgroundLayerGrid(loadedLayers);
}

function loadWMSWFSIntoMap(response: IWMSWFSDTO[]) {
  const loadedLayers: IWMSWFSDTO[] = [];
  blockIt($("#wmswfsListWrapper"));
  for (const entry of response) {
    //TODO: here i should check if the layer already exists, and remove it?
    if (entry.IncludedLayers !== undefined && entry.IncludedLayers.length > 0) {
      if (FeatureIsHidden(entry.WMSLayerGUID)) {
        //do nothing
      } else {
        const layerNames = entry.IncludedLayers.map(a => a.Name);
        if (entry.IsWFS) {
          //https://openlayers.org/en/latest/examples/vector-wfs-getfeature.html
          const wfsURL = new URI(entry.URL);
          wfsURL.search("");
          entry.Extent = getExtentOfWMSLayers(entry.IncludedLayers, false);
          const featureRequest = new WFS().writeGetFeature({
            srsName: "EPSG:3857",
            //featureNS: 'http://openstreemap.org',
            //featurePrefix: 'osm',
            featureTypes: layerNames,
            outputFormat: "application/json",
          });
 
          fetch(wfsURL.toString(), {
            method: "POST",
            body: new XMLSerializer().serializeToString(featureRequest),
          })
            .then(function(wfsCallResponse) {
              return wfsCallResponse.json();
            })
            .then(function(json) {
              const features = new GeoJSON().readFeatures(json);
              const wfsSource = new VectorSource();
              wfsSource.addFeatures(features);
              const wfsVector = new VectorLayer({
                source: wfsSource,
                //style: getStyleForFeature
                maxResolution: cobramap.GetResolutionForZoom(parseInt(entry.MinZoomLevel, 10)),
                minResolution: cobramap.GetResolutionForZoom(parseInt(entry.MaxZoomLevel, 10)),
              });
 
              (<any>wfsVector).WMSLayerGUID = entry.WMSLayerGUID;
              (<any>wfsVector).IsWFS = entry.IsWFS;
              (<any>wfsVector).WMSExtent = wfsSource.getExtent();
              wfsVector.setVisible(!FeatureIsHidden(entry.WMSLayerGUID));
              cobramap.AddLayer(wfsVector);
            });
        } else {
          ////https://openlayers.org/en/latest/apidoc/module-ol_source_TileWMS-TileWMS.html
          const tileWMSOptions = {
            url: entry.URL, //+ "?SERVICE=WMS&REQUEST=Getcapabilities", //&version=1.3.0",
            params: { LAYERS: layerNames.toString(), TILED: true },
            //crossOrigin: "Anonymous"
            //projection: 'EPSG:4326',
            //projection: 'EPSG:3857',
            //serverType: 'geoserver'
            //entry.WMSLayerName,
          };
          const layer = new TileLayer({
            source: new TileWMS(tileWMSOptions),
            //note this is switched.  weird huh?  Guess it was just a misunderstanding on how it was written up for thinkgeo vs openlayers
            maxResolution: cobramap.GetResolutionForZoom(parseInt(entry.MinZoomLevel, 10)),
            minResolution: cobramap.GetResolutionForZoom(parseInt(entry.MaxZoomLevel, 10)),
          });
          entry.Extent = getExtentOfWMSLayers(entry.IncludedLayers, false);
          (<any>layer).WMSLayerGUID = entry.WMSLayerGUID;
          (<any>layer).IsWFS = entry.IsWFS;
          (<any>layer).WMSExtent = entry.Extent;
          layer.setVisible(!FeatureIsHidden(entry.WMSLayerGUID));
          cobramap.AddLayer(layer);
        }
      }
 
      //if (isNaN(entry.Extent[0]) || isNaN(entry.Extent[1]) || isNaN(entry.Extent[2]) || isNaN(entry.Extent[3])) {
      //  entry.Extent = undefined;
      //}
      loadedLayers.push(entry);
    }
  }
  unblockit($("#wmswfsListWrapper"));
  LoadWMSWFSGrid(loadedLayers);
}
 
function getExtentOfWMSLayers(layers: IWMSWFSLayerDTO[], isWFS: boolean) {
  if (layers != null && layers.length > 0) {
    if (layers.length === 1) {
      return getExtentOfWMSLayer(layers[0], false);
    } else {
      let featExtent = null;
      for (const layer of layers) {
        const tmpExtent = getExtentOfWMSLayer(layer, false);
        if (tmpExtent !== null) {
          if (featExtent === null) {
            featExtent = tmpExtent;
          } else {
            extend(featExtent, tmpExtent);
          }
        }
      }
      return featExtent;
    }
  }
}
function getExtentOfWMSLayer(layer: IWMSWFSLayerDTO, invertExt: boolean) {
  if (layer !== undefined) {
    //should be [minx, miny, maxx, maxy]
    let ext: [number, number, number, number];
    //hate this, but can't figure out where in the WMS it says that you need to apply something like this
    //thought if i specified the version in the wms url, it would take care of it, but i do that admin side and no luck'
    //some tests should be placed here to see if we still need this invert, or it was just my beginning bungling.
    //http://docs.geoserver.org/latest/en/user/services/wms/basics.html
    ext = invertExt
      ? [parseFloat(layer.MinY), parseFloat(layer.MinX), parseFloat(layer.MaxY), parseFloat(layer.MaxX)]
      : [parseFloat(layer.MinX), parseFloat(layer.MinY), parseFloat(layer.MaxX), parseFloat(layer.MaxY)];
    //console.log(layer.Name);
    //console.log(layer.EPSG);
    //console.log(ext);
    if (layer.EPSG === null || layer.EPSG === undefined || layer.EPSG === "3857") {
      return ext;
    }
    if (layer.EPSG === "84") {
      //was getting this on some layers.  seems like a standardization issue
      layer.EPSG = "4326"; //http://spatialreference.org/ref/epsg/wgs-84/
    }
    try {
      const transExt = transformExtent(ext, getProjection("EPSG:" + layer.EPSG), getProjection("EPSG:3857"));
      //console.log(transExt);
      if (isNaN(transExt[0]) || isNaN(transExt[1]) || isNaN(transExt[2]) || isNaN(transExt[3])) {
        if (!invertExt) {
          return getExtentOfWMSLayer(layer, true);
        } else {
          return ext; //we've tried both ways.  give up.  maybe pass null?
        }
      }
      return transExt;
    } catch (err) {
      console.log("Could not find extent for EPSG " + layer.EPSG);
      return ext;
    }
  }
}

export function ShowHideFeature(itemType: FeatureDataType, itemGUID, secondaryKey) {
  //https://gis.stackexchange.com/questions/270387/show-hide-features-on-layer-in-openlayers
  if (itemType !== undefined) {
    const btnToSpin = "btn" + itemGUID;
    spinObject(btnToSpin, false);
    const index = hiddenFeatures.indexOf(itemGUID);
    if (index > -1) {
      hiddenFeatures.splice(index, 1);
    } else {
      hiddenFeatures.push(itemGUID);
    }
    if (itemType === FeatureDataType.WMSLayerGUID) {
      cobramap.GetLayers("WMSLayerGUID").forEach(function(lyr) {
        if (itemGUID === lyr.WMSLayerGUID) {
          lyr.setVisible(!FeatureIsHidden(itemGUID));
        }
      });
    } else if (itemType === FeatureDataType.BackgroundLayerGUID) {
      if (secondaryKey === undefined || secondaryKey === null) {
        cobramap.GetLayers("BackgroundLayerGUID").forEach(function(lyr) {
          if (itemGUID === lyr.BackgroundLayerGUID) {
            lyr.setVisible(!FeatureIsHidden(itemGUID));
          }
        });
      } else {
        RedrawFeatures(itemGUID, itemType, secondaryKey);
      }
    } else {
      RedrawFeatures(itemGUID, itemType, secondaryKey);
      if (itemType === FeatureDataType.LogEntryGUID) {
        cobramap.GetLayers("LogEntryGUID").forEach(function(lyr) {
          if (itemGUID === lyr.LogEntryGUID) {
            lyr.setVisible(!FeatureIsHidden(itemGUID));
          }
        });
      }
    }
    spinObject(btnToSpin, true);
    swapShowHideButtonInListView(itemGUID, FeatureIsHidden(itemGUID));
  }
  return false;
}

}
 
function RedrawFeaturesByFeatures(featuresToRedraw) {
  //whether or not the feature is hidden has already been saved, this is just redrawing.
  if (isRedrawing) {
    return;
  }
  isRedrawing = true;
  if (featuresToRedraw !== undefined && featuresToRedraw.length > 0) {
    for (const feat of featuresToRedraw) {
      const featDet = cobramap.GetFeatureDetails(feat);
      feat.setStyle(cobramap.GetStyleForFeatureWithDet(feat, featDet, hiddenFeatures));
    }
  }
  isRedrawing = false;
}
export function RedrawFeatures(itemGUID, dataType: FeatureDataType, secondaryKey) {
  let features;
  if (itemGUID !== undefined && itemGUID !== null) {
    features = getFeaturesByPrimaryKey(dataType, itemGUID, secondaryKey);
  } else {
    //redraw everything for that datatype
    switch (dataType) {
      case FeatureDataType.LogEntryGUID:
      case FeatureDataType.ProjectGUID:
      case FeatureDataType.RootOrganization:
        features = featEntrySource.getFeatures();
        break;
      case FeatureDataType.FacilityGUID:
        features = facSource.getFeatures();
        break;
      case FeatureDataType.BackgroundLayerGUID:
        features = [];
        cobramap.GetLayers("BackgroundLayerGUID").forEach(function(lyr) {
          features = features.concat(lyr.getSource().getFeatures());
        });
        break;
    }
  }
  RedrawFeaturesByFeatures(features);
}
