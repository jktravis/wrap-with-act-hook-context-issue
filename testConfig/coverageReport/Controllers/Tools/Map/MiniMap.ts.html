  export function LoadFeatureEntries() {
    let geoJsonUrl = _globalPrompts.baseMapAPI;
    if (
      _globalPrompts.ProjectGUID !== undefined &&
      _globalPrompts.ProjectGUID !== null &&
      _globalPrompts.ProjectGUID !== Guid.emptyGuidString
    ) {
      geoJsonUrl =
        geoJsonUrl +
        "ExportIncidentAsGeoJSON/" +
        _globalPrompts.ProjectGUID +
        "?MiniMapOnly=true&IncludeTool=" +
        _globalPrompts.IncludeTool;
    } else {
      geoJsonUrl = geoJsonUrl + "ExportOrganizationAsGeoJSON?orgName=" + _globalPrompts.RootOrganization;
    }
    featEntrySource = new VectorSource({
      format: new GeoJSON(),
      url: geoJsonUrl,
    });
 
    featEntryLayer = new VectorLayer({
      source: featEntrySource,
      style: getStyleForFeature,
    });
    const listenerKey = featEntrySource.on("change", function(e) {
      if (featEntrySource.getState() === "ready") {
        const features = featEntrySource.getFeatures();
        if (features !== undefined && features.length > 0) {
          const uri = new URI(window.location);
          if (!uri.hasQuery("CurrentMapZoom")) {
            let alreadyZoomed = false;
            if (features.length === 1) {
              //
              const geom = features[0].getGeometry();
              const geomType = geom.getType();
              if (geomType === "Point") {
                //this is a point then, not a polygon
                cobramap.AnimateOrFitToExtent(geom.getExtent(), defaultZoomLevel);

              cobramap.AnimateOrFitToExtent(featEntrySource.getExtent(), 0);
            }
          }
          LoadKMLFeatures(
            features.filter(function(item) {
              return item.get("KMLUrl") !== undefined;
            }),
          );
        }
      }
    });

  function writeOutMGRS(latVal, lonVal, updateTextBoxes: boolean) {
    if (usngConv !== null && usngConv !== "undefined") {
      if (updateTextBoxes === null || updateTextBoxes === undefined) {
        updateTextBoxes = true;
      }
      let mg = usngConv.LLtoUSNG(latVal, lonVal, 5); //3rd value is 5 in CoBARTS stuff
      /* NOT doing this anymore.  think it's confusing the with address acting differently.
      const mgRS = $('#MGRS');
      if (mgRS.length) {
        mgRS.val(mg);
      }*/
      if (updateTextBoxes) {
        mg = mg.replace(/\s+/g, ""); //remove all spaces in the response
        if (mg !== null && mg.length > 0) {
          const znTxt = getGridZoneObj();
          const weTxt = getGridWEObj();
          const nsTxt = getGridNSObj();

            weTxt.val(mg.substr(5, 4));
            nsTxt.val(mg.substr(9, 4));
          } else {
            znTxt.val(mg);
          }
        }
      }
    }
  }

  export function GetDistanceBetweenLocations(
    latValA: number,
    lonValA: number,
    latValB: number,
    lonValB: number,
    useMetric: boolean = false,
  ): IDistanceReturn {
    if (lonValA !== null && lonValB !== null && latValA !== null && latValB !== null) {
      const points = [[lonValA, latValA], [lonValB, latValB]];
      const distanceLine = new LineString(points);
      //https://openlayers.org/en/latest/apidoc/module-ol_sphere.html#.getLength
      const distance = getLength(distanceLine, { projection: "EPSG:4326" });
 
      return convertDistanceMeasurementToOutputUnits(distance, useMetric);
    }
    return null;
  }
  interface IDistanceReturn {
    distance: number;
    units: string;
  }
  function convertDistanceMeasurementToOutputUnits(distance: number, useMetric: boolean): IDistanceReturn {
    const rtnObj: IDistanceReturn = {
      distance: distance,
      units: "m",
    };
    if (!useMetric) {
      //in feet, do we switch to miles?
      const feet = distance * 3.28084;
      if (feet > 5280) {
        //switch to miles
        distance = feet / 5280;
        rtnObj.distance = Math.round(distance); //get nearest mile
        rtnObj.units = "mi";
      } else {
        rtnObj.distance = Math.round(distance / 10) * 10; //round to 10
        rtnObj.units = "ft";
      }
    } else {
      if (distance > 1000) {
        //greater than 1 meter
        rtnObj.distance = distance / 1000;
        rtnObj.units = "km";
      }
    }
    return rtnObj;
  }

    }
    //find a way to remove past KML layers

      format: new KML(),
    });
    const kmlVector = new VectorLayer({
      source: kmlSource,
    });
    kmlVector.IsKML = true;
    const listenerKey = kmlSource.on("change", function(e) {
      if (kmlSource.getState() === "ready") {

        if (kmlFeatures !== undefined && kmlFeatures.length > 0) {
          cobramap.AnimateOrFitToExtent(kmlSource.getExtent(), 0);
        }
      }
    });
    cobramap.AddLayer(kmlVector);
  }

    if (orgLayer) {

    orgSource = new VectorSource({

      url: orgURL,
    });
 
    orgLayer = new VectorLayer({
      source: orgSource,
      style: getStyleForMultiOrgs,
    });
    const listenerKey = orgSource.on("change", function(e) {
      if (orgSource.getState() === "ready") {
        const features = orgSource.getFeatures();
        if (features !== undefined && features.length > 0) {
          const uri = new URI(window.location);
          if (!uri.hasQuery("CurrentMapZoom")) {
            const prjLatLon = getDefaultLonLat();
            if (prjLatLon) {
              const baselonLatInWebMercator = fromLonLat(prjLatLon);
              cobramap.AnimateOrFitToExtent(baselonLatInWebMercator, defaultZoomLevel);
            } else {
              cobramap.AnimateOrFitToExtent(orgSource.getExtent(), 0);
            }
          }
        }
      }
    });
    cobramap.AddLayer(orgLayer);
  }

  export function getWKTFromMapExtent() {
    const mapExt = cobramap.mapView.calculateExtent();
    const poly = fromExtent(mapExt);
    const wktFormat = new WKT();
    const ret = wktFormat.writeGeometry(poly, {
      dataProjection: "EPSG:4326",
      featureProjection: "EPSG:3857",
    });
    return ret;
  }
