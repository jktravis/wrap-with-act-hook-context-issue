  private defaultZoom = 10;
 
  private buildMapData(defaultBaseMap, defaultLonLat: [number, number], target = "map") {
    let baselonLat: [number, number] = defaultLonLat;
    if (baselonLat === undefined) {
      baselonLat = [-95.7129, 37.0902]; //just whereever.  i need an initial spot to move from later on.  this is the middle of america
    }
    const baselonLatInWebMercator = fromLonLat(baselonLat);
 
    this.mapView = new View({
      center: baselonLatInWebMercator,
      zoom: this.defaultZoom,
    });
 
    this.mapView.on("propertychange", function(e) {
      switch (e.key) {
        case "resolution":
          $("#zoomvalue").text(e.oldValue);
          break;
      }
    });
 
    const wmtsLayer = new TileLayer({
      source: this.getMapDataSource(defaultBaseMap, null),
    });
 
    (<any>wmtsLayer).LayerType = defaultBaseMap;
    this.mapObj = new Map({
      controls: defaultControls().extend([new OverviewMap()]),
      target,
      layers: [wmtsLayer],
      view: this.mapView,
    });
 
    this.mapObj.on("pointerdrag", () => {
      this.mapObj.getViewport().style.cursor = "grabbing";
    });
 
    this.mapObj.on("pointerup", () => {
      this.mapObj.getViewport().style.cursor = "";
    });
 
    this.mapObj.isLoaded = false;
 
    const mousePositionControl = new MousePosition({
      coordinateFormat: this.createStringXYCustom,
      projection: "EPSG:4326",
      // comment the following two lines to have the mouse position
      // be placed within the map.
      className: "custom-mouse-position",
      target: document.getElementById("mouse-position"),
      undefinedHTML: "&nbsp;",
    });
    this.mapObj.addControl(mousePositionControl);
  }
 
  private createStringXYCustom = coordinate => {
    let ret = "<div><label class='badge badge-secondary'>" + toStringXY(coordinate, 4) + "</label></div>";
    if (Main.serverBuildType !== "Belgium" && this.usngConv !== undefined) {
      const mg = this.usngConv.LLtoUSNG(coordinate[1], coordinate[0], 5);
      ret = ret + "<div><label class='badge badge-secondary'>" + mg + "</label></div>";
    }
    return ret;
  };

  public ChangeWMTS(mapType: WMTSMapType) {
    if (this.mapObj !== undefined && this.mapObj.isLoaded) {
      let layerFound = false;
      const existingLayers = this.mapObj.getLayers();
      existingLayers.forEach(function(lyr) {
        if (lyr !== undefined && lyr.LayerType !== undefined) {
          if (lyr.LayerType === mapType) {
            layerFound = true;
            lyr.setVisible(true);
          } else {
            lyr.setVisible(false);
          }
        }
      });
      if (!layerFound) {
        const dataSource = this.getMapDataSource(mapType, existingLayers);
        if (dataSource !== null) {
          this.addWMTSFromSource(dataSource, mapType, existingLayers);
        }
      }
    }
  }
 
  private addWMTSFromSource(datasource, mapType: WMTSMapType, existingLayers) {
    const wmtsLayer = new TileLayer({ source: datasource });
    (<any>wmtsLayer).LayerType = mapType;
    this.mapObj.addLayer(wmtsLayer);
    let iWMTSCount = 0;
    let iOtherCount = 0;
    existingLayers.forEach(function(lyr) {
      if (lyr !== undefined && lyr.LayerType !== undefined) {
        lyr.setZIndex(iWMTSCount);
        iWMTSCount++;
      } else {
        lyr.setZIndex(iOtherCount + 10); //this maybe causes problems
        iOtherCount++;
      }
    });
  }
 
  //https://gis.stackexchange.com/questions/231629/openlayers-v4-0-1-support-google-maps-javascript-api
  private getMapDataSource(mapType, existingLayers) {
    switch (parseInt(mapType, 10)) {
      case WMTSMapType.GoogleRoads:
        return new TileImage({
          //Google Road Names
          //url: "https://mt1.google.com/vt/lyrs=h&x={x}&y={y}&z={z}",
          //Google Road Map
          url: "https://mt1.google.com/vt/lyrs=m&x={x}&y={y}&z={z}",
          projection: "EPSG:3857",
          crossOrigin: "Anonymous", //https://stackoverflow.com/questions/22710627/tainted-canvases-may-not-be-exported
        });
      case WMTSMapType.GoogleHybrid:
        return new TileImage({
          url: "https://mt1.google.com/vt/lyrs=y&x={x}&y={y}&z={z}",
          projection: "EPSG:3857",
          crossOrigin: "Anonymous",
        });
      case WMTSMapType.GoogleSat:
        return new TileImage({
          url: "https://mt1.google.com/vt/lyrs=s&hl=pl&&x={x}&y={y}&z={z}",
          projection: "EPSG:3857",
          crossOrigin: "Anonymous",
        });
      //Google Terrain & Roads
      //url: 'https://mt1.google.com/vt/lyrs=p&x={x}&y={y}&z={z}'
      //Google Road without Building
      //url: 'https://mt1.google.com/vt/lyrs=r&x={x}&y={y}&z={z}'
      //Google Terrain
      //url: 'https://mt1.google.com/vt/lyrs=t&x={x}&y={y}&z={z}'
      case WMTSMapType.NGI: // https://openlayers.org/en/latest/examples/wmts-layer-from-capabilities.html
        fetch(this._ngiURL)
          .then(function(response) {
            return response.text();
          })
          .then(function(text) {
            const wmtsparser = new WMTSCapabilities();
            const result = wmtsparser.read(text);
            const options = optionsFromCapabilities(result, {
              layer: "topo",
              //matrixSet: '3857',
              //layer: 'overlay',
              //matrixSet: '3812',
              projection: "EPSG:3857",
              crossOrigin: "Anonymous",
            });
            this.addWMTSFromSource(new WMTS(options), mapType, existingLayers);
          });
        return null;
      case WMTSMapType.Thinkgeo:
      case WMTSMapType.OSM:
      default:
        return new OSM(); //built in source provider for OSM
    }
  }

  public GetStyleForFeature(feature, hiddenFeatures) {
    const featDet = this.GetFeatureDetails(feature);
    return this.GetStyleForFeatureWithDet(feature, featDet, hiddenFeatures);
  }
 
  public GetStyleForFeatureWithDet(feature, featDet, hiddenFeatures) {
    if (
      hiddenFeatures !== undefined &&
      hiddenFeatures.length > 0 &&
      (hiddenFeatures.indexOf(featDet.PrimaryKey) > -1 || hiddenFeatures.indexOf(featDet.SecondaryKey) > -1)
    ) {
      return null;
    }
    //styling for KML will be handled by the KML itself.
    const kmlURL = feature.get("KMLUrl");
    if (kmlURL !== undefined && kmlURL.length > 0) {
      return null;
    }
 
    const featColor = this.getColorForFeature(feature);
 
    const styleRet = new Style({
      stroke: new Stroke({
        color: featColor.strokeColor,
        width: featColor.strokeWidth,
      }),
      fill: new Fill({
        color: featColor.fillColor,
      }),
      image: this.getIconStyle(feature, featColor),
      text: this.getTextStyle(feature),
    });
 
    return styleRet;
  }

  private getTextStyle(feature) {
    if ($("#showMapText").is(":checked")) {
      if (feature !== null && feature !== undefined) {

        if (title !== undefined && title.length > 0) {
          if (feature === this.selectedFeature) {

              fill: new Fill({

              }),
              text: title,
            });
          }
        }
      }
    }
    return null;
  }

  private getIconStyle(feature, featColor: IFeatureColor) {
    const iconURL = feature.get("marker-symbol");
    if (iconURL !== undefined && iconURL !== null && iconURL.length > 0) {
      return new Icon({
        //anchor: [0.5, 46],
        //anchorXUnits: 'fraction',
        //anchorYUnits: 'pixels',
        src: iconURL + "?w=" + this.imageSize() + "&404=logo.png",
        color: featColor.markerColor,
      });
    } else {
      return new Circle({
        radius: 5,
        fill: new Fill({
          color: featColor.fillColor,
        }),
      });
    }
  }
 
  private getColorForFeature(feature): IFeatureColor {
    let markerColor = feature.get("marker-color");
    let strokeColor = feature.get("stroke");
    let strokeWidth = feature.get("stroke-width");
    let fillColor = feature.get("fill");
    let defaultColor = "rgba(52,58,64,0.9)";
    if (feature === this.selectedFeature) {
      defaultColor = "rgba(255,0,0,0.9)";
    }
    if (markerColor === undefined || markerColor === null || markerColor.length <= 0) {
      //mrkColor = 'rgba(0, 0, 0, 0)';  //transparent for markers.
    }
    if (strokeColor === undefined || strokeColor === null || strokeColor.length <= 0) {
      strokeColor = defaultColor;
    }
    if (strokeWidth === undefined || strokeWidth === null || strokeWidth.length <= 0) {
      strokeWidth = 1;
    }
    if (fillColor === undefined || fillColor === null || fillColor.length <= 0) {
      fillColor = defaultColor;
    }
    return {
      markerColor,
      fillColor,
      strokeColor,
      strokeWidth,
    };
  }

  public GetResolutionForZoom(zoomVal) {
    const MAP_MIN_ZOOM = 0; //https://github.com/openlayers/openlayers/issues/5939
    const convertedRes = this.mapView.constrainResolution(this.mapView.getMaxResolution(), zoomVal - MAP_MIN_ZOOM, 0);
    return convertedRes;
  }
 
  public GetLayers(layerProperty) {
    const layerArray = [];
    this.mapObj.getLayers().forEach(function(lyr) {
      if (layerProperty === undefined) {
        //no property specified, just push all layers.
        layerArray.push(lyr);
      } else if (lyr.hasOwnProperty(layerProperty) && lyr[layerProperty] !== undefined) {
        layerArray.push(lyr);
      }
    });
    return layerArray;
  }

  private WireUpEvents(singleClickAction, selectAction) {
    if (singleClickAction !== null && singleClickAction !== undefined) {
      this.mapObj.on("singleclick", singleClickAction);
    }
    if (selectAction !== null && selectAction !== undefined) {
      const select = new Select({
        //https://openlayers.org/en/latest/apidoc/module-ol_interaction_Select.html  look to use the layers filter function
        condition: click,
        hitTolerance: this.hitTolerance,
        multi: true, //allows multiple features on select
        layers: function(layer) {
          return layer.CanSelect;
        },
        //style:'',//
      }); // ref to currently selected interaction
      this.mapObj.addInteraction(select);
      select.on("select", selectAction);
    }
    this.mapObj.on("moveend", this.clientExtentChanged);

      }
    }

   * @param {Object[]} features An array of features as provided by open layers

  setExactCoordDistanceOnFeature = (features, exactCoord) => {

  private getGenericPropertiesOfFeature(feature): string[] {
    const ignoredProperties = [
      "bbox",
      "geometry",
      "styleUrl",
      "Editable",
      "PARENTID",
      "LogEntryGUID",
      "Title",
      "Url",
      "FriendlyToolName",
      "ToolName",
      "FormattedDescription",
      "User",
      "CoBRAUserGUID",
      "Description",
      "DateCreated",
      "KMLUrl",
      "marker-symbol",
      "FileName",
    ];
    const featureDetails: string[] = [];
    if (feature !== null && feature !== undefined) {
      const props = feature.getProperties();
      if (props !== undefined) {
        featureDetails.push("<table class='table table-sm table-info p-2'>");
        for (const prop in props) {
          if (props.hasOwnProperty(prop)) {
            if (!(ignoredProperties.indexOf(prop) > -1) && props[prop].length > 0) {
              featureDetails.push("<tr><th>");
              featureDetails.push(prop);
              featureDetails.push("</th><td>");
              featureDetails.push(props[prop]);
              featureDetails.push("</td></tr>");
            }
          }
        }
        featureDetails.push("</table>");
      }
    }
    return featureDetails;
  }

  public TakeScreenshot() {
    this.mapObj.once("rendercomplete", function(event) {
      const canvas = event.context.canvas;
      if (navigator.msSaveBlob) {
