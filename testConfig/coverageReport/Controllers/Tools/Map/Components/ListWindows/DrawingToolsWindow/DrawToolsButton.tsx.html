import React, { Component, ComponentType } from "react";
import Button from "reactstrap/lib/Button";
import { DrawingStates, isDrawing } from "Controllers/Tools/Map/Components/MapDrawingTools/drawingFsm";
 
interface Props {
  Icon: ComponentType<any>;
  drawingState: DrawingStates;
  isActiveFn: (DrawingStates) => boolean;
  cancelFn: () => void;
  actionFn: () => void;
  title: string;
}
class DrawToolsButton extends Component<Props, any> {
  render() {
    const { Icon, drawingState, isActiveFn, cancelFn, actionFn, title } = this.props;
    const _isDrawing = isDrawing(drawingState);
    const isActive = isActiveFn(drawingState);
    return (
      <Button
        disabled={!isActive && _isDrawing}
        onClick={isActive ? cancelFn : actionFn}
        active={isActive}
        title={title}
        aria-label={title}
      >
        <Icon />
      </Button>
    );
  }
}
 
export default DrawToolsButton;
