import React, { Component } from "react";
import { MapLocalizationConsumer } from "Controllers/Tools/Map/Components/MapLocalization";
import ButtonGroup from "reactstrap/lib/ButtonGroup";
import Button from "reactstrap/lib/Button";
import {
  FaBan,
  FaBullseye,
  FaDharmachakra,
  FaDotCircle,
  FaDraftingCompass,
  FaDrawPolygon,
  FaMapPin,
} from "react-icons/fa";
import { MdTimeline } from "react-icons/md";
import {
  DrawingEvents,
  DrawingStates,
  isDrawing,
  isDrawingCircle,
  isDrawingFreehand,
  isDrawingPoint,
  isDrawingPolygon,
  isSavingCircle,
  isSavingFreehand,
  isSavingPoint,
  isSavingPolygon,
} from "Controllers/Tools/Map/Components/MapDrawingTools/drawingFsm";
import { either } from "ramda";
import DrawToolsButton from "Controllers/Tools/Map/Components/ListWindows/DrawingToolsWindow/DrawToolsButton";
import BasicListWindow from "Controllers/Tools/Map/Components/ListWindows/BasicListWindow";
 
interface ToolActions {
  onDrawPoint: () => void;
  onDrawCircle: () => void;
  onDrawFreehand: () => void;
  onDrawPolygon: () => void;
  onCancel: () => void;
}
 
interface Props {
  dismissFn: Function;
  drawingState: DrawingStates;
  toolActions: ToolActions;
}
 
class DrawingToolsWindow extends Component<Props, any> {
  render() {
    const { dismissFn, drawingState, toolActions } = this.props;
    const _isDrawing = isDrawing(drawingState);
    return (
      <MapLocalizationConsumer>
        {({
          MapSketchToolsTitle,
          DrawTitle,
          DrawPointSketchTitle,
          DrawFreehandTitle,
          DrawPolygonTitle,
          CancelDrawingTitle,
          DrawCircleTitle,
        }) => (
          <BasicListWindow headerText={MapSketchToolsTitle} dismissFn={dismissFn}>
            <div
              style={{
                display: "flex",
                flexDirection: "column",
                padding: ".5rem",
              }}
            >
              <div>{DrawTitle}</div>
              <div
                style={{
                  display: "flex",
                  justifyContent: "space-between",
                }}
              >
                <ButtonGroup size="sm">
                  <DrawToolsButton
                    Icon={FaMapPin}
                    drawingState={drawingState}
                    isActiveFn={either(isDrawingPoint, isSavingPoint)}
                    cancelFn={toolActions.onCancel}
                    actionFn={toolActions.onDrawPoint}
                    title={DrawPointSketchTitle}
                  />
 
                  <DrawToolsButton
                    Icon={MdTimeline}
                    drawingState={drawingState}
                    isActiveFn={either(isDrawingFreehand, isSavingFreehand)}
                    cancelFn={toolActions.onCancel}
                    actionFn={toolActions.onDrawFreehand}
                    title={DrawFreehandTitle}
                  />
 
                  <DrawToolsButton
                    Icon={FaDotCircle}
                    drawingState={drawingState}
                    isActiveFn={either(isDrawingCircle, isSavingCircle)}
                    cancelFn={toolActions.onCancel}
                    actionFn={toolActions.onDrawCircle}
                    title={DrawCircleTitle}
                  />
 
                  <DrawToolsButton
                    Icon={FaDrawPolygon}
                    drawingState={drawingState}
                    isActiveFn={either(isDrawingPolygon, isSavingPolygon)}
                    cancelFn={toolActions.onCancel}
                    actionFn={toolActions.onDrawPolygon}
                    title={DrawPolygonTitle}
                  />
                </ButtonGroup>
 
                <ButtonGroup size="sm">
                  <Button disabled>
                    <FaDharmachakra />
                  </Button>
                  <Button disabled>
                    <FaBullseye />
                  </Button>
                </ButtonGroup>
                <ButtonGroup size="sm">
                  <Button
                    color="warning"
                    disabled={!_isDrawing}
                    onClick={toolActions[DrawingEvents.onCancel]}
                    title={CancelDrawingTitle}
                    aria-label={CancelDrawingTitle}
                  >
                    <FaBan />
                  </Button>
                </ButtonGroup>
              </div>
            </div>
          </BasicListWindow>
        )}
      </MapLocalizationConsumer>
    );
  }
}
 
export default DrawingToolsWindow;
