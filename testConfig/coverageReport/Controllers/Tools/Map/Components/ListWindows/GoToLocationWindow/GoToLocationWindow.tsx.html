function handleSubmit(props, { searchType, address }, { setSubmitting }) {
  cond([
    [
      isCurrentLocation,
      () => {
        getCurrentLocation()
          .then(({ coords }) => {
            props.zoomFn([coords.longitude, coords.latitude]);
            setSubmitting(false);
          })
          .catch(error => {
            /* eslint-disable-next-line no-console */
            console.error(error);
            setSubmitting(false);
          });
      },
    ],
 
    [
      isAddress,
      () => {
        getGeoCodeFromGoogle({ address, key: props.googleApiKey })
          .then(result => {
            const coords: any = path(["results", "0", "geometry", "location"], result);
 
            if (!coords) {
              throw Error("Unable to locate address");
            }
 
            props.zoomFn([coords.lng, coords.lat]);
            setSubmitting(false);
          })
          .catch(error => {
            /* eslint-disable-next-line no-console */
            console.error(error);
            setSubmitting(false);
          });
      },
    ],
  ])(searchType);
}
