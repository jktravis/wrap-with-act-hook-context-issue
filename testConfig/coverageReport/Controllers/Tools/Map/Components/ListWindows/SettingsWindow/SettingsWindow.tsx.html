import React, { useContext } from "react";
import BasicListWindow from "../BasicListWindow";
import { MapLocalizationContext } from "Controllers/Tools/Map/Components/MapLocalization";
import FormGroup from "reactstrap/lib/FormGroup";
import Label from "reactstrap/lib/Label";
import Select from "react-select";
import { equals, prop, propOr } from "ramda";
import ReactSwitch from "react-switch";
import ButtonGroup from "reactstrap/lib/ButtonGroup";
import Button from "reactstrap/lib/Button";
import { FaQuestion } from "react-icons/fa";
import styled from "react-emotion";
import mapTypeOptions from "Controllers/Tools/Map/Components/utilities/mapTypeOptions";
import { IconSize } from "Controllers/Tools/Map/Components/utilities/getStyleForFeature";
 
const FormGroupFlex = styled(FormGroup)`
  display: flex;
  align-items: center;
  justify-content: space-between;
`;
 
const getValue = prop("value");
const getSetter = prop("setter");
 
interface ValueSetter {
  value: any;
  setter: (any) => void;
}
 
interface Props {
  mapSetting: ValueSetter;
  iconSetting: ValueSetter;
  showTextSetting: ValueSetter;
  dismissFn: (event: any) => void;
}
 
function SettingsWindow(props: Props) {
  const { SettingsTitle, BaseMapLabel, SelectPlaceholder, ChangeIconSize, ShowTextOnMap } = useContext(
    MapLocalizationContext,
  );
 
  const { dismissFn, mapSetting, iconSetting, showTextSetting } = props;
 
  const iconSize = getValue(iconSetting);
  const setIconSize = getSetter(iconSetting);
 
  const mapType = getValue(mapSetting);
  const setMapType = getSetter(mapSetting);
 
  const shouldShowText: boolean = propOr(false, "value", showTextSetting);
  const setShouldShowText: any = propOr(() => {}, "setter", showTextSetting);
 
  return (
    <BasicListWindow headerText={SettingsTitle} dismissFn={dismissFn}>
      <div
        style={{
          padding: ".5rem",
        }}
      >
        <FormGroupFlex>
          <Label>{BaseMapLabel}</Label>
          <div style={{ flexBasis: "50%" }}>
            <Select options={mapTypeOptions} value={mapType} onChange={setMapType} placholder={SelectPlaceholder} />
          </div>
        </FormGroupFlex>
 
        <FormGroupFlex>
          <Label>{ChangeIconSize}</Label>
 
          <ButtonGroup size="sm">
            <SetIconSizeButton currentSize={iconSize} setIconSize={setIconSize} targetSize={IconSize.small} />
            <SetIconSizeButton currentSize={iconSize} setIconSize={setIconSize} targetSize={IconSize.medium} />
            <SetIconSizeButton currentSize={iconSize} setIconSize={setIconSize} targetSize={IconSize.large} />
          </ButtonGroup>
        </FormGroupFlex>
 
        <FormGroupFlex>
          <Label>{ShowTextOnMap}</Label>
          <ReactSwitch checked={shouldShowText} onChange={setShouldShowText} />
        </FormGroupFlex>
      </div>
    </BasicListWindow>
  );
}
 
interface SetIconSizeButtonProps {
  currentSize: IconSize;
  targetSize: IconSize;
  setIconSize: (string) => void;
}
 
function SetIconSizeButton({ currentSize, setIconSize, targetSize }: SetIconSizeButtonProps) {
  return (
    <Button
      active={equals(targetSize, currentSize)}
      onClick={() => setIconSize(targetSize)}
      style={{ zIndex: "initial" }}
    >
      <FaQuestion style={{ fontSize: targetSize }} />
    </Button>
  );
}
 
export default SettingsWindow;
