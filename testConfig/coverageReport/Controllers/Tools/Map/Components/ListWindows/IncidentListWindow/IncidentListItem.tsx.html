import React, { Component } from "react";
import Button from "reactstrap/lib/Button";
import { FaEyeSlash, FaEye } from "react-icons/fa";
import styled from "react-emotion";
 
import truncateText from "Utilities/truncateText";
import ListItem from "../../styles/ListItem";
import { MapLocalizationConsumer } from "Controllers/Tools/Map/Components/MapLocalization";
 
const ListItemRow = ListItem.withComponent("div");
const ListItemFooter = styled("footer")`
  display: flex;
  flex-wrap: wrap;
  justify-content: space-between;
  font-size: 0.75em;
  margin-right: 2px;
  margin-bottom: 0.5rem;
`;
ListItemFooter.displayName = "ListItemFooter";
 
type Feature = {
  PrimaryKey: string;
  DataType?: string;
  Title: string;
  Description: string;
  User: string;
  FriendlyToolName: string;
  ToolName: string;
  DateCreated: string;
  UnixDateCreated: string;
  ImageURL: string;
  CanEdit: boolean;
  CanRemove: boolean;
  ShowDetail: boolean;
  Visible: boolean;
  IsKML: boolean;
  IncludedFeatures: any[];
};
 
type Props = {
  feature: Feature;
  showFeatureDetailFn: Function;
  toggleFeatureVisibilityFn: Function;
  zoomToFeatureFn: Function;
  panToFeatureFn: Function;
  removeDrawingFn?: (string) => void;
  isAlt?: boolean;
  style?: any;
};
 
class IncidentListItem extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    isAlt: false,
    style: {},
  };
 
  render() {
    const { feature, isAlt, showFeatureDetailFn, style } = this.props;
    return (
      <MapLocalizationConsumer>
        {({ ShowHideFeature, ZoomToExtent, PanTitle, ShowDetailsTitle, ExpandDetails, Edit, RemoveTitle }) => (
          <ListItemRow isAlt={isAlt} style={style}>
            <div className="media">
              <div className="media-body">
                <h5 className="mt-0">{feature.Title}</h5>
                <p>{feature.Description && truncateText(feature.Description, 100)}</p>
              </div>
              {feature.ImageURL && (
                <img className="ml-3" src={`${feature.ImageURL}?w=20`} alt={feature.FriendlyToolName} />
              )}
            </div>
            <div className="d-flex">
              <div className="mr-auto">
                <Button
                  title={ShowHideFeature}
                  size="sm"
                  color={feature.Visible ? "light" : "dark"}
                  onClick={() => this.props.toggleFeatureVisibilityFn(feature.PrimaryKey)}
                >
                  {feature.Visible ? <FaEye /> : <FaEyeSlash />}
                </Button>
              </div>
              <div className="btn-toolbar" role="toolbar">
                <div className="btn-group btn-group-sm">
                  <Button
                    size="sm"
                    color="info"
                    title={ZoomToExtent}
                    disabled={!feature.Visible}
                    onClick={() => {
                      this.props.zoomToFeatureFn(feature.PrimaryKey);
                    }}
                  >
                    <i className="fas fa-crosshairs" />
                  </Button>
                  <Button
                    size="sm"
                    color="info"
                    title={PanTitle}
                    disabled={!feature.Visible}
                    onClick={() => {
                      this.props.panToFeatureFn(feature.PrimaryKey);
                    }}
                  >
                    <i className="fas fa-arrows-alt" />
                  </Button>
                  <Button
                    size="sm"
                    color="info"
                    title={ShowDetailsTitle}
                    active={feature.ShowDetail}
                    disabled={!feature.Visible}
                    onClick={() => showFeatureDetailFn(feature.PrimaryKey)}
                  >
                    <i className="fas fa-info-circle" />
                  </Button>
 
                  {feature.CanRemove && (
                    <Button
                      color="danger"
                      title={RemoveTitle}
                      onClick={() => this.props.removeDrawingFn(feature.PrimaryKey)}
                    >
                      <i className="fas fa-trash" />
                    </Button>
                  )}
                </div>
 
                {feature.IncludedFeatures && feature.IncludedFeatures.length ? (
                  <div className="btn-group btn-group-sm ml-2">
                    <button
                      className="btn btn-sm btn-info"
                      title={ExpandDetails}
                      type="button"
                      data-toggle="collapse"
                      data-target="\\#incCollapse#=PrimaryKey#"
                      aria-expanded="false"
                      aria-controls="incCollapse#=PrimaryKey#"
                    >
                      <i className="fas fa-sort-down" />
                    </button>
                  </div>
                ) : null}
                {feature.CanEdit || feature.CanRemove ? (
                  <div className="btn-group btn-group-sm" role="group">
                    {feature.CanEdit && (
                      <button
                        className="btn btn-warning btn-sm"
                        title={Edit}
                        onClick={() => {
                          /* eslint-disable-next-line no-console*/
                          console.log("DrawTools.EditSketch('#:PrimaryKey#')");
                        }}
                      >
                        <i className="fas fa-edit" />
                      </button>
                    )}
                  </div>
                ) : null}
              </div>
            </div>
            {feature.IncludedFeatures && feature.IncludedFeatures.length ? (
              <div className="collapse" id="incCollapse#=PrimaryKey#">
                <table className="table table-striped table-info table-sm">
                  <tbody>
                    {feature.IncludedFeatures.map(feat => {
                      return (
                        <tr key={feat.PrimaryKey}>
                          <td>
                            {/*#if(CobraMap.FeatureIsHidden(IncludedFeatures[i].id)){#*/}
                            {/*<button id="btn#:IncludedFeatures[i].id#" title="` + _globalPrompts.ShowHideFeature + `" className="btn btn-dark btn-sm disabled" onclick="CobraMap.ShowHideFeature(0,'#:IncludedFeatures[i].id#','#:PrimaryKey#')">*/}
                            {/*<span className="fas fa-eye-slash"  />*/}
                            {/*</button>*/}
                            {/*#}else{#*/}

                              id="btn#:IncludedFeatures[i].id#"
                              title={ShowHideFeature}
                              className="btn btn-light btn-sm disabled"
                              onClick={() => {
                                /* eslint-disable-next-line no-console*/

                              <span className="fas fa-eye" />
                            </button>
