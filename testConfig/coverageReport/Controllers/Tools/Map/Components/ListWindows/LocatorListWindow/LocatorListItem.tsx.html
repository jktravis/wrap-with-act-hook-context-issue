import React, { Component } from "react";
import Button from "reactstrap/lib/Button";
import { FaEyeSlash, FaEye } from "react-icons/fa";
 
import { MapLocalizationConsumer } from "Controllers/Tools/Map/Components/MapLocalization";
import ListItemRow from "Controllers/Tools/Map/Components/styles/ListItemRow";
import ListItemFooter from "Controllers/Tools/Map/Components/styles/ListItemFooter";
 
type Props = {
  feature: any;
  showFeatureDetailFn: Function;
  toggleFeatureVisibilityFn: Function;
  zoomToFeatureFn: Function;
  panToFeatureFn: Function;
  isAlt?: boolean;
  style?: any;
};
 
class IncidentListItem extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    isAlt: false,
    style: {},
  };
 
  render() {
    const { feature, isAlt, showFeatureDetailFn, style } = this.props;
    return (
      <MapLocalizationConsumer>
        {({ ShowHideFeature, ZoomToExtent, PanTitle, ShowDetailsTitle, SerialNumber }) => (
          <ListItemRow isAlt={isAlt} style={style}>
            <div className="media">
              <div className="media-body">
                <h5 className="mt-0">{feature.Title}</h5>
                <p>{`${SerialNumber} : ${feature.LocatorSerialNumber}`}</p>
              </div>
              {feature.ImageURL && <img className="ml-3" src={`${feature.ImageURL}?w=20`} alt={feature.Title} />}
            </div>
            <ListItemFooter>
              <time data-format="lll" dateTime={feature.UnixDateCreated}>
                {feature.DateCreated}
              </time>
            </ListItemFooter>
            <div className="d-flex">
              <div className="mr-auto">
                <Button
                  title={ShowHideFeature}
                  size="sm"
                  color={feature.Visible ? "light" : "dark"}
                  onClick={() => this.props.toggleFeatureVisibilityFn(feature.PrimaryKey)}
                >
                  {feature.Visible ? <FaEye /> : <FaEyeSlash />}
                </Button>
              </div>
              <div className="btn-toolbar" role="toolbar">
                <div className="btn-group btn-group-sm">
                  <Button
                    size="sm"
                    color="info"
                    title={ZoomToExtent}
                    disabled={!feature.Visible}
                    onClick={() => {
                      this.props.zoomToFeatureFn(feature.PrimaryKey);
                    }}
                  >
                    <i className="fas fa-crosshairs" />
                  </Button>
                  <Button
                    size="sm"
                    color="info"
                    title={PanTitle}
                    disabled={!feature.Visible}
                    onClick={() => {
                      this.props.panToFeatureFn(feature.PrimaryKey);
                    }}
                  >
                    <i className="fas fa-arrows-alt" />
                  </Button>
                  <Button
                    size="sm"
                    color="info"
                    title={ShowDetailsTitle}
                    active={feature.ShowDetail}
                    disabled={!feature.Visible}
                    onClick={() => showFeatureDetailFn(feature.PrimaryKey)}
                  >
                    <i className="fas fa-info-circle" />
                  </Button>
                </div>
              </div>
            </div>
          </ListItemRow>
        )}
      </MapLocalizationConsumer>
    );
  }
}
 
export default IncidentListItem;
