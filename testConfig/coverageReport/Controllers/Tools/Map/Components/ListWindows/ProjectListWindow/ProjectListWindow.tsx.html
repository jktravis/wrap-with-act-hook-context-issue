import React, { Component, Fragment } from "react";
import styled from "react-emotion";
import { FaEye, FaEyeSlash, FaCrosshairs } from "react-icons/fa";
import ButtonGroup from "reactstrap/lib/ButtonGroup";
import Button from "reactstrap/lib/Button";
import Select from "react-select";
import isOdd from "Utilities/isOdd";
import { MapLocalizationConsumer } from "Controllers/Tools/Map/Components/MapLocalization";
import { curry } from "ramda";

import LoadSpinner from "Controllers/Tools/Map/Components/LoadSpinner";
import BasicListWindow from "Controllers/Tools/Map/Components/ListWindows/BasicListWindow";

`;
 
const Header = styled("header")`
  ${baseHeaderFooter};

Header.displayName = "Header";

  ${baseHeaderFooter};

Footer.displayName = "Footer";
 
interface Props {
  features: any[];
  dismissFn: Function;
  setFeatureToDisplayFn: Function;
  toggleFeatureVisibilityFn: Function;
  showAllFn: Function;
  hideAllFn: Function;
  zoomToFeatureFn: Function;
  panToFeatureFn: Function;
  zoomToAllFeaturesFn: Function;
  loadingStatus: string;
 
  /**
   * Options for sorting. The value key needs to match the key to the sort function
   * as found in the corresponding sortFnMap
   */
  sortOptions: any[];
  /**
   * A map of sort functions with each function keyed to the value from the sortOptions.
   */
  sortFnMap: {};
}
 
class ProjectListWindow extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    features: [],
    setFeatureToDisplayFn: Function.prototype,
    sortOptions: [],
    sortFnMap: {},
  };
 
  static getDerivedStateFromProps(nextProps, prevState) {
    if (!prevState.sortType && nextProps.sortOptions.length) {
      return {
        sortType: nextProps.sortOptions[0],
      };
    }
    return null;
  }
 
  state = {
    sortType: null,
  };
 
  handleSortChange = sortType => {
    this.setState(() => ({ sortType }));
  };
 
  sortItems(items) {
    const { sortFnMap } = this.props;
    const {
      sortType: { value },
    } = this.state;
    const sortFn = sortFnMap[value];
    if (sortFn) {
      return sortFn(items);
    }
    return items;
  }
 
  footerRenderer = () => {
    const { showAllFn, hideAllFn, zoomToAllFeaturesFn, sortOptions, features } = this.props;
    const { sortType } = this.state;
    return (
      <MapLocalizationConsumer>
        {({ ZoomToExtent, HideAll, ShowAll }) => (
          <Fragment>
            <ButtonGroup size="sm">
              <Button onClick={showAllFn as any} disabled={!features.length} title={ShowAll}>
                <FaEye />
              </Button>
              <Button onClick={hideAllFn as any} disabled={!features.length} title={HideAll}>
                <FaEyeSlash />
              </Button>
              <Button onClick={zoomToAllFeaturesFn as any} disabled={!features.length} title={ZoomToExtent}>
                <FaCrosshairs />
              </Button>
            </ButtonGroup>
            <div style={{ minWidth: "150px", marginLeft: ".5rem" }}>
              <Select value={sortType} options={sortOptions} menuPlacement="auto" onChange={this.handleSortChange} />
            </div>
          </Fragment>
        )}
      </MapLocalizationConsumer>
    );
  };
 
  rowRenderer = curry((features: any[], { index, style }) => {
    const feature = features[index];
    return (
      <IncidentListItem
        key={feature.PrimaryKey}
        isAlt={isOdd(index)}
        feature={feature}
        showFeatureDetailFn={this.props.setFeatureToDisplayFn}
        toggleFeatureVisibilityFn={this.props.toggleFeatureVisibilityFn}
        zoomToFeatureFn={this.props.zoomToFeatureFn}
        panToFeatureFn={this.props.panToFeatureFn}
        style={style}
      />
    );
  });
 
  render() {
    const { features, dismissFn, loadingStatus } = this.props;
    const sortedFeatures = this.sortItems(features);
    return (
      <MapLocalizationConsumer>
        {({ IncidentInfo, OrganizationColumn }) => (
          <BasicListWindow
            headerText={`${IncidentInfo} / ${OrganizationColumn}`}
            dismissFn={dismissFn}
            footerRenderer={this.footerRenderer}
          >
            {isLoading(loadingStatus) ? (
              <LoadSpinner />
            ) : (
              <VirtualizedFeatureList features={sortedFeatures} rowRenderer={this.rowRenderer(sortedFeatures)} />
            )}
          </BasicListWindow>
        )}
      </MapLocalizationConsumer>
    );
  }
}
 
export default ProjectListWindow;
