import React, { Component } from "react";
import ButtonGroup from "reactstrap/lib/ButtonGroup";
import Button from "reactstrap/lib/Button";
import { FaCrosshairs, FaEye, FaEyeSlash, FaPlus } from "react-icons/fa";
import WmsListItem from "./WmsListItem";
import isOdd from "Utilities/isOdd";
import { MapLocalizationConsumer } from "Controllers/Tools/Map/Components/MapLocalization";
import VirtualizedFeatureList from "Controllers/Tools/Map/Components/FeatureList";
import { curry } from "ramda";
import { isLoading } from "Controllers/Tools/Map/Components/utilities/loadingFsm/loadingFsm";
import LoadSpinner from "Controllers/Tools/Map/Components/LoadSpinner";
import BasicListWindow from "Controllers/Tools/Map/Components/ListWindows/BasicListWindow";
 
interface Props {
  features: any[];
  dismissFn: Function;
  toggleFeatureVisibilityFn: Function;
  showAllFn: Function;
  hideAllFn: Function;
  zoomToFeatureFn: Function;
  zoomToAllFeaturesFn: Function;
  toggleSelectorFn: Function;
  deleteFn: Function;
  loadingStatus: string;
}
 
class WmsListWindow extends Component<Props, any> {
  renderFooter = () => {
    const { showAllFn, hideAllFn, zoomToAllFeaturesFn, toggleSelectorFn, features } = this.props;
    return (
      <MapLocalizationConsumer>
        {({ ZoomToExtent, HideAll, ShowAll, AddWMSWFSLayertoSystem }) => (
          <ButtonGroup size="sm">
            <Button color="success" onClick={toggleSelectorFn as any} title={AddWMSWFSLayertoSystem}>
              <FaPlus />
            </Button>
            <Button onClick={showAllFn as any} disabled={!features.length} title={ShowAll}>
              <FaEye />
            </Button>
            <Button onClick={hideAllFn as any} disabled={!features.length} title={HideAll}>
              <FaEyeSlash />
            </Button>
            <Button onClick={zoomToAllFeaturesFn as any} disabled={!features.length} title={ZoomToExtent}>
              <FaCrosshairs />
            </Button>
          </ButtonGroup>
        )}
      </MapLocalizationConsumer>
    );
  };
 
  rowRenderer = curry((features: any, { style, index }) => {
    const feature = features[index];
    return (
      <WmsListItem
        key={feature.PrimaryKey}
        feature={feature}
        isAlt={isOdd(index)}
        toggleFeatureVisibilityFn={this.props.toggleFeatureVisibilityFn}
        zoomToFeatureFn={this.props.zoomToFeatureFn}
        deleteFn={this.props.deleteFn}
        style={style}
      />
    );
  });
 
  render() {
    const { features, dismissFn, loadingStatus } = this.props;
    return (
      <MapLocalizationConsumer>
        {({ WmsWfsLabel }) => (
          <BasicListWindow headerText={WmsWfsLabel} dismissFn={dismissFn} footerRenderer={this.renderFooter}>
            {isLoading(loadingStatus) ? (
              <LoadSpinner />
            ) : (
              <VirtualizedFeatureList features={features} rowRenderer={this.rowRenderer(features)} />
            )}
          </BasicListWindow>
        )}
      </MapLocalizationConsumer>
    );
  }
}
 
export default WmsListWindow;
