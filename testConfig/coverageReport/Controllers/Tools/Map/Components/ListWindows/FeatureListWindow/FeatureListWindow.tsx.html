import React, { Component } from "react";
import styled from "react-emotion";
import defaultHeaderRenderer from "./defaultHeaderRenderer";
 
const FeatureDiv = styled("div")`
  position: absolute;
  right: 3.1rem;
  margin-top: 0.5rem;
  border-radius: 4px;
  box-shadow: 0 2px 5px 0 #929292;
  background-color: #ffffff;
  width: 28vw;
  max-width: 390px;
`;
FeatureDiv.displayName = "FeatureDiv";
 
const baseHeaderFooter = `
  display: flex;
  justify-content: space-between;
  padding: 0.5rem;
  background-color: #f7f7f7;
`;
 
const Header = styled("header")`
  ${baseHeaderFooter};
  border-bottom: 1px solid #ccc;
`;
Header.displayName = "Header";
 
const Footer = styled("footer")`
  ${baseHeaderFooter};
  border-top: 1px solid #ccc;
`;
Footer.displayName = "Footer";
 
interface Props {
  headerText: string;
  dismissFn: Function;
  headerRenderer?: Function;
  footerRenderer?: Function;
}
 
class FeatureListWindow extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    headerRenderer: defaultHeaderRenderer,
    /**
     * no-op
     */
    footerRenderer: Function.prototype,
  };
 
  render() {
    const { dismissFn, headerText, children, headerRenderer, footerRenderer } = this.props;
    return (
      <FeatureDiv>
        <Header>{headerRenderer({ dismissFn, headerText })}</Header>
        {children}
        <Footer>{footerRenderer()}</Footer>
      </FeatureDiv>
    );
  }
}
 
export default FeatureListWindow;
