import React, { Component } from "react";
 
import ListItem from "../../styles/ListItem";
import ListItemHeader from "../../styles/ListItemHeader";
import truncateText from "Utilities/truncateText";
import Button from "reactstrap/lib/Button";
import { FaCaretDown, FaEye, FaEyeSlash } from "react-icons/fa";
import ButtonGroup from "reactstrap/lib/ButtonGroup";
import ButtonToolbar from "reactstrap/lib/ButtonToolbar";
import createDataToggle from "Utilities/createDataToggle";
import styled from "react-emotion";
import isOdd from "Utilities/isOdd";
import { MapLocalizationConsumer } from "Controllers/Tools/Map/Components/MapLocalization";
 
const SubFeaturesWrapper = styled("ul")`
  list-style: none;
  padding-left: 0;
  border-top: 1px solid gray;
`;
 
const SubFeature = styled("li")`
  display: flex;
  justify-content: space-between;
  align-items: center;
  padding: 0.25rem;
  border-bottom: 1px solid #868484;
  background-color: ${({ isAlt }: any) => (isAlt ? "#c5c5c5" : "#ececec")};
`;
 
type Props = {
  feature: any;
  showFeatureDetailFn: Function;
  toggleFeatureVisibilityFn: Function;
  toggleSubFeatureVisibilityFn: Function;
  zoomToFeatureFn: Function;
  zoomToSubFeatureFn: Function;
  deleteFn: Function;
  /**
   * This function is supplied by the List itself during render. If you're changing the heights dynamically, you'll
   * need to call this after the change to update the list.
   */
  recomputeHeightsFn: () => void;
  isAlt?: boolean;
  style?: any;
};
 
const ExpandableListItem = styled(ListItem)`
  border-left: ${({ showRelated }: any) => (showRelated ? "5px solid #17a2b8" : "inherit")};
`.withComponent("div");
 
class BackgroundListItem extends Component<Props, any> {
  state = {
    showSubFeatures: false,
  };
 
  toggleSubFeatures = () => this.setState(createDataToggle("showSubFeatures"), this.props.recomputeHeightsFn);
 
  render() {
    const { feature, isAlt, showFeatureDetailFn, deleteFn, zoomToSubFeatureFn, style } = this.props;
    const { showSubFeatures } = this.state;
    return (
      <MapLocalizationConsumer>
        {({ ShowHideFeature, ZoomToExtent, PanTitle, ShowDetailsTitle, ExpandDetails, Edit, RemoveTitle }) => (
          <ExpandableListItem isAlt={isAlt} showRelated={showSubFeatures} style={style}>
            <ListItemHeader>{feature.User && <small>{feature.User}</small>}</ListItemHeader>
            <div className="media">
              <div className="media-body">
                <h5 className="mt-0">{feature.Title}</h5>
                <p style={{ minHeight: "1rem" }}>{feature.Description && truncateText(feature.Description, 100)}</p>
              </div>
            </div>
            <div className="d-flex">
              <div className="mr-auto">
                <Button
                  title={ShowHideFeature}
                  size="sm"
                  color={feature.Visible ? "light" : "dark"}
                  onClick={() => this.props.toggleFeatureVisibilityFn(feature.PrimaryKey)}
                >
                  {feature.Visible ? <FaEye /> : <FaEyeSlash />}
                </Button>
              </div>
              <ButtonToolbar>
                <ButtonGroup size="sm">
                  <Button
                    color="info"
                    title={ZoomToExtent}
                    disabled={!feature.Visible}
                    onClick={() => {
                      this.props.zoomToFeatureFn(feature.PrimaryKey);
                    }}
                  >
                    <i className="fas fa-crosshairs" />
                  </Button>
                  <Button color="info" onClick={this.toggleSubFeatures} active={showSubFeatures}>
                    <FaCaretDown />
                  </Button>
                  <Button color="danger" title={RemoveTitle} onClick={() => deleteFn(feature.PrimaryKey)}>
                    <i className="fas fa-trash" />
                  </Button>
                </ButtonGroup>
              </ButtonToolbar>
            </div>
            {showSubFeatures && (
              <div>
                <span style={{ fontWeight: "bold" }}>SubFeatures</span>
                <SubFeaturesWrapper>
                  {Object.values(feature.Features).map((subFeature: any, idx) => (
                    <SubFeature key={subFeature.PrimaryKey} isAlt={isOdd(idx)}>
                      <Button
                        title={ShowHideFeature}
                        size="sm"
                        color={subFeature.Visible ? "light" : "dark"}
                        disabled={!feature.Visible}
                        onClick={() =>
                          this.props.toggleSubFeatureVisibilityFn(feature.PrimaryKey, subFeature.PrimaryKey)
                        }
                      >
                        {subFeature.Visible ? <FaEye /> : <FaEyeSlash />}
                      </Button>
                      <div style={{ flexGrow: 1, marginLeft: ".5rem", marginRight: ".5rem" }}>
                        {subFeature.SubTitle}
                      </div>
                      <ButtonToolbar>
                        <ButtonGroup size="sm">
                          <Button
                            color="info"
                            title={ZoomToExtent}
                            disabled={!feature.Visible || !subFeature.Visible}
                            onClick={() => {
                              zoomToSubFeatureFn(feature.PrimaryKey, subFeature.BackgroundFeatureGUID);
                            }}
                          >
                            <i className="fas fa-crosshairs" />
                          </Button>
                          <Button
                            color="info"
                            title={ShowDetailsTitle}
                            active={subFeature.ShowDetail}
                            disabled={!feature.Visible || !subFeature.Visible}
                            onClick={() => showFeatureDetailFn(feature.PrimaryKey, subFeature.BackgroundFeatureGUID)}
                          >
                            <i className="fas fa-info-circle" />
                          </Button>
                        </ButtonGroup>
                      </ButtonToolbar>
                    </SubFeature>
                  ))}
                </SubFeaturesWrapper>
              </div>
            )}
          </ExpandableListItem>
        )}
      </MapLocalizationConsumer>
    );
  }
}
 
export default BackgroundListItem;
