import React, { Component } from "react";
import { FaTimes } from "react-icons/fa";
import Button from "reactstrap/lib/Button";
import { ButtonGroup } from "reactstrap";
import ButtonDropdown from "reactstrap/lib/ButtonDropdown";
import DropdownToggle from "reactstrap/lib/DropdownToggle";
import DropdownItem from "reactstrap/lib/DropdownItem";
import DropdownMenu from "reactstrap/lib/DropdownMenu";
import styled from "react-emotion";
import { cond } from "ramda";
 
import Position from "Controllers/Tools/Map/Components/Position";
import { getItem, setItem } from "Api/localStorageApi";
import IncidentDetails from "Controllers/Tools/Map/Components/IncidentDetails";
import FacilityDetails from "Controllers/Tools/Map/Components/FacilityDetails";
import {
  isBackgroundDataSubFeatureType,
  isFacilityDataType,
  isIncidentDataType,
} from "Controllers/Tools/Map/Components/utilities/general";
import MapLocalization, { MapLocalizationConsumer } from "Controllers/Tools/Map/Components/MapLocalization";
import BackgroundSubFeatureDetail from "Controllers/Tools/Map/Components/BackgroundSubFeatureDetail";
 
interface Props {
  feature?: any;
  closeFn: Function;
}
 
const TruncatedHeader = styled("span")`
  width: 92%;
  white-space: nowrap;
  overflow: hidden;
  text-overflow: ellipsis;
  margin-right: 0.5rem;
`;
TruncatedHeader.displayName = "TruncatedHeader";
 
/**
 * Conditionally render the appropriate detail component
 * @link {https://ramdajs.com/docs/#cond}
 */
const renderDetailComponent = cond([
  [isFacilityDataType, facility => <FacilityDetails feature={facility} />],
  [isIncidentDataType, incident => <IncidentDetails feature={incident} />],
  [isBackgroundDataSubFeatureType, bg => <BackgroundSubFeatureDetail feature={bg} />],
]);
 
/**
 * The key used in local storage to keep the map position preference.
 */
const MAP_KEY = "cobra-map-position";
 
class FeatureDetails extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    closeFn: Function.prototype,
  };
 
  state = {
    position: "bottom-left",
    isDropDownOpen: false,
    /**
     * Initial rendering is always null do to finding the position.
     * This is to avoid jank when showing the item.
     */
    isGettingPosition: true,
  };
 
  componentDidMount() {
    getItem(MAP_KEY).then(position =>
      this.setState(state => ({
        position: position ? position : state.position,
        isGettingPosition: false,
      })),
    );
  }
 
  toggleDropDown = () =>
    this.setState(({ isDropDownOpen }) => ({
      isDropDownOpen: !isDropDownOpen,
    }));
 
  setPosition = position => () => {
    setItem(MAP_KEY, position).then(() => {
      this.setState(() => ({ position, isDropDownOpen: false }));
    });
  };
 
  render() {
    const { feature, closeFn } = this.props;
    const { isGettingPosition, position, isDropDownOpen } = this.state;
    if (!feature || isGettingPosition) {
      return null;
    }
    return (
      <Position position={position}>
        <div className="card bg-light">
          <div className="card-header">
            <div className="d-flex justify-content-between align-items-center">
              <TruncatedHeader>{feature.Title}</TruncatedHeader>
              <ButtonGroup>
                <Button size="sm" onClick={() => closeFn()}>
                  <FaTimes />
                </Button>
                <ButtonDropdown
                  isOpen={isDropDownOpen}
                  toggle={this.toggleDropDown}
                  size="sm"
                  aria-haspopup
                  aria-label="detail window position"
                  direction={position.endsWith("right") ? "left" : "down"}
                >
                  <DropdownToggle caret />
                  <MapLocalizationConsumer>
                    {({ BottomLeft, BottomRight, TopLeft, TopRight, Position }) => (
                      <DropdownMenu>
                        <DropdownItem header>{Position}</DropdownItem>
                        <DropdownItem disabled={position === "top-left"} onClick={this.setPosition("top-left")}>
                          {TopLeft}
                        </DropdownItem>
                        <DropdownItem disabled={position === "top-right"} onClick={this.setPosition("top-right")}>
                          {TopRight}
                        </DropdownItem>
                        <DropdownItem disabled={position === "bottom-left"} onClick={this.setPosition("bottom-left")}>
                          {BottomLeft}
                        </DropdownItem>
                        <DropdownItem disabled={position === "bottom-right"} onClick={this.setPosition("bottom-right")}>
                          {BottomRight}
                        </DropdownItem>
                      </DropdownMenu>
                    )}
                  </MapLocalizationConsumer>
                </ButtonDropdown>
              </ButtonGroup>
            </div>
          </div>
          {renderDetailComponent(feature)}
        </div>
      </Position>
    );
  }
}
 
export default FeatureDetails;
