import React from "react";
import ButtonGroup from "reactstrap/lib/ButtonGroup";
import Button from "reactstrap/lib/Button";
import { FaBan, FaSave } from "react-icons/fa";
import { MapLocalizationConsumer } from "Controllers/Tools/Map/Components/MapLocalization";
 
/**
 * Function wrapper to allow reuse of these buttons.
 * This function is manually curried. Call it once with the args. Then call
 * the result to get the component.
 * @param isSubmitting
 * @param isValid
 * @param cancelSaveFn
 * @returns {() => ReactNode}
 */
const createSaveCancelButtons = ({ isSubmitting, isValid, cancelSaveFn }) => () => {
  return (
    <MapLocalizationConsumer>
      {({ CancelText, SaveText }) => (
        <ButtonGroup>
          <Button
            color="warning"
            onClick={cancelSaveFn}
            disabled={isSubmitting}
            title={CancelText}
            aria-label={CancelText}
          >
            <FaBan /> {CancelText}
          </Button>
          <Button
            color="primary"
            type="submit"
            disabled={!isValid || isSubmitting}
            title={SaveText}
            aria-label={SaveText}
          >
            <FaSave /> {SaveText}
          </Button>
        </ButtonGroup>
      )}
    </MapLocalizationConsumer>
  );
};
 
export default createSaveCancelButtons;
