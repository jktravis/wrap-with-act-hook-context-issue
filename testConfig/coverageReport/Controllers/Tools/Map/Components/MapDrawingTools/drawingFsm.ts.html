enum DrawingStates {
  NotDrawing = "NotDrawing",
  DrawPoint = "DrawPoint",
  DrawCircle = "DrawCircle",
  DrawPolygon = "DrawPolygon",
  DrawFreehand = "DrawFreehand",
  DrawThirtyDegEvac = "DrawThirtyDegEvac",
  DrawConcentricCircles = "DrawConcentricCircles",
  SavePointDrawing = "SavePointDrawing",
  SaveCircleDrawing = "SaveCircleDrawing",
  SavePolygonDrawing = "SavePolygonDrawing",
  SaveFreehandDrawing = "SaveFreehandDrawing",
  SaveThirtyDegEvac = "SaveThirtyDegEvac",
  SaveConcentricCircles = "SaveConcentricCircles",
}
 
enum DrawingEvents {
  onDrawPoint = "onDrawPoint",
  onDrawCircle = "onDrawCircle",
  onDrawPolygon = "onDrawPolygon",
  onDrawFreehand = "onDrawFreehand",
  onDrawThirtyDegEvac = "onDrawThirtyDegEvac",
  onDrawConcentricCircles = "onDrawConcentricCircles",
  onCancel = "onCancel",
  onSave = "onSave",
  onComplete = "onComplete",
}
 
const drawingMachine = Machine({
  id: "root",
  states: {
    [DrawingStates.NotDrawing]: {
      id: DrawingStates.NotDrawing,
      on: {
        [DrawingEvents.onDrawFreehand]: `#${DrawingStates.DrawFreehand}`,
        [DrawingEvents.onDrawPoint]: `#${DrawingStates.DrawPoint}`,
        [DrawingEvents.onDrawCircle]: `#${DrawingStates.DrawCircle}`,

const isDrawing = state => {
  return [
    isDrawingPoint(state),
    isDrawingCircle(state),
    isDrawingPolygon(state),
    isDrawingFreehand(state),
    isDrawingThirtyDegEvac(state),
    isDrawingConcentricCircles(state),
  ].some(identity);
};
 
const isSaving = state => {
  return [
    isSavingPoint(state),
    isSavingCircle(state),
    isSavingPolygon(state),
    isSavingFreehand(state),
    isSavingThirtyDegEvac(state),
    isSavingConcentricCircles(state),
  ].some(identity);
};
