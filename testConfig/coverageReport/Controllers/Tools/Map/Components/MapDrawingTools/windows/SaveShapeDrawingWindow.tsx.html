import React, { Component } from "react";
import FormGroup from "reactstrap/lib/FormGroup";
import Label from "reactstrap/lib/Label";
import Input from "reactstrap/lib/Input";
import Form from "reactstrap/lib/Form";
import { Formik, ErrorMessage } from "formik";
import { createEditingStyle, createDefaultStyle } from "ol/style/Style";
import { Fill, Stroke, Style } from "ol/style";
import { fromCircle } from "ol/geom/Polygon";
import { curry } from "ramda";
import * as Yup from "yup";
 
import SaveDrawingWindow from "./SaveDrawingWindow";
import ColorPicker from "Controllers/Shared/Components/ColorSwatchButton";
import { addSketchToIncident } from "Api/mapApi";
import { convertColorObjectToRgba, getWktFromShape } from "../utilities";
import { MapLocalizationConsumer } from "Controllers/Tools/Map/Components/MapLocalization";
import createSaveCancelButtons from "Controllers/Tools/Map/Components/MapDrawingTools/SaveCancelButtons";
 
interface Props {
  projectId: string;
  feature: any;
  successSaveFn: () => void;
  cancelSaveFn: () => void;
  headerText?: string;
  getWktFn?: (any) => any;
  showFillEdit?: boolean;
}
 
class SaveShapeDrawingWindow extends Component<Props, any> {
  /**
   * Modified version of default `createDefaultStyle` from OL
   */
  static createStyle({ strokeColor, fillColor, strokeWidth }) {
    const fill = new Fill({
      color: fillColor,
    });
 
    const stroke = new Stroke({
      color: strokeColor,
      width: strokeWidth,
    });
 
    return new Style({
      fill: fill,
      stroke: stroke,
    });
  }
 
  static defaultProps: Partial<Props> = {
    headerText: "Save Shape",
    getWktFn: getWktFromShape,
    showFillEdit: false,
  };
 
  renderError = error => {
    return <div className="invalid-feedback"> {error} </div>;
  };
 
  /**
   * Sets the field's value.
   * @param {string} fieldName The Formik property to update
   * @param {(string, any) => void} setFieldValue Formik's setFieldValue function
   * @param {object} values Formik's values object containing current values
   * @returns {(object, object) => void} Change handler function used by react-color.
   */
  createHandleColorChange = curry((fieldName, setFieldValue, values) =>
    curry((color, event: any) => {
      if (fieldName === "StrokeWidth") {
        setFieldValue(fieldName, event.target.value);
      } else {
        setFieldValue(fieldName, color);
      }
      const strokeColor = fieldName === "StrokeColor" ? color : values.StrokeColor;
      const fillColor = fieldName === "FillColor" ? color : values.FillColor;
      const strokeWidth = fieldName === "StrokeWidth" ? event.target.value : values.StrokeWidth;
 
      const style = SaveShapeDrawingWindow.createStyle({
        strokeColor: convertColorObjectToRgba(strokeColor),
        fillColor: convertColorObjectToRgba(fillColor),
        strokeWidth: strokeWidth,
      });
 
      this.props.feature.setStyle(style);
    }),
  );
 
  handleFillColorChange = this.createHandleColorChange("FillColor");
  handleStrokeColorChange = this.createHandleColorChange("StrokeColor");
  handleStrokeWidthChange = this.createHandleColorChange("StrokeWidth");
 
  render() {
    const { feature, getWktFn, projectId, successSaveFn, showFillEdit, cancelSaveFn } = this.props;
 
    return (
      <MapLocalizationConsumer>
        {({
          SketchName,
          IsRequired,
          ValueNoLessThan1,
          AnnotationNamePlaceHolder,
          DescriptionTitle,
          BriefDescription,
          Line,
          Color,
          Width,
          Fill,
        }) => {
          return (
            <Formik
              initialValues={{
                StrokeColor: {
                  rgb: { r: 51, g: 153, b: 204, a: 1 },
                },
                FillColor: {
                  rgb: { r: 255, g: 255, b: 255, a: 0.4 },
                },
                IconURL: "",
                StrokeWidth: 1,
                SketchName: "",
                SketchDescription: "",
                WKT: getWktFn(feature),
                SymbolGUID: "",
                LogEntryGUID: "",
                DrawType: 0,
              }}
              validationSchema={Yup.object().shape({
                SketchName: Yup.string().required(`${SketchName} ${IsRequired}`),
                SketchDescription: Yup.string(),
                StrokeWidth: Yup.number().min(1, `${ValueNoLessThan1} ${IsRequired}`),
              })}
              onSubmit={(data, { setSubmitting, resetForm }) => {
                setSubmitting(true);
                const { StrokeColor, FillColor } = data;
                addSketchToIncident({
                  incidentId: projectId,
                  ...data,
                  StrokeColor: convertColorObjectToRgba(StrokeColor),
                  FillColor: convertColorObjectToRgba(FillColor),
                } as any).then(() => {
                  setSubmitting(false);
                  resetForm();
                  successSaveFn();
                });
              }}
            >
              {({ handleChange, handleBlur, values, setFieldValue, handleSubmit, isValid, isSubmitting }) => (
                <Form onSubmit={handleSubmit} style={{ position: "absolute", top: 0 }}>
                  <SaveDrawingWindow
                    footerRenderer={createSaveCancelButtons({ isValid, isSubmitting, cancelSaveFn })}
                    headerText={this.props.headerText}
                    dismissFn={cancelSaveFn}
                  >
                    <FormGroup>
                      <Label for="SketchName">{SketchName}</Label>
                      <Input
                        placeholder={AnnotationNamePlaceHolder}
                        id="SketchName"
                        name="SketchName"
                        onChange={handleChange}
                        value={values.SketchName}
                        disabled={isSubmitting}
                      />
                      <ErrorMessage name="SketchName" render={this.renderError} />
                    </FormGroup>
 
                    <FormGroup>
                      <Label for="SketchDescription">{DescriptionTitle}</Label>
                      <Input
                        type="textarea"
                        placeholder={BriefDescription}
                        id="SketchDescription"
                        name="SketchDescription"
                        onChange={handleChange}
                        value={values.SketchDescription}
                        disabled={isSubmitting}
                      />
                      <ErrorMessage name="SketchDescription" render={this.renderError} />
                    </FormGroup>
 
                    <div
                      style={{
                        display: "flex",
                        justifyContent: "space-between",
                        alignItems: "center",
                      }}
                    >
                      <FormGroup>
                        <Label>{`${Line} ${Color}`}</Label>
                        <ColorPicker
                          color={values.StrokeColor}
                          colorChangeFn={this.handleStrokeColorChange(setFieldValue, values)}
                        />
                      </FormGroup>
 
                      <FormGroup>
                        <Label>{`${Line} ${Width}`}</Label>
                        <Input
                          type="number"
                          value={values.StrokeWidth}
                          name="StrokeWidth"
                          min="1"
                          onChange={this.handleStrokeWidthChange(setFieldValue, values)("")}
                        />
                      </FormGroup>
                    </div>
 
                    {showFillEdit && (
                      <FormGroup>
                        <Label>{`${Fill} ${Color}`}</Label>
                        <ColorPicker
                          color={values.FillColor}
                          colorChangeFn={this.handleFillColorChange(setFieldValue, values)}
                        />
                      </FormGroup>
                    )}
                  </SaveDrawingWindow>
                </Form>
              )}
            </Formik>
          );
        }}
      </MapLocalizationConsumer>
    );
  }
}
 
export default SaveShapeDrawingWindow;
