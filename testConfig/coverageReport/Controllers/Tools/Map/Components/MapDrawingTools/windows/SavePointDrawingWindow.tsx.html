import React, { Component } from "react";
import FormGroup from "reactstrap/lib/FormGroup";
import Label from "reactstrap/lib/Label";
import Input from "reactstrap/lib/Input";
import Nav from "reactstrap/lib/Nav";
import NavItem from "reactstrap/lib/NavItem";
import NavLink from "reactstrap/lib/NavLink";
import TabContent from "reactstrap/lib/TabContent";
import styled, { cx } from "react-emotion";
import TabPane from "reactstrap/lib/TabPane";
import Row from "reactstrap/lib/Row";
import Col from "reactstrap/lib/Col";
import Form from "reactstrap/lib/Form";
import { Formik, ErrorMessage } from "formik";
import { createEditingStyle, createDefaultStyle } from "ol/style/Style";
import { Fill, Stroke, Style } from "ol/style";
import CircleStyle from "ol/style/Circle";
import { Icon } from "ol/style.js";
import { propOr, memoizeWith, identity } from "ramda";
import * as Yup from "yup";
 
import SaveDrawingWindow from "./SaveDrawingWindow";
import { getAvailableSymbols } from "Api/imageApi";
import SearchableImageGrid from "Controllers/Tools/Map/Components/MapDrawingTools/SearchableImageGrid";
import { getWkt } from "Controllers/Tools/Map/Components/MapDrawingTools/utilities";
import { baseUrl } from "Api/constants";
import { addSketchToIncident } from "Api/mapApi";
import {
  isLoading,
  LoadingStates,
  LoadingStatusEvents,
  transitionLoading,
} from "Controllers/Tools/Map/Components/utilities/loadingFsm/loadingFsm";
import LoadSpinner from "Controllers/Tools/Map/Components/LoadSpinner";
import { MapLocalizationConsumer } from "Controllers/Tools/Map/Components/MapLocalization";
import createSaveCancelButtons from "Controllers/Tools/Map/Components/MapDrawingTools/SaveCancelButtons";
 
const TabContentWrapper = styled("div")`
  background: white;
  padding: 0.25rem;
  border: 1px solid #dee2e6;
  border-top: none;
  border-radius: 4px;
  .image-grid-wrapper {
    background: #d2d2d2;
    padding: 0.25rem;
    border-radius: 4px;
  }
`;
 
const Footer = styled("footer")`
  display: flex;
  justify-content: space-between;
  padding: 0.5rem;
  background-color: #f7f7f7;
  border-top: 1px solid #ccc;
`;
Footer.displayName = "Footer";
 
const memoGetAvailableSymbols = memoizeWith(identity, projectId => {
  return getAvailableSymbols(projectId);
});
 
interface Props {
  projectId: string;
  feature: any;
  successSaveFn: () => void;
  cancelSaveFn: () => void;
}
 
class SavePointDrawingWindow extends Component<Props, any> {
  /**
   * Modified version of default `createDefaultStyle` from OL
   * @param img The Image/Symbol object
   */
  static createStyle(img) {
    const fill = new Fill({
      color: "rgba(255,255,255,0.4)",
    });
 
    const stroke = new Stroke({
      color: "#3399CC",
      width: 1.25,
    });
 
    const url = propOr("", "Url", img);
 
    return new Style({
      image: url
        ? new Icon({ src: `${url}?w=25` })
        : new CircleStyle({
            fill: fill,
            stroke: stroke,
            radius: 5,
          }),
      fill: fill,
      stroke: stroke,
    });
  }
 
  state = {
    activeTab: 0,
    favoriteSymbols: [],
    availableSymbols: [],
 
    selectedImage: null,
 
    loadImageStatus: LoadingStates.NotStarted,
  };
 
  componentDidMount() {
    this.setState(({ loadImageStatus }) => {
      return {
        loadImageStatus: transitionLoading(loadImageStatus, LoadingStatusEvents.onLoad),
      };
    });
 
    memoGetAvailableSymbols(this.props.projectId).then(availableSymbols => {
      this.setState(({ loadImageStatus }) => {
        const symbolsWithUrls = availableSymbols.map(sym => {
          sym.Url = `${baseUrl}/symimg/${sym.SymbolGUID}.png`;
          return sym;
        });
        return {
          favoriteSymbols: symbolsWithUrls.filter(symbol => symbol.SymbolSequence !== null),
          availableSymbols: symbolsWithUrls,
          loadImageStatus: transitionLoading(loadImageStatus, LoadingStatusEvents.onSuccess),
        };
      });
    });
  }
 
  toggle = activeTab => {
    this.setState(state => {
      if (state.activeTab === activeTab) {
        return null;
      }
      return {
        activeTab,
      };
    });
  };
 
  handleImageSelect = setFieldValue => img => {
    setFieldValue("SymbolGUID", propOr("", "SymbolGUID", img));
    setFieldValue("IconURL", propOr("", "Url", img));
    setFieldValue("SketchName", propOr("", "FSymbolName", img));
    setFieldValue("SketchDescription", propOr("", "FSymbolDescription", img));
 
    this.props.feature.setStyle(SavePointDrawingWindow.createStyle(img));
  };
 
  renderError = error => {
    return <div className="invalid-feedback"> {error} </div>;
  };
 
  render() {
    const { favoriteSymbols, availableSymbols, activeTab, loadImageStatus } = this.state;
    const { feature, cancelSaveFn, successSaveFn, projectId } = this.props;
 
    return (
      <MapLocalizationConsumer>
        {({
          SketchName,
          IsRequired,
          ValueNoLessThan1,
          AnnotationNamePlaceHolder,
          DescriptionTitle,
          BriefDescription,
          PointSketchAllSymbols,
          PointSketchFavorites,
          DrawPointSketchTitle,
        }) => {
          return (
            <Formik
              initialValues={{
                StrokeColor: "#3399CC",
                FillColor: "rgba(255,255,255,0.4)",
                IconURL: "",
                StrokeWidth: 1.25,
                SketchName: "",
                SketchDescription: "",
                WKT: getWkt(feature.getGeometry()),
                SymbolGUID: "",
                LogEntryGUID: "",
                DrawType: 0,
              }}
              validationSchema={Yup.object().shape({
                SketchName: Yup.string().required(`${SketchName} ${IsRequired}`),
                SketchDescription: Yup.string(),
              })}
              onSubmit={(data, { setSubmitting, resetForm }) => {
                setSubmitting(true);
                addSketchToIncident({ incidentId: projectId, ...data } as any).then(() => {
                  setSubmitting(false);
                  resetForm();
                  successSaveFn();
                });
              }}
            >
              {({ handleChange, handleBlur, values, setFieldValue, handleSubmit, isValid, isSubmitting }) => (
                <Form onSubmit={handleSubmit} style={{ position: "absolute", top: 0 }}>
                  <SaveDrawingWindow
                    footerRenderer={createSaveCancelButtons({ isValid, isSubmitting, cancelSaveFn })}
                    headerText={DrawPointSketchTitle}
                    dismissFn={cancelSaveFn}
                  >
                    <FormGroup>
                      <Label for="SketchName">{SketchName}</Label>
                      <Input
                        placeholder={AnnotationNamePlaceHolder}
                        id="SketchName"
                        name="SketchName"
                        onChange={handleChange}
                        value={values.SketchName}
                        disabled={isSubmitting}
                      />
                      <ErrorMessage name="SketchName" render={this.renderError} />
                    </FormGroup>
 
                    <FormGroup>
                      <Label for="SketchDescription">{DescriptionTitle}</Label>
                      <Input
                        type="textarea"
                        placeholder={BriefDescription}
                        id="SketchDescription"
                        name="SketchDescription"
                        onChange={handleChange}
                        value={values.SketchDescription}
                        disabled={isSubmitting}
                      />
                      <ErrorMessage name="SketchDescription" render={this.renderError} />
                    </FormGroup>
 
                    <Nav tabs>
                      <NavItem>
                        <NavLink
                          className={cx({ active: activeTab === 0 })}
                          onClick={() => {
                            this.toggle(0);
                          }}
                        >
                          {PointSketchFavorites}
                        </NavLink>
                      </NavItem>
                      <NavItem>
                        <NavLink
                          className={cx({ active: activeTab === 1 })}
                          onClick={() => {
                            this.toggle(1);
                          }}
                        >
                          {PointSketchAllSymbols}
                        </NavLink>
                      </NavItem>
                    </Nav>
 
                    <TabContent activeTab={activeTab}>
                      <TabPane tabId={0}>
                        <TabContentWrapper>
                          {isLoading(loadImageStatus) || isSubmitting ? (
                            <LoadSpinner />
                          ) : (
                            <Row>
                              <Col>
                                <div className="image-grid-wrapper">
                                  <SearchableImageGrid
                                    imageData={favoriteSymbols}
                                    onImageSelect={this.handleImageSelect(setFieldValue)}
                                  />
                                </div>
                              </Col>
                            </Row>
                          )}
                        </TabContentWrapper>
                      </TabPane>
                      <TabPane tabId={1}>
                        <TabContentWrapper>
                          {isLoading(loadImageStatus) || isSubmitting ? (
                            <LoadSpinner />
                          ) : (
                            <Row>
                              <Col>
                                <div className="image-grid-wrapper">
                                  <SearchableImageGrid
                                    imageData={availableSymbols}
                                    onImageSelect={this.handleImageSelect(setFieldValue)}
                                  />
                                </div>
                              </Col>
                            </Row>
                          )}
                        </TabContentWrapper>
                      </TabPane>
                    </TabContent>
                  </SaveDrawingWindow>
                </Form>
              )}
            </Formik>
          );
        }}
      </MapLocalizationConsumer>
    );
  }
}
 
export default SavePointDrawingWindow;
