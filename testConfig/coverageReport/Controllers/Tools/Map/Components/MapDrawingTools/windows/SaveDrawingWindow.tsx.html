/**
 * Window wrapper to position the drawing content
 */
import React, { Component, ReactNode } from "react";
import styled from "react-emotion";
import BasicSaveWindow from "./BasicSaveWindow";
 
const SaveWindowDiv = styled("div")`
  position: absolute;
  left: 3.1rem;
  margin-top: 0.5rem;
  border-radius: 4px;
  box-shadow: 0 2px 5px 0 #929292;
  background-color: #ebebeb;
  min-width: 380px;
  max-width: 680px;
  .content-wrapper {
    margin: 0.5rem;
  }
`;
SaveWindowDiv.displayName = "SaveWindowDiv";
 
const baseHeaderFooter = `
  display: flex;
  justify-content: space-between;
  padding: 0.5rem;
  background-color: #f7f7f7;
`;
 
const Header = styled("header")`
  ${baseHeaderFooter};
  border-bottom: 1px solid #ccc;
`;
Header.displayName = "Header";
 
const Footer = styled("footer")`
  ${baseHeaderFooter};
  border-top: 1px solid #ccc;
`;
Footer.displayName = "Footer";
 
interface Props {
  children: ReactNode;
  headerRenderer?: () => ReactNode;
  footerRenderer?: () => ReactNode;
  headerText?: string;
  dismissFn?: Function;
}
 
class SaveDrawingWindow extends Component<Props, any> {
  render() {
    const { children, headerRenderer, footerRenderer, headerText, dismissFn } = this.props;
    return (
      <BasicSaveWindow
        dismissFn={dismissFn}
        headerText={headerText}
        footerRenderer={footerRenderer}
        headerRenderer={headerRenderer}
      >
        <div className="content-wrapper">{children}</div>
      </BasicSaveWindow>
    );
  }
}
 
export default SaveDrawingWindow;
