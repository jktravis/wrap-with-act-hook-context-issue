import React, { Component } from "react";
import Input from "reactstrap/lib/Input";
import Select from "react-select";
 
import ImageGrid from "Controllers/Tools/Map/Components/MapDrawingTools/ImageGrid";
import { sortByLabelProp } from "Controllers/Tools/Map/Components/utilities/sortUtilities";
import { filter, allPass } from "ramda";
import { MapLocalizationConsumer } from "Controllers/Tools/Map/Components/MapLocalization";
 
interface Props {
  imageData: any[];
  onImageSelect: (any) => void;
}
 
class SearchableImageGrid extends Component<Props, any> {
  static isTheme = theme => data => {
    return data.FSymbolTheme === theme;
  };
 
  static isCategory = category => data => {
    return data.SymbolCategory === category;
  };
 
  static doesNameContain = name => data => {
    return data.FSymbolName.toLowerCase().includes(name.toLowerCase());
  };
 
  state = {
    searchText: "",
    theme: null,
    category: null,
  };
 
  extractThemesAndCategories = data => {
    return data.reduce(
      (result, curr) => {
        if (!result.themes.some(f => f.value === curr.FSymbolTheme)) {
          result.themes = sortByLabelProp([...result.themes, { value: curr.FSymbolTheme, label: curr.FSymbolTheme }]);
        }
 
        if (!result.categories.some(f => f.value === curr.SymbolCategory)) {
          result.categories = sortByLabelProp([
            ...result.categories,
            { value: curr.SymbolCategory, label: curr.SymbolCategory },
          ]);
        }
        return result;
      },
      { themes: [], categories: [] },
    );
  };
 
  handleTextChange = event => {
    const { value } = event.target;
    this.setState(() => ({ searchText: value }));
  };
 
  createHandleSelectChange = stateKey => option => {
    this.setState(() => ({ [stateKey]: option }));
  };
 
  handleThemeChange = this.createHandleSelectChange("theme");
  handleCategoryChange = this.createHandleSelectChange("category");
 
  filterData = data => {
    const { searchText, category, theme } = this.state;
    const fns = [];
    if (searchText) {
      fns.push(SearchableImageGrid.doesNameContain(searchText));
    }
    if (category) {
      fns.push(SearchableImageGrid.isCategory(category.label));
    }
    if (theme) {
      fns.push(SearchableImageGrid.isTheme(theme.label));
    }
 
    if (fns.length) {
      return filter(allPass(fns), data);
    }
    return data;
  };
 
  render() {
    const { imageData } = this.props;
    const { searchText, category, theme } = this.state;
    /**
     * Extract the themes and categories out of the imageData
     */
    const { themes, categories } = this.extractThemesAndCategories(imageData);
 
    const filteredImageData = this.filterData(imageData);
 
    return (
      <MapLocalizationConsumer>
        {({ PointSymbolNameSearch, SelectPlaceholder }) => (
          <div>
            <ImageGrid imageData={filteredImageData} onImageSelect={this.props.onImageSelect} />
            <div
              style={{
                marginTop: ".5rem",
                display: "flex",
                justifyContent: "space-between",
              }}
            >
              <div
                style={{
                  flexBasis: "39%",
                }}
              >
                <Input
                  placeholder={PointSymbolNameSearch}
                  value={searchText}
                  onChange={this.handleTextChange}
                  id="image-name-search"
                />
              </div>
 
              <div
                style={{
                  flexBasis: "30%",
                }}
              >
                <Select
                  options={themes}
                  menuPlacement="auto"
                  value={theme}
                  onChange={this.handleThemeChange}
                  isClearable
                  id="theme-selector"
                  placeholder={SelectPlaceholder}
                />
              </div>
 
              <div
                style={{
                  flexBasis: "30%",
                }}
              >
                <Select
                  options={categories}
                  menuPlacement="auto"
                  value={category}
                  onChange={this.handleCategoryChange}
                  isClearable
                  id="category-selector"
                  placeholder={SelectPlaceholder}
                />
              </div>
            </div>
          </div>
        )}
      </MapLocalizationConsumer>
    );
  }
}
 
export default SearchableImageGrid;
