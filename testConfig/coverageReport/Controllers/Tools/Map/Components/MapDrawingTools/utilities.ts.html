import WKT from "ol/format/WKT";
import { Draw, Modify, Snap } from "ol/interaction";
import { fromCircle } from "ol/geom/Polygon";
import { getLength, getArea } from "ol/sphere.js";
import rgba from "polished/lib/color/rgba";
import { invoker, compose } from "ramda";
import { format } from "d3";
 
function getWkt(geometry) {
  const wktFormat = new WKT();
  return wktFormat.writeGeometry(geometry, {
    dataProjection: "EPSG:4326",
    featureProjection: "EPSG:3857",
  } as any);
}
 
function calculateArea(polygon) {
  const area = getArea(polygon);
  const output = { value: 0, unit: "" };
  if (Main.UseMetric()) {
    if (area > 10000) {
      output.value = Math.round((area / 1000000) * 100) / 100;
      output.unit = "km2";
    } else {
      output.value = Math.round(area * 100) / 100;
      output.unit = "m2";
    }
  } else {
    const feetSq = area * 10.7639;
    if (feetSq > 107639.1) {
      const miSq = feetSq * 3.587e-8;
      output.value = Math.round(miSq * 100) / 100;
      output.unit = "mi2";
    } else {
      output.value = Math.round(feetSq * 100) / 100;
      output.unit = "ft2";
    }
  }
  return output;
}
 
function calculateLength(line) {
  const length = getLength(line);
  let output = { value: 0, unit: "" };
  if (Main.UseMetric()) {
    if (length > 100) {
      output.value = Math.round((length / 1000) * 100) / 100;
      output.unit = "km";
    } else {
      output.value = Math.round(length * 100) / 100;
      output.unit = "m";
    }
  } else {
    //in feet, do we switch to miles?
    const feet = length * 3.28084;
    if (feet > 5280) {
      //switch to miles
      const miles = feet / 5280;
      output.value = Math.round(miles * 100) / 100;
      output.unit = "mi";
    } else {
      output.value = Math.round(feet * 100) / 100;
      output.unit = "ft";
    }
  }
  return output;
}
 
/**
 * Simply extract and join the unit and value from the object.
 * Useful in conjunction with the formatLength and formatArea
 * @param unit
 * @param value
 */
function joinUnitsValue({ unit, value }) {
  return `${format(",")(value)} ${unit}`;
}
 
const formatArea = compose(
  joinUnitsValue,
  calculateArea,
);
 
const formatLength = compose(
  joinUnitsValue,
  calculateLength,
);
 
/**
 * The color object is in the format from react-color. This primarily extracts the `rbg` object
 * and converts it to a string.
 * @param color
 */
function convertColorObjectToRgba(color) {
  return rgba(color.rgb.r, color.rgb.g, color.rgb.b, color.rgb.a);
}

const getWktFromShape = compose(

const formatLengthOfCircle = compose(
