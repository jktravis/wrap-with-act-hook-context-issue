import React, { Component, createRef } from "react";
import MousePosition from "ol/Control/MousePosition.js";
import styled from "react-emotion";
 
import convertCoordinates from "Controllers/Tools/Map/Components/utilities/convertCoordinates";
 
const MouseIndicatorDiv = styled("div")`
  position: absolute;
  right: 3.1rem;
  bottom: 50px;
  padding: 0.25rem;
  border-radius: 4px;
  background: rgba(255, 255, 255, 0.4);
`;
 
const MouseIndicatorText = styled("div")`
  border: 1px solid rgba(165, 165, 165, 0.5);
  background-color: #fff;
  padding: 0.25rem;
  border-radius: 4px;
`;
 
interface Props {
  /**
   * ol/map
   */
  map: any;
}
class MouseIndicator extends Component<Props, any> {
  mousePositionRef: React.RefObject<any>;
 
  constructor(props) {
    super(props);
 
    this.mousePositionRef = createRef();
  }
 
  componentDidMount() {
    const mousePositionControl = new MousePosition({
      coordinateFormat: convertCoordinates,
      projection: "EPSG:4326",
      target: this.mousePositionRef.current,
      /**
       * Don't use their crappy style
       */
      className: "",
    });
 
    this.props.map.addControl(mousePositionControl);
  }
  render() {
    return (
      <MouseIndicatorDiv>
        <MouseIndicatorText innerRef={this.mousePositionRef} />
      </MouseIndicatorDiv>
    );
  }
}
 
export default MouseIndicator;
