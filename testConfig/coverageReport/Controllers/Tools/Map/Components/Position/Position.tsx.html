import React, { Component } from "react";
import styled from "react-emotion";
 
const topMargin = ".5rem";
const bottomMargin = ".5rem";
const marginLeft = "3rem";
const marginRight = "3.1rem";
 
const StyledDetail = styled("div")`
  position: absolute;
  z-index: 10;
  border-radius: 4px;
  box-shadow: 0 0.188em 0.125em #777;
  max-width: 33vw;
`;
 
StyledDetail.displayName = "StyledDetail";
 
const TopLeftStyledDetail = styled(StyledDetail)`
  top: ${topMargin};
  left: ${marginLeft};
`;
TopLeftStyledDetail.displayName = "TopLeftStyledDetail";
 
const TopRightStyledDetail = styled(StyledDetail)`
  top: ${topMargin};
  right: ${marginRight};
`;
TopRightStyledDetail.displayName = "TopRightStyledDetail";
 
const BottomRightStyledDetail = styled(StyledDetail)`
  bottom: ${bottomMargin};
  right: ${marginRight};
`;
BottomRightStyledDetail.displayName = "BottomRightStyledDetail";
 
const BottomLeftStyledDetail = styled(StyledDetail)`
  bottom: ${bottomMargin};
  left: ${marginLeft};
`;
BottomLeftStyledDetail.displayName = "BottomLeftStyledDetail";
 
interface Props {
  position?: string;
  children?: any;
}
 
const positions = {
  "top-left": TopLeftStyledDetail,
  "top-right": TopRightStyledDetail,
  "bottom-left": BottomLeftStyledDetail,
  "bottom-right": BottomRightStyledDetail,
};
 
class Position extends Component<Props, undefined> {
  static defaultProps: Partial<Props> = {
    position: "bottom-left",
  };
 
  render() {
    const { position: pos, children } = this.props;
    const Comp = positions[pos] ? positions[pos] : positions[Position.defaultProps.position];
    return <Comp>{children}</Comp>;
  }
}
 
export default Position;
