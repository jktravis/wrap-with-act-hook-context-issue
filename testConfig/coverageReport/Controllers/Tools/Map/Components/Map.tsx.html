    this.setState(({ wmsLoadStatus }) => ({
      wmsLoadStatus: loadingMachine.transition(wmsLoadStatus, LoadingStatusEvents.onLoad).value,
    }));
    getIncidentServices(this.props.project.id)
      .then(data => {
        const { wmsLayers: currentWms } = this.state;
 
        /**
         * Map the data to promises so that we can mitigate any issue with nested async calls.
         * The result of these will be an array of objects that can be keyed and stored in state.
         */
        Promise.all(
          data.map(datum => {
            return new Promise((resolve, reject) => {
              const { WMSLayerGUID: wmsId } = datum;
 
              /**
               * If the item is _currently_ in state, just leave it alone, and do nothing else.
               */
              const stateWmsItem = currentWms[wmsId];
              if (stateWmsItem) {
                resolve(stateWmsItem);
              }
 
              /**
               * We'll add new layers at the end, so just make sure to remove any previous versions.
               */
              removeLayerByKeyVal(this.map, "WMSLayerGUID", wmsId);
 
              const layerNames = datum.IncludedLayers.map(({ Name }) => Name);
 
              if (datum.IsWFS) {
                /**
                 * Process WFSssss
                 */
                const featureRequest = new WFS().writeGetFeature({
                  srsName: "EPSG:3857",
                  featureTypes: layerNames,
                  outputFormat: "application/json",
                } as any);
 
                getWfsFeature(datum.URL, new XMLSerializer().serializeToString(featureRequest)).then(data => {
                  const features = new GeoJSON().readFeatures(data);
 
                  const source = new VectorSource({
                    features,
                  });
 
                  const wmsLayer = new VectorLayer({
                    source,
                    /**
                     * Note that it's _min_ zoom level that is used for _max_ resolution
                     */
                    maxResolution: getResolutionForZoom(this.mapView, parseInt(datum.MinZoomLevel, 10)),
                    /**
                     * Note that it's _max_ zoom level that is used for _min_ resolution
                     */
                    minResolution: getResolutionForZoom(this.mapView, parseInt(datum.MaxZoomLevel, 10)),
                  });
 
                  wmsLayer.WMSLayerGUID = datum.WMSLayerGUID;
                  wmsLayer.IsWFS = datum.IsWFS;
                  wmsLayer.WMSExtent = datum.WMSExtent = source.getExtent();
 
                  this.map.addLayer(wmsLayer);
 
                  resolve(mapWmsData(datum));
                });
              } else {
                /**
                 * Process WMSssss
                 */
                const tileWMSOptions = {
                  url: datum.URL,
                  params: { LAYERS: layerNames.toString(), TILED: true },
                };
 
                const layer = new TileLayer({
                  source: new TileWMS(tileWMSOptions),
                  /**
                   * Note that it's _min_ zoom level that is used for _max_ resolution
                   */
                  maxResolution: getResolutionForZoom(this.mapView, parseInt(datum.MinZoomLevel, 10)),
                  /**
                   * Note that it's _max_ zoom level that is used for _min_ resolution
                   */
                  minResolution: getResolutionForZoom(this.mapView, parseInt(datum.MaxZoomLevel, 10)),
                });
 
                layer.WMSLayerGUID = datum.WMSLayerGUID;
                layer.IsWFS = datum.IsWFS;
 
                /**
                 * WMS data is wild west. So, check that the layers have values good enough to make
                 * an extent.
                 */
                const layersWithValidExtents = datum.IncludedLayers.filter(isExtentValid);
                const extent = getExtentFromMinMaxXY(layersWithValidExtents);
 
                /**
                 * Leave extent empty so we can disable the button when no extent is present.
                 */

                  layer.WMSExtent = datum.WMSExtent = extent;
                }
 
                this.map.addLayer(layer);
                resolve(mapWmsData(datum));
              }
            });

  /**
   * Creates a new function that will be used to handle the buttons from the drawing tools window.
   * @param {string} type The "type" of drawing as described by OpenLayers
   * @param {DrawingEvents} drawingEvent The event to be used in the transition when activating.
   */
  createDrawFn = (type: string, drawingEvent: DrawingEvents) => () => {
    const currentDrawing = new Draw({ source: this.drawingSource, type });
 
    /**
     * I hate uninitialized variables that are to be set at some later
     * time, but you have no idea when, really. But here we are... 
     * It's needed here so it can be set in the drawstart, and removed
     * in the drawend.
     */
    let listener;
 
    currentDrawing.on("drawstart", ({ feature, coordinate }) => {
      this.measureTooltip.setPosition(coordinate);
 
      /**
       * Update the measure tooltip when the geometry changes.
       */
      listener = feature.getGeometry().on("change", geometry => {
        this.measureTooltipElement.innerHTML = this.getGeometricMeasurement(geometry);
      });
 
      this.setState(() => ({ drawingFeature: feature }));
    });
 
    currentDrawing.on("drawend", ({ feature }) => {
      this.map.removeInteraction(currentDrawing);
 
      /**
       * Stop listening to the change event set in drawstart
       */
      unByKey(listener);
 
      this.handleDoubleClickZoomDrawing();
 
      this.measureTooltipElement.className = "tooltip tooltip-static";
 
      this.setState(({ drawingStateValue }) => ({
        drawingFeature: feature,
        drawingStateValue: transitionDrawing(drawingStateValue, DrawingEvents.onSave),
      }));
    });
 
    this.map.addInteraction(currentDrawing);
 
    this.createMeasureTooltip();
    this.createHelpTooltip();
 
    this.setState(({ drawingStateValue }) => {
      return {
        currentDrawing,
        drawingStateValue: transitionDrawing(drawingStateValue, drawingEvent),
      };
    });
  };
