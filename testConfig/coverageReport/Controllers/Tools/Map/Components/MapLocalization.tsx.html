import React, { Component } from "react";
import LocalizationContext from "./MapLocalizationContext";
import { getTranslations } from "Api/translationApi";
 
const { Provider: MapLocalizationProvider, Consumer: MapLocalizationConsumer }: any = LocalizationContext;
 
class MapLocalization extends Component<any, any> {
  state = {
    //RootResource
    UserDetailTitle: "",
    ViewInToolTitle: "",
    ShowDetailsTitle: "",
    ExpandDetails: "",
    Edit: "",
    RemoveTitle: "",
    BottomLeft: "Bottom Left",
    BottomRight: "Bottom Right",
    TopLeft: "Top Left",
    TopRight: "Top Right",
    Position: "Position",
    NoItemsToDisplay: "No items to display",
    IncidentInfo: "Incident Information",
    OrganizationColumn: "Organization",
 
    // Mapping
    ShowHideFeature: "",
    ZoomTitle: "",
    PanTitle: "",
    BackgroundDataLayersLabel: "",
    FacilityWindowLabel: "",
    IncidentDataLayerList: "",
    FacilityTitle: "Facility",
    CategoryTitle: "Category",
  };
 
  componentDidMount() {
    Promise.all([getTranslations(Object.keys(this.state)), getTranslations(Object.keys(this.state), "Mapping")]).then(
      ([root, mapping]) => {
        this.setState(() => ({ ...root, ...mapping }));
      },
    );
  }
 
  render() {
    return <MapLocalizationProvider value={this.state}>{this.props.children}</MapLocalizationProvider>;
  }
}
 
export { MapLocalizationConsumer };
export default MapLocalization;
