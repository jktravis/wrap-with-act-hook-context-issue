import React, { Component } from "react";
import styled from "react-emotion";
import { darken } from "polished";
 
import CobraLogoSvg from "./CobraLogoSvg";
 
const ToolbarLogoButton = styled("button")`
  background-color: #3e4246;
  display: inline-block;
  font-weight: 400;
  text-align: center;
  vertical-align: middle;
  user-select: none;
  border: 1px solid hsla(7, 58%, 38%, 1);
  font-size: 1rem;
  width: 42px;
  overflow: hidden;
  padding: 0;
  &:focus {
    outline: 0;
    box-shadow: 0 0 0 0.2rem rgba(130, 138, 145, 0.5);
  }
`;
 
ToolbarLogoButton.displayName = "ToolbarLogoButton";
 
interface Props {
  clickFn: Function;
  isActive?: boolean;
}
 
class CobraLogo extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    isActive: false,
    clickFn: Function.prototype,
  };
 
  state = {
    backgroundColor: "hsl(4, 70%, 50%)",
    hover: false,
  };
 
  hoverIn = () => {
    this.setState(() => ({ hover: true }));
  };
 
  hoverOut = () => {
    this.setState(() => ({ hover: false }));
  };
 
  render() {
    const { backgroundColor, hover } = this.state;
    const { clickFn, isActive } = this.props;
    const color = isActive || hover ? darken(0.1, backgroundColor) : backgroundColor;
    return (
      <ToolbarLogoButton onMouseEnter={this.hoverIn} onMouseLeave={this.hoverOut} onClick={clickFn as any}>
        <CobraLogoSvg backgroundColor={color} />
      </ToolbarLogoButton>
    );
  }
}
 
export default CobraLogo;
