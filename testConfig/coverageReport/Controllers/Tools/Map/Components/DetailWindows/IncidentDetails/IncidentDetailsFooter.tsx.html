import React from "react";
 
import getCurrentUrlPrefix from "Utilities/getCurrentUrlPrefix";
import { FaLink } from "react-icons/fa";
import { MapLocalizationConsumer } from "Controllers/Tools/Map/Components/MapLocalization";
 
function IncidentDetailsFooter({ feature }) {
  return (
    <div
      className="d-flex justify-content-between align-items-center"
      style={{
        width: "100%",
        padding: "0 0.5rem",
      }}
    >
      {feature.User && (
        <MapLocalizationConsumer>
          {({ UserDetailTitle }) => (
            <a
              href={`${getCurrentUrlPrefix()}Admin/User/GetUserDetailedPageByGUID?CoBRAUserGUID=${
                feature.CoBRAUserGUID
              }`}
              title={UserDetailTitle}
              aria-label={UserDetailTitle}
              target="_blank"
            >
              <span>{feature.User}</span>
            </a>
          )}
        </MapLocalizationConsumer>
      )}
      <small>
        <time data-format="lll" dateTime={feature.UnixDateCreated}>
          {feature.DateCreated}
        </time>
      </small>
      {feature.Url && (
        <MapLocalizationConsumer>
          {({ ViewInToolTitle }) => (
            <a href={feature.Url} title={ViewInToolTitle} aria-label={ViewInToolTitle} target="_blank">
              <FaLink />
            </a>
          )}
        </MapLocalizationConsumer>
      )}
    </div>
  );
}
 
export default IncidentDetailsFooter;
