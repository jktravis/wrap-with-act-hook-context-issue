import React, { Component, Fragment } from "react";
 
import Media from "reactstrap/lib/Media";
import { not, allPass, isNil, either, isEmpty, prop, compose } from "ramda";
import { MapLocalizationConsumer } from "Controllers/Tools/Map/Components/MapLocalization";
import styled from "react-emotion";
 
const has = propName =>
  compose(
    not,
    either(isNil, isEmpty),
    prop(propName),
  );
 
const hasImageURL = has("ImageURL");
const hasSubTitle = has("SubTitle");
const hasFormattedDescription = has("FormattedDescription");
const hasDescription = has("Description");
 
const shouldShowDefault = allPass([
  compose(
    not,
    hasImageURL,
  ),
  compose(
    not,
    hasSubTitle,
  ),
  compose(
    not,
    hasFormattedDescription,
  ),
  compose(
    not,
    hasDescription,
  ),
]);
 
const NoDetail = styled("div")`
  display: flex;
  justify-content: center;
  align-items: center;
  width: 100%;
`;
 
interface Props {
  feature: any;
}
 
class IncidentDetails extends Component<Props, undefined> {
  render() {
    const { feature } = this.props;
    return (
      <Media style={{ overflow: "auto", padding: "1rem" }}>
        {shouldShowDefault(feature) ? (
          <MapLocalizationConsumer>
            {({ NoDetailAvailable }) => <NoDetail>{NoDetailAvailable}</NoDetail>}
          </MapLocalizationConsumer>
        ) : (
          <Fragment>
            {feature.ImageURL && (
              <div className="mr-2">
                <img src={`${feature.ImageURL}?w=35&mode=max`} alt={feature.FileName} />
              </div>
            )}
            <Media body>
              {feature.SubTitle && <small>{feature.SubTitle}</small>}
              {feature.FormattedDescription ? (
                <div
                  dangerouslySetInnerHTML={{
                    __html: feature.FormattedDescription,
                  }}
                />
              ) : null}
              <p
                dangerouslySetInnerHTML={{
                  __html: feature.Description,
                }}
              />
            </Media>
          </Fragment>
        )}
      </Media>
    );
  }
}
 
export default IncidentDetails;
