import React, { useContext } from "react";
import {
  FaAtlas,
  FaBuilding,
  FaCamera,
  FaClipboardList,
  FaCogs,
  FaEdit,
  FaExpand,
  FaGlobeAmericas,
  FaInfo,
  FaMapMarker,
} from "react-icons/fa";
import { IoIosCellular } from "react-icons/io";
import styled from "react-emotion";
import MenuButtonCheck from "Controllers/Tools/Map/Components/Toolbar/MenuButtonCheck";
import Button from "reactstrap/lib/Button";
import { MapLocalizationContext } from "Controllers/Tools/Map/Components/MapLocalization";
import { MdFullscreenExit } from "react-icons/md";
 
const ToolbarContainer = styled("div")`
  display: flex;
  flex-direction: column;
  justify-content: space-between;
  background-color: #6d747d;
`;
ToolbarContainer.displayName = "ToolbarContainer";
 
const ToolDropArea = styled("div")`
  min-height: 9em;
  border: 0.3em #ffcc00 dashed;
  width: 100%;
`;
// @ts-ignore
ToolDropArea.displayName = ToolDropArea;
 
interface ToggleProps {
  isOpen: boolean;
  toggleFn: (any) => void;
}
 
interface HasValuesProps {
  hasValues: boolean;
}
 
interface ToggleHasValuesProps extends ToggleProps, HasValuesProps {}
 
interface Props {
  incident?: ToggleHasValuesProps;
  facility?: ToggleHasValuesProps;
  background?: ToggleHasValuesProps;
  wms?: ToggleHasValuesProps;
  project?: ToggleHasValuesProps;
  locator?: ToggleHasValuesProps;
  drawing?: ToggleProps;
  settings?: ToggleProps;
  goToLocation?: ToggleProps;
  fullScreen: ToggleProps;
}
 
function Toolbar(props: Props) {
  const { incident, facility, background, wms, project, locator, drawing, settings, goToLocation, fullScreen } = props;
  const {
    BackgroundDataLayersLabel,
    ExpandContract,
    FacilityWindowLabel,
    GoToLocation,
    IncidentDataLayerList,
    IncidentInfo,
    LocatorData,
    MapSketchToolsTitle,
    SaveMapAsAttachment,
    SettingsTitle,
    WmsWfsLabel,
  } = useContext(MapLocalizationContext);
 
  return (
    <ToolbarContainer>
      <div className="btn-group-vertical">
        {incident ? (
          <MenuButtonCheck
            clickFn={incident.toggleFn}
            title={IncidentDataLayerList}
            Icon={FaClipboardList}
            isActive={incident.isOpen}
            showCheck={incident.hasValues}
          />
        ) : null}
 
        {facility ? (
          <MenuButtonCheck
            clickFn={facility.toggleFn}
            title={FacilityWindowLabel}
            Icon={FaBuilding}
            isActive={facility.isOpen}
            showCheck={facility.hasValues}
          />
        ) : null}
 
        {background ? (
          <MenuButtonCheck
            clickFn={background.toggleFn}
            title={BackgroundDataLayersLabel}
            Icon={FaAtlas}
            isActive={background.isOpen}
            showCheck={background.hasValues}
          />
        ) : null}
 
        {wms ? (
          <MenuButtonCheck
            clickFn={wms.toggleFn}
            title={WmsWfsLabel}
            Icon={FaGlobeAmericas}
            isActive={wms.isOpen}
            showCheck={wms.hasValues}
          />
        ) : null}
 
        {locator && locator.hasValues ? (
          <MenuButtonCheck
            clickFn={locator.toggleFn}
            title={LocatorData}
            Icon={IoIosCellular}
            isActive={locator.isOpen}
            showCheck={locator.hasValues}
          />
        ) : null}
 
        {project ? (
          <MenuButtonCheck
            clickFn={project.toggleFn}
            title={IncidentInfo}
            Icon={FaInfo}
            isActive={project.isOpen}
            showCheck={project.hasValues}
          />
        ) : null}
      </div>
 
      <div>
        <div className="btn-group-vertical">
          {drawing ? (
            <Button title={MapSketchToolsTitle} onClick={drawing.toggleFn} active={drawing.isOpen}>
              <FaEdit />
            </Button>
          ) : null}
          <button className="btn btn-secondary" title={SaveMapAsAttachment}>
            <FaCamera />
          </button>
 
          {fullScreen && (
            <Button title={ExpandContract} onClick={fullScreen.toggleFn}>
              {fullScreen.isOpen ? <MdFullscreenExit /> : <FaExpand />}
            </Button>
          )}
 
          {goToLocation ? (
            <Button id="Location" title={GoToLocation} onClick={goToLocation.toggleFn} active={goToLocation.isOpen}>
              <FaMapMarker />
            </Button>
          ) : null}
 
          {settings ? (
            <Button id="Settings" title={SettingsTitle} onClick={settings.toggleFn} active={settings.isOpen}>
              <FaCogs />
            </Button>
          ) : null}
        </div>
      </div>
    </ToolbarContainer>
  );
}
 
export default Toolbar;
