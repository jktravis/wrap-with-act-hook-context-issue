import React, { Component, ComponentType } from "react";
import Button from "reactstrap/lib/Button";
import { FaCheck } from "react-icons/fa";
 
interface Props {
  clickFn: Function;
  title: string;
  Icon: ComponentType<any>;
  isActive?: boolean;
  showCheck?: boolean;
}
 
class MenuButtonCheck extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    isActive: false,
    showCheck: false,
  };
 
  render() {
    const { isActive, showCheck, clickFn, Icon, title } = this.props;
    return (
      <Button title={title} active={isActive} onClick={clickFn as any}>
        <Icon
          style={{
            position: showCheck ? "absolute" : "relative",
            top: showCheck ? "12px" : "0",
          }}
        />
        {showCheck && <FaCheck color="#1ddc98" style={{ position: "relative", right: "-4px", top: "2px" }} />}
      </Button>
    );
  }
}
 
export default MenuButtonCheck;
