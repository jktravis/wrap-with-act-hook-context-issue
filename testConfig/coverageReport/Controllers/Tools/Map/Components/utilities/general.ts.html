/**
 * Checks that the IHasMinMaxXY extent value actually has values at runtime.
 * Funny how type-checking can't help here. ðŸ˜’
 */
function isExtentValid({ MinX, MinY, MaxX, MaxY }) {
  return [MinX, MinY, MaxX, MaxY].every(
    compose(
      not,
      isNil,
    ),
  );
}
 
interface IHasMinMaxXY {
  MinX: number | string;
  MinY: number | string;
  MaxX: number | string;
  MaxY: number | string;
  EPSG: string;
}
/**
 * Extracts an "extent" suitable for OpenLayers
 * @param {} layers
 */
function getExtentFromMinMaxXY(layers: IHasMinMaxXY[]) {
  if (layers.length === 1) {
    const [layer] = layers;
    return createAndConvertExtent(layer);
  } else Eif (layers.length > 1) {
    return layers.reduce((result: any, layer: any) => {
      if (result) {
        return immutableExtend(result, createAndConvertExtent(layer));
      }
      return createAndConvertExtent(layer);
    }, null);
  }
}
 
/**
 * NOTE that 4326 has a flipped x/y.
 */
function createExtentFromMinMaxValues({ MinX, MinY, MaxX, MaxY, ...rest }) {
  const extent = getEpsg(rest) === "4326" ? [MinY, MinX, MaxY, MaxX] : [MinX, MinY, MaxX, MaxY];
  return extent.map(parseFloat);
}
 
/**
 * We curry here so we can pre-load the layer, and just wait
 * for the extent later, allowing us to compose.
 */
const convertExtentTo3857 = curry((layer, ext) => {
  return tryCatch(
    sublayer =>
      transformExtent(ext, getProjection(`EPSG:${convertEpsg84To4326(getEpsg(sublayer))}`), getProjection("EPSG:3857")),
    always(ext),
  )(layer);
});
 
/**
 * Simple getter that provides a default if the value doesn't exist
 */
const getEpsg = compose(
  ifElse(equals(""), always("3857"), identity),
  propOr("3857", "EPSG"),
);
 
/**
 * A function that takes the EPSG value and either returns the converted
 * value from 84 -> 4326, or the value that it was given (aka, identity)
 */
const convertEpsg84To4326 = ifElse(equals("84"), () => "4326", identity);
 
/**
 * A function composition that brings all of these together
 */
const createAndConvertExtent = layer => {
  return compose(
    convertExtentTo3857(layer),
    createExtentFromMinMaxValues,
  )(layer);
};

 * There's some issue that this is supposed to solve in lieu of using the one provided by

}
