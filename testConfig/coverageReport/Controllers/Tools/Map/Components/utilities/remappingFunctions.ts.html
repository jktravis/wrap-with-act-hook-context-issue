import uuid from "uuid";
import { compose, prop, indexBy, curry, equals } from "ramda";
import produce from "immer";
 
import FeatureDataType from "./FeatureDataTypeEnum";
 
/**
 * Maps the Visible property to false when rootOrgName equals the Title of the feature.
 * @param {string} rootOrgName The root organization's name with which to compare.
 * @param {{Title: string, Visible: boolean}} feature The feature to check
 * @returns {Function | Feature} This function is curried and will return a new function if provided fewer args than
 * params.
 */
const mapRootOrgDefaultHidden = curry((rootOrgName: string, feature: { Title: string; Visible: boolean }) => {
  return produce(feature, draft => {
    if (equals(feature.Title, rootOrgName)) {
      draft.Visible = false;
    }
  });
});
 
function mapBackgroundLayerProperties(feature) {
  return produce(feature, draft => {
    draft.PrimaryKey = draft.BackgroundLayerGUID;
    draft.Title = draft.BackgroundLayerName;
    draft.Visible = true;
    draft.ShowDetail = false;
    draft.DataType = FeatureDataType.BackgroundLayerGUID;
    draft.BackgroundGeoJson = {
      ...draft.Features,
    };
  });
}
 
function mapBackgroundSubFeature(feature) {
  return produce(feature, (draft: any) => {
    draft.PrimaryKey = draft.get("BackgroundFeatureGUID");
    draft.BackgroundLayerGUID = draft.get("BackgroundLayerGUID");
    draft.BackgroundFeatureGUID = draft.get("BackgroundFeatureGUID");
    draft.BackgroundFeatureName = draft.get("SubTitle");
    draft.DataType = FeatureDataType.BackgroundLayerSubFeature;
    draft.Title = draft.get("Title");
    draft.SubTitle = draft.get("SubTitle");
    draft.UnixDateCreated = draft.get("UnixDateCreated");
    draft.DateCreated = draft.get("DateCreated");
    draft.Visible = true;
    draft.ShowDetail = false;
  });
}

/**
 * Since I have all these specific mapping functions, do I really need one that
 * tries to figure out the type? Can an incident have different kinds of Ids?
 * @param feature
 */
function mapFeatureToId(feature) {
  return produce(feature, draft => {
    if (draft.get("LogEntryGUID")) {
      draft.PrimaryKey = draft.get("LogEntryGUID");
      draft.DataType = FeatureDataType.LogEntryGUID;
    } else if (draft.get("ProjectGUID")) {
      draft.PrimaryKey = draft.get("ProjectGUID");
      draft.DataType = FeatureDataType.ProjectGUID;
    } else if (draft.get("RootOrganization")) {
      draft.PrimaryKey = draft.get("RootOrganization");
      draft.DataType = FeatureDataType.RootOrganization;
    } else if (draft.get("FacilityGUID")) {
      draft.PrimaryKey = draft.get("FacilityGUID");
      draft.DataType = FeatureDataType.FacilityGUID;
    } else if (draft.get("BackgroundLayerGUID")) {
      draft.PrimaryKey = draft.get("BackgroundFeatureGUID");
      draft.SecondaryKey = draft.get("BackgroundLayerGUID");
      draft.DataType = FeatureDataType.BackgroundLayerGUID;
    } else if (draft.get("LocatorSerialNumber")) {
      draft.PrimaryKey = draft.get("LocatorSerialNumber");
      draft.DataType = FeatureDataType.LocatorData;
    } else {
      draft.PrimaryKey = uuid.v4();
      draft.DataType = FeatureDataType.Unknown;
    }
  });
}
 
function mapFeatureIsKml(feature) {
  return produce(feature, draft => {
    const kmlURL = draft.get("KMLUrl");
    draft.IsKML = !!(kmlURL && kmlURL.length);
  });
}

}
