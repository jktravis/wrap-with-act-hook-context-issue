import { curry, prop, assoc, compose, map, tap } from "ramda";
 
/**
 * Map over an array adding new "value" and "key" properties with values from
 * the corresponding keys that already exist on the object.
 * @param {string} valueKey
 * @param {string} labelKey
 * @param {any[]} data The data that will be mapped over.
 * @returns {any[]} A new array with the new mappings.
 */
const mapValueLabelKeys = curry((valueKey: string, labelKey: string, data: any[]) => {
  return data.map(item => {
    return compose(
      assoc("value", prop(valueKey, item)),
      assoc("label", prop(labelKey, item)),
    )(item);
  });
});
 
export default mapValueLabelKeys;
