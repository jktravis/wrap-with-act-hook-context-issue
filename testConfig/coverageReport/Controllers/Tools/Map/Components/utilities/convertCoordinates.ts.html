/* global
Main,
*/
import { toStringXY } from "ol/coordinate.js";
import usng from "usng/usng.js";
import { css } from "emotion";
import { compose, join, reverse, split } from "ramda";
 
const tableClass = css`
  font-size: 0.75rem;
`;
 
function convertCoordinates(coordinate) {
  const [lon, lat] = coordinate;
  let coords = toStringXY(coordinate, 4);
  let mgrs;
  if (Main && Main.serverBuildType !== "Belgium") {
    const converter = new usng.Converter();
    mgrs = converter.LLtoMGRS(lat, lon, 5);
  }
  return coordinateTemplate(coords, mgrs);
}
 
function coordinateTemplate(coords, mgrs) {
  return `
<table class="${tableClass}">
  <tbody>
    <tr>
      <td>LAT/LON:</td>
      <td>${flipLonLat(coords)}</td>
    </tr>
    ${
      mgrs
        ? `
      <tr>
        <td>MGRS:</td>
        <td>${mgrs}</td>
      </tr>
    `
        : ""
    }
  </tbody>
</table>
  `;
}
 
const flipLonLat = compose(
  join(", "),
  // @ts-ignore
  reverse,
  split(","),
);
 
export default convertCoordinates;
