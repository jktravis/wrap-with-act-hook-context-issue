function getColorForFeature(feature) {
  const defaultColor = feature.ShowDetail ? "rgba(255,0,0,0.9)" : "rgba(52,58,64,0.9)";
 
  return {
    markerColor: getMarkerColor(feature),
    fillColor: feature.get("fill") || defaultColor,
    strokeColor: feature.get("stroke") || defaultColor,
    strokeWidth: feature.get("stroke-width") || 1,
  };
}
 
/**
 * Get the marker color. If it's one of the "badColors", return undefined to disable the color.
 * @param feature
 */
function getMarkerColor(feature) {
  const badColors = ["Blank", "Light"];
  return !badColors.includes(feature.get("marker-color")) ? feature.get("marker-color") : undefined;
}
 
function getTextStyle(iconSize: IconSize, feature) {
  const text = feature.Title;
  if (text) {
    return new Text({
      font: `${iconSize}px bold Gruppo,Lato`,
      fill: new Fill({
        color: "#212121",
      }),
      offsetY: -iconSize,
      text,
    });
  }
}
 
const noStyle = new Style(null);
 
const getStyleForFeature = curry((iconSize: number, showText: boolean, feature: any) => {
  /**
   * Don't style hidden or KML features.
   * But _do_ style features with a missing `Visible` prop
   */
  if ((feature.Visible !== undefined && !feature.Visible) || feature.isKML) {
    return noStyle;
  }
 
  const featColor = getColorForFeature(feature);
 
  return new Style({
    stroke: new Stroke({
      color: featColor.strokeColor,
      width: featColor.strokeWidth,
    }),
    fill: new Fill({
      color: featColor.fillColor,
    }),
    image: getIconStyle(feature, featColor, iconSize),
    text: showText ? getTextStyle(iconSize, feature) : undefined,
  });
});
