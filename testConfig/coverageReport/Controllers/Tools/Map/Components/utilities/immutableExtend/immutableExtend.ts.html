/**
 * A replacement for the ol/extend module that does not mutate the array, but instead
 * returns a new array with the extended extent.
 * @link https://github.com/openlayers/openlayers/blob/9285b283c0481d50af0195471d38d269432aa590/src/ol/extent.js#L306
 */
function immutableExtend(extent1, extent2) {
  const result1 = [...extent1];
  const result2 = [...extent2];
  if (result2[0] < result1[0]) {
    result1[0] = result2[0];
  }
  if (result2[2] > result1[2]) {
    result1[2] = result2[2];
  }
  if (result2[1] < result1[1]) {
    result1[1] = result2[1];
  }
  if (result2[3] > result1[3]) {
    result1[3] = result2[3];
  }
  return result1;
}
 
export default immutableExtend;
