import TileLayer from "ol/layer/Tile";
import TileImage from "ol/source/TileImage";
import OSM from "ol/source/OSM";
 
const mapTypeOptions = Object.freeze(
  [
    { value: "OSM", label: "Open Street Maps", layer: new TileLayer({ source: new OSM() }) },
    {
      value: "GoogleRoads",
      label: "Google Roads",
      layer: new TileLayer({
        source: new TileImage({
          url: "http://mt1.google.com/vt/lyrs=m&x={x}&y={y}&z={z}",
          projection: "EPSG:3857",
          /**
           * @link {https://stackoverflow.com/questions/22710627/tainted-canvases-may-not-be-exported}
           */
          crossOrigin: "Anonymous",
        }),
      }),
    },
    /**
     * Google Roads and Hybrid both appear to be configured the same from looking at previous art.
     */
    {
      value: "GoogleHybrid",
      label: "Google Hybrid",
      layer: new TileLayer({
        source: new TileImage({
          url: "http://mt1.google.com/vt/lyrs=y&x={x}&y={y}&z={z}",
          projection: "EPSG:3857",
          crossOrigin: "Anonymous",
        }),
      }),
    },
    {
      value: "GoogleSatellite",
      label: "Google Satellite",
      layer: new TileLayer({
        source: new TileImage({
          url: "http://mt1.google.com/vt/lyrs=s&hl=pl&&x={x}&y={y}&z={z}",
          projection: "EPSG:3857",
          crossOrigin: "Anonymous",
        }),
      }),
    },
  ].map(Object.freeze),
);
 
export default mapTypeOptions;
