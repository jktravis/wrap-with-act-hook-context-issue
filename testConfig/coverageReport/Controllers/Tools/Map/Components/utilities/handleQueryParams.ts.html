import { fromLonLat, transformExtent } from "ol/proj";
import {
  all,
  always,
  both,
  compose,
  curry,
  equals,
  identity,
  ifElse,
  isEmpty,
  length,
  map,
  not,
  pathOr,
  split,
} from "ramda";
 
import qs from "query-string";
 
const lengthEq2 = compose(
  equals(2),
  length,
);
 
const notEmpty = compose(
  not,
  isEmpty,
);
 
const areNoneEmpty = all(notEmpty);
 
const isLength2AndNoneEmpty = both(lengthEq2, areNoneEmpty);
 
const notNaN = compose(
  not,
  isNaN,
);
 
const areNoneNaN = all(notNaN);
 
const isLength2AndNoneNaN = both(lengthEq2, areNoneNaN);
 
function handleQueryParams(mapView, history) {
  if (hasLonLatParam(history)) {
    const center = fromLonLat(getLatLonParamFromHistory(history));
    mapView.animate({
      center,
      duration: 1000,
    });
    return;
  }
  if (hasCurrentMapCenter(history) && hasCurrentMapZoom(history)) {
    const center = getCurrentMapCenterFromHistory(history);
    const zoom = getCurrentMapZoomParamFromHistory(history);
    mapView.animate({
      center,
      zoom,
      duration: 1000,
    });
  }
}
 
const getCurrentMapZoomParamFromHistory = compose(
  Number,
  pathOr("", ["CurrentMapZoom"]),
  qs.parse,
  pathOr("", ["location", "search"]),
);
 
const getCurrentMapCenterFromHistory = compose(
  map(Number),
  ifElse(isLength2AndNoneEmpty, identity, always([])),
  split(","),
  pathOr("", ["CurrentMapCenter"]),
  qs.parse,
  pathOr("", ["location", "search"]),
);
 
const hasCurrentMapCenter = compose(
  isLength2AndNoneNaN,
  getCurrentMapCenterFromHistory,
);
 
const hasCurrentMapZoom = compose(
  notNaN,
  getCurrentMapZoomParamFromHistory,
);
 
/**
 * Creates a function that will determine if the current coords is the map center
 * in the query parameters. The final function takes the history object as created
 * from the history library, extracts the "location.search" prop, and parses the
 * result. If "CurrentMapCenter" exists, it will be split, converted and compared,
 * with the lon/lat provided.
 *
 * This function provides it's own conversion of lon/lat. No need to do that
 * explicitly.
 *
 * @param {number[]} coords The non-converted longitude and latitude array.
 * @returns {(history: {location: {search: string}}) => boolean} Final function
 */
const isUsingCoords = curry(coords =>
  compose(
    equals(fromLonLat(coords)),
    map(Number),
    split(","),
    pathOr("", ["CurrentMapCenter"]),
    qs.parse,
    pathOr("", ["location", "search"]),
  ),
);
 
const getLatLonParamFromHistory = compose(
  map(Number),
  ifElse(isLength2AndNoneEmpty, identity, always([])),
  split(","),
  pathOr("", ["LonLat"]),
  qs.parse,
  pathOr("", ["location", "search"]),
);
 
const hasLonLatParam = compose(
  isLength2AndNoneNaN,
  getLatLonParamFromHistory,
);
 
export { getLatLonParamFromHistory, hasLonLatParam, isUsingCoords };
export default handleQueryParams;
