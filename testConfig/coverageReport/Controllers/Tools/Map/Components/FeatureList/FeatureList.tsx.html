import React, { Component, createRef, RefObject } from "react";
import styled from "react-emotion";
import { MapLocalizationConsumer } from "Controllers/Tools/Map/Components/MapLocalization";
import { List, CellMeasurer, CellMeasurerCache } from "react-virtualized";
 
interface Props {
  features: any[];
  /**
   * See https://github.com/bvaughn/react-virtualized/blob/master/docs/List.md#rowrenderer
   * for parameters.
   */
  rowRenderer: Function;
}
 
const NoItems = styled("div")`
  padding: 2rem;
  text-align: center;
  color: #8c8b8b;
`;
 
class FeatureList extends Component<Props, undefined> {
  static listWidth = 380;
 
  static getListHeight = featureLength => {
    return featureLength ? 425 : 88;
  };
 
  list: RefObject<List>;
  cache: CellMeasurerCache;
 
  constructor(props) {
    super(props);
 
    this.list = createRef();
 
    this.cache = new CellMeasurerCache({
      defaultHeight: 200,
      minHeight: 120,
      fixedWidth: true,
    });
  }
 
  /**
   * Clear the measurer the cache and recalculate the row heights.
   * This is only needed by your rowRenderer if you are changing the heights dynamically, e.g. expand/collapse.
   * If not, you do not need to call this in your renderer. Doing so will incur a bit of a performance cost.
   */
  recomputeHeights = () => {
    this.cache.clearAll();
    this.list.current.recomputeRowHeights();
  };
 
  render() {
    const { features, rowRenderer } = this.props;
    return (
      <List
        className="map__feature-list"
        ref={this.list}
        width={FeatureList.listWidth}
        height={FeatureList.getListHeight(features.length)}
        rowCount={features.length}
        rowHeight={this.cache.rowHeight}
        deferredMeasurementCache={this.cache}
        rowRenderer={args => {
          const { index, parent, key } = args;
          return (
            <CellMeasurer cache={this.cache} columnIndex={0} key={key} parent={parent} rowIndex={index}>
              {rowRenderer({ ...args, recomputeHeights: this.recomputeHeights })}
            </CellMeasurer>
          );
        }}
        noRowsRenderer={() => (
          <MapLocalizationConsumer>
            {({ NoItemsToDisplay }) => <NoItems>{NoItemsToDisplay}</NoItems>}
          </MapLocalizationConsumer>
        )}
        overscanRowCount={5}
        style={{
          outline: "none",
        }}
      />
    );
  }
}
 
export default FeatureList;
