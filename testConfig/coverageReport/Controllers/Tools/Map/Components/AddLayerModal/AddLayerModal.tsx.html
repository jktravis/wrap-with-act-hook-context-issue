import React, { Component } from "react";
import Modal from "reactstrap/lib/Modal";
import ModalHeader from "reactstrap/lib/ModalHeader";
import ModalBody from "reactstrap/lib/ModalBody";
import ModalFooter from "reactstrap/lib/ModalFooter";
import Button from "reactstrap/lib/Button";
import FormGroup from "reactstrap/lib/FormGroup";
import Select from "react-select";
import { FaBan, FaSave } from "react-icons/fa";
 
interface MenuOption {
  label: string;
  value: string;
}
 
interface Props {
  isOpen: boolean;
  toggleFn: () => void;
  headerText: string;
  addLayerFn: Function;
  /**
   * Data needs to have two specific keys. {label: string, value: string}.
   * Value needs to be unique across the entire collection.
   */
  getDataFn: () => Promise<any>;
}
 
class AddLayerModal extends Component<Props, any> {
  state = {
    layer: null,
    options: [],
  };
 
  componentDidMount() {
    this.props.getDataFn().then(options => this.setState(() => ({ options })));
  }
 
  handleAddClick = () => {
    const { layer } = this.state;
    this.props.addLayerFn(layer);
    this.props.toggleFn();
    this.resetState();
  };
 
  resetState = () => {
    this.setState(() => ({ layer: null }));
  };
 
  handleChange = layer => {
    this.setState(() => ({ layer }));
  };
 
  render() {
    const { isOpen, toggleFn, headerText } = this.props;
    const { options, layer } = this.state;
    return (
      <Modal isOpen={isOpen} toggle={toggleFn}>
        <ModalHeader toggle={toggleFn}>{headerText}</ModalHeader>
        <ModalBody>
          <FormGroup>
            <Select value={layer} options={options} onChange={this.handleChange} />
          </FormGroup>
        </ModalBody>
        <ModalFooter>
          <Button color="warning" onClick={toggleFn}>
            <FaBan /> Cancel
          </Button>
          <Button color="primary" disabled={!layer} onClick={this.handleAddClick}>
            <FaSave /> Add
          </Button>
        </ModalFooter>
      </Modal>
    );
  }
}
 
export default AddLayerModal;
