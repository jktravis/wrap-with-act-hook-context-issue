import React, { Component } from "react";
import Row from "reactstrap/lib/Row";
 
import getCurrentUrlPrefix from "Utilities/getCurrentUrlPrefix";
import { FaLink } from "react-icons/fa";
import { MapLocalizationConsumer } from "Controllers/Tools/Map/Components/MapLocalization";
 
interface Props {
  feature: any;
}
 
class IncidentDetails extends Component<Props, undefined> {
  render() {
    const { feature } = this.props;
    return (
      <div className="card-body">
        <Row>
          <div>{feature.SubTitle && <small>{feature.SubTitle}</small>}</div>
          <div className="col-auto">
            <small>
              <time data-format="lll" dateTime={feature.UnixDateCreated}>
                {feature.DateCreated}
              </time>
            </small>
          </div>
        </Row>
        <div className="media">
          {feature.ImageURL && (
            <div className="mr-3">
              <img className="detailImage" src={`${feature.ImageURL}?w=35&mode=max`} alt={feature.FileName} />
            </div>
          )}
          <div className="media-body">
            <div
              dangerouslySetInnerHTML={{
                __html: feature.FormattedDescription,
              }}
            />
            <p
              dangerouslySetInnerHTML={{
                __html: feature.Description,
              }}
            />
            <div className="d-flex justify-content-between">
              {feature.User && (
                <div>
                  <MapLocalizationConsumer>
                    {({ UserDetailTitle }) => (
                      <a
                        href={`${getCurrentUrlPrefix()}Admin/User/GetUserDetailedPageByGUID?CoBRAUserGUID=${
                          feature.CoBRAUserGUID
                        }`}
                        className="btn btn-link btn-sm"
                        title={UserDetailTitle}
                        aria-label={UserDetailTitle}
                        target="_blank"
                      >
                        <span>{feature.User}</span>
                      </a>
                    )}
                  </MapLocalizationConsumer>
                </div>
              )}
              {feature.Url && (
                <div>
                  <MapLocalizationConsumer>
                    {({ ViewInToolTitle }) => (
                      <a href={feature.Url} title={ViewInToolTitle} aria-label={ViewInToolTitle} target="_blank">
                        <FaLink />
                      </a>
                    )}
                  </MapLocalizationConsumer>
                </div>
              )}
            </div>
          </div>
        </div>
      </div>
    );
  }
}
 
export default IncidentDetails;
