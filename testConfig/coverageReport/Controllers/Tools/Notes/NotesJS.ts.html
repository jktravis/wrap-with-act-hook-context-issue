  function saveNoteWithoutAttachment(baseurl) {
    const rmObj = buildLogObject();
    const strTmp = JSON.stringify(rmObj);
    $Ajax.ajaxAntiForgery({
      url: baseurl,
      type: "POST",
      data: {
        modelString: strTmp,
      },
      success: function(response) {
        //first remove any validation errors that happened before
        $(".input-validation-error").each(function() {
          let el = $(this);
          el.removeClass("input-validation-error");
        });
        if (response.success === false) {
          Main.handleValErr($("#txtValSummary"), response);
          window.scrollTo(0, 0);
        } else {
          Dashboard.jqueryclosemodal();
        }
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }

  function buildLogObject() {
    let ret: NoteObject = new NoteObject();
    ret.NoteText = Main.htmlEncode($("#NoteText").val());
 
    if ($("#Lat")) {
      ret.Lat = $("#Lat").val();
    }
    if ($("#Lon")) {
      ret.Lon = $("#Lon").val();
    }
    const tmp = $("#RequestedBy_RoleGUIDs").data("kendoMultiSelect");
    //if ((tmp !==null) && (tmp !== 'undefined')) {
    if (typeof tmp === "undefined") {
      ret.RequestedBy_RoleGUIDs = JSON.stringify($("#RequestedBy_RoleGUIDs").val());
      ret.RequestedBy_UserGUIDs = JSON.stringify($("#RequestedBy_UserGUIDs").val());
    } else {
      ret.RequestedBy_RoleGUIDs = JSON.stringify(
        $("#RequestedBy_RoleGUIDs")
          .data("kendoMultiSelect")
          .value(),
      );
      ret.RequestedBy_UserGUIDs = JSON.stringify(
        $("#RequestedBy_UserGUIDs")
          .data("kendoMultiSelect")
          .value(),
      );
    }
    ret.RequestedBy_Other = $("#RequestedBy_Other").val();
    ret.LogEntryGUID = $("#LogEntryGUID").val();
    ret.StaticGUID = $("#StaticGUID").val();
    ret.Street1 = $("#Street1").val();
    ret.Street2 = $("#Street2").val();
    ret.City = $("#City").val();
    const stCombo = $("#State_Province").data("kendoComboBox");
    ret.State_Province = stCombo.text();
    ret.Zip_PostalCode = $("#Zip_PostalCode").val();
    ret.County_Region = $("#County_Region").val();
    ret.Country = $("#Country").val();
    ret.LinkUrl = $("#LinkUrl").val();
    ret.LogType = $("#LogType").val();
    ret.CopiedToSigEv = copyLogIsCheckedModal();
    ret.LinkedLogEntriesAsString = $("#LinkedLogEntriesAsString").val();
    ret.DateCreated = $("#DateCreated").val();
    return ret;
  }

  angular.module("cobraapp").factory("notesRepository", function() {
    return {
      getDataSource: function(hostURL, projectGUID, isRelevant) {
        return new kendo.data.DataSource({
          transport: {
            read: {
              url: hostURL + "api/NotesAPI/GetLogListJSON/" + projectGUID + "?isrelevant=" + isRelevant,
            },
          },
          sort: {
            field: "FormattedDateCreated",
            dir: "desc",
          },
          pageSize: 8,
          schema: {
            model: {
              id: "LogEntryGUID",
              fields: {
                LogEntryGUID: { editable: false },
                StaticGUID: { editable: false },
                FriendlyUserName: { editable: false },
                RequestedBy: { editable: false },
                NoteText: { editable: false },
                FormattedDateCreated: { type: "date", editable: false },
              },
            },
          },
        });
      },
    };
  });
