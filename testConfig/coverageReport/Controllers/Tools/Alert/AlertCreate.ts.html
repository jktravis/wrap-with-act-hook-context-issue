namespace AlertCreate {
  export function init(alertType) {
    $("#divUserTargetList").hide();
    SetRadioButtons(alertType);
    $("[name=AlertType]").click(function() {
      const val = $(this).val();
      if (val === "UsersInIncident" || val === "UsersInOrganization") {
        $("#divUserTargetList").show();
      } else {
        $("#divUserTargetList").hide();
      }
    });
    const tmp = $("#DetailsCollection").val();
    if (tmp.length > 0) {
      let detailsCollection = JSON.parse(tmp);
      buildDetailList(detailsCollection);
    }
 
    let multiselect = $("#TargetUsers").data("kendoMultiSelect"),
      setSearch = function(e) {
        if (e.type !== "keypress" || kendo.keys.ENTER === e.keyCode) {
          multiselect.search($("#word").val());
          $("#word").val("");
        }
      };
 
    $("#find").click(setSearch);
    $("#word").keypress(setSearch);
    if (multiselect !== null && multiselect !== undefined) {
      multiselect.options.filter = "contains";
    }
 
    $("#isLinkCheckbox").change(function() {
      if ($(this).is(":checked")) {
        $("#txtDetailText").prop("placeholder", "Enter Link");
      } else {
        $("#txtDetailText").prop("placeholder", "Enter Detail");
      }
    });
    function filterTypeOnChanged() {
      multiselect.options.filter = $("#filter").val();
    }
 
    $("#myModal").on("hidden.bs.modal", function() {
      //reset save button back to add new detail (instead of a potential edit detail action)
      $("#SaveDetailButton").attr("onclick", "addnewdetail();");
    });
  }

  function SetRadioButtons(alertType) {
    let showUsers = false;
    if (alertType === "IncidentWide") {
      $("#IncidentWideRB").prop("checked", true);
    } else if (alertType === "OrganizationWide") {
      $("#OrganizationWideRB").prop("checked", true);
    } else if (alertType === "UsersInIncident") {
      $("#UsersInIncidentRB").prop("checked", true);
      showUsers = true;
    }
    if (alertType === "UsersInOrganization") {
      $("#UsersInOrganizationRB").prop("checked", true);
      showUsers = true;
    }
 
    if (showUsers) {
      $("#divUserTargetList").show();
    } else {
      $("#divUserTargetList").hide();
    }
  }
 
  function buildDetailList(details) {
    let cList = $("#detailsListPlaceholder");
    cList.empty();
    $.each(details, function(i) {
      let li = $("<li/>")
        .addClass("list-group-item")
        .appendTo(cList);
      let div = $("<div/>")
        .addClass("input-group")
        .appendTo(li);
      let lbl = $("<label />")
        .text(details[i].DisplayTitle + ":")
        .appendTo(div);
      let detailLbl = $("<em />")
        .addClass("mr-4")
        .text(details[i].Text)
        .attr({ style: "text-overflow: ellipsis; display: inline-block; word-wrap: break-word;" })
        .appendTo(div);
      let btnCol = $("<div/>")
        .addClass("input-group-append btn-group-sm")
        .appendTo(div);
      let edit = $("<button/>")
        .addClass("btn btn-primary")
        .text(" Edit")
        .attr({ type: "button", value: i, onclick: "AlertCreate.editButtonClick(" + i + ");" })
        .appendTo(btnCol);
      let deleteBtn = $("<button/>")
        .addClass("btn btn-danger")
        .text(" Delete")
        .attr({ type: "button", value: i, onclick: "AlertCreate.deleteButtonClick(" + i + ");" })
        .appendTo(btnCol);
    });
  }
 
  export function deleteButtonClick(index) {
    const tmp = $("#DetailsCollection").val();
    let detailsCollection = [];
    if (tmp.length > 0) {
      detailsCollection = JSON.parse(tmp);
      detailsCollection.splice(index, 1);
      buildDetailList(detailsCollection);
      $("#DetailsCollection").val(JSON.stringify(detailsCollection));
    }
  }
 
  export function editButtonClick(index) {
    var tmp = $("#DetailsCollection").val();
    var detailsCollection = [];
    if (tmp.length > 0) {
      detailsCollection = JSON.parse(tmp);
      $("#SaveDetailButton").attr("onclick", "AlertCreate.saveDetailButtonClick(" + index + ");");
      $("#txtDetailHeader").val(detailsCollection[index].DisplayTitle);
      $("#txtDetailText").val(detailsCollection[index].Text);
      $("#ModalHeader").text("Edit Detail");
 
      if (detailsCollection[index].IsLink) {
        $("#isLinkCheckbox").prop("checked", true);
      } else {
        $("#isLinkCheckbox").prop("checked", false);
      }
      $("#myModal").modal("show");
    }
  }
 
  export function saveDetailButtonClick(index) {
    $("#SaveDetailButton").attr("onclick", "AlertCreate.addnewdetail();");
    $("#ModalHeader").text("Add New Detail");
    $("#myModal").modal("hide");
    let txtHead = $("#txtDetailHeader").val();
    let txtDet = $("#txtDetailText").val();
    if (txtHead.length > 0 && txtDet.length > 0) {
      let tmp = $("#DetailsCollection").val();
      let detailsCollection = [];
      if (tmp.length > 0) {
        detailsCollection = JSON.parse(tmp);
        detailsCollection[index].DisplayTitle = txtHead;
        detailsCollection[index].Text = txtDet;
        if ($("#isLinkCheckbox").is(":checked")) {
          detailsCollection[index].IsLink = true;
        }
        buildDetailList(detailsCollection);
        $("#DetailsCollection").val(JSON.stringify(detailsCollection));
      }
    }
    $("#txtDetailHeader").val("");
    $("#txtDetailText").val("");
  }
 
  export function addnewdetail() {
    let txtHead = $("#txtDetailHeader").val();
    let txtDet = $("#txtDetailText").val();
    if (txtHead.length > 0 && txtDet.length > 0) {
      let alertDetail: any = {};
      alertDetail.DisplayTitle = txtHead;
      alertDetail.Text = txtDet;
      if ($("#isLinkCheckbox").is(":checked")) {
        alertDetail.IsLink = true;
      } else {
        alertDetail.IsLink = false;
      }
      let tmp = $("#DetailsCollection").val();
      let detailsCollection = [];
      if (tmp.length > 0) {
        detailsCollection = JSON.parse(tmp);
      }
      detailsCollection.push(alertDetail);
      buildDetailList(detailsCollection);
 
      $("#DetailsCollection").val(JSON.stringify(detailsCollection));
    }
    $("#myModal").modal("hide");
    $("#txtDetailHeader").val("");
    $("#txtDetailText").val("");
  }
