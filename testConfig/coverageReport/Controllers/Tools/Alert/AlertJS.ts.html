  interface IAlertScope extends ng.IScope {
    alertDataSourceL: kendo.data.DataSource;
    alertDataSourceM: kendo.data.DataSource;

  declare function initAlertLocalization();
  export function saveAlert(baseurl) {
    const alerttext = $("#EventDescriptionModal").val();
    let latval;
    if ($("#Lat")) {
      latval = $("#Lat").val();
    }
    let lonval;
    if ($("#Lon")) {
      lonval = $("#Lon").val();
    }
    const logguid = $("#LogEntryGUID").val();
    const origToolName = $("#origToolName").val();
    if (alerttext !== null && alerttext.length > 0) {
      const url = baseurl;
 
      $Ajax.ajaxAntiForgery({
        url: url,
        type: "POST",
        data: {
          Lat: latval,
          Lon: lonval,
          AlarmText: alerttext,
          LogEntryGUID: logguid,
          origToolName: origToolName,
        },
        success: function(response) {
          if (response !== null) {
            Dashboard.jqueryclosemodal();
            //add refresh to notes window here if you like
            //refreshAllalert(true);
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }
 
  function refreshAllalert() {
    if (angular.element("#alertControllerID").length) {
      (<IAlertScope>angular.element("#alertControllerID").scope()).alertDataSourceL.read();
    }
    //would like to figure out a way to make this repository reside in the dashboard controller somehow, so that i
    //only need to refresh this once.
    if (angular.element("#alertControllerIDM").length) {
      (<IAlertScope>angular.element("#alertControllerIDM").scope()).alertDataSourceM.read();
    }
  }
 
  angular.module("cobraapp").factory("alertRepository", function() {
    return {
      getDataSourceM: function(hostURL, projectGUID, configProjectGUID) {
        return new kendo.data.DataSource({
          transport: {
            read: {
              url: hostURL + "Tools/alert/_GetAlertListJSON/" + projectGUID + "?ConfigProjectGUID=" + configProjectGUID,
              //url: hostURL + "Tools/api/AlertAPI/GetAlerts/" + projectGUID + "?ConfigProjectGUID=" + configProjectGUID
            },
            destroy: {
              url: hostURL + "Tools/alert/_DeleteAlert/" + projectGUID,
              type: "POST",
              dataType: "json",
            },
          },
          schema: {
            model: {
              id: "AlarmGUID",
              fields: {
                AlarmGUID: { editable: false },
                AlertType: { editable: false },
                AlertText: { editable: false },
                FriendlyUserName: { editable: false },
                PercentAcknowledged: { editable: false },
                FormattedDateCreated: { type: "date", editable: false },
              },
            },
          },
        });
      },
      getDataSourceL: function(hostURL, projectGUID, configProjectGUID) {
        return new kendo.data.DataSource({
          transport: {
            read: {
              url: hostURL + "Tools/alert/_GetAlertListJSON/" + projectGUID + "?ConfigProjectGUID=" + configProjectGUID,
              //url: hostURL + "Tools/api/AlertAPI/GetAlerts/" + projectGUID + "?ConfigProjectGUID=" + configProjectGUID
            },
            destroy: {
              url: hostURL + "Tools/alert/_DeleteAlert/" + projectGUID,
              type: "POST",
              dataType: "json",
            },
          },
          pageSize: 7,
          schema: {
            model: {
              id: "AlarmGUID",
              fields: {
                AlarmGUID: { editable: false },
                AlertType: { editable: false },
                AlertText: { editable: false },
                FriendlyUserName: { editable: false },
                PercentAcknowledged: { editable: false },
                FormattedDateCreated: { type: "date", editable: false },
              },
            },
          },
        });
      },
    };
  });
