  export function saveSensor(baseurl) {
    var nmval = $("#SensorName").val();
    var typval = $("#SensorType").val();
    var lvlval = $("#Level").val();
    var unitval = $("#Units").val();
    var agentval = $("#Agent").val();
    var latval;
    if ($("#Lat")) {
      latval = $("#Lat").val();
    }
    var lonval;
    if ($("#Lon")) {
      lonval = $("#Lon").val();
    }
    var alarmval = alarmIsChecked();
    if (nmval !== null && nmval.length > 0) {
      var url = baseurl; //+ "?NewNoteText=" + notetext + "&HighPriority=" + highpri; //+ "&Lat=" + latval + "&Lon=" + lonval;
 
      $Ajax.ajaxAntiForgery({
        url: url,
        type: "POST",
        data: {
          Lat: latval,
          Lon: lonval,
          SensorName: nmval,
          SensorType: typval,
          CheckIfAlarm: alarmval,
          Level: lvlval,
          Units: unitval,
          Agent: agentval,
        },
        success: function(response) {
          if (response !== null) {
            Dashboard.jqueryclosemodal();
            //add refresh to notes window here if you like
            //refreshAllsensor(true);
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }
  function alarmIsChecked() {
    if ($("#CheckIfAlarm").is(":checked")) {
      return true;
    } else {
      return false;
    }
  }
  function refreshAllsensor() {
    if (angular.element("#sensorControllerID").length) {
      (<ISensorScope>angular.element("#sensorControllerID").scope()).sensorDataSourceL.read();
    }
    //would like to figure out a way to make this repository reside in the dashboard controller somehow, so that i
    //only need to refresh this once.
    if (angular.element("#sensorControllerIDM").length) {
      (<ISensorScope>angular.element("#sensorControllerIDM").scope()).sensorDataSourceM.read();
    }
  }
 
  angular.module("cobraapp").factory("sensorRepository", function() {
    return {
      getDataSourceM: function(hostURL, projectGUID, configProjectGUID) {
        return new kendo.data.DataSource({
          transport: {
            read: {
              url: hostURL + "Tools/Sensor/_GetSensorsJSON/" + projectGUID + "?ConfigProjectGUID=" + configProjectGUID,
            },
            destroy: {
              url:
                hostURL +
                "Tools/Sensor/_DeleteAllSensorsByGUID/" +
                projectGUID +
                "?ConfigProjectGUID=" +
                configProjectGUID,
              type: "POST",
              dataType: "json",
            },
          },
          schema: {
            model: {
              id: "LogEntryGUID",
              fields: {
                LogEntryGUID: { editable: false },
                SensorGUID: { editable: false },
                SensorType: { editable: false },
                Level: { editable: true },
                Agent: { editable: true },
                User: { editable: true },
                SensorReadingAmount: { type: "int", editable: true },
                SensorStatus: { editable: true },
                LocationString: { editable: true },
                FormattedTimeOfReading: { type: "date", editable: false },
              },
            },
          },
        });
      },
      getDataSourceL: function(hostURL, projectGUID, configProjectGUID) {
        return new kendo.data.DataSource({
          transport: {
            read: {
              url: hostURL + "Tools/Sensor/_GetSensorsJSON/" + projectGUID + "?ConfigProjectGUID=" + configProjectGUID,
            },
            destroy: {
              url:
                hostURL +
                "Tools/Sensor/_DeleteAllSensorsByGUID/" +
                projectGUID +
                "?ConfigProjectGUID=" +
                configProjectGUID,
              type: "POST",
              dataType: "json",
            },
          },
          pageSize: 13,
          schema: {
            model: {
              id: "LogEntryGUID",
              fields: {
                LogEntryGUID: { editable: false },
                SensorGUID: { editable: false },
                SensorType: { editable: false },
                Level: { editable: true },
                Agent: { editable: true },
                User: { editable: true },
                SensorReadingAmount: { type: "int", editable: true },
                SensorStatus: { editable: true },
                LocationString: { editable: true },
                FormattedTimeOfReading: { type: "date", editable: false },
              },
            },
          },
        });
      },
    };
  });
