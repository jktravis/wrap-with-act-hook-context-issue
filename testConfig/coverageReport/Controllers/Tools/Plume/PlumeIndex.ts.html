import { MiniMap } from "../Map/MiniMap";
 
export namespace PlumeIndex {
  interface IGlobalParams {
    ToolName: string;
    DateText: string;
    PlumeName: string;
  }
  let _globalParams: IGlobalParams;
 
  export function init(globalParams) {
    _globalParams = globalParams;
    Main.cbraPush.client.logentryForToolUpdated = PlumeIndex.refreshListView;
    $.connection.hub.start().done(function() {
      Main.cbraPush.server.watchTool(_globalParams.ToolName, Main.currentProjectGUID);
    });
    $(window).resize();
 
    let sortBy;
    if (Main.serverBuildType === "Belgium") {
      sortBy = [
        { text: _globalParams.DateText, value: "Date" },
        { text: _globalParams.PlumeName, value: "Name" },
        { text: "Chemical", value: "Chemical" },
        { text: "CAS Number", value: "CAS" },
        { text: "Calculation State", value: "CalcState" },
      ];
    } else {
      sortBy = [
        { text: _globalParams.DateText, value: "Date" },
        { text: _globalParams.PlumeName, value: "Name" },
        { text: "Plume Type", value: "PlumeType" },
      ];
    }
    $("#selectDropdown").kendoDropDownList({
      dataTextField: "text",
      dataValueField: "value",
      dataSource: sortBy,
      index: 0,
      change: onChange,
    });
 
    const searchBy = [{ text: _globalParams.PlumeName, value: "Name" }, { text: "Chemical", value: "Chemical" }];
 
    $("#searchByDropDown").kendoDropDownList({
      dataTextField: "text",
      dataValueField: "value",
      dataSource: searchBy,
      index: 0,
    });
 
    document.getElementById("searchBtn").addEventListener("click", searchSaferList, false);
 
    function onChange() {
      const selection = $("#selectDropdown").val();
      const list = $("#lstPlumes").data("kendoListView").dataSource;
 
      saferPlumeListSort(selection, list);
    }
  }
  export function refreshListView() {
    $("#lstPlumes")
      .data("kendoListView")
      .dataSource.read();
    Main.notifySomethingHasChanged();
  }
  export function askDeletePlume(plumeID) {
    $("#plumeGUID").val(plumeID);
    $("#deleteConfirmModal").modal("show");
  }
 
  export function deletePlume(url) {
    const plumeID = $("#plumeGUID").val();
    $Ajax.ajaxAntiForgery({
      url: url + plumeID,
      type: "POST",
      success: function(response) {
        // The AJAX request succeeded and the result variable
        //should be the list of plumes in system
        if (response.success === true) {
          $("#lstPlumes")
            .data("kendoListView")
            .dataSource.read();
          //need to refresh map after delete
          MiniMap.LoadFeatureEntries();
        } else {
          //should alert the user somehow?
        }
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
    $("#deleteConfirmModal").modal("hide");
  }
 
  export function onPlumeChange(e) {
    const ds = $("#lstPlumes").data("kendoListView");
    const index = ds.select().index(),
      dataItem = ds.dataSource.view()[index];
    const lgGUID = dataItem.LogEntryGUID;
    MiniMap.ZoomToFeature(lgGUID);
  }
 
  export function ResizeMap() {
    const headerHeight = $("#headDiv").outerHeight(true);
    const statusHeight = $("#divstatusinfo").outerHeight(true);
    const footerHeight = $("#footdiv").outerHeight(true);
    const mainPaddingTop = parseInt($("#main").css("padding-top"));
    const mainPaddingBottom = parseInt($("#main").css("padding-bottom"));
    const mousecoordHeight = $("#mouseCoordinates").outerHeight(true);
    const plumeDataHeight = $("#divPlumeData").outerHeight(true);
    const windowHeight = $(window).height();
 
    const mapHeight =
      windowHeight -
      headerHeight -
      footerHeight -
      mainPaddingTop -
      mainPaddingBottom -
      mousecoordHeight -
      statusHeight -
      plumeDataHeight -
      130;
    $("#divmap").height(mapHeight);
  }
 
  function saferPlumeListSort(selection, list) {
    switch (selection) {
      case "Date":
        list.sort({ field: "UnixDateCreated", dir: "asc" });
        break;
      case "Name":
        list.sort({ field: "PlumeName", dir: "asc" });
        break;
      case "CAS":
        list.sort({ field: "Agent_CAS", dir: "asc" });
        break;
      case "Chemical":
        list.sort({ field: "Chemical", dir: "asc" });
        break;
      case "CalcState":
        list.sort({ field: "CalcState", dir: "asc" });
        break;
      case "PlumeType":
        list.sort({ field: "PlumeType", dir: "asc" });
        break;
      default:
        break;
    }
  }
 
  function searchSaferList() {
    const searchValue = $("#searchBar").val();
    const selection = $("#searchByDropDown").val();
    const list = $("#lstPlumes").data("kendoListView").dataSource;
 
    if (!searchValue) {
      list.filter({});
    } else {
      /* eslint-disable no-case-declarations */
      switch (selection) {
        case "Name":
          list.filter({ field: "PlumeName", operator: "contains", value: searchValue });
          break;
        case "Chemical":
          const filter = { logic: "or", filters: [] };
          filter.filters.push({ field: "Chemical", operator: "contains", value: searchValue });
          filter.filters.push({ field: "Chemical", operator: "isempty", value: searchValue });
          list.filter(filter);
          break;
        default:
          break;
      }
      /* eslint-enable no-case-declarations */
    }
  }
}
