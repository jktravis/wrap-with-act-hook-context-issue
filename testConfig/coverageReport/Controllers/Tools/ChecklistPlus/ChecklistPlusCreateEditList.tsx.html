import React, { Component } from "react";
import ChecklistPlusImportModal from "./ChecklistPlusImportModal";
import ReactTable from "react-table";
import {
  importChecklistFromFile,
  createChecklistTemplate,
  getChecklistByGUID,
  updateChecklistTemplate,
} from "../../../Api/ChecklistPlusApi";
import { Button, ButtonGroup, Form, FormGroup, Label, Input, CustomInput } from "reactstrap";
import "../../../../Content/ExtraStyles/ChecklistPlus.scss";
 
interface Props {
  toggleNewChecklist?: Function;
  toggleEditChecklist?: Function;
  isEdit?: boolean;
  toEditGuid: string;
  selectedOrg: string;
}
 
class ChecklistPlusCreateEditList extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    toggleNewChecklist: Function.prototype,
    toggleEditChecklist: Function.prototype,
    isEdit: false,
    toEditGuid: "",
    selectedOrg: "",
  };
 
  options = { weekday: "long", year: "numeric", month: "long", day: "numeric" };
  today = new Date();
 
  makeOptions = opt => {
    return <option key={opt}>{opt}</option>;
  };
 
  updateRowData = (row, target) => {
    let tempData = this.state.data;
    {
      row.value[1] === "Checkbox"
        ? row.value[0] === "on"
          ? (tempData[row.index]["status"][0] = "")
          : (tempData[row.index]["status"][0] = "on")
        : (tempData[row.index]["status"][0] = target.value);
    }
    tempData[row.index]["status"][0] = row.value[0];
    this.setState(() => {
      return { data: tempData, dataString: this.dataToDataString(tempData) };
    });
  };
 
  state = {
    columns: [
      {
        Header: "TITLE",
        accessor: "checktitle",
        headerStyle: { fontWeight: "bold" },
        columns: [
          {
            Header: "Item #",
            id: "rowNum",
            style: { whiteSpace: "unset" },
            Cell: row => {
              return <div>{row.index + 1}</div>;
            },
          },
          {
            Header: "OPR",
            accessor: "opr",
            style: { whiteSpace: "unset" },
          },
          {
            Header: "Action/Task",
            accessor: "acttask",
            style: { whiteSpace: "unset" },
          },
          {
            Header: "Status",
            accessor: "status",
            style: { whiteSpace: "unset" },
            Cell: row =>
              row.value[1] !== "Checkbox" ? (
                <FormGroup>
                  <CustomInput
                    value={this.state.data[row.index]["status"][0]}
                    onChange={e => {
                      this.updateRowData(row, e.target);
                    }}
                    type="select"
                    name="select"
                    bsSize="sm"
                    id={`ddown${row.index + 1}}`}
                  >
                    {row.value.slice(1).map(this.makeOptions)}
                  </CustomInput>
                </FormGroup>
              ) : (
                <FormGroup>
                  <CustomInput
                    checked={this.state.data[row.index]["status"][0]}
                    unselectable="on"
                    onChange={e => {
                      this.updateRowData(row, e.target);
                    }}
                    type="checkbox"
                    id={`chkbx${row.index + 1}}`}
                    label="Task Complete"
                  />
                </FormGroup>
              ),
          },
          {
            Header: "Date Modified",
            accessor: "datemod",
            style: { whiteSpace: "unset" },
          },
        ],
      },
      {
        Header: "Checklist OPR: CEX",
        accessor: "userheader",
        headerStyle: { fontWeight: "bold" },
        columns: [
          {
            Header: "User and Positions",
            accessor: "userpos",
            style: { whiteSpace: "unset" },
          },
        ],
      },
      {
        Header: `Current as of ${this.today.toLocaleDateString("en-US")}`,
        accessor: "checktitle",
        headerStyle: { fontWeight: "bold" },
        columns: [
          {
            Header: "Remarks/Notes",
            accessor: "notes",
            style: { whiteSpace: "unset" },
            Cell: row => (
              <FormGroup>
                <Input
                  type="textarea"
                  name="notesinput"
                  id="notesinput"
                  bsSize="sm"
                  style={{ resize: "none", height: 75 }}
                />
              </FormGroup>
            ),
          },
        ],
      },
      {
        Header: "",
        accessor: "deleteEditBtns",
        width: 50,
        style: { whiteSpace: "unset" },
        Cell: row => (
          <ButtonGroup vertical>
            <Button id={`saveBtn${row.index}`} color="info">
              <i className="fas fa-edit" />
            </Button>
            <Button
              id={`delBtn${row.index}`}
              onClick={() => {
                if (window.confirm("Are you sure you wish to delete this item?")) {
                  this.deleteRow(row.index);
                }
              }}
              color="danger"
            >
              <i className="fa fa-trash-alt" />
            </Button>
          </ButtonGroup>
        ),
      },
    ],
    data: [],
    showImportModal: false,
    glowButton: true,
    title: "Create New Checklist",
    description: "Template Description",
    dataString: "",
    hasDetails: false,
    hasUploaded: false,
  };
 
  componentDidMount() {
    if (this.props.isEdit) {
      this.getTableData(this.props.toEditGuid);
    }
  }
 
  handleDataString = dataIn => {
    const data = JSON.parse(dataIn);
    return data.map(datum => {
      return { opr: datum[0], acttask: datum[1], status: datum[2].split(","), datemod: datum[3], userpos: datum[4] };
    });
  };
 
  dataToDataString = dataIn => {
    return JSON.stringify(
      dataIn.map(({ opr, acttask, status, datemod, userpos }) => [
        opr,
        acttask,
        status.join(","),
        datemod,
        userpos,
        this.state.title,
      ]),
    );
  };
 
  handleImportFromFile = async input => {
    let data = await importChecklistFromFile(input);
    const dec = new TextDecoder("utf-8");
    data = dec.decode(data);
    const dataList = this.handleDataString(data);
    this.setState(() => {
      return { glowButton: false, title: data[0][5], data: dataList, dataString: this.dataToDataString(dataList) };
    });
  };
 
  getTableData = async guid => {
    const tableData = await getChecklistByGUID({ SelectedGuidString: guid });
    const dataStr = tableData[0][3];
    const dataList = this.handleDataString(dataStr);
    this.setState(() => {
      return {
        glowButton: false,
        title: tableData[0][0],
        data: dataList,
        dataString: this.dataToDataString(dataList),
        hasDetails: true,
      };
    });
  };
 
  handleModalClose = () => {
    if (this.state.title && this.state.data) {
      this.setState({ hasUploaded: true, showImportModal: false, hasDetails: true });
    } else {
      this.setState({ hasUploaded: true, showImportModal: false, hasDetails: false });
    }
  };
 
  openImportModal = () => {
    this.setState({ showImportModal: true });
  };
 
  deleteRow = index => {
    const out = [...this.state.data];
    out.splice(index, 1);
    this.setState({ data: out });
  };
 
  clearChecklist = () => {
    this.setState({ data: [] });
  };
 
  setTitleDesc = (titleIn, descIn) => {
    if (titleIn && descIn) {
      this.setState({ title: titleIn, description: descIn });
    } else if (titleIn) {
      this.setState({ title: titleIn });
    }
  };
 
  updateChecklist = async () => {
    if (this.state.hasDetails) {
      const { dataString, title, description } = this.state;
      const { toEditGuid, selectedOrg } = this.props;
      await updateChecklistTemplate({
        templateData: dataString,
        templateName: title,
        OrganizationName: selectedOrg,
        templateDescription: description,
        templateGUIDstr: toEditGuid,
      });
      this.props.toggleEditChecklist();
    } else {
      $.notify({ message: "Checklist requires a title and data to be uploaded." }, { type: "danger" });
    }
  };
 
  addChecklist = async () => {
    if (this.state.hasDetails) {
      const { dataString, title, description } = this.state;
      const { toEditGuid, selectedOrg } = this.props;
      await createChecklistTemplate({
        OrganizationName: selectedOrg,
        checklistTitle: title,
        data: dataString,
        description: description,
      });
      this.props.toggleNewChecklist();
    } else {
      $.notify({ message: "Checklist requires a title and data to be uploaded." }, { type: "danger" });
    }
  };
 
  render() {
    return (
      <div>
        <div>
          <h3>{!this.props.isEdit ? this.state.title : "EDIT: " + this.state.title}</h3>
        </div>
        <div>
          <ReactTable
            data={this.state.data}
            noDataText={`No Checklist Items Available --- Import From File to Add More`}
            columns={this.state.columns}
            defaultPageSize={5}
            className="-striped -highlight"
          />
        </div>
        <div style={{ display: "flex", justifyContent: "center" }}>
          {!this.props.isEdit && !this.state.hasUploaded ? (
            <Button
              className="glowing"
              onClick={() => this.openImportModal()}
              color="primary"
              style={{ marginTop: 10, marginLeft: 10, marginRight: 10 }}
            >
              Import From File
            </Button>
          ) : (
            <Button
              onClick={() => this.openImportModal()}
              color="primary"
              style={{ marginTop: 10, marginLeft: 10, marginRight: 10 }}
            >
              Update Title / Description
            </Button>
          )}
          {this.state.showImportModal ? (
            <ChecklistPlusImportModal
              isEdit={this.props.isEdit}
              hasUploaded={this.state.hasUploaded}
              setTitleDesc={this.setTitleDesc}
              closeModalfn={this.handleModalClose}
              importFromFile={this.handleImportFromFile}
            />
          ) : null}
          <Button
            style={{ marginTop: 10, marginLeft: 10, marginRight: 10 }}
            onClick={this.clearChecklist}
            color="warning"
          >
            Clear Checklist Items
          </Button>
          <Button
            style={{ marginTop: 10, marginLeft: 10, marginRight: 10 }}
            onClick={this.props.isEdit ? this.updateChecklist : this.addChecklist}
            color="info"
          >
            Save Template
          </Button>
        </div>
      </div>
    );
  }
}
 
export default ChecklistPlusCreateEditList;
