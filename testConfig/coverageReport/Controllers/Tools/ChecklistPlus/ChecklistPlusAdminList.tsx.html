import React, { Component } from "react";
import ReactTable from "react-table";
import OrgSelector from "../../Shared/Components/OrgSelector/OrgSelector";
import ChecklistPlusCreateEditAdminModal from "./ChecklistPlusCreateEditAdminModal";
import { currentOrg } from "Api/constants";
import { Button, ButtonGroup } from "reactstrap";
import { getChecklists, removeChecklist } from "Api/checklistplusApi";
import "../../../../Content/ExtraStyles/ChecklistPlus.scss";
 
class ChecklistPlusAdminList extends Component<{}, any> {
  toggleNewChecklist = () => {
    {
      const { selectedOption, newChecklist } = this.state;
      this.getTableData(selectedOption.value);
      this.setState({ newChecklist: !newChecklist });
    }
  };
 
  toggleEditChecklist = guid => {
    {
      const { selectedOption, editChecklist } = this.state;
      this.getTableData(selectedOption.value);
      this.setState({ selectedGuid: guid, editChecklist: !editChecklist });
    }
  };
 
  options = { weekday: "long", year: "numeric", month: "long", day: "numeric" };
  today = new Date();
 
  makeOptions = opt => {
    return <option>{opt}</option>;
  };
 
  state = {
    columns: [
      {
        Header: "Checklist Title",
        accessor: "templateName",
        style: { whiteSpace: "unset" },
      },
      {
        Header: "Template Description",
        accessor: "templateDescription",
        style: { whiteSpace: "unset" },
      },
      {
        Header: "Date Modified",
        accessor: "datemod",
        style: { whiteSpace: "unset" },
      },
      {
        Header: "",
        accessor: "deleteEditBtns",
        width: 73,
        style: { whiteSpace: "unset" },
        Cell: row => (
          <ButtonGroup>
            <Button
              id={`saveBtn${row.index}`}
              onClick={() => {
                this.toggleEdit(row.index);
              }}
              color="info"
              size="sm"
            >
              <i className="fas fa-edit" />
            </Button>
            <Button
              id={`delBtn${row.index}`}
              onClick={() => {
                if (window.confirm("Are you sure you wish to delete this item?")) {
                  this.deleteRow(row.index);
                }
              }}
              color="danger"
              size="sm"
            >
              <i className="fa fa-trash-alt" />
            </Button>
          </ButtonGroup>
        ),
      },
    ],
    data: [],
    showImportModal: false,
    glowButton: true,
    title: "Create New Checklist",
    selectedOption: { label: currentOrg, value: currentOrg },
    newChecklist: false,
    editChecklist: false,
    selectedGuid: "",
  };
 
  handleSelectedOrg = option => {
    this.setState({ selectedOption: option }, () => {
      if (this.state.selectedOption) {
        this.getTableData(this.state.selectedOption.value);
      }
    });
  };
 
  getTableData = async orgName => {
    const tableData = await getChecklists({ OrganizationName: orgName });
    this.setState(() => {
      const dataList = tableData.map(datum => {
        return {
          templateName: datum.TemplateTitle,
          templateDescription: datum.TemplateDescription,
          datemod: datum.DateLastModified,
          guid: datum.ChecklistPlusTemplateGUID,
          dataString: datum.TemplateData,
        };
      });
      return { data: dataList };
    });
  };
 
  componentDidMount() {
    this.getTableData(this.state.selectedOption.value);
    if (this.state.data.length <= 0) {
      this.setState({ glowbutton: true });
    }
  }
 
  handleModalClose = () => {
    this.setState({ showImportModal: false });
  };
 
  openImportModal = () => {
    this.setState({ showImportModal: true });
  };
 
  toggleEdit = index => {
    const tempGuid = this.state.data[index].guid;
    this.toggleEditChecklist(tempGuid);
  };
 
  deleteRow = async index => {
    await removeChecklist({ SelectedGuidString: this.state.data[index].guid });
    const out = [...this.state.data];
    out.splice(index, 1);
    this.setState({ data: out });
  };
 
  render() {
    return (
      <div>
        <div>
          <h3>{this.state.title}</h3>
        </div>
        <OrgSelector changeFn={this.handleSelectedOrg} selectedOption={this.state.selectedOption} />
        <div>
          <ReactTable
            data={this.state.data}
            noDataText={`No Checklist Templates Available --- Create a New One Below`}
            columns={this.state.columns}
            defaultPageSize={5}
            className="-striped -highlight"
          />
        </div>
        <div style={{ display: "flex", justifyContent: "center" }}>
          {this.state.glowButton ? (
            <Button
              className="glowing"
              onClick={() => this.toggleNewChecklist()}
              color="primary"
              style={{ marginTop: 10, marginLeft: 10, marginRight: 10 }}
            >
              Add New Template
            </Button>
          ) : (
            <Button
              onClick={() => this.toggleNewChecklist()}
              color="primary"
              style={{ marginTop: 10, marginLeft: 10, marginRight: 10 }}
            >
              Add New Template
            </Button>
          )}
        </div>
 
        {this.state.newChecklist ? (
          <ChecklistPlusCreateEditAdminModal
            selectedOrg={this.state.selectedOption.value}
            toggleNewChecklist={this.toggleNewChecklist}
          />
        ) : this.state.editChecklist ? (
          <ChecklistPlusCreateEditAdminModal
            selectedOrg={this.state.selectedOption.value}
            isEdit={true}
            toEditGuid={this.state.selectedGuid}
            toggleEditChecklist={this.toggleEditChecklist}
          />
        ) : null}
      </div>
    );
  }
}
 
export default ChecklistPlusAdminList;
