import React, { Component } from "react";
import ChecklistPlusImportModal from "./ChecklistPlusImportModal";
import ReactTable from "react-table";
import produce from "immer";
import { importChecklistFromFile, getChecklists, getChecklistEntry, updateChecklistEntry } from "Api/checklistplusApi";
import { Button, ButtonGroup, FormGroup, Input, CustomInput, Modal, ModalHeader, ModalBody } from "reactstrap";
import "../../../../Content/ExtraStyles/ChecklistPlus.scss";
 
interface Props {
  toggleNewChecklist?: Function;
  toggleEditChecklist?: Function;
  addChecklist?: Function;
  isEdit?: boolean;
  toEditGuid: string;
  selectedOrg: string;
  colsInput: object;
  currentProj: string;
}
 
class ChecklistPlusCreateEditIncidentModal extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    toggleNewChecklist: Function.prototype,
    toggleEditChecklist: Function.prototype,
    addChecklist: Function.prototype,
    isEdit: false,
    toEditGuid: "",
    selectedOrg: "",
    colsInput: [],
    currentProj: "",
  };
 
  options = { weekday: "long", year: "numeric", month: "long", day: "numeric" };
  today = new Date();
 
  makeOptions = opt => {
    return <option key={opt}>{opt}</option>;
  };
 
  makeOptionsList = (opt, index, rowIndex) => {
    return (
      <div key={opt}>
        {opt}
        <Button close onClick={() => this.removeOption(index, rowIndex)} />
      </div>
    );
  };
 
  removeOption = (index, rowIndex) => {
    this.setState(() => {
      const nextData = produce(this.state.data, tempData => {
        tempData[rowIndex].status.splice(index + 1, 1);
      });
      return { data: nextData, dataString: this.dataToDataString(nextData) };
    });
  };
 
  addOption = (index, rowIndex) => {
    this.setState(() => {
      const nextData = produce(this.state.data, tempData => {
        const value = (document.getElementById(`newOption${rowIndex}`) as HTMLInputElement).value;
        tempData[rowIndex].status.push(value);
      });
      return { data: nextData, dataString: this.dataToDataString(nextData) };
    });
  };
 
  updateRowData = (row, target) => {
    this.setState(() => {
      const nextData = produce(this.state.data, tempData => {
        row.value[1] === "Checkbox"
          ? row.value[0] === "on"
            ? (tempData[row.index].status[0] = "")
            : (tempData[row.index].status[0] = "on")
          : (tempData[row.index].status[0] = target.value);
        tempData[row.index].datemod = this.today.toUTCString();
      });
      return { data: nextData, dataString: this.dataToDataString(nextData) };
    });
  };
 
  toggleEditRow = rowIndex => {
    this.setState(() => {
      const nextData = produce(this.state.data, tempData => {
        tempData[rowIndex].status[0] !== "_EDIT_"
          ? (tempData[rowIndex].status[0] = "_EDIT_")
          : (tempData[rowIndex].status[0] = "");
      });
      return { data: nextData, dataString: this.dataToDataString(nextData) };
    });
  };
 
  state = {
    data: [],
    showImportModal: false,
    glowButton: true,
    title: "Create New Checklist",
    description: "Template Description",
    dataString: "",
    hasDetails: false,
    hasUploaded: false,
    checklistAdded: false,
  };
 
  columns = [
    {
      Header: "Template Details",
      accessor: "checktitle",
      headerStyle: { fontWeight: "bold" },
      columns: [
        {
          Header: "Item #",
          id: "rowNum",
          style: { whiteSpace: "unset" },
          Cell: row => {
            return <div>{row.index + 1}</div>;
          },
        },
        {
          Header: "OPR",
          accessor: "opr",
          style: { whiteSpace: "unset" },
        },
        {
          Header: "Action/Task",
          accessor: "acttask",
          style: { whiteSpace: "unset" },
        },
        {
          Header: "Status Options",
          accessor: "status",
          style: { whiteSpace: "unset" },
          Cell: row =>
            row.value[1] !== "Checkbox" ? (
              <FormGroup>
                <CustomInput
                  value={
                    row.value[0] !== "_EDIT_"
                      ? this.state.data[row.index].status[0]
                      : this.state.data[row.index].status[1]
                  }
                  onChange={e => {
                    this.updateRowData(row, e.target);
                  }}
                  type="select"
                  name="select"
                  bsSize="sm"
                  id={`ddown${row.index + 1}}`}
                >
                  {row.value.slice(1).map(this.makeOptions)}
                </CustomInput>
              </FormGroup>
            ) : (
              <FormGroup>
                <CustomInput
                  checked={this.state.data[row.index].status[0]}
                  unselectable="on"
                  onChange={e => {
                    this.updateRowData(row, e.target);
                  }}
                  type="checkbox"
                  id={`chkbx${row.index + 1}}`}
                  label="Task Complete"
                />
              </FormGroup>
            ),
        },
        {
          Header: "Date Modified",
          accessor: "datemod",
          style: { whiteSpace: "unset" },
        },
      ],
    },
    {
      Header: "",
      accessor: "userheader",
      headerStyle: { fontWeight: "bold" },
      columns: [
        {
          Header: "User and Positions",
          accessor: "userpos",
          style: { whiteSpace: "unset" },
        },
      ],
    },
    {
      Header: `Current as of ${this.today.toLocaleDateString("en-US")}`,
      accessor: "checktitle",
      headerStyle: { fontWeight: "bold" },
      columns: [
        {
          Header: "Remarks/Notes",
          accessor: "notes",
          style: { whiteSpace: "unset" },
          Cell: row => (
            <FormGroup>
              <Input
                type="textarea"
                name="notesinput"
                id="notesinput"
                bsSize="sm"
                style={{ resize: "none", height: 38 }}
              />
            </FormGroup>
          ),
        },
      ],
    },
    {
      Header: "",
      accessor: "deleteEditBtns",
      width: 40,
      style: { whiteSpace: "unset" },
      Cell: row => (
        <ButtonGroup className="float-right">
          <Button
            id={`delBtn${row.index}`}
            onClick={() => {
              if (window.confirm("Are you sure you wish to delete this item?")) {
                this.deleteRow(row.index);
              }
            }}
            color="danger"
            size="sm"
          >
            <i className="fa fa-trash-alt" />
          </Button>
        </ButtonGroup>
      ),
    },
  ];
  searchColumns = [
    {
      Header: "Checklist Title",
      accessor: "templateName",
      style: { whiteSpace: "unset" },
    },
    {
      Header: "Template Description",
      accessor: "templateDescription",
      style: { whiteSpace: "unset" },
    },
    {
      Header: "",
      accessor: "deleteEditBtns",
      width: 40,
      style: { whiteSpace: "unset" },
      Cell: row => (
        <ButtonGroup>
          <Button
            id={`addBtn${row.index}`}
            onClick={async () => {
              await this.addChecklistEntry(row.index);
            }}
            color="primary"
            size="sm"
          >
            <i className="fas fa-plus" />
          </Button>
        </ButtonGroup>
      ),
    },
  ];
 
  componentDidMount() {
    this.props.isEdit
      ? this.getTableData(this.props.toEditGuid, null)
      : this.getTableData(null, this.props.selectedOrg);
  }
 
  handleDataString = dataIn => {
    const data = JSON.parse(dataIn);
    return data.map(datum => {
      return { opr: datum[0], acttask: datum[1], status: datum[2].split(","), datemod: datum[3], userpos: datum[4] };
    });
  };
 
  dataToDataString = dataIn => {
    return JSON.stringify(
      dataIn.map(({ opr, acttask, status, datemod, userpos }) => [
        opr,
        acttask,
        status.join(","),
        datemod,
        userpos,
        this.state.title,
      ]),
    );
  };
 
  handleImportFromFile = async input => {
    let data = await importChecklistFromFile(input);
    const dec = new TextDecoder("utf-8");
    data = dec.decode(data);
    const dataList = this.handleDataString(data);
    this.setState(() => {
      return { glowButton: false, title: data[0][5], data: dataList, dataString: this.dataToDataString(dataList) };
    });
  };
 
  getTableData = async (guid, orgName) => {
    if (guid) {
      const tableData = await getChecklistEntry({ entryGUID: guid });
      const dataStr = tableData[0][3];
      const dataList = this.handleDataString(dataStr);
      this.setState(() => {
        return {
          glowButton: false,
          title: tableData[0][0],
          description: tableData[0][2],
          data: dataList,
          dataString: this.dataToDataString(dataList),
          hasDetails: true,
        };
      });
    } else {
      const tableData = await getChecklists({ OrganizationName: orgName });
      this.setState(() => {
        if (tableData) {
          const dataList = tableData.map(datum => {
            return {
              templateName: datum.TemplateTitle,
              templateDescription: datum.TemplateDescription,
              datemod: datum.DateLastModified,
              guid: datum.ChecklistPlusTemplateGUID,
              dataString: datum.TemplateData,
            };
          });
          return { data: dataList };
        } else {
          return { data: [] };
        }
      });
    }
  };
 
  handleModalClose = () => {
    if (this.state.title && this.state.data) {
      this.setState({ hasUploaded: true, showImportModal: false, hasDetails: true });
    } else {
      this.setState({ hasUploaded: true, showImportModal: false, hasDetails: false });
    }
  };
 
  openImportModal = () => {
    this.setState({ showImportModal: true });
  };
 
  deleteRow = index => {
    const out = [...this.state.data];
    out.splice(index, 1);
    this.setState({ data: out });
  };
 
  clearChecklist = () => {
    this.setState({ data: [] });
  };
 
  addChecklistEntry = async index => {
    const { data } = this.state;
    await this.props.addChecklist(data[index].dataString, this.props.selectedOrg, data[index].guid);
    this.setState({ checklistAdded: true });
  };
 
  updateChecklistEntry = () => {
    const { data, title, description } = this.state;
    const { toEditGuid, selectedOrg } = this.props;
    updateChecklistEntry({
      templateData: this.dataToDataString(data),
      templateName: title,
      OrganizationName: selectedOrg,
      templateDescription: description,
      templateGUIDstr: toEditGuid,
      projGUID: this.props.currentProj,
    });
    this.setState({ checklistAdded: true });
  };
 
  setTitleDesc = (titleIn, descIn) => {
    if (titleIn && descIn) {
      this.setState({ title: titleIn, description: descIn });
    } else if (titleIn) {
      this.setState({ title: titleIn });
    }
  };
 
  toggle = () => {
    {
      this.props.isEdit ? this.props.toggleEditChecklist() : this.props.toggleNewChecklist();
    }
  };
 
  render() {
    return (
      <div>
        <Modal isOpen={true} backdrop={true} size={`lg`}>
          <ModalHeader>{!this.props.isEdit ? "Select a Checklist Template" : "EDIT: " + this.state.title}</ModalHeader>
          <ModalBody>
            <div>
              <ReactTable
                data={this.state.data}
                noDataText={`No Checklist Items Available --- Import From File to Add More`}
                columns={!this.props.isEdit ? this.searchColumns : this.columns}
                defaultPageSize={5}
                className="-striped -highlight"
              />
            </div>
 
            <div className="btnRow">
              {!this.props.isEdit && !this.state.hasUploaded ? null : (
                <Button className="btnSpacing" onClick={() => this.openImportModal()} color="primary">
                  Edit Title / Description
                </Button>
              )}
              {this.props.isEdit ? (
                <Button className="btnSpacing" onClick={this.updateChecklistEntry} color="primary">
                  Update Template
                </Button>
              ) : null}
              {this.state.checklistAdded ? (
                <Button className="buttonFlexRight" onClick={this.toggle} color="primary">
                  Close
                </Button>
              ) : (
                <Button className="buttonFlexRight" onClick={this.toggle} color="warning">
                  Cancel
                </Button>
              )}
            </div>
            {this.state.showImportModal && (
              <ChecklistPlusImportModal
                isEdit={this.props.isEdit}
                hasUploaded={this.state.hasUploaded}
                setTitleDesc={this.setTitleDesc}
                closeModalfn={this.handleModalClose}
                importFromFile={this.handleImportFromFile}
              />
            )}
          </ModalBody>
        </Modal>
      </div>
    );
  }
}
 
export default ChecklistPlusCreateEditIncidentModal;
