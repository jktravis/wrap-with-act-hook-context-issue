import React, { Component } from "react";
import ChecklistPlusImportModal from "./ChecklistPlusImportModal";
import ReactTable from "react-table";
import produce from "immer";
import {
  importChecklistFromFile,
  createChecklistTemplate,
  getChecklistByGUID,
  updateChecklistTemplate,
} from "Api/checklistplusApi";
import {
  Button,
  ButtonGroup,
  FormGroup,
  Input,
  CustomInput,
  InputGroupAddon,
  Modal,
  ModalHeader,
  ModalBody,
} from "reactstrap";
import "../../../../Content/ExtraStyles/ChecklistPlus.scss";
import InputGroup from "reactstrap/lib/InputGroup";
 
interface Props {
  toggleNewChecklist?: Function;
  toggleEditChecklist?: Function;
  isEdit?: boolean;
  toEditGuid: string;
  selectedOrg: string;
}
 
class ChecklistPlusCreateEditAdminModal extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    toggleNewChecklist: Function.prototype,
    toggleEditChecklist: Function.prototype,
    isEdit: false,
    toEditGuid: "",
    selectedOrg: "",
  };
 
  options = { weekday: "long", year: "numeric", month: "long", day: "numeric" };
  today = new Date();
 
  makeOptions = opt => {
    return <option key={opt}>{opt}</option>;
  };
 
  makeOptionsList = (opt, index, rowIndex) => {
    return (
      <div key={opt}>
        {opt}
        <Button close onClick={() => this.removeOption(index, rowIndex)} />
      </div>
    );
  };
 
  removeOption = (index, rowIndex) => {
    this.setState(() => {
      const nextData = produce(this.state.data, tempData => {
        tempData[rowIndex].status.splice(index + 1, 1);
      });
      return { data: nextData, dataString: this.dataToDataString(nextData) };
    });
  };
 
  addOption = (index, rowIndex) => {
    this.setState(() => {
      const nextData = produce(this.state.data, tempData => {
        const value = (document.getElementById(`newOption${rowIndex}`) as HTMLInputElement).value;
        tempData[rowIndex].status.push(value);
      });
      return { data: nextData, dataString: this.dataToDataString(nextData) };
    });
  };
 
  updateRowData = (row, target) => {
    this.setState(() => {
      const nextData = produce(this.state.data, tempData => {
        row.value[1] === "Checkbox"
          ? row.value[0] === "on"
            ? (tempData[row.index].status[0] = "")
            : (tempData[row.index].status[0] = "on")
          : (tempData[row.index].status[0] = target.value);
        tempData[row.index].datemod = this.today.toUTCString();
      });
      return { data: nextData, dataString: this.dataToDataString(nextData) };
    });
  };
 
  toggleEditRow = rowIndex => {
    this.setState(() => {
      const nextData = produce(this.state.data, tempData => {
        tempData[rowIndex].status[0] !== "_EDIT_"
          ? (tempData[rowIndex].status[0] = "_EDIT_")
          : (tempData[rowIndex].status[0] = "");
      });
      return { data: nextData, dataString: this.dataToDataString(nextData) };
    });
  };
 
  state = {
    data: [],
    showImportModal: false,
    glowButton: true,
    title: "Create New Checklist",
    description: "Template Description",
    dataString: "",
    hasDetails: false,
    hasUploaded: false,
  };
 
  columns = [
    {
      Header: "Template Details",
      accessor: "checktitle",
      headerStyle: { fontWeight: "bold" },
      columns: [
        {
          Header: "Item #",
          id: "rowNum",
          style: { whiteSpace: "unset" },
          Cell: row => {
            return <div>{row.index + 1}</div>;
          },
        },
        {
          Header: "OPR",
          accessor: "opr",
          style: { whiteSpace: "unset" },
        },
        {
          Header: "Action/Task",
          accessor: "acttask",
          style: { whiteSpace: "unset" },
        },
        {
          Header: "Status Options",
          accessor: "status",
          style: { whiteSpace: "unset" },
          Cell: row =>
            row.value[1] !== "Checkbox" ? (
              row.value[0] !== "_EDIT_" ? (
                <FormGroup>
                  <CustomInput
                    value={this.state.data[row.index].status[0]}
                    onChange={e => {
                      this.updateRowData(row, e.target);
                    }}
                    disabled={true}
                    type="select"
                    name="select"
                    bsSize="sm"
                    id={`ddown${row.index + 1}}`}
                  >
                    {row.value.slice(1).map(this.makeOptions)}
                  </CustomInput>
                </FormGroup>
              ) : (
                <div>
                  {row.value.slice(1).map((item, index) => this.makeOptionsList(item, index, row.index))}
                  <InputGroup size="sm">
                    <Input id={`newOption${row.index}`} />
                    <InputGroupAddon addonType="append">
                      <Button onClick={index => this.addOption(index, row.index)}>
                        <i className="fas fa-plus" />
                      </Button>
                    </InputGroupAddon>
                  </InputGroup>
                </div>
              )
            ) : (
              <FormGroup>
                <CustomInput
                  checked={this.state.data[row.index].status[0]}
                  disabled={true}
                  unselectable="on"
                  onChange={e => {
                    this.updateRowData(row, e.target);
                  }}
                  type="checkbox"
                  id={`chkbx${row.index + 1}}`}
                  label="Task Complete"
                />
              </FormGroup>
            ),
        },
      ],
    },
    {
      Header: "",
      accessor: "userheader",
      headerStyle: { fontWeight: "bold" },
      columns: [
        {
          Header: "User and Positions",
          accessor: "userpos",
          style: { whiteSpace: "unset" },
        },
      ],
    },
    {
      Header: "",
      accessor: "deleteEditBtns",
      width: 73,
      style: { whiteSpace: "unset" },
      Cell: row => (
        <ButtonGroup className="float-right">
          {this.state.data[row.index].status[1] !== "Checkbox" ? (
            <Button id={`saveBtn${row.index}`} color="info" size="sm" onClick={() => this.toggleEditRow(row.index)}>
              {this.state.data[row.index].status[0] !== "_EDIT_" ? (
                <i className="fas fa-edit" />
              ) : (
                <i className="fas fa-save" />
              )}
            </Button>
          ) : null}
          <Button
            id={`delBtn${row.index}`}
            onClick={() => {
              if (window.confirm("Are you sure you wish to delete this item?")) {
                this.deleteRow(row.index);
              }
            }}
            color="danger"
            size="sm"
          >
            <i className="fa fa-trash-alt" />
          </Button>
        </ButtonGroup>
      ),
    },
  ];
 
  componentDidMount() {
    if (this.props.isEdit) {
      this.getTableData(this.props.toEditGuid);
    }
  }
 
  handleDataString = (data, first?) => {
    return data.TemplateData.map(datum => {
      let status = [];
      if (first) {
        status = datum.Status.split(",");
        status.unshift("");
      } else {
        status = JSON.parse(datum.Status);
      }
      return {
        opr: datum.OPR,
        acttask: datum.ActionTask,
        status: status,
        datemod: datum.DateLastModified,
        userpos: datum.User,
      };
    });
  };
 
  dataToDataString = dataIn => {
    return dataIn.map(({ opr, acttask, status, datemod, userpos }) => {
      return {
        OPR: opr,
        ActionTask: acttask,
        Status: JSON.stringify(status),
        DateLastModified: datemod,
        User: userpos,
        Title: this.state.title,
      };
    });
  };
 
  handleImportFromFile = async input => {
    let data = await importChecklistFromFile(input);
    const dec = new TextDecoder("utf-8");
    data = JSON.parse(dec.decode(data));
    const dataList = this.handleDataString(data, true);
    this.setState(() => {
      return { glowButton: false, title: data.Title, data: dataList, dataString: this.dataToDataString(dataList) };
    });
  };
 
  getTableData = async guid => {
    const tableData = await getChecklistByGUID({ SelectedGuidString: guid });
    const dataList = this.handleDataString(tableData[0]);
    this.setState(() => {
      return {
        glowButton: false,
        title: tableData[0].TemplateTitle,
        description: tableData[0].TemplateDescription,
        data: dataList,
        dataString: this.dataToDataString(dataList),
        hasDetails: true,
      };
    });
  };
 
  handleModalClose = () => {
    if (this.state.title && this.state.data) {
      this.setState({ hasUploaded: true, showImportModal: false, hasDetails: true });
    } else {
      this.setState({ hasUploaded: true, showImportModal: false, hasDetails: false });
    }
  };
 
  openImportModal = () => {
    this.setState({ showImportModal: true });
  };
 
  deleteRow = index => {
    const out = [...this.state.data];
    out.splice(index, 1);
    this.setState({ data: out });
  };
 
  clearChecklist = () => {
    this.setState({ data: [] });
  };
 
  setTitleDesc = (titleIn, descIn) => {
    if (titleIn && descIn) {
      this.setState({ title: titleIn, description: descIn });
    } else if (titleIn) {
      this.setState({ title: titleIn });
    }
  };
 
  updateChecklist = async () => {
    if (this.state.hasDetails) {
      const { data, title, description } = this.state;
      const { toEditGuid, selectedOrg } = this.props;
      await updateChecklistTemplate({
        templateData: this.dataToDataString(data),
        organizationName: selectedOrg,
        templateTitle: title,
        templateDescription: description,
        checklistPlusTemplateGUID: toEditGuid,
      });
      this.props.toggleEditChecklist();
    } else {
      $.notify({ message: "Checklist requires a title and data to be uploaded." }, { type: "danger" });
    }
  };
 
  addChecklist = async () => {
    if (this.state.hasDetails) {
      const { data, title, description } = this.state;
      const { selectedOrg } = this.props;
      await createChecklistTemplate({
        templateData: this.dataToDataString(data),
        organizationName: selectedOrg,
        templateTitle: title,
        templateDescription: description,
      });
      this.props.toggleNewChecklist();
    } else {
      $.notify({ message: "Checklist requires a title and data to be uploaded." }, { type: "danger" });
    }
  };
 
  toggle = () => {
    {
      this.props.isEdit ? this.props.toggleEditChecklist() : this.props.toggleNewChecklist();
    }
  };
 
  render() {
    return (
      <div>
        <Modal isOpen={true} backdrop={true} size={`lg`}>
          <ModalHeader>{!this.props.isEdit ? this.state.title : "EDIT: " + this.state.title}</ModalHeader>
          <ModalBody>
            <div>
              <ReactTable
                data={this.state.data}
                noDataText={`No Checklist Items Available --- Import From File to Add More`}
                columns={this.columns}
                defaultPageSize={5}
                className="-striped -highlight"
              />
            </div>
 
            <div className="btnRow">
              {!this.props.isEdit && !this.state.hasUploaded ? (
                <Button className="glowing btnSpacing" onClick={() => this.openImportModal()} color="primary">
                  Import From File
                </Button>
              ) : (
                <Button className="btnSpacing" onClick={() => this.openImportModal()} color="primary">
                  Update Title / Description
                </Button>
              )}
              <Button
                className="btnSpacing"
                onClick={this.props.isEdit ? this.updateChecklist : this.addChecklist}
                color="primary"
              >
                Save Template
              </Button>
              <Button className="buttonFlexRight" onClick={this.clearChecklist} color="danger">
                Clear Checklist Items
              </Button>
              <Button className="btnSpacing" onClick={this.toggle} color="warning">
                Cancel
              </Button>
            </div>
            {this.state.showImportModal && (
              <ChecklistPlusImportModal
                isEdit={this.props.isEdit}
                hasUploaded={this.state.hasUploaded}
                setTitleDesc={this.setTitleDesc}
                closeModalfn={this.handleModalClose}
                importFromFile={this.handleImportFromFile}
              />
            )}
          </ModalBody>
        </Modal>
      </div>
    );
  }
}
 
export default ChecklistPlusCreateEditAdminModal;
