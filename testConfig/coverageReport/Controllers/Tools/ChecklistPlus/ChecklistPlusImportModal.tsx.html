import React, { Component } from "react";
import { FormGroup, Input, Modal, ModalHeader, ModalBody, ModalFooter, Button, Label } from "reactstrap";
import { FilePond, File, registerPlugin } from "react-filepond";
import "../../../../Content/ExtraStyles/ChecklistPlus.scss";
import "filepond/dist/filepond.min.css";
 
interface Props {
  closeModalfn?: Function;
  importFromFile?: Function;
  setTitleDesc?: Function;
  isEdit?: boolean;
  hasUploaded?: boolean;
}
 
class ChecklistPlusImportModal extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    closeModalfn: Function.prototype,
    importFromFile: Function.prototype,
    setTitleDesc: Function.prototype,
    isEdit: false,
    hasUploaded: false,
  };
 
  state = {
    selectedFile: null,
    modalTitle: "Import Checklist from File",
    uploadTitle: "Upload",
    title: "",
    description: "",
  };
 
  componentDidMount() {
    if (this.props.isEdit || this.props.hasUploaded) {
      this.setState({ modalTitle: "Update Title / Description", uploadTitle: "Update" });
    } else {
      this.setState({ modalTitle: "Import Checklist from File", uploadTitle: "Upload" });
    }
  }
 
  toggle = () => {
    this.props.closeModalfn();
  };
 
  upload = async () => {
    const { selectedFile } = this.state;
 
    const formData = new FormData();
    formData.append("filename", selectedFile.name);
    formData.append("file", selectedFile);
 
    await this.props.importFromFile(formData);
 
    this.props.setTitleDesc(this.state.title, this.state.description);
    this.toggle();
  };
 
  update = () => {
    this.props.setTitleDesc(this.state.title, this.state.description);
    this.toggle();
  };
 
  render() {
    return (
      <div>
        <Modal isOpen={true} backdrop={true}>
          <ModalHeader>{this.state.modalTitle}</ModalHeader>
          <ModalBody>
            {!this.props.isEdit && !this.props.hasUploaded ? (
              <FilePond
                allowMultiple={false}
                onupdatefiles={fileItems => {
                  this.setState({
                    selectedFile: fileItems[0].file,
                  });
                }}
              >
                <File key={this.state.selectedFile} src={this.state.selectedFile} origin="local" />
              </FilePond>
            ) : null}
            <FormGroup>
              <Label>Checklist Template Title</Label>
              <Input
                type="text"
                name="notesinput"
                id="titleInput"
                bsSize="sm"
                style={{ resize: "none" }}
                onBlur={e => this.setState({ title: e.target.value })}
              />
              <Label>Checklist Description</Label>
              <Input
                type="textarea"
                name="notesinput"
                id="titleInput"
                bsSize="sm"
                style={{ resize: "none" }}
                onBlur={e => this.setState({ description: e.target.value })}
              />
            </FormGroup>
          </ModalBody>
          <ModalFooter>
            <Button color="primary" onClick={!this.props.isEdit && !this.props.hasUploaded ? this.upload : this.update}>
              {this.state.uploadTitle}
            </Button>
            <Button color="warning" onClick={this.toggle}>
              Cancel
            </Button>
          </ModalFooter>
        </Modal>
      </div>
    );
  }
}
 
export default ChecklistPlusImportModal;
