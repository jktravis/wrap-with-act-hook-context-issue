import React, { Component } from "react";
import ReactTable from "react-table";
import ChecklistPlusCreateEditIncidentModal from "./ChecklistPlusCreateEditIncidentModal";
import { Button, ButtonGroup } from "reactstrap";
import { addChecklistEntry, GetChecklistInstanceList, removeChecklistEntry } from "Api/checklistplusApi";
import "../../../../Content/ExtraStyles/ChecklistPlus.scss";
 
interface Props {
  currentOrg: string;
  currentProj: string;
}
 
class ChecklistPlusIncidentList extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    currentOrg: "",
    currentProj: "",
  };
 
  toggleNewChecklist = () => {
    {
      const { newChecklist } = this.state;
      this.getTableData(this.props.currentProj);
      this.setState({ newChecklist: !newChecklist });
    }
  };
 
  toggleEditChecklist = guid => {
    {
      const { editChecklist } = this.state;
      this.getTableData(this.props.currentProj);
      this.setState({ selectedGuid: guid, editChecklist: !editChecklist });
    }
  };
 
  options = { weekday: "long", year: "numeric", month: "long", day: "numeric" };
  today = new Date();
 
  makeOptions = opt => {
    return <option>{opt}</option>;
  };
 
  state = {
    data: [],
    showImportModal: false,
    glowButton: true,
    title: "Create New Checklist",
    selectedOption: { label: null, value: null },
    newChecklist: false,
    editChecklist: false,
    selectedGuid: "",
  };
 
  columns = [
    {
      Header: "Checklist Title",
      accessor: "templateName",
      style: { whiteSpace: "unset" },
    },
    {
      Header: "Template Description",
      accessor: "templateDescription",
      style: { whiteSpace: "unset" },
    },
    {
      Header: "Date Modified",
      accessor: "datemod",
      style: { whiteSpace: "unset" },
    },
    {
      Header: "",
      accessor: "deleteEditBtns",
      width: 73,
      style: { whiteSpace: "unset" },
      Cell: row => (
        <ButtonGroup>
          <Button
            id={`saveBtn${row.index}`}
            onClick={() => {
              this.toggleEdit(row.index);
            }}
            color="info"
            size="sm"
          >
            <i className="fas fa-edit" />
          </Button>
          <Button
            id={`delBtn${row.index}`}
            onClick={() => {
              if (window.confirm("Are you sure you wish to delete this item?")) {
                this.deleteRow(row.index);
              }
            }}
            color="danger"
            size="sm"
          >
            <i className="fa fa-trash-alt" />
          </Button>
        </ButtonGroup>
      ),
    },
  ];
 
  addChecklist = async (_templateData, _OrganizationName, _templateGUIDstr) => {
    if (_templateData && _OrganizationName && _templateGUIDstr) {
      const newGUID = await addChecklistEntry(
        {
          templateData: _templateData,
          organizationName: _OrganizationName,
          checklistPlusTemplateGUID: _templateGUIDstr,
        },
        this.props.currentProj,
      );
    }
  };
 
  handleSelectedOrg = option => {
    this.setState({ selectedOption: option }, () => {
      if (this.state.selectedOption) {
        this.getTableData(this.props.currentProj);
      }
    });
  };
 
  getTableData = async _projGUID => {
    let tableData = null;
    if (_projGUID) {
      tableData = await GetChecklistInstanceList({ id: _projGUID });
    }
    this.setState(() => {
      if (tableData) {
        const dataList = tableData.map(datum => {
          return {
            templateName: datum[0],
            templateDescription: datum[2],
            datemod: datum[5],
            guid: datum[4],
            dataString: datum[3],
          };
        });
        return { data: dataList };
      } else {
        return { data: [] };
      }
    });
  };
 
  componentDidMount() {
    this.getTableData(this.props.currentProj);
    if (this.state.data.length <= 0) {
      this.setState({ glowbutton: true });
    }
  }
 
  handleModalClose = () => {
    this.setState({ showImportModal: false });
  };
 
  openImportModal = () => {
    this.setState({ showImportModal: true });
  };
 
  toggleEdit = index => {
    const tempGuid = this.state.data[index].guid;
    this.toggleEditChecklist(tempGuid);
  };
 
  deleteRow = index => {
    removeChecklistEntry({
      projGUID: this.props.currentProj,
      templateGUIDstr: this.state.data[index].guid,
    });
 
    const out = [...this.state.data];
    out.splice(index, 1);
    this.setState({ data: out });
  };
 
  render() {
    return (
      <div>
        <div>
          <h3>{this.state.title}</h3>
        </div>
        <div>
          <ReactTable
            data={this.state.data}
            noDataText={`No Checklist Templates Available --- Create a New One Below`}
            columns={this.columns}
            defaultPageSize={5}
            className="-striped -highlight"
          />
        </div>
        <div style={{ display: "flex", justifyContent: "center" }}>
          {this.state.glowButton ? (
            <Button
              className="glowing"
              onClick={() => this.toggleNewChecklist()}
              color="primary"
              style={{ marginTop: 10, marginLeft: 10, marginRight: 10 }}
            >
              Add New Checklist
            </Button>
          ) : (
            <Button
              onClick={() => this.toggleNewChecklist()}
              color="primary"
              style={{ marginTop: 10, marginLeft: 10, marginRight: 10 }}
            >
              Add New Checklist
            </Button>
          )}
        </div>
 
        {this.state.newChecklist ? (
          <ChecklistPlusCreateEditIncidentModal
            addChecklist={this.addChecklist}
            colsInput={this.columns}
            selectedOrg={this.props.currentOrg}
            toggleNewChecklist={this.toggleNewChecklist}
          />
        ) : this.state.editChecklist ? (
          <ChecklistPlusCreateEditIncidentModal
            selectedOrg={this.props.currentOrg}
            currentProj={this.props.currentProj}
            isEdit={true}
            toEditGuid={this.state.selectedGuid}
            toggleEditChecklist={this.toggleEditChecklist}
          />
        ) : null}
      </div>
    );
  }
}
 
export default ChecklistPlusIncidentList;
