namespace KMLExport {
  let _postKMLUrl;
  let _showKMLUrl;
  export function init(postKMLUrl, showKMLUrl) {
    // Bind handler to click event of tree checkboxes to toggle visibility of layers
    $("#layertreeview").on("click", ".k-checkbox input:checkbox", function() {
      var state = "unchecked";
      if ($(this).is(":checked")) {
        state = "checked";
      }
      var itemId = $(this)
        .closest("li")
        .attr("data-id");
      var itemName = encodeURIComponent(
        $(this)
          .closest("li")
          .find(".k-in")
          .html(),
      );
      if (
        (itemId.search("ParentNode") && $(this).closest("li")[0].childNodes[1] != null) ||
        $(this)
          .closest("li")
          .children("ul.k-group").length > 0
      ) {
        var nodeIds = new Array();
        var nodeNames = new Array();
        //first add the checked node
        nodeIds.push(itemId);
        nodeNames.push(itemName);
        $(this)
          .closest("li")
          .find("li.k-item")
          .each(function() {
            nodeIds.push($(this).attr("data-id"));
            //nodeNames.push($(this).find(".k-in").html());
            var name = $(this)
              .find(".k-in")
              .html();
            name = encodeURIComponent(name);
            nodeNames.push(name);
          });
        AddOrRemoveNodes(nodeIds, nodeNames, state);
      } else {
        AddOrRemoveNode(itemId, itemName, state);
      }
    });
    if (Main.currentProjectGUID.length > 0) {
      $.connection.hub.start();
    }
  }
 
  function AddOrRemoveNode(itemUid, itemName, state) {
    var ids = [];
    ids.push(itemUid);
    var names = [];
    names.push(itemName);
    AddOrRemoveNodes(ids, names, state);
  }
 
  function AddOrRemoveNodes(itemUids, itemNames, state) {
    var nodesToIgnore = $("#nodesToIgnore").val();
    if (nodesToIgnore == "" || nodesToIgnore == null) {
      nodesToIgnore = "[]";
    }
    var namesToIgnore = $("#nodeNamesToIgnore").val();
    if (namesToIgnore == "" || namesToIgnore == null) {
      namesToIgnore = "[]";
    }
    var NodesToUpdate;
    NodesToUpdate.NodeIds = JSON.stringify(itemUids);
    NodesToUpdate.NodeNames = JSON.stringify(itemNames);
    NodesToUpdate.NodesToIgnore = nodesToIgnore; //JSON.stringify(nodesToIgnore);
    NodesToUpdate.NodeNamesToIgnore = namesToIgnore; //JSON.stringify(namesToIgnore);
 
    if (state == "checked") {
      NodesToUpdate.CheckState = true;
    } else {
      NodesToUpdate.CheckState = false;
    }
    //var params = { nodeIDs: itemUid, itemNames: itemName, stringState: state, nodesToIgnore: nodesToIgnore};
    Main.blockit($("#submitDiv"));
    $.ajax({
      url: _postKMLUrl,
      data: JSON.stringify(NodesToUpdate),
      //contentType: 'application/json; charset=utf-8',
      type: "POST",
      success: function(response) {
        $("#nodesToIgnore").val(response[0]);
        $("#nodeNamesToIgnore").val(response[1]);
        Main.unblockit($("#submitDiv"));
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }
 
  function getSelectedURL(asNetworkLink, includeOrgData) {
    //var dwnType = $('#DownloadType').val();
    if (includeOrgData === null || includeOrgData === undefined) {
      includeOrgData = false;
    }
    var url;
    if (asNetworkLink == true) {
      url = _showKMLUrl + "&UseExplicitKML=false&IncludeOrgData=" + includeOrgData;
    } else {
      url = _showKMLUrl + "&UseExplicitKML=true&IncludeOrgData=" + includeOrgData;
    }
    return url;
  }
  export function GetKML(asNetworkLink) {
    var url = getSelectedURL(asNetworkLink, $("#includeOrgData").prop("checked"));
    window.location.href = url;
  }
}
