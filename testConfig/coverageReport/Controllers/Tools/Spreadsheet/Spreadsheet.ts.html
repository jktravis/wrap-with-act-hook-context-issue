  interface IGlobalPrompts {
    MissingNameMessage: string;
    uploadURL: string;
    deleteURL: string;
  }
  var _globalPrompts: IGlobalPrompts;
  export function init(globalPrompts: IGlobalPrompts, templateData: string) {
    _globalPrompts = globalPrompts;
    //loadImportUpload();
    if ($("#spreadsheet").length > 0) {
      var spreadsheet = $("#spreadsheet").data("kendoSpreadsheet");
      if (templateData != null && templateData.length > 0) {
        spreadsheet.fromJSON($.parseJSON(templateData));
      }
      $("form").on("submit", function(e) {
        e.preventDefault();
        if (document.activeElement.id === "btnSubmit") {
          $(this)
            .find("input[type=submit]")
            .prop("disabled", true);
          if (validateForm(e)) {
            //populate hidden fields
            $("#TemplateData").val(JSON.stringify(spreadsheet.toJSON()));
            this.submit();
          } else {
            $(this)
              .find("input[type=submit]")
              .prop("disabled", false);
          }
        }
      });
    }
  }
  function loadImportUpload() {
    var ALLOWED_EXTENSIONS = [".xlsx", ".csv", ".txt", ".json"];
    $("#upload").kendoUpload({
      async: {
        saveUrl: _globalPrompts.uploadURL,
      },
      multiple: false,
      localization: {
        select: "Select file to import...",
      },
      select: function(e) {
        var extension = e.files[0].extension.toLowerCase();
        if (ALLOWED_EXTENSIONS.indexOf(extension) === -1) {
          alert("Please, select a supported file format");
          e.preventDefault();
        }
      },
      success: function(e) {
        var spreadsheet = $("#spreadsheet").data("kendoSpreadsheet");
        spreadsheet.fromJSON(e.response);
      },
    });
  }
  function validateForm(e): Boolean {
    var errResponse = {
      Messages: [],
    };
 
    var tempName = $("#TemplateName").val();
    if (tempName.length > 0) {
      Main.clearObjectErrorStatus($("#TemplateName"));
    } else {
      Main.markObjectAsError($("#TemplateName"));
      errResponse.Messages.push(_globalPrompts.MissingNameMessage);
    }
    if (errResponse.Messages.length > 0) {
      Main.handleValErr(null, errResponse);
      return false;
    } else {
      return true;
    }
  }
  export function deleteTemplate(templateGUID) {
    $Ajax.ajaxAntiForgery({
      url: _globalPrompts.deleteURL,
      type: "POST",
      data: {
        templateGUID: templateGUID,
      },
      success: function(response) {
        if (response.success === false) {
          $.notify({ message: response.Message }, { type: "danger" });
        } else {
          var gridDataSource = $("#sprdlistgrid").data("kendoGrid").dataSource;
          gridDataSource.read();
        }
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }
}
