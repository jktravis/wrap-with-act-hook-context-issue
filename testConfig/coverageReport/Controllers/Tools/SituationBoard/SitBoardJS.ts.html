namespace Dashboard_SitBoard {
  interface ISitBoardScope extends ng.IScope {
    currentHeight: number;
    getCategoryData(parID: string, isLarge: boolean): void;
  }
  function refreshAllsitboard() {
    if (angular.element("#sitboardControllerID").length) {
      //$("#SelectedParent").data("kendoDropDownList").dataSource.read();
      parchanged();
    }
    //would like to figure out a way to make this repository reside in the dashboard controller somehow, so that i
    //only need to refresh this once.
    if (angular.element("#sitboardControllerIDM").length) {
      //$("#SelectedParentM").data("kendoDropDownList").dataSource.read();
      parchangedM();
    }
  }
  export function parchanged() {
    if ($("#SelectedParent").length > 0) {
      const drp = $("#SelectedParent").data("kendoDropDownList");
      if (drp) {
        const parID = drp.value();
        if (angular.element("#sitboardControllerID").length) {
          (<ISitBoardScope>angular.element("#sitboardControllerID").scope()).getCategoryData(parID, true);
        }
      }
    }
  }
  export function parchangedM() {
    if ($("#SelectedParentM").length > 0) {
      const drp = $("#SelectedParentM").data("kendoDropDownList");
      if (drp) {
        const parID = drp.value();
        (<ISitBoardScope>angular.element("#sitboardControllerIDM").scope()).currentHeight = $(
          "#sitMlst",
        )[0].offsetHeight;
        (<ISitBoardScope>angular.element("#sitboardControllerIDM").scope()).getCategoryData(parID, false);
      }
    }
  }
  angular.module("cobraapp").factory("sitboardsRepository", function($http) {
    return {
      getCategories: function(hostURL, projectGUID, parentID) {
        return $http
          .get(hostURL + "Tools/SituationBoard/GetSitBoardData/" + projectGUID + "?parentID=" + parentID)
          .then(function(res) {
            return res.data;
          });
      },
    };
  });
 
  angular.module("cobraapp").controller("sitboardController", function($scope, sitboardsRepository) {
    $scope.loadcard = function(ocard) {
      $scope.currentCard = ocard;
      if (ocard.SelectedSize === 1) {
        parchangedM();
      } else {
        parchanged();
      }
    };
    $scope.$on("initModalJS", function() {
      //do nothing
    });
    $scope.$on("refreshSelf", function(event, toolname) {
      if (toolname === $scope.currentCard.toolName) {
        refreshAllsitboard();
      }
    });
 
    $scope.getCategoryData = function(parentID, islarge) {
      sitboardsRepository.getCategories($scope.hostURL, $scope.projectGUID, parentID).then(function(categories) {
        $scope.categories = categories;
        if (islarge === true) {
          $scope.cardFinishedLoading($scope.largecard);
        } else {
          //var tmp = $("#sitMlst")[0].offsetHeight;
          //if ((tmp !== $scope.currentHeight) && ($scope.currentCard.isLoading === false)) //not working for some reason
          //{
          //Dashboard.refreshMasonry();
          //}
          $scope.cardFinishedLoading($scope.currentCard);
        }
      });
    };
  });
