import React, { Component } from "react";
import { getLocatorBySerial } from "Api/LocatorApi";
import Modal from "reactstrap/lib/Modal";
import ModalHeader from "reactstrap/lib/ModalHeader";
import ModalBody from "reactstrap/lib/ModalBody";
import Label from "reactstrap/lib/Label";
import { Button, Form, Input, ButtonGroup, Col, Row } from "reactstrap";
import "filepond/dist/filepond.min.css";
import "../../../../Content/ExtraStyles/Locator.scss";
import FormGroup from "reactstrap/lib/FormGroup";
 
type Props = {
  translatedFields?: any;
  buttonColor?: string;
  closeModalfn?: Function;
};
 
class LocatorLookupModal extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    translatedFields: {},
    buttonColor: "primary",
    closeModalfn: Function.prototype,
  };
 
  state = {
    selectedLocator: null,
    searchText: null,
    wasSearched: false,
  };
 
  toggle = () => {
    this.props.closeModalfn();
  };
 
  async handleSearch(e) {
    e.preventDefault();
    const locSerial = e.target[0].value;
    if (locSerial) {
      const loc = await getLocatorBySerial({ serialNum: locSerial });
      this.setState({ selectedLocator: loc, wasSearched: true });
    }
  }
 
  render() {
    return (
      <div>
        <Modal isOpen={true}>
          <ModalHeader>{this.props.translatedFields.LocatorLookupTitle}</ModalHeader>
          <ModalBody>
            <Form onSubmit={this.handleSearch.bind(this)}>
              <Row>
                <Col>
                  <FormGroup>
                    <Label>
                      <b>{this.props.translatedFields.SerialNumber}</b>
                    </Label>
                    <Input type="text" name="SerialNumber" />
                  </FormGroup>
                </Col>
              </Row>
              {this.state.selectedLocator ? (
                <Row>
                  <Col>
                    <FormGroup>
                      <h1>
                        <Label>{this.props.translatedFields.LocatorName}:</Label>
                        <br />
                        {this.state.selectedLocator.LocatorName}
                        <br />
                        <br />
                        <br />
                        <Label>{this.props.translatedFields.OrganizationColumn}:</Label>
                        <br />
                        {this.state.selectedLocator.OrganizationName}
                      </h1>
                    </FormGroup>
                  </Col>
                </Row>
              ) : this.state.wasSearched ? (
                <h1>
                  <Label>{this.props.translatedFields.NotFoundTitle}</Label>
                </h1>
              ) : null}
              <Row>
                <Col>
                  <ButtonGroup>
                    <Button type="submit" name="Submit" color="primary">
                      {this.props.translatedFields.SearchTitle}
                    </Button>
                    <Button color="warning" onClick={this.toggle}>
                      {this.props.translatedFields.CancelText}
                    </Button>
                  </ButtonGroup>
                </Col>
              </Row>
            </Form>
          </ModalBody>
        </Modal>
      </div>
    );
  }
}
 
export default LocatorLookupModal;
