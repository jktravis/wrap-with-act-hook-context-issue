import React, { Component } from "react";
import Select from "react-select";
import LocatorPageIDEnum from "./LocatorPageIDEnum";
import { Button, Form, Input, Col } from "reactstrap";
import "../../../../Content/ExtraStyles/Locator.scss";
import Label from "reactstrap/lib/Label";
import OrgSelector from "../../Shared/Components/OrgSelector/OrgSelector";
import { baseUrl, IsSysAdmin, currentOrg } from "Api/constants";
import Row from "reactstrap/lib/Row";
import ButtonGroup from "reactstrap/lib/ButtonGroup";
import Modal from "reactstrap/lib/Modal";
import ModalHeader from "reactstrap/lib/ModalHeader";
import ModalBody from "reactstrap/lib/ModalBody";
import { createLocatorType, UpdateLocatorType, getListOfLocatorTypesImages } from "Api/LocatorApi";
 
type Props = {
  pageChangeFn?: Function;
  translatedFields?: any;
  currentSelection?: any;
};
 
const imagePoolUrl = baseUrl + "/riskimg/";
 
function getOptionLabel({ label, value }) {
  return (
    <div>
      {value ? <img src={imagePoolUrl + value + ".png?w=25"} alt={label} /> : null}
      <span>{label}</span>
    </div>
  );
}
 
class LocatorTypeCreateEditModal extends Component<any, any> {
  static defaultProps: Partial<Props> = {
    currentSelection: null,
    translatedFields: {},
  };
 
  state = {
    fields: {
      OrganizationName: currentOrg,
      LocatorTypeGUID: null,
      LocatorTypeName: null,
      LocatorTypeImagePoolGUID: null,
    },
    errors: {},
    selectedOrg: { label: currentOrg, value: currentOrg },
    isEdit: false,
    availableImages: [],
    validationGood: false,
    isGlobalType: false,
  };
 
  validate = () => {
    const fields = this.state.fields;
    const errors: any = {};
    let formIsValid = true;
 
    //LocatorTypeName
    if (!fields.LocatorTypeName) {
      formIsValid = false;
      errors.LocatorTypeName =
        this.props.translatedFields.LocatorTypeName + " " + this.props.translatedFields.IsRequired;
    }
 
    //Organization Name
    if (!IsSysAdmin) {
      fields.OrganizationName = currentOrg;
    }
 
    if (this.state.isGlobalType) {
      fields.OrganizationName = "";
    }
 
    //Image
    if (!fields.LocatorTypeImagePoolGUID) {
      formIsValid = false;
      errors.LocatorTypeImagePoolGUID = "Image is Required";
    }
 
    this.setState({ validationGood: formIsValid });
    return errors;
  };
 
  handleCreateLocatorType = async parent => {
    parent.persist();
    parent.preventDefault();
    const errors = this.validate();
    if (this.state.validationGood) {
      const { LocatorTypeName, LocatorTypeGUID, OrganizationName, LocatorTypeImagePoolGUID } = this.state.fields;
      if (!this.state.isEdit) {
        await createLocatorType(LocatorTypeName, OrganizationName, LocatorTypeImagePoolGUID);
      } else {
        await UpdateLocatorType(LocatorTypeGUID, LocatorTypeName, LocatorTypeImagePoolGUID);
      }
      this.toggle();
    } else {
      this.setState({ validationGood: false, errors: errors });
    }
  };
 
  handleChange(field, e) {
    const fields = this.state.fields;
    fields[field] = e.target.value;
    this.setState({ fields });
    this.validate();
  }
 
  handleSelectedImage = option => {
    const fields = this.state.fields;
    fields.LocatorTypeImagePoolGUID = option.value;
    this.setState({ fields: fields });
    this.validate();
  };
 
  handleSelectedOrg = option => {
    const fields = this.state.fields;
    fields.OrganizationName = option.value;
    this.setState({ fields: fields, selectedOrg: option });
    this.updateImageDropdown();
    this.validate();
  };
 
  updateImageDropdown = async () => {
    let orgUsed = currentOrg;
    if (this.state.selectedOrg !== null) {
      orgUsed = this.state.selectedOrg.value;
    }
    const avail = await getListOfLocatorTypesImages({ OrganizationName: orgUsed });
    if (avail !== null) {
      this.setState(() => ({
        availableImages: avail.map(d => {
          return { label: d.FileName, value: d.RiskImagePoolGUID };
        }),
      }));
    }
  };
 
  async componentDidMount() {
    const fields = this.state.fields;
    if (this.props.currentSelection !== null) {
      fields.LocatorTypeGUID = this.props.currentSelection.LocatorTypeGUID;
      fields.OrganizationName = this.props.currentSelection.OrganizationName;
      fields.LocatorTypeName = this.props.currentSelection.LocatorTypeName;
      fields.LocatorTypeImagePoolGUID = this.props.currentSelection.LocatorTypeImagePoolGUID;
      this.setState({
        fields: fields,
        selectedOrg: { value: fields.OrganizationName, label: fields.OrganizationName },
        isEdit: true,
        validationGood: true,
      });
    }
    this.updateImageDropdown();
  }
 
  toggle = () => {
    this.props.closeModalfn();
  };
 
  render() {
    const errors: any = this.state.errors;
    let selectedImage = {};
 
    if (
      this.state.availableImages !== undefined &&
      this.state.availableImages !== null &&
      this.state.availableImages.length > 0 &&
      this.state.fields.LocatorTypeImagePoolGUID
    ) {
      const value = this.state.availableImages.find(d => {
        return d.value === this.state.fields.LocatorTypeImagePoolGUID;
      });
      if (value !== null) {
        selectedImage = { label: value.label, value: value.value };
      }
    }
 
    return (
      <div>
        <Modal isOpen={true} toggle={this.toggle}>
          <ModalHeader>{this.props.translatedFields.RegisterNewLocator}</ModalHeader>
          <ModalBody>
            <Form onSubmit={this.handleCreateLocatorType} name={LocatorPageIDEnum.LocatorTypePage} className="col-12">
              <Row>
                <Col>
                  <Label>
                    <b>{this.props.translatedFields.LocatorTypeName}</b>
                    <span style={{ color: "red" }}>*</span>
                    <Input
                      type="text"
                      name="LocatorTypeName"
                      value={this.state.fields.LocatorTypeName}
                      onChange={this.handleChange.bind(this, "LocatorTypeName") as any}
                    />
                    <span style={{ color: "red" }}>
                      <b>{errors.LocatorTypeName}</b>
                    </span>
                  </Label>
                </Col>
                {IsSysAdmin && !this.state.isEdit ? (
                  <Col>
                    <Label for="GlobalCheckbox">
                      <b>{this.props.translatedFields.Global}</b>
                    </Label>
                    <br />
                    <input
                      name="GlobalCheckbox"
                      type="checkbox"
                      checked={this.state.isGlobalType}
                      onChange={() => {
                        this.setState({ isGlobalType: !this.state.isGlobalType });
                      }}
                    />
                  </Col>
                ) : null}
              </Row>
              {IsSysAdmin && !this.state.isEdit ? (
                <Row>
                  {!this.state.isGlobalType ? (
                    <Col>
                      <Label for="OrgSelector">
                        <b>{this.props.translatedFields.OrganizationColumn}</b>
                        <span style={{ color: "red" }}>*</span>
                      </Label>
                      <OrgSelector changeFn={this.handleSelectedOrg} selectedOption={this.state.selectedOrg} />
                      <span style={{ color: "red" }}>
                        <b>{errors.OrganizationName}</b>
                      </span>
                    </Col>
                  ) : null}
                </Row>
              ) : null}
              <Row>
                <Col>
                  <Label for="ImageSelector">
                    <b>{this.props.translatedFields.AssociatedImageName}</b>
                    <span style={{ color: "red" }}>*</span>
                  </Label>
                  <Select
                    id="ImageSelector"
                    options={this.state.availableImages}
                    value={selectedImage}
                    onChange={this.handleSelectedImage as any}
                    getOptionLabel={getOptionLabel}
                  />
                  <span style={{ color: "red" }}>
                    <b>{errors.LocatorTypeImagePoolGUID}</b>
                  </span>
                </Col>
              </Row>
              <br />
              <Row>
                <Col>
                  <ButtonGroup>
                    <Button disabled={!this.state.validationGood} type="submit" name="Submit" color="primary">
                      {this.props.translatedFields.SubmitButton}
                    </Button>
                    <Button onClick={this.toggle} color="warning">
                      {this.props.translatedFields.CancelText}
                    </Button>
                  </ButtonGroup>
                </Col>
              </Row>
            </Form>
          </ModalBody>
        </Modal>
      </div>
    );
  }
}
 
export default LocatorTypeCreateEditModal;
