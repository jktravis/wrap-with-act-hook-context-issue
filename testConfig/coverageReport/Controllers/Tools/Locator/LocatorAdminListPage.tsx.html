import React, { Component } from "react";
import OrgSelector from "../../Shared/Components/OrgSelector/OrgSelector";
import ReactTable from "react-table";
import { getListOfLocators, removeLocator } from "Api/LocatorApi";
import { baseUrl, currentOrg } from "Api/constants";
import LocatorPageIDEnum from "./LocatorPageIDEnum";
import Button from "reactstrap/lib/Button";
import "../../../../Content/ExtraStyles/Locator.scss";
import LocatorMassImportModal from "./LocatorMassImportModal";
import LocatorCreateEditModal from "./LocatorCreateEditModal";
import LocatorLookupModal from "./LocatorLookupModal";
import BatteryBar from "./BatteryBar";
import { ButtonGroup } from "reactstrap";
 
const imagePoolUrl = baseUrl + "/riskimg/";
const intervalRefreshTimer = 3000;
 
type Props = {
  pageChangeFn?: Function;
  translatedFields?: any;
};
 
class LocatorAdminListPage extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    pageChangeFn: Function.prototype,
    translatedFields: {},
  };
 
  state = {
    selectedOption: { label: currentOrg, value: currentOrg },
    tableData: [],
    selectedLocator: null,
    showLocatorEditModal: false,
    showMassImportModal: false,
    showLookupModal: false,
    interval: 0,
  };
 
  handleSelectedOrg = option => {
    this.setState(
      () => ({ selectedOption: option }),
      () => {
        if (this.state.selectedOption) {
          this.getTableData(this.state.selectedOption.value);
        }
      },
    );
  };
 
  async getTableData(orgName) {
    const data = await getListOfLocators({ OrganizationName: orgName });
    this.setState(() => ({ tableData: data }));
  }
 
  handleModalClose = () => {
    if (this.state.selectedOption) {
      this.getTableData(this.state.selectedOption.value);
    } else {
      this.getTableData(currentOrg);
    }
    this.setState({
      showLocatorEditModal: false,
      showMassImportModal: false,
      showLookupModal: false,
      selectedLocator: null,
    });
  };
 
  async componentDidMount() {
    this.getTableData(currentOrg);
    const interval = setInterval(() => {
      this.getTableData(currentOrg);
    }, intervalRefreshTimer);
    this.setState({ interval });
  }
 
  async componentWillUnmount() {
    clearInterval(this.state.interval);
  }
 
  async deleteLocator(serialNumber) {
    const result = confirm(this.props.translatedFields.DeleteConfirm);
    if (result) {
      try {
        await removeLocator(serialNumber);
      } catch (e) {
        alert(this.props.translatedFields.FailedDelete);
      }
      this.getTableData(currentOrg);
    }
  }
 
  async loadLocatorEditPage(row) {
    const values = row.original;
    this.setState({ selectedLocator: values, showLocatorEditModal: true });
  }
 
  render() {
    const { pageChangeFn, translatedFields } = this.props;
 
    const columns = [
      {
        Header: "",
        accessor: "LocatorTypeImagePoolGUID",
        width: 30,
        filterable: false,
        sortable: false,
        Cell: row => <img src={imagePoolUrl + row.value + ".png?w=25"} alt={row.original.FileName} />,
      },
      {
        Header: this.props.translatedFields.LocatorName,
        accessor: "LocatorName",
      },
      {
        Header: this.props.translatedFields.SerialNumber,
        accessor: "LocatorSerialNum",
      },
      {
        Header: this.props.translatedFields.OrganizationColumn,
        accessor: "OrganizationName",
      },
      {
        Header: this.props.translatedFields.TimeOfReading,
        accessor: "MostRecentReadingDate",
        Cell: row => <span>{row.value ? new Date(row.value).toLocaleString() : null}</span>,
      },
      {
        Header: this.props.translatedFields.BatteryPercent,
        accessor: "MostRecentBatteryLevel",
        width: 100,
        Cell: row => <BatteryBar batteryLevel={row.value} batteryMaxValue={80} maxWidth={50} />,
      },
      {
        Header: "",
        accessor: "LocatorSerialNum",
        width: 100,
        filterable: false,
        sortable: false,
        Cell: row => (
          <ButtonGroup>
            <Button onClick={() => this.loadLocatorEditPage(row) as any} color="warning">
              <i className="fa fa-edit" />
            </Button>
            <Button onClick={() => this.deleteLocator(row.value) as any} color="danger">
              <i className="fa fa-trash" />
            </Button>
          </ButtonGroup>
        ),
      },
    ];
    return (
      <div>
        <OrgSelector changeFn={this.handleSelectedOrg} selectedOption={this.state.selectedOption} />
        <br />
        <ReactTable
          data={this.state.tableData}
          columns={columns}
          defaultPageSize={10}
          filterable
          defaultSorted={[{ id: "LocatorName", desc: false }]}
        />
        <br />
        <div className="rowC">
          <Button onClick={pageChangeFn as any} name={LocatorPageIDEnum.LocatorTypePage} color="primary">
            {this.props.translatedFields.ViewLocatorTypes}
          </Button>
          <Button
            onClick={() => {
              this.setState({ showLookupModal: true });
            }}
            color="primary"
          >
            {this.props.translatedFields.LocatorLookupTitle}
          </Button>
          <Button
            onClick={() => {
              this.setState({ showLocatorEditModal: true });
            }}
            color="primary"
            className="buttonFlexRight"
          >
            {this.props.translatedFields.RegisterNewLocator}
          </Button>
          <Button
            onClick={() => {
              this.setState({ showMassImportModal: true });
            }}
            color="primary"
          >
            {this.props.translatedFields.ImportFromFile}
          </Button>
        </div>
        {this.state.showLocatorEditModal ? (
          <LocatorCreateEditModal
            closeModalfn={this.handleModalClose}
            translatedFields={this.props.translatedFields}
            currentSelection={this.state.selectedLocator}
          />
        ) : null}
        {this.state.showMassImportModal ? (
          <LocatorMassImportModal closeModalfn={this.handleModalClose} translatedFields={translatedFields} />
        ) : null}
        {this.state.showLookupModal ? (
          <LocatorLookupModal closeModalfn={this.handleModalClose} translatedFields={translatedFields} />
        ) : null}
      </div>
    );
  }
}
 
export default LocatorAdminListPage;
