import LocatorIncidentHomePage from "./LocatorIncidentHomePage";
import { getTranslations } from "../../../Api/translationApi";

  "LocatorName",

interface Props {
  project: string;
}
 
class LocatorApp extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    project: "",
  };
 
  state = {
    pageID: LocatorPageIDEnum.LocatorPage,
    values: null,
    translatedFields: {},
  };
 
  handleClick = event => {
    const { name } = event.target;
    $("html, body").animate({ scrollTop: 0 }, "fast");
    this.setState(() => ({ pageID: name, values: null }));
  };
 
  async componentDidMount() {
    const data = await getTranslations(requiredTranslatedFields);
    this.setState(() => ({ translatedFields: data }));
  }
 
  render() {
    return (
      <div>
        {this.props.project ? (
          this.state.pageID === LocatorPageIDEnum.LocatorPage ? (
            <LocatorIncidentHomePage
              translatedFields={this.state.translatedFields}
              pageChangeFn={this.handleClick}
              project={this.props.project}
            />
          ) : null
        ) : this.state.pageID === LocatorPageIDEnum.LocatorPage ? (
          <LocatorAdminListPage translatedFields={this.state.translatedFields} pageChangeFn={this.handleClick} />
        ) : this.state.pageID === LocatorPageIDEnum.LocatorTypePage ? (
          <LocatorTypeListPage translatedFields={this.state.translatedFields} pageChangeFn={this.handleClick} />
        ) : null}
      </div>
    );
  }
}
 
export default LocatorApp;
