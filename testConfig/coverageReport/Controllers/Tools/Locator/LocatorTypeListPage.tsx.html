import React, { Component } from "react";
import OrgSelector from "../../Shared/Components/OrgSelector/OrgSelector";
import ReactTable from "react-table";
import { getListOfLocatorTypes, removeLocatorType } from "../../../Api/LocatorApi";
import { baseUrl, IsSysAdmin, currentOrg } from "../../../Api/constants";
import LocatorPageIDEnum from "./LocatorPageIDEnum";
import LocatorTypeCreateEditModal from "./LocatorTypeCreateEditModal";
import LocatorTypeImagePoolList from "./LocatorTypeImagePoolList";
import { Button, ButtonGroup, Form, Input, Col, Row } from "reactstrap";
import "../../../../Content/ExtraStyles/Locator.scss";
 
type Props = {
  pageChangeFn?: Function;
  translatedFields?: any;
};
 
const imagePoolUrl = baseUrl + "/riskimg/";
 
class LocatorTypeListPage extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    pageChangeFn: Function.prototype,
    translatedFields: {},
  };
 
  state = {
    selectedOption: { label: currentOrg, value: currentOrg },
    tableData: [],
    showCreateEditModal: false,
    selectedLocatorType: null,
  };
 
  handleSelectedOrg = option => {
    this.setState(
      () => ({ selectedOption: option }),
      () => {
        if (this.state.selectedOption) {
          this.getTableData(this.state.selectedOption.value);
        }
      },
    );
  };
 
  async getTableData(orgName) {
    let data = await getListOfLocatorTypes({ OrganizationName: orgName });
    if (data === null || data === undefined) {
      data = [];
    }
    this.setState(() => ({ tableData: data }));
  }
 
  async componentDidMount() {
    this.getTableData(currentOrg);
  }
 
  async deleteLocatorType(guid) {
    const result = confirm(this.props.translatedFields.DeleteConfirm);
    if (result) {
      try {
        await removeLocatorType(guid);
        this.getTableData(currentOrg);
      } catch (e) {
        alert(this.props.translatedFields.FailedDelete);
      }
    }
  }
 
  async loadLocatorTypeEditPage(row) {
    const values = row.original;
    this.setState({ selectedLocatorType: values, showCreateEditModal: true });
  }
 
  handleModalClose = () => {
    if (this.state.selectedOption) {
      this.getTableData(this.state.selectedOption.value);
    } else {
      this.getTableData(currentOrg);
    }
    this.setState({ showCreateEditModal: false, selectedLocatorType: null });
  };
 
  render() {
    const { pageChangeFn } = this.props;
    const columns = [
      {
        Header: "",
        accessor: "LocatorTypeImagePoolGUID",
        width: 50,
        Cell: row => (
          <img src={imagePoolUrl + row.original.LocatorTypeImagePoolGUID + ".png?w=25"} alt={row.original.FileName} />
        ),
      },
      {
        Header: this.props.translatedFields.LocatorTypeName,
        accessor: "LocatorTypeName",
      },
      {
        Header: this.props.translatedFields.OrganizationColumn,
        accessor: "OrganizationName",
      },
      {
        Header: "",
        accessor: "LocatorTypeGUID",
        width: 100,
        Cell: row => (
          <ButtonGroup>
            <Button onClick={() => this.loadLocatorTypeEditPage(row) as any} color="warning">
              <i className="fa fa-edit" />
            </Button>
            <Button onClick={() => this.deleteLocatorType(row.value) as any} color="danger">
              <i className="fa fa-trash" />
            </Button>
          </ButtonGroup>
        ),
      },
    ];
    return (
      <div>
        <OrgSelector changeFn={this.handleSelectedOrg} selectedOption={this.state.selectedOption} />
        <br />
        <Row>
          <Col>
            <ReactTable data={this.state.tableData} columns={columns} defaultPageSize={10} />
          </Col>
          <Col>
            <LocatorTypeImagePoolList
              translatedFields={this.props.translatedFields}
              selectedOrg={this.state.selectedOption.value}
            />
          </Col>
        </Row>
        <br />
        <div className="rowC">
          <Button
            onClick={() => {
              this.setState({ showCreateEditModal: true });
            }}
            color="primary"
          >
            {this.props.translatedFields.CreateLocatorType}
          </Button>
          <Button
            className="buttonFlexRight"
            onClick={pageChangeFn as any}
            name={LocatorPageIDEnum.LocatorPage}
            color="primary"
          >
            {this.props.translatedFields.ViewLocators}
          </Button>
        </div>
        {this.state.showCreateEditModal ? (
          <LocatorTypeCreateEditModal
            closeModalfn={this.handleModalClose}
            translatedFields={this.props.translatedFields}
            currentSelection={this.state.selectedLocatorType}
          />
        ) : null}
      </div>
    );
  }
}
 
export default LocatorTypeListPage;
