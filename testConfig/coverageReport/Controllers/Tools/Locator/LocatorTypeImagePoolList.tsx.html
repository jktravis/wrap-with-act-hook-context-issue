import React, { Component } from "react";
import ReactTable from "react-table";
import { getListOfLocatorTypesImages, removeLocatorTypeImage, addLocatorTypeImage } from "Api/LocatorApi";
import { baseUrl, currentOrg } from "Api/constants";
import Button from "reactstrap/lib/Button";
import "../../../../Content/ExtraStyles/Locator.scss";
import Modal from "reactstrap/lib/Modal";
import ModalFooter from "reactstrap/lib/ModalFooter";
import ModalHeader from "reactstrap/lib/ModalHeader";
import ModalBody from "reactstrap/lib/ModalBody";
import { FilePond, File, registerPlugin } from "react-filepond";
import "filepond/dist/filepond.min.css";
 
const imagePoolUrl = baseUrl + "/riskimg/";
 
type Props = {
  selectedOrg?: string;
  translatedFields?: any;
};
 
class LocatorTypeImagePoolList extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    selectedOrg: "",
    translatedFields: {},
  };
 
  state = {
    tableData: [],
    fileUploadResultText: "",
    selectedFiles: null,
    showUploadModal: false,
  };
 
  async getTableData(orgName) {
    const data = await getListOfLocatorTypesImages({ OrganizationName: orgName });
    this.setState(() => ({ tableData: data }));
  }
 
  async componentDidMount() {
    this.getTableData(this.props.selectedOrg);
  }
 
  async deleteLocatorImage(guid) {
    const result = confirm(this.props.translatedFields.DeleteConfirm);
    if (result) {
      try {
        await removeLocatorTypeImage(guid);
        this.getTableData(currentOrg);
      } catch (e) {
        alert(this.props.translatedFields.FailedDelete);
      }
    }
  }
 
  upload = async () => {
    const files = this.state.selectedFiles;
 
    const formData = new FormData();
    for (let x = 0; x < files.length; x++) {
      formData.append(files[x].name, files[x]);
    }
    let response;
    try {
      response = await addLocatorTypeImage(formData, { OrganizationName: currentOrg });
    } catch (err) {
      response = err.response.data.ExceptionMessage;
    }
 
    this.setState({ selectedFiles: null });
    this.toggle();
  };
 
  toggle = () => {
    this.getTableData(currentOrg);
    this.setState({ showUploadModal: false });
  };
 
  render() {
    const { translatedFields } = this.props;
    const columns = [
      {
        Header: translatedFields.LocatorImagePoolTitle,
        columns: [
          {
            Header: "",
            accessor: "RiskImagePoolGUID",
            width: 50,
            Cell: row => (
              <img src={imagePoolUrl + row.original.RiskImagePoolGUID + ".png?w=25"} alt={row.original.FileName} />
            ),
          },
          {
            Header: "",
            accessor: "FileName",
          },
          {
            Header: "",
            accessor: "RiskImagePoolGUID",
            width: 50,
            Cell: row => (
              <Button onClick={() => this.deleteLocatorImage(row.value) as any} color="danger">
                <i className="fa fa-trash" />
              </Button>
            ),
          },
        ],
      },
    ];
    return (
      <div>
        <ReactTable data={this.state.tableData} columns={columns} defaultPageSize={10} />
        <div className="rowC">
          <Button
            className="buttonFlexCenter"
            onClick={() => {
              this.setState({ showUploadModal: true });
            }}
            color="primary"
          >
            {this.props.translatedFields.Upload}
          </Button>
        </div>
        <span style={{ color: "red", justifyContent: "center" }}>
          <b>{this.state.fileUploadResultText}</b>
        </span>
        {this.state.showUploadModal ? (
          <Modal isOpen={true}>
            <ModalHeader>{this.props.translatedFields.Upload}</ModalHeader>
            <ModalBody>
              <FilePond
                allowMultiple={true}
                onupdatefiles={fileItems => {
                  this.setState({
                    selectedFiles: fileItems.map(fileItem => fileItem.file),
                  });
                }}
              >
                {this.state.selectedFiles
                  ? this.state.selectedFiles.map(file => <File key={file} src={file} origin="local" />)
                  : null}
              </FilePond>
            </ModalBody>
            <ModalFooter>
              <Button color="primary" onClick={this.upload}>
                {this.props.translatedFields.Upload}
              </Button>{" "}
              <Button color="warning" onClick={this.toggle}>
                {this.props.translatedFields.CancelText}
              </Button>
            </ModalFooter>
          </Modal>
        ) : null}
      </div>
    );
  }
}
 
export default LocatorTypeImagePoolList;
