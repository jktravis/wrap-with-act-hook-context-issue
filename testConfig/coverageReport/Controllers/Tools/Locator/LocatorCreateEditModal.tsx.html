import React, { Component } from "react";
import { Button, Form, Input, ButtonGroup, Col } from "reactstrap";
import "../../../../Content/ExtraStyles/Locator.scss";
import Modal from "reactstrap/lib/Modal";
import ModalHeader from "reactstrap/lib/ModalHeader";
import ModalBody from "reactstrap/lib/ModalBody";
import Label from "reactstrap/lib/Label";
import OrgSelector from "../../Shared/Components/OrgSelector/OrgSelector";
import { baseUrl, IsSysAdmin, currentOrg } from "Api/constants";
import { getListOfLocatorTypes, createLocator, UpdateLocator } from "Api/LocatorApi";
import Row from "reactstrap/lib/Row";
import FormGroup from "reactstrap/lib/FormGroup";
import Select from "react-select";
 
const imagePoolUrl = baseUrl + "/riskimg/";
 
type Props = {
  translatedFields: any;
  currentSelection?: any;
  closeModalfn?: Function;
};
 
function getOptionLabel({ label, imageGUID }) {
  return (
    <div>
      {imageGUID ? <img src={imagePoolUrl + imageGUID + ".png?w=25"} alt={label} /> : null}
      <span>{label}</span>
    </div>
  );
}
 
class LocatorCreateEditModal extends Component<any, any> {
  static defaultProps: Partial<Props> = {
    currentSelection: null,
    translatedFields: {},
    closeModalfn: Function.prototype(),
  };
 
  state = {
    fields: { OrganizationName: currentOrg },
    errors: {},
    selectedOrg: { label: currentOrg, value: currentOrg },
    availableTypes: [],
    selectedLocatorType: null,
    isEdit: false,
    isMobile: false,
    validationGood: false,
  };
 
  validate = () => {
    const fields: any = this.state.fields;
    const errors: any = {};
    let formIsValid = true;
 
    //LocatorName
    if (!fields.LocatorName) {
      formIsValid = false;
      errors.LocatorName = this.props.translatedFields.LocatorName + " " + this.props.translatedFields.IsRequired;
    }
 
    //Organization Name
    if (IsSysAdmin) {
      if (!fields.OrganizationName) {
        formIsValid = false;
        errors.OrganizationName =
          this.props.translatedFields.OrganizationColumn + " " + this.props.translatedFields.IsRequired;
      }
    } else {
      fields.OrganizationName = currentOrg;
    }
 
    //LocatorTypeGUID
    if (!fields.LocatorTypeGUID) {
      formIsValid = false;
      errors.LocatorTypeGUID = this.props.translatedFields.LocatorType + " " + this.props.translatedFields.IsRequired;
    }
 
    if (!fields.LocatorSerialNumber) {
      formIsValid = false;
      errors.LocatorSerialNumber =
        this.props.translatedFields.LocatorSerialNumber + " " + this.props.translatedFields.IsRequired;
    }
 
    this.setState({ validationGood: formIsValid });
  };
 
  handleCreateLocator = async event => {
    event.preventDefault();
    const fields: any = this.state.fields;
 
    try {
      if (this.state.validationGood) {
        //LocatorSerialNumber
        //[DYNS]*[A-Z]{2}[0-9](0[0-9]|[1-5]+[0-9]+)[a-zA-Z0-9]{6}
        //Regex for serial number format, if confirmed
        if (
          !(
            fields.LocatorSerialNumber &&
            /[DYNS]*[A-Z]{2}[0-9](0[0-9]|[1-5]+[0-9]+)[a-zA-Z0-9]{6}/.test(fields.LocatorSerialNumber)
          )
        ) {
          if (confirm(this.props.translatedFields.ValidSerialNumConf + " " + fields.LocatorSerialNumber)) {
            //Moved this here so live validation doesnt constantly query the user.
            await this.handleCreateLocatorResponse(event);
            this.toggle();
          }
        } else {
          await this.handleCreateLocatorResponse(event);
          this.toggle();
        }
      } else {
        this.setState({ errors: false });
      }
    } catch (err) {
      const errors: any = {};
      errors.LocatorSerialNumber = err.response.data.ExceptionMessage;
      this.setState({ errors: errors });
      return errors;
    }
  };
 
  async handleCreateLocatorResponse(event) {
    let response = { Key: "", Value: false };
    const fields: any = this.state.fields;
    if (this.state.isEdit) {
      await UpdateLocator(fields.LocatorID, fields.LocatorName, fields.OrganizationName, fields.LocatorTypeGUID);
    } else {
      response = await createLocator(
        fields.OrganizationName,
        fields.LocatorSerialNumber,
        fields.LocatorName,
        fields.LocatorTypeGUID,
      );
    }
    if (response.Value) {
      const errors: any = {};
      errors.LocatorSerialNumber = "There is already a Locator with this Serial Number registered";
      this.setState({ errors: errors });
    } else {
      this.toggle();
    }
  }
 
  handleChange(field, e) {
    const fields = this.state.fields;
    fields[field] = e.target.value;
    this.setState({ fields });
    this.validate();
  }
 
  handleSelectedOrg = option => {
    const fields = this.state.fields;
    fields.OrganizationName = option.value;
    this.setState(
      () => ({ fields: fields, selectedOrg: option }),
      () => {
        this.getLocatorTypeData(this.state.selectedOrg.value);
      },
    );
    this.validate();
  };
 
  handleSelectedLocatorType = option => {
    const fields: any = this.state.fields;
    fields.LocatorTypeGUID = option.value;
    this.setState({ fields: fields, selectedLocatorType: option });
    this.validate();
  };
 
  async getLocatorTypeData(orgName) {
    let data = await getListOfLocatorTypes({ OrganizationName: orgName });
    let currentType = null;
    const errors: any = this.state.errors;
    if (this.state.selectedLocatorType !== null) {
      const value = data.find(d => {
        return d.LocatorTypeGUID === this.props.currentSelection.LocatorTypeGUID;
      });
      if (value !== null && value !== undefined) {
        currentType = { value: value.LocatorTypeGUID, label: value.LocatorTypeName };
      } else {
        errors.LocatorTypeGUID = this.props.translatedFields.PrevDeletedLocType;
      }
    }
 
    if (!data) {
      data = [];
    }
 
    this.setState(() => ({
      availableTypes: data.map(d => {
        return { label: d.LocatorTypeName, value: d.LocatorTypeGUID, imageGUID: d.LocatorTypeImagePoolGUID };
      }),
      selectedLocatorType: currentType,
      errors: errors,
    }));
  }
 
  async componentDidMount() {
    this.getLocatorTypeData(currentOrg);
    const fields: any = this.state.fields;
    if (this.props.currentSelection !== null) {
      fields.LocatorID = this.props.currentSelection.LocatorID;
      fields.OrganizationName = this.props.currentSelection.OrganizationName;
      fields.LocatorTypeGUID = this.props.currentSelection.LocatorTypeGUID;
      fields.LocatorName = this.props.currentSelection.LocatorName;
      fields.LocatorSerialNumber = this.props.currentSelection.LocatorSerialNum;
      this.setState({
        fields: fields,
        selectedOrg: { value: fields.OrganizationName, label: fields.OrganizationName },
        selectedLocatorType: { value: fields.LocatorTypeGUID, label: "" },
        isEdit: true,
        isMobile: this.props.currentSelection.IsAutoGen,
        validationGood: true,
      });
    }
  }
 
  toggle = () => {
    this.props.closeModalfn();
  };
 
  render() {
    const fields: any = this.state.fields;
    const errors: any = this.state.fields;
    return (
      <div>
        <Modal isOpen={true} toggle={this.toggle}>
          <ModalHeader>{this.props.translatedFields.RegisterNewLocator}</ModalHeader>
          <ModalBody>
            <Form onSubmit={this.handleCreateLocator}>
              <Row>
                <Col>
                  <FormGroup>
                    <Label>
                      <b>{this.props.translatedFields.LocatorName}</b>
                      <span style={{ color: "red" }}>*</span>
                    </Label>
                    <Input
                      type="text"
                      name="LocatorName"
                      value={fields.LocatorName}
                      onChange={this.handleChange.bind(this, "LocatorName") as any}
                    />
                    <span style={{ color: "red" }}>
                      <b>{errors.LocatorName}</b>
                    </span>
                  </FormGroup>
                </Col>
              </Row>
              {IsSysAdmin && !this.state.isMobile ? (
                <Row>
                  <Col>
                    <FormGroup>
                      <Label for="OrgSelector">
                        <b>{this.props.translatedFields.OrganizationColumn}</b>
                        <span style={{ color: "red" }}>*</span>
                      </Label>
                      <OrgSelector changeFn={this.handleSelectedOrg} selectedOption={this.state.selectedOrg} />
                      <span style={{ color: "red" }}>
                        <b>{errors.OrganizationName}</b>
                      </span>
                    </FormGroup>
                  </Col>
                </Row>
              ) : null}
              {!this.state.isMobile ? (
                <Row>
                  <Col>
                    <FormGroup>
                      <Label for="LocatorTypeSelector">
                        <b>{this.props.translatedFields.LocatorType}</b>
                        <span style={{ color: "red" }}>*</span>
                      </Label>
                      <Select
                        id="LocatorTypeSelector"
                        options={this.state.availableTypes}
                        value={this.state.selectedLocatorType}
                        onChange={this.handleSelectedLocatorType as any}
                        getOptionLabel={getOptionLabel}
                      />
                      <span style={{ color: "red" }}>
                        <b>{errors.LocatorTypeGUID}</b>
                      </span>
                    </FormGroup>
                  </Col>
                </Row>
              ) : null}
              {!this.state.isMobile ? (
                <Row>
                  <Col>
                    <FormGroup>
                      <Label>
                        <b>{this.props.translatedFields.SerialNumber}</b>
                        <span style={{ color: "red" }}>*</span>
                      </Label>
                      <Input
                        type="text"
                        name="LocatorSerialNumber"
                        value={fields.LocatorSerialNumber}
                        onChange={this.handleChange.bind(this, "LocatorSerialNumber") as any}
                        disabled={this.state.isEdit}
                      />
                      <span style={{ color: "red" }}>
                        <b>{errors.LocatorSerialNumber}</b>
                      </span>
                    </FormGroup>
                  </Col>
                </Row>
              ) : null}
              <Row>
                <Col>
                  <ButtonGroup>
                    <Button disabled={!this.state.validationGood} type="submit" name="Submit" color="primary">
                      {this.props.translatedFields.SubmitButton}
                    </Button>
                    <Button onClick={this.toggle} color="warning">
                      {this.props.translatedFields.CancelText}
                    </Button>
                  </ButtonGroup>
                </Col>
              </Row>
            </Form>
          </ModalBody>
        </Modal>
      </div>
    );
  }
}
 
export default LocatorCreateEditModal;
