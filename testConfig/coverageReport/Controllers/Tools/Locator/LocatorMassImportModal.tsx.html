import React, { Component } from "react";
import { massImportLocatorsFromFile } from "Api/LocatorApi";
import Modal from "reactstrap/lib/Modal";
import ModalFooter from "reactstrap/lib/ModalFooter";
import ModalHeader from "reactstrap/lib/ModalHeader";
import Button from "reactstrap/lib/Button";
import ModalBody from "reactstrap/lib/ModalBody";
import { FilePond, File, registerPlugin } from "react-filepond";
import "filepond/dist/filepond.min.css";
import "../../../../Content/ExtraStyles/Locator.scss";
import LoadSpinner from "../Map/Components/LoadSpinner";
 
type Props = {
  translatedFields?: any;
  buttonColor?: string;
  closeModalfn?: Function;
};
 
class LocatorMassImportModal extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    translatedFields: {},
    buttonColor: "primary",
    closeModalfn: Function.prototype,
  };
 
  state = {
    selectedFile: null,
    processing: false,
    errorMessage: "",
    prevUploadFailed: false,
  };
 
  toggle = () => {
    this.props.closeModalfn();
  };
 
  upload = async () => {
    const { selectedFile } = this.state;
    let errorMessage = "";
 
    const formData = new FormData();
    formData.append("filename", selectedFile.name);
    formData.append("file", selectedFile);
 
    this.setState({ processing: true });
 
    const response = await massImportLocatorsFromFile(formData);
 
    if (response) {
      alert(this.props.translatedFields.ImportSuccess);
      // The actual download
      const blob = new Blob([response], { type: "application/vnd.ms-excel" });
      const link = document.createElement("a");
      link.href = window.URL.createObjectURL(blob);
      link.download = "MassImportResults.xlsx";
 
      document.body.appendChild(link);
 
      link.click();
 
      document.body.removeChild(link);
 
      this.toggle();
    } else {
      alert(this.props.translatedFields.ImportFailure);
      errorMessage = this.props.translatedFields.ImportFailure;
    }
    this.setState({ processing: false, errorMessage: errorMessage });
  };
 
  render() {
    return (
      <Modal isOpen={true}>
        <ModalHeader>{this.props.translatedFields.LocatorFileImport}</ModalHeader>
        <ModalBody>
          {this.state.processing ? (
            <LoadSpinner />
          ) : (
            <FilePond
              allowMultiple={false}
              onupdatefiles={fileItems => {
                this.setState({
                  selectedFile: fileItems[0].file,
                });
              }}
            >
              {this.state.selectedFile ? (
                <File key={this.state.selectedFile} src={this.state.selectedFile} origin="local" />
              ) : null}
            </FilePond>
          )}
          {this.state.prevUploadFailed ? (
            <span style={{ color: "red" }}>
              <b>{this.state.errorMessage}</b>
            </span>
          ) : null}
        </ModalBody>
        <ModalFooter>
          <Button color="primary" onClick={this.upload} disabled={this.state.processing}>
            {this.props.translatedFields.Upload}
          </Button>{" "}
          <Button color="warning" onClick={this.toggle}>
            {this.props.translatedFields.CancelText}
          </Button>
        </ModalFooter>
      </Modal>
    );
  }
}
 
export default LocatorMassImportModal;
