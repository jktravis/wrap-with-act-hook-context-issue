import React, { Component } from "react";
import {
  getListOfLocators,
  addActiveLocatorToProject,
  removeActiveLocatorFromProject,
  getListOfActiveLocatorsForProject,
} from "Api/LocatorApi";
import { currentOrg } from "Api/constants";
import { CardBody, Card, CardHeader } from "reactstrap";
import "../../../../Content/ExtraStyles/Locator.scss";
import Row from "reactstrap/lib/Row";
import Col from "reactstrap/lib/Col";
import { FaHeartbeat } from "react-icons/fa";
import ReactCSSTransitionGroup from "react-addons-css-transition-group";
 
const FIVE_MIN = 5 * 60 * 1000; //In ms
 
type Props = {
  translatedFields?: any;
  project?: string;
};
 
class LocatorIncidentListPage extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    translatedFields: {},
    project: "",
  };
 
  state = {
    selectedOption: { label: currentOrg, value: currentOrg },
    tableData: [],
    selectedTableData: [],
  };
 
  constructor(props) {
    super(props);
 
    this.moveToSelectedList = this.moveToSelectedList.bind(this);
    this.moveToAvailableList = this.moveToAvailableList.bind(this);
  }
 
  async getTableData(orgName) {
    let data = await getListOfLocators({ OrganizationName: orgName });
    let selectedLocators = [];
    if (this.props.project) {
      selectedLocators = await getListOfActiveLocatorsForProject({ id: this.props.project });
 
      //This for loop removes all Selected locators, IE Active locators in an incident
      //From the data grid, so that active locators do not appear in the Inactive table
      if (selectedLocators && selectedLocators.length > 0) {
        data = data.filter(function(value, index, err) {
          for (let k = 0; k < selectedLocators.length; k++) {
            if (selectedLocators[k].LocatorSerialNum === value.LocatorSerialNum) {
              return false;
            }
          }
          return true;
        });
      }
    }
 
    this.setState(() => ({
      tableData: data,
      selectedTableData: selectedLocators,
    }));
  }
 
  async componentDidMount() {
    this.getTableData(currentOrg);
  }
 
  async moveToSelectedList(rowVal) {
    this.setState(({ tableData, selectedTableData }) => {
      const selectedItem = tableData.find(function(value, index, err) {
        return value.LocatorSerialNum === rowVal;
      });
 
      if (this.props.project) {
        const result = addActiveLocatorToProject(selectedItem.LocatorSerialNum, this.props.project).then(response => {
          return true;
        });
      }
 
      tableData = tableData.filter(function(value, index, err) {
        return value.LocatorSerialNum !== rowVal;
      });
      selectedTableData.push(selectedItem);
      selectedTableData.sort(function(a, b) {
        if (a.LocatorName > b.LocatorName) {
          return 1;
        } else {
          return -1;
        }
      });
 
      tableData.sort(function(a, b) {
        if (a.LocatorName > b.LocatorName) {
          return 1;
        } else {
          return -1;
        }
      });
 
      return {
        tableData: tableData,
        selectedTableData: selectedTableData,
      };
    });
  }
 
  async moveToAvailableList(rowVal) {
    this.setState(({ tableData, selectedTableData }) => {
      const selectedItem = selectedTableData.find(function(value, index, err) {
        return value.LocatorSerialNum === rowVal;
      });
 
      if (this.props.project) {
        const result = removeActiveLocatorFromProject(selectedItem.LocatorSerialNum, this.props.project).then(
          response => {
            return true;
          },
        );
      }
 
      selectedTableData = selectedTableData.filter(function(value, index, err) {
        return value.LocatorSerialNum !== rowVal;
      });
      tableData.push(selectedItem);
      tableData.sort(function(a, b) {
        if (a.LocatorName > b.LocatorName) {
          return 1;
        } else {
          return -1;
        }
      });
 
      selectedTableData.sort(function(a, b) {
        if (a.LocatorName > b.LocatorName) {
          return 1;
        } else {
          return -1;
        }
      });
 
      return {
        tableData: tableData,
        selectedTableData: selectedTableData,
      };
    });
  }
 
  /**
   *  If current time minus the time of the reading is less than 5 minutes, return true.
   */
  validateMostRecentReading(time) {
    return !time && Date.now() - new Date(time).getUTCMilliseconds() < FIVE_MIN;
  }

  render() {
    const { translatedFields, project } = this.props;
    const { tableData, selectedTableData } = this.state;
 
    //Saved for future use of the DateRangePicker
    //   <DateRangePicker
    //   locale={{ weekLabel: row.LocatorSerialNum }}
    //   onApply={this.handleDateRangePickerEvent as any}
    //   startDate={row.StartReadingsDate ? new Date(row.StartReadingsDate) : new Date()}
    //   endDate={row.EndReadingsDate ? new Date(row.EndReadingsDate) : new Date()}
    // >
    //   <i className="fas fa-calendar-alt" />
    // </DateRangePicker>
    let availableElementsList = null;
    if (tableData && tableData.length > 0) {
      availableElementsList = tableData.map((item, index) => {
        return (
          <Row key={item.LocatorSerialNum} style={{ position: "relative", marginBottom: "10px" }}>
            <Col xs="1">
              <input
                type="checkbox"
                defaultChecked={false}
                onChange={() => this.moveToSelectedList(item.LocatorSerialNum)}
                style={{ position: "absolute", top: "35%" }}
              />
            </Col>
            <Col xs="10">
              <b>{translatedFields.LocatorName}:</b> {item.LocatorName}
              <br />
              <b>{translatedFields.SerialNumber}:</b> {item.LocatorSerialNum}
            </Col>
            <Col xs="1">
              {this.validateMostRecentReading(item.TimeOfReading) ? (
                <FaHeartbeat style={{ color: "green", position: "absolute", top: "40%" }} />
              ) : (
                <FaHeartbeat style={{ color: "red", position: "absolute", top: "40%" }} />
              )}
            </Col>
          </Row>
        );
      });
    }
 
    let selectedElementsList = null;
    if (selectedTableData && selectedTableData.length > 0) {
      selectedElementsList = selectedTableData.map((item, index) => {
        return (
          <Row key={item.LocatorSerialNum} style={{ position: "relative", marginBottom: "10px" }}>
            <Col xs="1">
              <input
                type="checkbox"
                defaultChecked={true}
                onChange={() => this.moveToAvailableList(item.LocatorSerialNum)}
                style={{ position: "absolute", top: "35%" }}
              />
            </Col>
            <Col xs="10">
              <b>{translatedFields.LocatorName}:</b> {item.LocatorName}
              <br />
              <b>{translatedFields.SerialNumber}:</b> {item.LocatorSerialNum}
            </Col>
            <Col xs="1">
              {this.validateMostRecentReading(item.TimeOfReading) ? (
                <FaHeartbeat style={{ color: "green", position: "absolute", top: "40%" }} />
              ) : (
                <FaHeartbeat style={{ color: "red", position: "absolute", top: "40%" }} />
              )}
            </Col>
          </Row>
        );
      });
    }
 
    return (
      <div>
        <Row>
          <Col xs="6">
            <Card>
              <CardHeader style={{ textAlign: "center" }}>
                <b>{translatedFields.OrgLocators}</b>
              </CardHeader>
              <CardBody>
                {availableElementsList && availableElementsList.length > 0 ? (
                  <ReactCSSTransitionGroup
                    transitionName="active"
                    transitionEnterTimeout={300}
                    transitionLeaveTimeout={300}
                  >
                    {availableElementsList}
                  </ReactCSSTransitionGroup>
                ) : null}
              </CardBody>
            </Card>
          </Col>{" "}
          <Col xs="6">
            <Card>
              <CardHeader style={{ textAlign: "center" }}>
                <b>{translatedFields.SelectedLocators}</b>
              </CardHeader>
              <CardBody>
                {selectedElementsList && selectedElementsList.length > 0 ? (
                  <ReactCSSTransitionGroup
                    transitionName="active"
                    transitionEnterTimeout={300}
                    transitionLeaveTimeout={300}
                  >
                    {selectedElementsList}
                  </ReactCSSTransitionGroup>
                ) : null}
              </CardBody>
            </Card>
          </Col>
        </Row>
      </div>
    );
  }
}
