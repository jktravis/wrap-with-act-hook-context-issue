ï»¿import React, { Component } from "react";
import { FaBolt } from "react-icons/fa";
 
interface Props {
  batteryLevel: number;
  batteryMaxValue?: number;
  maxWidth?: number;
}
 
class BatteryBar extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    batteryLevel: 0,
    batteryMaxValue: 100,
    maxWidth: 100,
  };
 
  render() {
    const { batteryLevel, batteryMaxValue, maxWidth } = this.props;
    return (
      <div
        style={{
          width: `${maxWidth}`,
          height: "80%",
          backgroundColor: "#dadada",
          borderRadius: "2px",
          position: "relative"
        }}
      >
        {(batteryLevel / batteryMaxValue) * 100 > 1 ? (
          <div>
            <div
              style={{
                width: `${(batteryLevel / batteryMaxValue) * 100}%`,
                height: "100%",
                backgroundColor:
                  batteryLevel / batteryMaxValue > 60 / 100
                    ? "#85cc00"
                    : batteryLevel / batteryMaxValue > 20 / 100
                    ? "#ffbf00"
                    : "#ff2e00",
                borderRadius: "2px",
                transition: "all .2s ease-out",
                position: "absolute"
              }}
            />
            <div style={{ position: "absolute", left: "35%" }}>
              <b>{(batteryLevel / batteryMaxValue) * 100 > 100 ? 100 : (batteryLevel / batteryMaxValue) * 100}%</b>
            </div>
          </div>
        ) : (
          <div>
            <div
              style={{
                width: `100%`,
                height: "100%",
                backgroundColor: "#000000",
                borderRadius: "2px",
                transition: "all .2s ease-out",
                position: "absolute"
              }}>
              <FaBolt style={{ position: "absolute", color: "white", left: "45%", top: "20%" }} />
            </div>
          </div>
        )}
      </div>
    );
  }
}
 
export default BatteryBar;
