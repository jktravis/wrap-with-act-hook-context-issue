  export function validateForm(): boolean {
    const errResponse = {
      Messages: [],
    };
    const nameVal = $("#enValue").val();
    if (!(nameVal.length > 0)) {
      Main.markObjectAsError($("#enValue"));
      errResponse.Messages.push(_globalPrompts.MissingNameMessage);
    } else {
      Main.clearObjectErrorStatus($("#enValue"));
    }
 
    if (errResponse.Messages.length > 0) {
      Main.handleValErr(null, errResponse);
      return false;
    } else {
      return true;
    }
  }
 
  function loadLevelGrid() {
    if ($("#lvlGrid").length > 0) {
      const levelsAsJSON = $("#DataStructure").val();
      if (levelsAsJSON.length > 0) {
        levels = JSON.parse(levelsAsJSON);
      }
      levels.sort(function(a, b) {
        return a.GridIndex > b.GridIndex ? 1 : b.GridIndex > a.GridIndex ? -1 : 0;
      });
      const dataSource = new kendo.data.DataSource({
        transport: {
          read: function(e) {
            e.success(levels);
          },
        },
        batch: false,
        schema: {
          model: {
            id: "ID",
            fields: {
              ID: { type: "number", editable: false },
              Description: { type: "string", editable: true },
              frDescription: { type: "string", editable: true },
              deDescription: { type: "string", editable: true },
              nlDescription: { type: "string", editable: true },

      grid.setDataSource(dataSource);
      reloadLevels();
    }

  function getIndexById(id) {

  export function saveLevel() {
    const txtLevel = $("#txtLevelText").val();
    const existingID = $("#ExistingLevelID").val();
    if (
      txtLevel.length > 0 &&
      $("#lvlIcon")
        .data("kendoDropDownList")
        .value() !== ""
    ) {
      if (existingID) {
        //Update existing object
        const index = getIndexById(Number(existingID));
        levels[index].LevelText = txtLevel;
        levels[index].frText = $("#txtLevelTextFR").val();
        levels[index].nlText = $("#txtLevelTextNL").val();
        levels[index].deText = $("#txtLevelTextDE").val();
        levels[index].Description = $("#txtDescription").val();
        levels[index].frDescription = $("#txtDescriptionFR").val();
        levels[index].nlDescription = $("#txtDescriptionNL").val();
        levels[index].deDescription = $("#txtDescriptionDE").val();
        levels[index].RiskImagePoolGUID = $("#lvlIcon")
          .data("kendoDropDownList")
          .value();
        levels[index].GridIndex = index;
      } else {
        //create object
        const obj: Level = new Level();
        obj.LevelText = txtLevel;
        obj.frText = $("#txtLevelTextFR").val();
        obj.nlText = $("#txtLevelTextNL").val();
        obj.deText = $("#txtLevelTextDE").val();
        obj.Description = $("#txtDescription").val();
        obj.frDescription = $("#txtDescriptionFR").val();
        obj.nlDescription = $("#txtDescriptionNL").val();
        obj.deDescription = $("#txtDescriptionDE").val();
        //obj.MitigationText = $("#txtMitigationText").val();
        obj.RiskImagePoolGUID = $("#lvlIcon")
          .data("kendoDropDownList")
          .value();
        //add to shelterChar array
        if (levels !== null && levels !== undefined) {
          obj.ID = levels.length + 1;
          obj.GridIndex = levels.length;
        } else {
          obj.ID = 1;
          obj.GridIndex = 0;
        }
        levels.push(obj);
      }
 
      reloadLevels();
      $("#LevelModal").modal("hide");
    } else {
      $.notify({ message: _globalPrompts.RiskSelectIcon }, { type: "danger", z_index: 3000 });
    }
  }
 
  function moveRiskLevel(dataItem, newIndex, oldIndex, grid) {
    if (newIndex !== oldIndex) {
      levels[oldIndex].GridIndex = newIndex;
      levels[newIndex].GridIndex = oldIndex;
 
      const temp = levels[newIndex];
      levels[newIndex] = levels[oldIndex];
      levels[oldIndex] = temp;
 
      grid.dataSource.remove(dataItem);
      grid.dataSource.insert(newIndex, dataItem);
    }
  }
 
  export function moveLevelUp(id) {
    const grid = $("#lvlGrid").data("kendoGrid");
    const dataItem = grid.dataSource.getByUid(id);
    const index = grid.dataSource.indexOf(dataItem);
    const newIndex = Math.max(0, index - 1);
 
    moveRiskLevel(dataItem, newIndex, index, grid);
  }
  export function moveLevelDown(id) {
    const grid = $("#lvlGrid").data("kendoGrid");
    const dataItem = grid.dataSource.getByUid(id);
    const index = grid.dataSource.indexOf(dataItem);
    const length = grid.dataSource.total();
    const newIndex = Math.min(length - 1, index + 1);
 
    moveRiskLevel(dataItem, newIndex, index, grid);
  }

  export function saveLevel(riskGUID) {
    //$.notify({ message: "not yet implemented" }, { type: 'danger' });
    const dropDown = $("#drpSelLevel").data("kendoDropDownList");
    const selItem = dropDown.dataItem();
    $Ajax.ajaxAntiForgery({
      url: _globalPrompts.saveLevelURL,
      type: "POST",
      data: {
        riskGUID: riskGUID,
        levelText: selItem.LevelText,
      },
      success: function(response) {
        if (response.success === false) {
          $.notify({ message: response.Message }, { type: "danger" });
        } else {
          const gridDataSource = $("#lvlGrid").data("kendoGrid").dataSource;
          gridDataSource.read();
          if (selItem !== undefined) {
            $("#curLevelDesc").text(selItem.Description);

            $("#curIcon").attr("alt", selItem.LevelText);

        Main.handlejqerr(xhr, status, error);
      },
    });
  }
