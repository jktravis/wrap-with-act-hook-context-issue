  export function BuildIAPInstanceList(workspaceDiv) {
    if (workspaceDiv === null || !workspaceDiv) {
      workspaceDiv = $("#workspaceDiv");
    }
    let instanceList = workspaceDiv
      .find("li")
      .map(function() {
        let templateGUID = $(this).data("templateguid");
        if (templateGUID === undefined || templateGUID.length === 0) {
          //no templateGUID!
          templateGUID = null;
        }
        return {
          ObjectType: $(this).data("objecttype"),
          TemplateGUID: $(this).data("templateguid"),
          Id: $(this).data("id"),
        };
      })
      .get();
    return instanceList;
  }
