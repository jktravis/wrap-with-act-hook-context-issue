  function formDropped($item, workspaceDiv) {
    if ($item.parent().attr("id") !== "IAPList") {
      $item.fadeOut(function() {
        //var $workspaceDiv = $('#workspaceDiv');
        let $list = $("ul", workspaceDiv).length
          ? $("ul", workspaceDiv)
          : $("<ul class='gallery ui-helper-reset'/>").appendTo(workspaceDiv);
        let clone = $item.clone();
        let id = clone.attr("id");
        clone.attr("id", id + "IAP");
        setExclamationPoint(clone, $("#infoDiv").data("opperiodguid"));
        let $removeButton = $("<button>", { title: "Remove From IAP", class: "removeInstanceButton " });
        let $span = $("<span>", { class: "" }); //used to be fa-layers fa-fw.  doesn't exist anymore
        $span.append($("<i>", { class: "fas fa-trash-alt" })); //TODO: tom had a fa-2x.  had to remove it, but now it's a data-fa-transform="grow-4" if you want to add it back
        $removeButton.append($span);
        if (!(currentIAP.isLocked === "False")) {
          $removeButton.prop("disabled", true);
        }
        let btnGroup = clone.children(".row:first").children(".btn-group:first");
        if (btnGroup !== undefined && btnGroup.length > 0) {
          btnGroup.append($removeButton);
        } else {
          clone.append($removeButton);
        }
        clone.removeClass("no-child-items").end();
        //.append($removeButton)
        clone.appendTo($list).fadeIn();
        afterDropBinding();
      });
    } else {
      //reorder?
    }
  }

  function setExclamationPoint(elem, curOpPeriodGUID) {
    //initially remove any that are there
    let item;
 
    if (elem instanceof jQuery) {
      item = elem;
    } else {
      item = $("#" + elem.id);
    }
 
    item.find("span.fa-exclamation").remove();
 
    if (curOpPeriodGUID) {
      //compare against current op period, if different add exclamation point
 
      if (item.data("parent") !== curOpPeriodGUID) {
        let firstDivGroup = item.children(".col-md-6:first");
        firstDivGroup.prepend(
          $("<span>", {
            title: "This form is in a different operational period.",
            class: "fa fa-exclamation text-danger",
          }),
        );
      }
    }
  }

  function removeListItemFromIAP(li, id) {
    li.fadeOut(function() {
      li.find("button.fa-trash-alt")
        .remove()
        .find("i.fa-exclamation")
        .remove()
        .end();
    });
    //var $list = $("ul", '#workspaceDiv');
    //$list.find(id).remove();
    id = id.substring(0, id.length - 3);
    if (li.data("objecttype") === "Form") {
      let parentID = li.data("parent");
      $("#" + parentID)
        .find($("#" + id))
        .fadeIn();
    } else {
      $("#AvailableAttachmentsDiv")
        .find($("#" + id))
        .fadeIn();
    }
    li.empty();
    li.remove();
  }

  }

  function BuildReport(formList) {
    // clear out div that holds report
    let $reportDiv = $("#reportDiv");
    $reportDiv.html("");
    $("#instanceHistoryDiv").html("");
 
    if (!formList) {
      return;
    }
    let cantProcessList = [];
    let parsed = JSON.parse(formList);
    let iCount = 0;
    $.each(parsed, function() {
      let $div;
      if (iCount > 0) {
        let $break = $("<div>", { class: "page-break" });
        $reportDiv.append($break);
      }
      if (this.Type === "Form") {
        $div = $("<div>", { id: this.InstanceGUID, class: "page-break" });
        //add each form
        $div.append(this.FormTemplate);
        FormBase.HydrateHTMLWithInputValues(this.FormData, $div, false);
        FormBase.ReplaceInputWithSpan($div);
        $reportDiv.append($div);
        //need sometthing here to resize textboxes in forms, like we do with resizeAllTextAreas in forms.ts
      } else if (this.Type === "Image") {
        //handle attachment?
        $div = $("<div>", { id: this.InstanceGUID, class: "page-break" });
        $div.append("<h4>" + this.text + "</h4>");
        $div.append($("<img>", { src: this.url, class: "col-md-12 img-fluid" }));
        $reportDiv.append($div);
      } else if (this.Type === "Text") {
        $div = $("<div>", { id: this.InstanceGUID, class: "page-break" });
        $div.append("<h4>" + this.text + "</h4>");
        $div.append("<p>" + this.data + "</p>");
        $reportDiv.append($div);
      } else {
        //general attachment add it at the end
        cantProcessList.push(this);
      }
      iCount++;
    });
 
    $reportDiv.append(buildAttachmentList(cantProcessList));
    $reportDiv.removeClass("hidden");
    Main.resizeAllTextAreas();
    (<any>$reportDiv).print({
      addGlobalStyles: true,
      stylesheet: null,
      rejectWindow: true,
      noPrintSelector: ".no-print",
      iframe: true,
      append: null,
      prepend: null,
    });
    $reportDiv.addClass("hidden");
  }

  function changeActiveIAP(newIapGUID) {

  function changeIAPState(iapGUID, currentState) {
    let url = iapVars.setIAPStateURL;
    let state = "locked";
    if (currentState && currentState !== "Draft") {
      state = "unlocked";
    }
    //show confirmation dialog!
 
    $Ajax.ajaxAntiForgery({
      url: url,
      type: "POST",
 
      data: {
        iapGUID: iapGUID,
        state: state,
      },
      success: function(result) {
        if (result && result.success) {
          refreshDetailsPanel(result.iapGUID);
        } else {
          $.notify({ message: result.Message }, { type: "danger" });
        }
        //need to refresh the list of IAPs
        //probably should make this IAP active
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
