  export function sendModalChat(maxMessLen, baseurl) {
    const typVal = $("#SelectedRoomModal").val();
    const msg = $("#EventDescriptionModal").val();
    const linkedLog = $("#LinkedLogEntriesAsString").val();
    let latval;
    if ($("#Lat")) {
      latval = $("#Lat").val();
    }
    let lonval;
    if ($("#Lon")) {
      lonval = $("#Lon").val();
    }
    if (msg !== null && msg.length > 0) {
      if (msg.length > maxMessLen) {
        $.notify(
          {
            message:
              "Message length cannot be longer than " +
              maxMessLen +
              " characters.  Your current message length is " +
              msg.length +
              ".",
          },
          { type: "danger" },
        );
        return;
      }
      const url = baseurl; //+ "?SelectedRoom=" + typVal + "&MessageText=" + msg;
 
      $Ajax.ajaxAntiForgery({
        url: url,
        type: "POST",
        data: {
          Lat: latval,
          Lon: lonval,
          SelectedRoom: typVal,
          MessageText: msg,
          LinkedLogEntriesAsString: linkedLog,
        },
        success: function(response) {
          if (response !== null) {
            Dashboard.jqueryclosemodal();
            //add refresh to chat window here if you like
            (<IChatScope>angular.element("#chatControllerIDM").scope()).shouldResetChatPositionM = true;
            //refreshAllchat();
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }
