  export function roomselectionsChanged(preURL) {
    const multiselectObj = $("#SelectedRooms").data("kendoMultiSelect");
    if (multiselectObj !== null) {
      const multiselectVal = multiselectObj.value();
      if (multiselectVal !== null) {
        Main.blockit($("#divRoomSection"));
        $("#btnSend").prop("disabled", false);
        const last24Val = Chat_Multiple.checkboxIsChecked();
        const theurl = preURL + "?last24Val=" + last24Val + "&includeBaseWide=false";
        $Ajax.ajaxAntiForgery({
          url: theurl,
          type: "POST",
          data: {
            SelectedRooms: multiselectVal,
          },
          success: function(response) {
            if (response !== null) {
              $.each(multiselectVal, function(index, value) {
                Chat_Multiple.elshouldScroll.push(value);
              });
              $("#divRoomSection").html(response);
              Main.unblockit($("#divRoomSection"));
              return;
            }
          },
          error: function(xhr, status, error) {
            Main.handlejqerr(xhr, status, error);
          },
        });
      }
    }
    $("#btnSend").prop("disabled", true);
  }

  export function SendMessage(maxMessLen, preUrl, selectedRoom) {
    const msg = $("#msg" + selectedRoom).val();
    if (msg !== null && msg.length > 0) {
      if (msg.length > maxMessLen) {
        $.notify(
          {
            message:
              "Message length cannot be longer than " +
              maxMessLen +
              " characters.  Your current message length is " +
              msg.length +
              ".",
          },
          { type: "danger" },
        );
        return;
      }
      const url = preUrl + "?SelectedRoom=" + selectedRoom;
 
      $Ajax.ajaxAntiForgery({
        url: url,
        async: true,
        type: "POST",
        data: {
          MessageText: msg,
        },
        success: function(response) {
          if (response !== null) {
            elshouldScroll.push(selectedRoom);
          }
        },
        error: function(xhr, status, error) {
          //Main.handlejqerr(xhr, status, error);
 
          $.notify({ message: "Could not send message.  Have you lost your internet connection?" }, { type: "danger" });
        },
      });
      $("#msg" + selectedRoom).val("");
      $("#msg" + selectedRoom).focus();
    }
  }
