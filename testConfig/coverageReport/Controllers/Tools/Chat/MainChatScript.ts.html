  export function refreshChatListView() {
    if (shouldScroll !== true) {
      //only bother with this check if you're not going to automatically scroll down anyways
      const curhgt = $("#lstMessages")[0].scrollTop;
      const scrolledtobottom = curhgt + $("#lstMessages").height() === $("#lstMessages")[0].scrollHeight;
      if (scrolledtobottom === true) {
        shouldScroll = true;
      }
    }
    $("#lstMessages")
      .data("kendoListView")
      .dataSource.read();
  }
  function checkPrintButton() {
    if ($("#btnPrint").length > 0) {
      const total = $("#lstMessages")
        .data("kendoListView")
        .dataSource.total();
      if (total > 0) {
        $("#btnPrint").prop("disabled", false);
      } else {
        $("#btnPrint").prop("disabled", true);
      }
    }
  }
  function meanscrollbottom() {
    $("#lstMessages").prop({ scrollTop: $("#lstMessages").prop("scrollHeight") - $("#lstMessages").height() });
  }
  function nicescrollbottom() {
    //http://stackoverflow.com/questions/6850401/maintain-scroll-position-for-a-div-based-on-scroll-bar-position
    $("#lstMessages").animate({ scrollTop: $("#lstMessages").prop("scrollHeight") - $("#lstMessages").height() }, 3000);
  }
  let shouldScroll;
  function chatdatabound(e) {
    if (shouldScroll === true) {
      shouldScroll = false;
      meanscrollbottom();
    }
    Main.qtipSection("lstMessages");
    const myTextEl = document.getElementById("lstMessages");
    myTextEl.innerHTML = link(myTextEl.innerHTML);
    checkPrintButton();
  }
  export function SendMessage(maxMessLen, preUrl) {
    const typVal = $("#SelectedRoom").val();
    if (typVal === null) {
      return;
    }
    const msg = $("#MessageToSend").val();
    if (msg !== null && msg.length > 0) {
      if (msg.length > maxMessLen) {
        $.notify(
          {
            message:
              "Message length cannot be longer than " +
              maxMessLen +
              " characters.  Your current message length is " +
              msg.length +
              ".",
          },
          { type: "danger" },
        );
        return;
      }
      const url = preUrl + "?SelectedRoom=" + typVal;
 
      $Ajax.ajaxAntiForgery({
        url: url,
        type: "POST",
        data: {
          MessageText: msg,
        },
        success: function(response) {
          if (response !== null) {
            //refreshChatListView();    don't need this cause of signalR
            shouldScroll = true;
          }
        },
        error: function(xhr, status, error) {
          //Main.handlejqerr(xhr, status, error);
          $.notify({ message: "Could not send message.  Have you lost your internet connection?" }, { type: "danger" });
        },
      });
      $("#MessageToSend").val("");
      $("#MessageToSend").focus();
    }
  }
  export function roomchanged(preURL) {
    const typVal = $("#SelectedRoom").val();
    if (typVal === null) {
      $("#btnSend").prop("disabled", true);
    } else {
      $("#btnSend").prop("disabled", false);
      const last24Val = checkboxIsChecked();
      const theurl = preURL + "?roomGUID=" + typVal + "&last24=" + last24Val + "&includeBaseWide=true";
      const dataSource = new kendo.data.DataSource({
        transport: {
          read: {
            url: theurl,
            dataType: "json",
          },
        },
        schema: {
          model: {
            id: "LogEntryGUID",
            fields: {
              LogEntryGUID: { editable: false },
              StaticGUID: { editable: false },
              FriendlyUserName: { editable: false },
              FormattedDateCreated: { type: "date" },
              EventDescription: { editable: false },
            },
          },
        },
      });
      const listView = $("#lstMessages").data("kendoListView");
      listView.setDataSource(dataSource);
      shouldScroll = true;
      //show in the header what room is selected
      if ($("#currentRoom").length > 0) {
        const textvalues: string[] = new Array<string>();
        $("#SelectedRoom :selected").each(function(i, selected) {
          textvalues[i] = $(selected).text();
        });
        $("#currentRoom").text(textvalues.join(", "));
      }
    }
  }
