  export function addSelectedToIncident(urlToPost) {
    const dsCO = $("#lsttweets").data("kendoListView");
    const data = dsCO.dataSource.view(),
      selected = $.map(dsCO.select(), function(item) {
        return data[$(item).index()].StatusID;
      });
    if (selected.length > 0) {
      $Ajax.ajaxAntiForgery({
        url: urlToPost,
        type: "POST",
        data: {
          twitIDs: selected,
        },
        success: function(response) {
          $("#lblResp").html("Tweets added at " + new Date($.now()));
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }
  function onChange(arg) {
    const selected = $.map(this.select(), function(item) {
      return $(item);
    });
    if (selected.length === 0 || selected.length > 1) {
      $("#otherdiv").html("");
    } else {
      const dsCO = $("#lsttweets").data("kendoListView");
      const index = dsCO.select().index(),
        dataItem = dsCO.dataSource.view()[index];
      const usrid = dataItem.User.UserIDResponse;
      $.ajax({
        url: _globalPrompts.GetUserInfoPanelUrl,
        type: "GET",
        data: {
          UserID: usrid,
        },
        success: function(response) {
          $("#otherdiv").html(response);
          const myTextEl = document.getElementById("otherdiv");
          myTextEl.innerHTML = Autolinker.link(myTextEl.innerHTML);
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
 
    //kendoConsole.log("Selected: " + selected.length + " item(s), [" + selected.join(", ") + "]");
  }
