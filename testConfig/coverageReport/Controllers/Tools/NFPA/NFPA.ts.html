namespace NFPA {
  let _getNFPAUrl;
  export function init(toolname, getNFPAUrl) {
    // Change the MouseCoordinate display projection.
    _getNFPAUrl = getNFPAUrl;
    drawShape();
    //$(window).resize();
    if (Main.currentProjectGUID.length > 0) {
      Main.cbraPush.client.logentryForToolUpdated = refreshHistGrid;
      $.connection.hub.start().done(function() {
        Main.cbraPush.server.watchTool(toolname, Main.currentProjectGUID);
      });
    }
  }
 
  function refreshHistGrid() {
    if ($("#hstgrid").length > 0) {
      $("#hstgrid")
        .data("kendoGrid")
        .dataSource.read();
      Main.notifySomethingHasChanged();
    }
  }
  export function SpecChanged() {
    var healthval = $("#Health").val();
    var flameval = $("#Flame").val();
    var reactval = $("#React").val();
    var specval = $("#Special").val();
    var urlToSend =
      _getNFPAUrl +
      "?theNFPAType=Special&HealthVal=" +
      healthval +
      "&ReactVal=" +
      reactval +
      "&FlamVal=" +
      flameval +
      "&SpecVal=" +
      specval;
 
    $Ajax.ajaxAntiForgery({
      url: urlToSend,
      type: "POST",
      success: function(response) {
        if (response.success === true) {
          $("#SpecialText").val(response.data);
        } else {
          console.log(response);
        }
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }

  function drawShape() {
    var myCanvas = <HTMLCanvasElement>document.getElementById("myCanvas");
    if (myCanvas !== null) {
      var ctxRed = myCanvas.getContext("2d");
      ctxRed.beginPath();
      ctxRed.moveTo(200, 0);
      ctxRed.lineTo(300, 100);
      ctxRed.lineTo(200, 200);
      ctxRed.lineTo(100, 100);
      ctxRed.closePath();
      ctxRed.fillStyle = "red";
      ctxRed.fill();
      ctxRed.stroke();
 
      var ctxBlue = myCanvas.getContext("2d");
      ctxBlue.beginPath();
      ctxBlue.moveTo(100, 100);
      ctxBlue.lineTo(200, 200);
      ctxBlue.lineTo(100, 300);
      ctxBlue.lineTo(0, 200);
      ctxBlue.closePath();
      ctxBlue.fillStyle = "blue";
      ctxBlue.fill();
      ctxBlue.stroke();
 
      var ctxYellow = myCanvas.getContext("2d");
      ctxYellow.beginPath();
      ctxYellow.moveTo(300, 100);
      ctxYellow.lineTo(400, 200);
      ctxYellow.lineTo(300, 300);
      ctxYellow.lineTo(200, 200);
      ctxYellow.closePath();
      ctxYellow.fillStyle = "yellow";
      ctxYellow.fill();
      ctxYellow.stroke();
 
      var ctxWhite = myCanvas.getContext("2d");
      ctxWhite.beginPath();
      ctxWhite.moveTo(200, 200);
      ctxWhite.lineTo(300, 300);
      ctxWhite.lineTo(200, 400);
      ctxWhite.lineTo(100, 300);
      ctxWhite.closePath();
      ctxWhite.fillStyle = "white";
      ctxWhite.fill();
      ctxWhite.stroke();
    }
  }
