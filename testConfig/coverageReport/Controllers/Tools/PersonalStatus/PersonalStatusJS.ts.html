  export function savestatus(preurl) {
    const notetext = $("#EventDescriptionModal").val();
    let latval;
    if ($("#Lat")) {
      latval = $("#Lat").val();
    }
    let lonval;
    if ($("#Lon")) {
      lonval = $("#Lon").val();
    }
    const statcolor = $("#StatusColorModal").val();
    if (statcolor !== null && statcolor !== "") {
      $Ajax.ajaxAntiForgery({
        url: preurl,
        type: "POST",
        data: {
          Lat: latval,
          Lon: lonval,
          StatusText: notetext,
          StatusColor: statcolor,
        },
        success: function(response) {
          if (response !== null) {
            Dashboard.jqueryclosemodal();
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }
  function refreshAllpersonalstatus() {
    if (angular.element("#personalstatusControllerID").length) {
      (<IPersonalStatusScope>angular.element("#personalstatusControllerID").scope()).personalstatusDataSourceL.read();
    }
    //would like to figure out a way to make this repository reside in the dashboard controller somehow, so that i
    //only need to refresh this once.
    if (angular.element("#personalstatusControllerIDM").length) {
      (<IPersonalStatusScope>angular.element("#personalstatusControllerIDM").scope()).personalstatusDataSourceM.read();
    }
  }
 
  angular.module("cobraapp").factory("personalstatusRepository", function() {
    return {
      getDataSourceM: function(hostURL, projectGUID, configProjectGUID) {
        return new kendo.data.DataSource({
          transport: {
            read: {
              url: hostURL + "Tools/personalstatus/_GetPersonalStatusListJSON/" + projectGUID,
            },
          },
          schema: {
            model: {
              id: "CoBRAUserGUID",
              fields: {
                CoBRAUserGUID: { editable: false },
                CurrentLogGUID: { editable: false },
                FriendlyUserName: { editable: false },
                FriendlyStatus: { editable: false },
                StatusClass: { editable: false },
                StatusBorder: { editable: false },
                StatusColor: { editable: false },
                FormattedDateCreated: { type: "date", editable: false },
              },
            },
          },
        });
      },
      getDataSourceL: function(hostURL, projectGUID, configProjectGUID) {
        return new kendo.data.DataSource({
          transport: {
            read: {
              url: hostURL + "Tools/personalstatus/_GetPersonalStatusListJSON/" + projectGUID,
            },
          },
          pageSize: 10,
          schema: {
            model: {
              id: "CoBRAUserGUID",
              fields: {
                CoBRAUserGUID: { editable: false },
                CurrentLogGUID: { editable: false },
                FriendlyUserName: { editable: false },
                FriendlyStatus: { editable: false },
                StatusClass: { editable: false },
                StatusBorder: { editable: false },
                StatusColor: { editable: false },
                FormattedDateCreated: { type: "date", editable: false },
              },
            },
          },
        });
      },
    };
  });
