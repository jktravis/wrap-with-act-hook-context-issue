  export function sendnote(url: string, cb: () => void) {
    const notetext = $("#NewNoteText").val();
    if (notetext != null && notetext.length > 0) {
      Main.blockit($("#multiLogInput"));
      const logType = $("#NewLogType").val();
      let currentRoom = (<any>angular.element("#multiControllerID").scope()).selectedRoom;
      if (typeof currentRoom === "undefined") {
        currentRoom = (<any>angular.element("#multiControllerID").scope()).basewideguid;
      }
      $("#txtValSummaryLogL").text("");
      $Ajax.ajaxAntiForgery({
        url: url,
        type: "POST",
        data: {
          NewNoteText: Main.htmlEncode(notetext),
          roomGUIDs: JSON.stringify(
            $("#NewCopyLog")
              .data("kendoMultiSelect")
              .value(),
          ),
          RoomGUID: currentRoom,
          LogType: logType,
        },
        success: function(response) {
          if (response.success === true) {
            $("#NewNoteText").val("");
            $("#NewNoteText").focus();
            //$("#NewCopyLog").data("kendoMultiSelect").value([]);
          } else {
            Main.handleValErr($("#txtValSummaryLogL"), response);
            $("#txtValSummaryLogL").show();
          }
          Main.unblockit($("#multiLogInput"));
          if (typeof cb === "function") {
            cb();
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
 
          if (typeof cb === "function") {
            cb();
          }
        },
      });
    }
  }

  function saveNoteWithoutAttachment(baseurl) {
    let rmObj = buildLogObject();
    if (rmObj != null) {
      const strTmp = JSON.stringify(rmObj);
      let roomguids;
      if ($("#CopyLog").length > 0 && $("#CopyLog").data("kendoMultiSelect") !== null) {
        roomguids = JSON.stringify(
          $("#CopyLog")
            .data("kendoMultiSelect")
            .value(),
        );
      }
      $Ajax.ajaxAntiForgery({
        url: baseurl,
        type: "POST",
        data: {
          modelString: strTmp,
          roomGUIDs: roomguids,
        },
        success: function(response) {
          //first remove any validation errors that happened before
          $(".input-validation-error").each(function() {
            let el = $(this);
            el.removeClass("input-validation-error");
          });
          if (response.success === false) {
            Main.handleValErr($("#txtValSummary"), response);
            window.scrollTo(0, 0);
          } else {
            Dashboard.jqueryclosemodal();
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }

  function buildLogObject() {
    let ret: NoteObject = new NoteObject();
    const ntText = $("#NoteText").val();
    if (ntText != null && ntText.length > 0) {
      ret.NoteText = Main.htmlEncode(ntText);
 
      if ($("#Lat")) {
        ret.Lat = $("#Lat").val();
      }
      if ($("#Lon")) {
        ret.Lon = $("#Lon").val();
      }
      const tmp = $("#RequestedBy_RoleGUIDs").data("kendoMultiSelect");
      //if ((tmp !==null) && (tmp !== 'undefined')) {
      if (typeof tmp === "undefined") {
        ret.RequestedBy_RoleGUIDs = JSON.stringify($("#RequestedBy_RoleGUIDs").val());
        ret.RequestedBy_UserGUIDs = JSON.stringify($("#RequestedBy_UserGUIDs").val());
      } else {
        ret.RequestedBy_RoleGUIDs = JSON.stringify(
          $("#RequestedBy_RoleGUIDs")
            .data("kendoMultiSelect")
            .value(),
        );
        ret.RequestedBy_UserGUIDs = JSON.stringify(
          $("#RequestedBy_UserGUIDs")
            .data("kendoMultiSelect")
            .value(),
        );
      }
      ret.RequestedBy_Other = $("#RequestedBy_Other").val();
      ret.LogEntryGUID = $("#LogEntryGUID").val();
      ret.StaticGUID = $("#StaticGUID").val();
      ret.Street1 = $("#Street1").val();
      ret.Street2 = $("#Street2").val();
      ret.City = $("#City").val();
      const stCombo = $("#State_Province").data("kendoComboBox");
      ret.State_Province = stCombo.text();
      ret.Zip_PostalCode = $("#Zip_PostalCode").val();
      ret.County_Region = $("#County_Region").val();
      ret.Country = $("#Country").val();
      ret.LinkUrl = $("#LinkUrl").val();
      ret.LogType = $("#LogType").val();
      ret.LinkedLogEntriesAsString = $("#LinkedLogEntriesAsString").val();
      ret.DateCreated = $("#DateCreated").val();
      ret.RoomGUID = $("#RoomGUID").val();
 
      if (!ret.RoomGUID || ret.RoomGUID === "00000000-0000-0000-0000-000000000000") {
        const largeScope = angular.element("#multiControllerID").scope();
        const medScope = angular.element("#multiControllerIDM").scope();
        $("#LargeLogRoomDropdown")
          .data("kendoDropDownList")
          .trigger("change");
        if ($("#wasDragged").length > 0 && $("#wasDragged").val() === "true" && typeof medScope !== "undefined") {
          //If it was dragged, we use medScope or nothing
          ret.RoomGUID = (<any>medScope).selectedRoomM;
        } else if (
          typeof largeScope !== "undefined" &&
          (<any>largeScope).selectedRoom !== undefined &&
          (<any>largeScope).selectedRoom !== null
        ) {
          //If largeScope is defined and the selectionRoom is defined and NOT null, check to see if it was dragged
          ret.RoomGUID = (<any>largeScope).selectedRoom;
        } else if (typeof medScope !== "undefined") {
          //If medScope is undefined, we do nothing
          ret.RoomGUID = (<any>medScope).selectedRoomM;
        }
      }
      if ($("#wasDragged").length > 0) {
        $("#wasDragged").val("false");
      }
      return ret;
    } else {
      const errText: string = "The Log Entry field is requried";
      $("#txtValSummary").text(errText);
      return null;
    }
  }

  angular.module("cobraapp").factory("multiLogbookRepository", function() {
    return {
      getDataSource: function(hostURL, projectGUID, roomGUID) {
        return new kendo.data.DataSource({
          transport: {
            read: {
              url: hostURL + "api/MultiLogbookAPI/GetMultiLogListJSON/" + projectGUID,
              data: {
                RoomGUID: function() {
                  return roomGUID;
                },
              },
            },
          },
          sort: {
            field: "FormattedDateCreated",
            dir: "desc",
          },
          schema: {
            model: {
              id: "LogEntryGUID",
              fields: {
                LogEntryGUID: { editable: false },
                StaticGUID: { editable: false },
                FriendlyUserName: { editable: false },
                RequestedBy: { editable: false },
                NoteText: { editable: false },
                FormattedDateCreated: { type: "date", editable: false },
              },
            },
          },
        });
      },
      getRoomDataSource: function(hostURL, projectGUID) {
        return {
          transport: {
            read: {
              url: hostURL + "Tools/MultiLogbook/GetMultiLogBookRoomsYouHaveAccessTo/" + projectGUID,
            },
          },
        };
      },
    };
  });
