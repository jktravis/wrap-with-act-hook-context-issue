  export function LogListExport(e) {
    $.ajax({
      url: _logTypeURL,
      type: "GET",
      async: false,
      success: function(response) {
        response.forEach(function(item, index) {
          //Start at 1 so we skip the first row, which is all headers in the excel document
          for (var i = 1; i < e.workbook.sheets[0].rows.length; i++) {
            //Position cells[1], because the logtype is the second column in the grid.
            //Not a fan of the absolute here, but no sure how else to dynamically check it.
            //The column title is based on current culture translations from the resx, which
            //I dont know how to get easily from a ts file.
            //Possibility for the future?
            if (e.workbook.sheets[0].rows[i].cells[1].value === item.LogTypeKey) {
              e.workbook.sheets[0].rows[i].cells[1].value = item.CurrentCultureTranslation;
            }
            e.workbook.sheets[0].rows[i].cells[3].value = Main.htmlUnescape(
              e.workbook.sheets[0].rows[i].cells[3].value,
            );
            e.workbook.sheets[0].rows[i].cells[4].value = Main.htmlUnescape(
              e.workbook.sheets[0].rows[i].cells[4].value,
            );
          }
        });
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }
 
  export function sendDetailedNote() {
    var url = _sendDetailedNoteURL;
    var notetext = $("#NewNoteText").val();
    var copyLog = copyLogIsChecked();
    var logType = $("#LogType").val();
    var roomGUIDs = getSelectedRooms();
    var newUrl = url + "?CopyLog=" + copyLog;
    newUrl = newUrl + "?SelectedRooms=" + roomGUIDs;
    if (logType.length > 0) {
      newUrl = newUrl + "&LogType=" + logType;
    }
    if (notetext.length > 0) {
      newUrl = newUrl + "&NewNoteText=" + notetext;
    }
    newUrl = newUrl + "&RoomGUID=" + currentRoomValue();
    window.location.href = newUrl;
  }
  export function sendnote() {
    var notetext = $("#NewNoteText").val();
 
    if (notetext.length > 0) {
      Main.blockit($("#divEntry"));
      var copyLog = copyLogIsChecked();
      var logType = $("#LogType").val();
      var url = _sendNoteURL;
 
      $Ajax.ajaxAntiForgery({
        url: url,
        type: "POST",
        data: {
          NewNoteText: Main.htmlEncode(notetext),
          CopyLog: copyLog,
          roomGUIDs: getSelectedRooms(),
          RoomGUID: currentRoomValue(),
          LogType: logType,
        },
        success: function(response) {
          if (response.success === true) {
            $("#NewNoteText").val("");
            $("#NewNoteText").focus();
            uncheckCopyLog();
          } else {
            $.notify({ message: response.Message }, { type: "danger" });
          }
          Main.unblockit($("#divEntry"));
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }

  function getSelectedRooms() {
    if ($("#CopyLog") != null && $("#CopyLog").length > 0) {
