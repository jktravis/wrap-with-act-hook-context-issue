    function getFormTemplate(hostURL, projectGUID, instanceGUID) {

        });
    }
 
    function getFormData(hostURL, id, templateGUID, instanceGUID) {
      //GetFormInstanceData(Guid id, Guid templateGUID, Guid instanceGUID)
      return $http
        .get(hostURL + formsURL + "GetFormInstanceData/" + id, {
          params: {
            templateGUID: templateGUID,
            instanceGUID: instanceGUID,
          },
        })
        .then(function(res) {
          return res.data;
        });
    }
    class FormObject {
      templateGUID: string;
      templateHTML: string;
      formData: any;
    }
    function getFormInstance(hostURL, projGUID, instanceGUID) {
      let deferred = $q.defer();
 
      getFormTemplate(hostURL, projGUID, instanceGUID).then(function(data) {
        let result: FormObject = new FormObject();
        result.templateGUID = data.templateGUID;
        result.templateHTML = data.templateHTML;
        getFormData(hostURL, projGUID, result.templateGUID, instanceGUID).then(function(formData) {
          result.formData = formData.data;
          return deferred.resolve(result);
        });
      });
 
      return deferred.promise;
    }

  function MakeFormReadonly(divToMakeReadonly) {
    if (divToMakeReadonly === null) {
      return;
    }
    // var divToMakeReadonly = $('#' + startingDivName);
    let inputs = divToMakeReadonly.find(":input");
 
    inputs.each(function() {
      $(this).attr("readonly", "true");
      let type = $(this).attr("type");
 
      if (type === "checkbox" || type === "radio") {
        $(this).attr("disabled", "true");
      }
    });
  }
 
  function HydrateHTMLWithInputValues(data, div, setInitialValues) {
    if (!data) {
      return;
    }
    if (!div) {
      div = $("#formInstanceDiv");
    }
    let parsed = JSON.parse(data);
    $.each(parsed, function() {
      //var $el = $('[id="' + this.Id + '"]'),
      //    type = $el.attr('type');
      let $el = div.find('[id="' + this.Id + '"]'),
        type = $el.attr("type");
 
      switch (type) {
        case "checkbox":
          $el.attr("checked", "checked");
          break;
        case "radio":
          $el.filter('[value="' + this.Value + '"]').attr("checked", "checked");
          break;
        default:
          $el.val(this.Value);
      }
    });
 
    if (setInitialValues === null || setInitialValues === undefined) {
      setInitialValues = true;
    }
    if (setInitialValues) {
      SetInitialValues(div);
    }
  }
