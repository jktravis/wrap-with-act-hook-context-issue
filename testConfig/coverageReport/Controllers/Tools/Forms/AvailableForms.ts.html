  function setActiveGUIDInURI(instanceGUID, templateGUID) {
    if (history.pushState) {
      const uri = new URI(window.location);
      if (instanceGUID !== undefined && instanceGUID !== null && instanceGUID !== "") {
        //this doesn't work, but should be investigated.  https://medialize.github.io/URI.js/
        //if (uri.query('formID') === instanceGUID) {
        //          return; //why do anything?  it's already set
        //      }
        uri.setQuery("formID", instanceGUID);
      } else {
        if (uri.hasQuery("formID") === true) {
          //id already exists, but no new one coming in, so remove from uri
          uri.removeQuery("formID");
        }
      }
      if (templateGUID !== undefined && templateGUID !== null && templateGUID !== "") {
        uri.setQuery("templateID", templateGUID);
      }
      //window.history.pushState({ path: newurl }, '', newurl);
      window.history.pushState({ path: uri.toString() }, "", uri.toString());
    }
  }
 
  export function init(cnfg, formNavDiv) {
    if (formNavDiv === undefined || formNavDiv === null) {
      formNavDiv = $("#formsNavigationDiv");
    }
    formsNavigationDiv = formNavDiv;
    config = cnfg;
  }
 
  export function setActiveInMenu(instanceGUID: string, selectorDivName: string) {
    //clear all active items
    let active;
    if (selectorDivName === undefined || selectorDivName === null) {
      active = $("#formSelectorDiv .active");
    } else {
      active = $("#" + selectorDivName + " .active");
    }
 
    if (active && active.length > 0) {
      $.each(active, function() {
        //var $element = $('[id="' + this.Id + '"]');
        //$element.removeClass('active');
        $(this).removeClass("active");
      });
    }
    if (instanceGUID === null || instanceGUID === undefined) {
      if (activeInstanceGUID !== null && activeInstanceGUID !== undefined) {
        instanceGUID = activeInstanceGUID;
      }
    }
    //find node with id
    if (instanceGUID) {
      activeInstanceGUID = instanceGUID;
      active = $("#" + instanceGUID);
      if (active && active.length > 0) {
        active.addClass("active");
        //make sure parent div is expanded
        const parent = active.closest("div.collapse");
        if (parent && parent.length > 0) {
          //parent.collapse('show');
          //with the forms grouped by op period and template, need to check to see if parent's parent needs to be expanded as well
          const grandParent = parent.closest("div.collapse");
          if (grandParent && grandParent.length > 0) {
            //grandParent.collapse('show');
          }
        }
      } else {
        activeInstanceGUID = null;
      }
    } else {
      activeInstanceGUID = null;
    }
 
    setActiveGUIDInURI(activeInstanceGUID, null);
  }
 
  export function refreshInstanceMenu(instanceGUID, selectorDivName, clearActiveGUID: boolean) {
    if (clearActiveGUID !== undefined && clearActiveGUID) {
      activeInstanceGUID = null;
    }
    //will find all nodes that are open and remember them
    //if instanceGUID is not empty, it will set that one to active
    //otherwise it will remember the last active node and make it active again
    $.ajax({
      url: config.getFormsNavigationURL,
      data: {
        //will eventually pass the date/time of last load, so we can track changes
        allowNewForm: true,
      },
      type: "GET",
      cache: false,
      success: function(response: any) {
        //$('#formsNavigationDiv').html(response);
        formsNavigationDiv.html(response);
        setActiveInMenu(instanceGUID, selectorDivName);
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }
 
  export function refreshInstanceMenuWithCurrentInstance(selectorDivName) {
    if (activeInstanceGUID === null || activeInstanceGUID === undefined) {
      refreshInstanceMenu(null, selectorDivName, true);
    }
    refreshInstanceMenu(activeInstanceGUID, selectorDivName, false);
  }
