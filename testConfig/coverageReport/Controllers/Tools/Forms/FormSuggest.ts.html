//import { Main } from '../../../basicCoBRAScript';
/*export*/ namespace formSuggest {
  let instanceDiv: any;
  let formVars: IFormsVars;
  let suggestInputs = [];
  let cobraInputs = [];
  let reservedCoBRAIds = [];
  let isInitialized = false;
 
  export function init(globalVars: IFormsVars, iDiv) {
    //console.log('in initialize');
    instanceDiv = iDiv;
    formVars = globalVars;
    //getReservedIds();
  }
 
  ///looks through all inputs in the form, finding fields that have reserved key word
  /// or data-suggest===true.
  export function processForm(formInstanceDiv, instanceGUID) {
    //console.log('in processForm');
    instanceDiv = formInstanceDiv;
    //find all inputs, then append a datalist to each one that has a match in reservedCoBRAIds
    //or has the suggest (or suggest=true) data attribute
    getReservedIds().done(function() {
      //console.log('reserved ids have been gotten' + reservedCoBRAIds.length);
      loopThroughInputs(instanceGUID);
    });
  }
 
  function loopThroughInputs(instanceGUID) {
    if (instanceDiv === null || instanceDiv === undefined) {
      return;
    }
 
    let inputs = instanceDiv.find(":input");
 
    //inputs.each(function () {
 
    //});
    let rtnJSON = inputs.each(function() {
      let reslt;
      let $item = $(this);
      let type = $item.attr("type");
      if (type === null || type !== "hidden") {
        let hasSuggest = false;
        let suggest;
        //if ($.hasData($item)) {
        suggest = $item.data("suggest");
        if (suggest === undefined && $item.is("[suggest]")) {
          suggest = $item.attr("suggest");
        }
 
        if (suggest !== null && suggest !== undefined && suggest) {
          hasSuggest = true;
        }
 
        let reservedWordAttr = "";
        reservedWordAttr = $item.data("cobrareserved");
        if (reservedWordAttr === undefined && $item.is("[cobrareserved]")) {
          reservedWordAttr = $item.attr("cobrareserved");
        }
 
        if (
          reservedWordAttr !== null &&
          reservedWordAttr !== undefined &&
          reservedWordAttr &&
          reservedWordAttr !== ""
        ) {
          let matchFound = $.grep(reservedCoBRAIds, function(e) {
            return e.InputID.toLowerCase() === reservedWordAttr.toLowerCase();
          });
          if (!matchFound || matchFound.length === 0) {
            reservedWordAttr = "";
          }
        }
 
        let id = $item.attr("id");
 
        if (hasSuggest) {
          suggestInputs.push(id);
        }
 
        let foundReservedId = $.grep(reservedCoBRAIds, function(e) {
          return e.InputID.toLowerCase() === id.toLowerCase();
          //return e.InputID === id;
        });
        if (foundReservedId && foundReservedId.length > 0) {
          cobraInputs.push(id);
          reslt = buildCoBRAIdDatalist(id, id, $item, hasSuggest, instanceGUID, buildSuggestDatalist);
        } else if (reservedWordAttr !== undefined && reservedWordAttr !== "") {
          if ($.inArray(id, cobraInputs) === -1) {
            reslt = buildCoBRAIdDatalist(id, reservedWordAttr, $item, hasSuggest, instanceGUID, buildSuggestDatalist);
          }
        } else if (hasSuggest) {
          //the else statement is here because buildSuggestDataList is a followup action in the previous method.
          reslt = buildSuggestDatalist(id, $item, false);
        }
      }
    });
  }
 
  function buildSuggestDatalist(id, item, hasDatalist) {
    let dListName = id + "_DataList";
    item.attr("list", dListName);
    //$item.attr('list', dListName);
    if (hasDatalist === undefined || !hasDatalist) {
      $('<datalist id="' + dListName + '"/>').insertAfter(item);
    }
    let dataList = $("#" + dListName);
    //populate datalist
    getSuggestValues(id, dataList);
  }
 
  function buildCoBRAIdDatalist(id, reservedWord, item, hasSuggest, instanceGUID, next) {
    let dListName = id + "_DataList";
    item.attr("list", dListName);
    //$item.attr('list', dListName);
    $('<datalist id="' + dListName + '"/>').insertAfter(item);
    let dataList = $("#" + dListName);
    //populate datalist
 
    if (hasSuggest && next) {
      getReservedValues(id, reservedWord, dataList, instanceGUID, next(id, item, true));
      //next(id, item, true);
    } else {
      getReservedValues(id, reservedWord, dataList, instanceGUID, null);
    }
  }
 
  function addDataListValue(dataList, value) {
    let opt = $("<option></option>").attr("value", value);
    let exists = dataList.find("option[value='" + value + "']");
 
    if (!exists || exists.length === 0) {
      dataList.append(opt);
    }
  }
 
  function getReservedValues(id, reservedWord, dataList, instanceGUID, next) {
    //look in reservedCoBRAIds and get how to retrieve the list
 
    let data = {
      InputID: "",
      ToolName: "",
      DependentID: "",
    };
    let found = $.grep(reservedCoBRAIds, function(e) {
      if (e.InputID.toLowerCase() === id.toLowerCase()) {
        return true;
      } else if (e.InputID.toLowerCase() === reservedWord.toLowerCase()) {
        return true;
      }
      return false;
    });
 
    if (found && found !== undefined && found.length > 0) {
      data = found[0];
      if (data.DependentID === undefined || data.DependentID === null) {
        data.DependentID = "";
      }
    }
 
    $.getJSON(formVars.GetEngineValuesForInputURL, {
      inputId: data.InputID,
      toolName: data.ToolName,
      dependentIds: data.DependentID,
      instanceGUID: instanceGUID,
    })
      .done(function(result) {
        if (result.success === true) {
          let arr = JSON.parse(result.data);
          $.each(arr, function(index, item) {
            addDataListValue(dataList, item);
          });
          if (next) {
            //next;
          }
        } else {
          //console.log(result.Message);
        }
      })
      .fail(function() {
        console.log("error retrieving CoBRA Reserved Values");
      });
  }
 
  function getSuggestValues(id, dataList) {
    $.getJSON(formVars.getSuggestionValuesURL, { inputId: id })
      .done(function(result) {
        let arr = JSON.parse(result.data);
        $.each(arr, function(index, item) {
          addDataListValue(dataList, item);
        });
      })
      .fail(function(error) {
        console.log("error retrieving CoBRA Suggest Values");
      });
  }
 
  function getReservedIds() {
    let deferred = $.Deferred();
    setTimeout(function() {
      if (!isInitialized) {
        $.getJSON(formVars.getCoBRAReservedIdsURL)
          .done(function(result) {
            reservedCoBRAIds = JSON.parse(result.data);
            isInitialized = true;
            deferred.resolve(reservedCoBRAIds);
          })
          .fail(function() {
            console.log("error retrieving CoBRA reserved Ids");
            isInitialized = true; //error state but better to have form load than hang.
            deferred.resolve([]);
          });
      } else {
        deferred.resolve(reservedCoBRAIds);
      }
    }, 3000);
    return deferred.promise();
  }
}
