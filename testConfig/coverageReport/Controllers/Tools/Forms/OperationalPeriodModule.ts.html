//import { Main } from '../../../basicCoBRAScript';
/*export*/ namespace operationalPeriodModule {
  let config = {};
  export function init(cnfg) {
    config = cnfg;
  }
 
  export function showModal(modalID, comboboxID, saveButtonID, instanceGUID, val, done) {
    //opPeriodChangeComboBox
    let dropDown = $("#" + comboboxID);
    populateOpPeriodDropDown(dropDown, val, null);
 
    let modalWindow = $("#" + modalID);
 
    let deferred = $.Deferred<void>();
    let promise = deferred.promise();
    promise.done(function() {
      done(instanceGUID, dropDown.data("kendoDropDownList").value());
    });
    //promise.fail(dismissAndContinueAction);
 
    promise.always(function() {
      //close the dialog
      modalWindow.modal("hide");
    });
 
    // DOM interaction
    $("#" + saveButtonID).click(function() {
      deferred.resolve();
 
      return false; // prevent default form behavior
    });
 
    //$('#opPeriodFormModal').modal('show');
    modalWindow.modal("show");
  }
 
  export function populateOpPeriodDropDown(dropdown, currentVal, onChangeHandler) {
    if ($.isNumeric(currentVal)) {
      //is an index
      if (currentVal <= 0) {
        dropdown.data("kendoDropDownList").value("00000000-0000-0000-0000-000000000000");
      } else {
        dropdown.data("kendoDropDownList").select(currentVal);
      }
    } else {
      //is a guid (or emptystring)
      dropdown.data("kendoDropDownList").value(currentVal);
    }
  }
 
  function showChangeOpPeriodDialog(instanceGUID) {
    //bring up modal popup to ask for new name
    $("#opChangeFormInstanceGUID").val(instanceGUID);
    let val = $("#formInfoDiv").data("opGuid");
 
    populateOpPeriodDropDown($("#opPeriodChangeComboBox"), val, null);
    // dropdownlist.value($("#value").val());
    $("#opPeriodFormModal").modal("show");
  }
  $("#opPeriodFormModal").on("shown.bs.modal", function() {
    $("#opPeriodChangeComboBox").focus();
  });
}
