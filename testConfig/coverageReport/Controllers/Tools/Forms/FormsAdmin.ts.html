  export function DeleteTemplate(e) {
    e.preventDefault();
    //e.preventDefault ? e.preventDefault() : e.returnValue = false;
    let tr = $(e.target).closest("tr"); //get the row for deletion
    let data = this.dataItem(tr); //get the row data so it can be referred later
 
    ShowTemplateDeleteDialog(data);
  }
 
  function ShowTemplateDeleteDialog(data) {
    $("#confirmModal").modal("show");
 
    // DOM interaction
    $("#confirmOKButton").click(function() {
      $("#confirmModal").modal("hide");
      let deleteTemplateURL = formVars.deleteTemplateURL;
      $Ajax.ajaxAntiForgery({
        url: deleteTemplateURL,
        data: {
          templateGUID: data.FormTemplateGUID,
        },
        type: "POST",
        success: function(response) {
          if (response.success === true) {
            location.reload();
          } else {
            $.notify({ message: response.Message }, { type: "danger" });
          }
        },
        failure: function(response) {
          $("#deletedDiv").html("<span class='alert alert-warning'>Error loading form template</span>");
          //$("#plumeEditorDiv").unblock();
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    });
  }

  export function getFileInfo(e) {
    return $.map(e.files, function(file) {
      let info = file.name;
 
      // File size is not available in all browsers
      if (file.size > 0) {
        info += " (" + Math.ceil(file.size / 1024) + " KB)";
      }
      return info;
    }).join(", ");
  }
 
  export function getNameFromFileUpload(e) {
    let imagePath = getFileInfo(e);
    imagePath = imagePath.substr(0, imagePath.lastIndexOf(".")) || imagePath;
 
    $("#formTemplateName").val(imagePath);
  }
