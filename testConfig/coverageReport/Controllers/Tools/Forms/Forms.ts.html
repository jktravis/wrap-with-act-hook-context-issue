  ///will find all the input elements and then build a key/value pair of the id and value
  ///these key/value pairs will be bundled in a JSON object and returned from the function
  function RetrieveAllInputValues(divToScrape) {
    if (divToScrape === null) {
      return null;
    }
    let inputs = divToScrape.find(":input");
 
    //inputs.each(function () {
 
    //});
    let rtnJSON = inputs
      .map(function() {
        // $(this) is used more than once; cache it for performance.
        let $item = $(this);
        let val = $.trim($item.val());
        let type = $item.attr("type");
        if (val && (type === null || type !== "hidden")) {
          if (type === "checkbox" || type === "radio") {
            if ($item.is(":checked")) {
              return {
                // Note: using .data() to read HTML5 data- attributes
                //  requires jQuery 1.4.3+. Use attr() in older versions.
                id: $item.attr("id"),
                value: val,
              };
            }
          } else {
            return {
              // Note: using .data() to read HTML5 data- attributes
              //  requires jQuery 1.4.3+. Use attr() in older versions.
              id: $item.attr("id"),
              value: val,
            };
          }
        }
      })
      .get();
    return rtnJSON;
  }
 
  export function HydrateHTMLWithInputValues(data, div, setInitialValues) {
    if (!data) {
      return;
    }
    if (!div) {
      div = $("#formInstanceDiv");
    }
    let parsed = JSON.parse(data);
    $.each(parsed, function() {
      //var $el = $('[id="' + this.Id + '"]'),
      //    type = $el.attr('type');
      let $el = div.find('[id="' + this.Id + '"]'),
        type = $el.attr("type");
 
      switch (type) {
        case "checkbox":
          $el.attr("checked", "checked");
          break;
        case "radio":
          $el.filter('[value="' + this.Value + '"]').attr("checked", "checked");
          break;
        default:
          $el.val(this.Value);
      }
    });
 
    if (setInitialValues === null || setInitialValues === undefined) {
      setInitialValues = true;
    }
    if (setInitialValues) {
      SetInitialValues(div);
    }
  }
  function SetInitialValues(div) {
    let inputs = div.find(":input");
 
    inputs.each(function() {
      let $item = $(this);
      let val: any = $.trim($item.val());
      let type = $item.attr("type");
      if (type === "checkbox") {
        //val = $item.attr('checked');
        val = $item.is(":checked");
      } else if (type === "radio") {
        //            val = $item.attr('checked');
        val = $item.is(":checked");
      }
 
      if (val === undefined || val === null) {
        val = "";
      }
      if (type === "hidden") {
        val = "";
      }
 
      $(this).data("initialValue", val);
    });
  }

  function HasUnsavedChanges(divToCheck) {
    if (divToCheck === null) {
      return null;
    }
 
    let inputs = divToCheck.find(":input");
    let isDirty = false;
 
    inputs.each(function() {
      let initialValue = $(this).data("initialValue");
      let type = $(this).attr("type");
      let currentValue: any = $.trim($(this).val());
      if (type === "checkbox") {
        //currentValue = $(this).attr('checked');
        currentValue = $(this).is(":checked");
      } else if (type === "radio") {
        //currentValue = $(this).attr('checked');
        currentValue = $(this).is(":checked");
      }
 
      if (currentValue === "undefined" || currentValue === null) {
        currentValue = "";
      }
 
      if (initialValue !== currentValue) {
        if (type !== "hidden") {
          isDirty = true;
          return false;
        }
      }
    });
    return isDirty;
  }
