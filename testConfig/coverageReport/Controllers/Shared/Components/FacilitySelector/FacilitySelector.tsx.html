import React, { Component, createRef } from "react";
import Modal from "reactstrap/lib/Modal";
import ModalHeader from "reactstrap/lib/ModalHeader";
import ModalBody from "reactstrap/lib/ModalBody";
import FormGroup from "reactstrap/lib/FormGroup";
import { Form } from "reactstrap";
import Label from "reactstrap/lib/Label";
import Input from "reactstrap/lib/Input";
import InputGroupAddon from "reactstrap/lib/InputGroupAddon";
import { FaSearch } from "react-icons/fa";
import ModalFooter from "reactstrap/lib/ModalFooter";
import Button from "reactstrap/lib/Button";
import ButtonGroup from "reactstrap/lib/ButtonGroup";
import InputGroup from "reactstrap/lib/InputGroup";
import ReactTable from "react-table";
import BeatLoader from "react-spinners/BeatLoader";
import styled from "react-emotion";
 
import { searchAllFacilities } from "../../../../Api/facilityApi";
 
const FlexCenter = styled("div")`
  display: flex;
  justify-content: space-around;
  align-items: center;
`;
FlexCenter.displayName = "FlexCenter";
 
const columns = [
  {
    Header: "Facility",
    accessor: "FacilityName",
  },
  {
    Header: "Organization",
    accessor: "FriendlyOrgName",
  },
  {
    Header: "Description",
    accessor: "ScrubbedFacilityDescription",
  },
  {
    Header: "Category",
    accessor: "FacilityCategory",
  },
  {
    Header: "Sub Category",
    accessor: "FacilitySubCategory",
  },
];
 
interface OptionalSearchArgs {
  orListString?: string;
  isLoading?: boolean;
  wktString?: string;
}
 
interface Props {
  isOpen: boolean;
  toggleFn: Function;
  selectFacilityFn: (FacilityGUID) => void;
 
  /**
   * Optional args for the search API.
   * @see searchAllFacilities
   */
  optionalSearchArgs?: OptionalSearchArgs;
}
 
class FacilitySelector extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    optionalSearchArgs: {},
  };
 
  static getPageSize(len) {
    return len < 10 && len > 0 ? len : 10;
  }
  input: any;
  state = {
    facilityData: [],
    facilityName: "",
    selectedItemId: null,
    loadingSearchResults: false,
  };
 
  constructor(props) {
    super(props);
    this.input = createRef();
  }
 
  componentDidMount() {}
 
  /**
   * Submit the search
   * @param event
   */
  handleSubmit = event => {
    event.preventDefault();
    this.setState(() => ({ loadingSearchResults: true }));
    searchAllFacilities({ ...this.props.optionalSearchArgs, query: this.state.facilityName }).then(facilityData =>
      this.setState(() => ({ facilityData, loadingSearchResults: false })),
    );
  };
 
  reset = () => {
    this.setState(() => ({
      facilityData: [],
      facilityName: "",
      selectedItemId: null,
    }));
  };
 
  handleChange = event => {
    const { name, value } = event.target;
    this.setState(() => ({ [name]: value }));
  };
 
  toggleSelection = id => {
    this.setState(({ selectedItemId }) => {
      if (selectedItemId === id) {
        return {
          selectedItemId: null,
        };
      }
      return { selectedItemId: id };
    });
  };
 
  isSelected = key => {
    return key === this.state.selectedItemId;
  };
 
  getTrProps = (state, row) => {
    const selected = this.isSelected(row.original.FacilityGUID);
    const style = selected ? { backgroundColor: "#f4c109" } : {};
    return {
      style: {
        ...style,
      },
      onClick: () => {
        this.toggleSelection(row.original.FacilityGUID);
      },
    };
  };
 
  selectAndContinue = () => {
    this.props.selectFacilityFn(this.state.selectedItemId);
    this.reset();
  };
 
  selectAndClose = () => {
    this.props.selectFacilityFn(this.state.selectedItemId);
    this.props.toggleFn();
    this.reset();
  };
 
  render() {
    const { facilityData, facilityName, selectedItemId, loadingSearchResults } = this.state;
    const { isOpen, toggleFn } = this.props;
    return (
      <Modal isOpen={isOpen} toggle={toggleFn as any} size="lg">
        <ModalHeader toggle={toggleFn as any}>Facility Selector</ModalHeader>
        <ModalBody>
          <p>Search for an existing facility within the system.</p>
          <Form onSubmit={this.handleSubmit} style={{ width: "50%" }} autoComplete="off">
            <FormGroup>
              <Label for="facility-name">Facility Name</Label>
              <InputGroup>
                <Input
                  id="facility-name"
                  name="facilityName"
                  ref={this.input}
                  onChange={this.handleChange}
                  value={facilityName}
                />
                <InputGroupAddon addonType="append">
                  <Button color="primary" disabled={!facilityName.length || loadingSearchResults}>
                    <FaSearch /> Search
                  </Button>
                </InputGroupAddon>
              </InputGroup>
            </FormGroup>
          </Form>
          {loadingSearchResults ? (
            <FlexCenter style={{ padding: "1rem", margin: "1rem" }}>
              <BeatLoader />
            </FlexCenter>
          ) : facilityData.length ? (
            <ReactTable
              className="-striped -highlight"
              data={this.state.facilityData}
              columns={columns}
              pageSize={FacilitySelector.getPageSize(facilityData.length)}
              showPageSizeOptions={false}
              getTrProps={this.getTrProps}
            />
          ) : null}
        </ModalBody>
        <ModalFooter>
          <ButtonGroup>
            <Button color="warning" onClick={this.props.toggleFn as any}>
              Close
            </Button>
            <Button color="primary" onClick={this.selectAndContinue} disabled={!selectedItemId}>
              Select Facility and Continue Searching
            </Button>
            <Button color="success" onClick={this.selectAndClose} disabled={!selectedItemId}>
              Select Facility
            </Button>
          </ButtonGroup>
        </ModalFooter>
      </Modal>
    );
  }
}
 
export default FacilitySelector;
