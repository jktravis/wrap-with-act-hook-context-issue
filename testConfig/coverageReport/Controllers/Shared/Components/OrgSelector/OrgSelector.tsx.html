import React, { Component } from "react";
import Select from "react-select";
import { getListOfPossibleOrganizations } from "../../../../Api/organziationApi";
 
type OrgSelectorState = {
  availableOptions: any[];
};
 
type Props = {
  selectedOption?: any;
  changeFn?: Function;
};
 
class OrgSelector extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    selectedOption: null,
    changeFn: Function.prototype,
  };
 
  state: OrgSelectorState = {
    availableOptions: [],
  };
 
  async componentDidMount() {
    const data = await getListOfPossibleOrganizations();
    this.setState(() => ({
      availableOptions: data.map(d => {
        return { label: d.OrganizationName, value: d.OrganizationName };
      }),
    }));
  }
 
  render() {
    const { availableOptions } = this.state;
    const { selectedOption, changeFn } = this.props;
 
    return (
      <Select
        id="OrgSelector"
        options={availableOptions}
        value={selectedOption}
        onChange={changeFn as any}
        isClearable
      />
    );
  }
}
 
export default OrgSelector;
