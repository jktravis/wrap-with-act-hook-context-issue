import React from "react";
import { ChromePicker } from "react-color";
import ReactDOM from "react-dom";
import { Manager, Popper, Reference } from "react-popper";
import styled from "react-emotion";
import { css } from "emotion";
import createDataToggle from "Utilities/createDataToggle";
import rgba from "polished/lib/color/rgba";
import { FaTimes } from "react-icons/fa";
import Button from "reactstrap/lib/Button";
 
const colorPickerSwatch = css`
  background: #fff;
  border-color: #d2d2d2;
  padding: 0.4rem;
 
  .color-picker__color {
    width: 38px;
    height: 24px;
    border-radius: 2px;
    background: whitesmoke;
  }
`;
 
const ColorWindowHeader = styled("div")`
  background: white;
  display: flex;
  justify-content: flex-end;
  padding: 0.15rem 0.25rem;
  border-top: 1px solid;
  border-left: 1px solid;
  border-right: 1px solid;
  border-color: #cccccc;
  border-radius: 4px 4px 0px 0px;
`;
 
const ColorPickerSwatchPopover = styled("div")`
  .color-picker-popover__color {
    border: 1px solid #ccc;
  }
`;
 
interface Props {
  colorChangeFn: (color, event) => void;
  color: any;
}
 
class ColorSwatchButton extends React.Component<Props, any> {
  static defaultProps: Partial<Props> = {
    color: {
      rgb: { r: 0, g: 0, b: 0, a: 0.7 },
    },
  };
 
  popperMount: HTMLElement;
  state = {
    showPicker: false,
  };
 
  constructor(props) {
    super(props);
    this.popperMount = document.createElement("div");
  }
 
  togglePicker = () => this.setState(createDataToggle("showPicker"));
 
  componentDidMount() {
    const body = document.querySelector("body");
    body.appendChild(this.popperMount);
  }
 
  componentWillUnmount() {
    const body = document.querySelector("body");
    body.removeChild(this.popperMount);
  }
 
  render() {
    const { colorChangeFn, color } = this.props;
    const { showPicker } = this.state;
    const backgroundColor = rgba(color.rgb.r, color.rgb.g, color.rgb.b, color.rgb.a);
    return (
      <div>
        <Manager>
          <Reference>
            {({ ref }) => (
              <button
                type="button"
                className={`btn btn-light ${colorPickerSwatch}`}
                onClick={this.togglePicker}
                ref={ref}
              >
                <div className="color-picker__color" style={{ backgroundColor }} />
              </button>
            )}
          </Reference>
          {ReactDOM.createPortal(
            <Popper placement="bottom-start">
              {({ placement, ref, style }) => {
                if (showPicker) {
                  return (
                    <div style={{ display: showPicker ? "block" : "none" }}>
                      <ColorPickerSwatchPopover
                        className="color-picker-popover"
                        innerRef={ref}
                        style={{ ...style, zIndex: 4 }}
                        data-placement={placement}
                      >
                        <ColorWindowHeader>
                          <Button color="close" size="sm" onClick={this.togglePicker}>
                            <FaTimes />
                          </Button>
                        </ColorWindowHeader>
                        <div className="color-picker-popover__color">
                          <ChromePicker color={color ? color.rgb : undefined} onChangeComplete={colorChangeFn as any} />
                        </div>
                      </ColorPickerSwatchPopover>
                    </div>
                  );
                }
                return null;
              }}
            </Popper>,
            this.popperMount,
          )}
        </Manager>
      </div>
    );
  }
}
 
export default ColorSwatchButton;
