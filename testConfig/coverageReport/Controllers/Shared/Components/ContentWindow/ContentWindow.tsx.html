import React, { ReactNode, useState } from "react";
import styled from "react-emotion";
import defaultHeaderRenderer from "./defaultHeaderRenderer";
 
const ContentWindowContainer = styled("div")`
  position: absolute;
  border-radius: 4px;
  box-shadow: 0 2px 5px 0 #929292;
  background-color: rgba(235, 235, 235);
  will-change: opacity;
  transition: opacity 0.25s ease;
  opacity: ${({ isHovering }: any) => (isHovering ? 1 : 0.8)};
  .content-window-container__header,
  .content-window-container__footer {
    display: flex;
    justify-content: space-between;
    align-items: center;
    padding: 0.5rem;
    background-color: rgba(247, 247, 247);
  }
  .content-window-container__header {
    border-bottom: 1px solid #ccc;
  }
  .content-window-container__footer {
    border-top: 1px solid #ccc;
  }
`;
ContentWindowContainer.displayName = "ContentWindowContainer";
 
interface ContentWindowProps {
  headerText?: string;
  /**
   * Function to dismiss the window.
   * This should call back to a parent component that is showing/hiding the window.
   */
  dismissFn?: Function;
  /**
   * Override the header render.
   * Uses the `defaultHeaderRenderer` if no renderer is provided.
   * It will be given the save `dismissFn` and `headerText` that was provided to
   * the component.
   */
  headerRenderer?: Function;
  /**
   * Override the footer.
   * No default is provided
   */
  footerRenderer?: Function;
  /**
   * Override the style
   */
  style?: any;
  /**
   * Add a class.
   */
  className?: string;
  /**
   * The content you want to show in the window.
   */
  children?: ReactNode;
}
 
function ContentWindow(props: ContentWindowProps) {
  const {
    dismissFn = () => null,
    headerText,
    children,
    headerRenderer = defaultHeaderRenderer,
    footerRenderer = () => null,
    style,
    className,
  } = props;
 
  const [isHovering, setHovering] = useState(false);
 
  return (
    <ContentWindowContainer
      style={style}
      isHovering={isHovering}
      className={className}
      onMouseOver={() => setHovering(true)}
      onMouseOut={() => setHovering(false)}
    >
      <div className="content-window-container__header">{headerRenderer({ dismissFn, headerText })}</div>
      {children}
      <div className="content-window-container__footer">{footerRenderer()}</div>
    </ContentWindowContainer>
  );
}
 
/* eslint-disable-next-line no-undef */
export { ContentWindowProps };
export default ContentWindow;
