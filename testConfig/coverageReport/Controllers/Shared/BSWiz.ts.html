export namespace BSWiz {
  //Activate the target tab.
  //* @param { jQuery } Collection of tabs
  //  * @targetIdx { Number } The target index from the collection
  //    * /
  export function activateTab($tabs, targetIdx) {
    const $next = $($tabs.get(targetIdx));
    $next.addClass("active").attr("aria-selected", "true");
    $($next.attr("href"))
      .addClass("active")
      .addClass("show");
  }
 
  /**
   * Gets the active tab.
   * @returns {jQuery} The active tab as a jQuery object.
   */
  export function getActiveTab() {
    return $(".nav-pills a.active").first();
  }
 
  /**
   * Disable all the active tab and content panel
   */
  export function disableTab() {
    const $activeTab = getActiveTab()
      .removeClass("active")
      .addClass("disabled")
      .attr("aria-selected", "false");
 
    $($activeTab.attr("href"))
      .first()
      .removeClass("active")
      .removeClass("show");
  }
  /**
   * Higher order export function that will handle the tab switching.
   * The `innerFn` is the export function that will ultimately be executed.
   * @returns {export function} A export function that will execute the tab switch.
   */
  export function handleTabSwitch(forward, startingIdx, validateCheck: () => string) {
    return function innerFn() {
      if (validateCheck !== null) {
        const resp = validateCheck();
        if (resp !== undefined && resp.length > 0) {
          if ($("#validationError").length > 0) {
            $("#validationError").text(resp);
          } else {
            $.notify({ message: resp }, { type: "danger" });
          }
          return;
        }
      }
 
      if ($("#validationError").length > 0) {
        $("#validationError").text("");
      }
      const $activeTab = getActiveTab();
      const $tabs = $(".tabHeader a.nav-link");
 
      disableTab();
 
      const idx = $tabs.index($activeTab);
      //checks if the startingIdx is set. this will only happen when the modal is being reset on close
      const targetIdx = startingIdx !== null ? startingIdx : forward ? idx + 1 : idx - 1;
 
      activateTab($tabs, targetIdx);
 
      const showFinish = shouldShowFinish($tabs, targetIdx); //this returns a bool
      const showPrev = shouldShowPrev(targetIdx); //this returns a bool
 
      togglePrev(showPrev); //dependant on the bool val, shows or hides previous button
      toggleFinish(showFinish); // depentant on the bool val, shows or hides the finish button and next button
    };
  }
  export function shouldShowFinish($tabs, idx) {
    return $tabs.length === idx + 1; //idx + 1 because the index will always be one off, do to the tabs.length = total pills and idx = current tab, not target
  }
 
  export function shouldShowPrev(targetIdx) {
    return targetIdx <= 0; //if this is not the first pill, show the prev button
  }
  //show or hide the prev button, dependant on the pill
  export function togglePrev(showPrev) {
    if (showPrev) {
      $(".btn-previous").hide();
    } else {
      $(".btn-previous").show();
    }
  }
  //handles showing and hiding both the finish and the next btns
  export function toggleFinish(showFinish) {
    if (showFinish) {
      $(".btn-next").hide();
      $(".btn-finish").show();
    } else {
      $(".btn-finish").hide();
      $(".btn-next").show();
    }
  }
}
