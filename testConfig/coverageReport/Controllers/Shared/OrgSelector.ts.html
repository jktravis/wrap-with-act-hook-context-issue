  function addOrgToDiv(org, callingButton) {
    const btnId = $(callingButton).attr("id");
    const orglist = $("#" + btnId)
      .closest(".card")
      .children(".org-list");
    if (orglist.length > 0) {
      const div = $(orglist[0]); //$('#' + divName);
      let data = div.data("orglist");
      if (_isSingleSelection) {
        data = [];
        data.push(org.OrganizationName);
      } else {
        if (!data || !$.isArray(data)) {
          data = [];
        }
        if ($.inArray(org.OrganizationName, data) === -1) {
          data.push(org.OrganizationName);
        }
      }
      div.data("orglist", data);
      RefreshDiv(div);
    }
  }
 
  function wireTemplateEventListeners(div) {
    const divId = $(div).attr("id");
    $("#" + divId + " .remove-org").each(function(i, elem) {
      $(elem).on("click", removeOrg);
    });
    let isReadonly: boolean = false;
    const readonlyVal: any = $(div).data("readonly");
    if (readonlyVal !== undefined) {
      isReadonly = <boolean>readonlyVal;
    }
    if (isReadonly) {
      //hide delete button
      $("#" + divId + " .remove-org").hide();
    }
  }
 
  function removeOrg(e) {
    const orgName = $(e.currentTarget).data("organizationname");
    const btnId = $(e.currentTarget).attr("id");
    //find closest user list
    //remove user list from array
    const orglist = $(e.currentTarget)
      .closest(".card")
      .children(".org-list");
    if (orglist.length > 0) {
      const div = $(orglist[0]); //$('#' + divName);
      let data = div.data("orglist");
      if (_isSingleSelection) {
        data = [];
      } else {
        if ($.inArray(orgName, data) !== -1) {
          data = $.grep(data, function(value) {
            return value !== orgName;
          });
        }
      }
      div.data("orglist", data);
      RefreshDiv(div);
      handleOrgReturn(null, null);
    }
  }

    if (!_orgsDS) {

          read: {
            url: function() {
              if (!_orgSearchURL) {

              } else {
                return _orgSearchURL;
              }
            },
            dataType: "json", // "jsonp" is required for cross-domain requests; use "json" for same-domain requests
            //data: _addressSearchParams,
            cache: true,
          },
          parameterMap: function(options, operation) {
            if (operation === "read") {
              //var st = $("#streetSelector").data('kendoAutoComplete');
              //_addressSearchParams.filters = st.dataSource.filter().filters;
              const name = $("#textOrgSearch").val();
              let tmpIsLoading: boolean = _isLoading;
              if (tmpIsLoading === false && !isModalShown()) {
                tmpIsLoading = true;
              }
              return {
                query: name,
                orgListString: "",
                isLoading: tmpIsLoading,
              };
            }
          }, //parameterMap
        },
        requestEnd: function() {
          $(".orgSearchProcessing").hide();
          $("#orgContainer").removeClass("blurredDiv");
        },
        requestStart: function() {
          $(".orgSearchProcessing").show();
          $("#orgContainer").addClass("blurredDiv");
        },

              row = grid.tbody.find(">tr:not(.k-grouping-row)").eq(0);
            grid.select(row);
