  function getSelectedAddress() {
    const rtnAddress: IAddress = {
      AddressGUID: "",
      HouseNumber: "",
      StreetName: "",
      ZipCode: "",
      City: "",
      State: "",
      Country: "",
      Latitude: null,
      Longitude: null,
      AdminSubDivisionGUID: "",
      DivisionName: "",
    };
 
    //if grid has selected item, use it
    const grid = $("#addressList").data("kendoGrid");
    let sel: IAddressObservable = <IAddressObservable>grid.dataItem(grid.select());
    //else if grid has 1 item, use it,
    if (sel === null) {
      //try to get it from first item
      sel = grid.dataSource.data()[0];
    }
    if (sel !== null && sel !== undefined) {
      rtnAddress.AddressGUID = sel.AddressGUID;
      rtnAddress.StreetName = sel.StreetName;
      rtnAddress.HouseNumber = sel.HouseNumber;
      rtnAddress.ZipCode = sel.ZipCode;
      rtnAddress.AdminSubDivisionGUID = sel.AdminSubDivisionGUID;
      rtnAddress.City = sel.DivisionName;
      rtnAddress.Latitude = sel.Latitude;
      rtnAddress.Longitude = sel.Longitude;
      rtnAddress.Country = sel.Country;
      //if (sel.Province !== null) {
      rtnAddress.State = sel.ProvinceName;
      //}
    } else {
      //else build address from selected fields in autocomplete boxes
      rtnAddress.AddressGUID = "";
      rtnAddress.StreetName = $("#streetSelector")
        .data("kendoAutoComplete")
        .value();
      rtnAddress.HouseNumber = $("#houseSelector")
        .data("kendoAutoComplete")
        .value();
      rtnAddress.ZipCode = $("#zipSelector")
        .data("kendoAutoComplete")
        .value();
      const cityVal = $("#citySelector")
        .data("kendoAutoComplete")
        .dataItem(0);
      if (cityVal !== null && cityVal !== undefined) {
        rtnAddress.City = cityVal.DivisionName;
        rtnAddress.AdminSubDivisionGUID = cityVal.AdminSubDivisionGUID;
      } else {
        rtnAddress.City = $("#citySelector")
          .data("kendoAutoComplete")
          .value();
      }
      rtnAddress.Latitude = null;
      rtnAddress.Longitude = null;
    }
 
    return rtnAddress;
  }

  function configureFromFieldsObj(fieldNames) {
    if (fieldNames !== null && fieldNames !== undefined) {
      if (fieldNames.City !== null && fieldNames.City !== undefined) {
        _fieldNames.City = fieldNames.City;
      }
      if (fieldNames.State !== null && fieldNames.State !== undefined) {
        _fieldNames.State = fieldNames.State;
      }
      if (fieldNames.Zip !== null && fieldNames.Zip !== undefined) {
        _fieldNames.Zip = fieldNames.Zip;
      }
      if (fieldNames.Street !== null && fieldNames.Street !== undefined) {
        _fieldNames.Street = fieldNames.Street;
      }
      if (fieldNames.House !== null && fieldNames.House !== undefined) {
        _fieldNames.House = fieldNames.House;
      }
    }
    const ellipses: string = "...";
    if (!_fieldNames.House.Prompt.endsWith(ellipses)) {
      _fieldNames.House.Prompt = _fieldNames.House.Prompt + ellipses;
    }
    if (!_fieldNames.State.Prompt.endsWith(ellipses)) {
      _fieldNames.State.Prompt = _fieldNames.State.Prompt + ellipses;
    }
    if (!_fieldNames.City.Prompt.endsWith(ellipses)) {
      _fieldNames.City.Prompt = _fieldNames.City.Prompt + ellipses;
    }
    if (!_fieldNames.Street.Prompt.endsWith(ellipses)) {
      _fieldNames.Street.Prompt = _fieldNames.Street.Prompt + ellipses;
    }
    if (!_fieldNames.Zip.Prompt.endsWith(ellipses)) {
      _fieldNames.Zip.Prompt = _fieldNames.Zip.Prompt + ellipses;
    }
  }
