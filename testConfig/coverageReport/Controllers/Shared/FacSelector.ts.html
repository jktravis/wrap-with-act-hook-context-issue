  function selectFac(hidePopup) {
    const Fac = getSelectedFac();

  function addFacToDiv(Fac, callingButton) {
    const btnId = $(callingButton).attr("id");
    const FacList = $("#" + btnId)
      .closest(".card")
      .children(".fac-list");
    if (FacList.length > 0) {
      const div = $(FacList[0]); //$('#' + divName);
      let data = div.data("faclist");
      if (_isSingleSelection) {
        data = [];
        data.push(Fac.FacilityGUID);
      } else {
        if (!data || !$.isArray(data)) {
          data = [];
        }
        if ($.inArray(Fac.FacilityGUID, data) === -1) {
          data.push(Fac.FacilityGUID);
        }
      }
      div.data("faclist", data);
      RefreshDiv(div);
    }
  }
 
  function wireTemplateEventListeners(div) {
    const divId = $(div).attr("id");
    $("#" + divId + " .remove-Fac").each(function(i, elem) {
      $(elem).on("click", removeFac);
    });
    let isReadonly: boolean = false;
    const readonlyVal: any = $(div).data("readonly");
    if (readonlyVal !== undefined) {
      isReadonly = <boolean>readonlyVal;
    }
    if (isReadonly) {
      //hide delete button
      $("#" + divId + " .remove-fac").hide();
    }
  }
 
  function removeFac(e) {
    const FacGuid = $(e.currentTarget).data("facguid");
    const btnId = $(e.currentTarget).attr("id");
    //find closest Fac list
    //remove Fac list from array
    const FacList = $(e.currentTarget)
      .closest(".card")
      .children(".fac-list");
    if (FacList.length > 0) {
      const div = $(FacList[0]); //$('#' + divName);
      let data = div.data("faclist");
      if (_isSingleSelection) {
        data = [];
      } else {
        if ($.inArray(FacGuid, data) !== -1) {
          data = $.grep(data, function(value) {
            return value !== FacGuid;
          });
        }
      }
      div.data("faclist", data);
      RefreshDiv(div);
    }
  }
 
  function getSelectedFac() {
    const rtnFac: IFac = {
      FacilityGUID: "",
    };
 
    //if grid has selected item, use it
    const grid = $("#FacList").data("kendoGrid");
    let sel: IFacObservable = <IFacObservable>grid.dataItem(grid.select());
    //else if grid has 1 item, use it,
    if (sel === null) {
      //try to get it from first item
      sel = grid.dataSource.data()[0];
    }
    if (sel !== null && sel !== undefined) {
      rtnFac.FacilityGUID = sel.FacilityGUID;
    } else {
      rtnFac.FacilityGUID = "";
    }
 
    return rtnFac;
  }

  function configureFacGrid() {
    //configure ds for addresses
    if (!_FacsDS) {
      _FacsDS = new kendo.data.DataSource({
        //type: 'aspnetmvc-ajax',
        transport: {
          read: {
            url: function() {
              if (!_facSearchURL) {
                return _serviceMapping.FacSearch;
              } else {
                return _facSearchURL;
              }
            },
            dataType: "json", // "jsonp" is required for cross-domain requests; use "json" for same-domain requests
            //data: _addressSearchParams,
            cache: true,
          },
          parameterMap: function(options, operation) {
            if (operation === "read") {
              const name = $("#facilityTextSearch").val();
              let tmpIsLoading: boolean = _isLoading;
              if (tmpIsLoading === false && !isModalShown()) {
                tmpIsLoading = true;
              }
              let wktString = "";
              if (filterByMapExtentChecked() && getWKTFromMapExtent) {
                wktString = getWKTFromMapExtent();
              }
              return {
                query: name,
                orgListString: _filterOrgs,
                isLoading: tmpIsLoading,
                wktString: wktString,
              };
            }
          }, //parameterMap
        },
        requestEnd: function() {
          $(".facSearchProcessing").hide();
          $("#facContainer").removeClass("blurredDiv");
        },
        requestStart: function() {
          $(".facSearchProcessing").show();
          $("#facContainer").addClass("blurredDiv");
        },
        change: function(e) {
          const data = this.data();
          //console.log(data.length); // how many records found
          if (data.length === 1) {
            //select first row
            const grid = $("#FacList").data("kendoGrid"),
              row = grid.tbody.find(">tr:not(.k-grouping-row)").eq(0);
            grid.select(row);
          }
        },
        pageSize: 10,
        serverFiltering: false,
        serverPaging: false,
        schema: {
          type: "json",
          data: "Data",
          model: {
            fields: {
              CurrentCultureTranslation: { field: "CurrentCultureTranslation", type: "string" },
              FriendlyOrgName: { field: "FriendlyOrgName", type: "string" },
              ScrubbedFacilityDescription: { field: "ScrubbedFacilityDescription", type: "string" },
              FacilityCategory: { field: "FacilityCategory", type: "string" },
              FacilitySubCategory: { field: "FacilitySubCategory", type: "string" },
              FacilityGUID: { field: "FacilityGUID", type: "guid" },
            },
          },
          total: function(response) {
            response.Data.forEach(function(item, index) {
              if (!item.FacilityGUID) {
                response.splice(index, 1);
              }
            });
            return response.Total;
          },
        },
      });
    }
