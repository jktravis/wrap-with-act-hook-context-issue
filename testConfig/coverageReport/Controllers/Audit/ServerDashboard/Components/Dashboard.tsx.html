import React, { Component } from "react";
import { CardColumns, Button } from "reactstrap";
import { GetServerDashboardList } from "Api/serverDashboardApi";
import ServerCard from "./ServerCard";
import AddServerModal from "./AddServerModal";
import { getTranslations } from "../../../../Api/translationApi";
interface Props {
  baseApi: string;
  translatedFields?: any;
}
 
const requiredTranslatedFields = ["ServerName", "UserName", "Password"];
 
class Dashboard extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    translatedFields: {},
  };
  state = {
    servers: [],
    showAddServerModal: false,
  };
  async componentDidMount() {
    const data = await getTranslations(requiredTranslatedFields);
    this.setState(() => ({ translatedFields: data }));
    this.loadServers();
  }
 
  async loadAddServerModal() {
    this.setState({ showAddServerModal: true });
  }
  handleAddServerModalClose = () => {
    this.setState({
      showAddServerModal: false,
    });
  };
 
  loadServers = async () => {
    const serverListData = await GetServerDashboardList();
    this.setState(() => {
      if (serverListData) {
        return { servers: serverListData };
      } else {
        return { servers: [] };
      }
    });
  };
 
  render() {
    const { servers } = this.state;
    return (
      <div>
        <Button onClick={() => this.loadAddServerModal() as any} color="info">
          <i className="fa fa-edit" />
        </Button>
        <CardColumns>
          {servers.map(server => (
            <ServerCard key={server.ServerDashboardLoginInfoGUID} server={server} />
          ))}
        </CardColumns>
        {this.state.showAddServerModal ? (
          <AddServerModal
            closeModalfn={this.handleAddServerModalClose}
            translatedFields={this.props.translatedFields}
          />
        ) : null}
      </div>
    );
  }
}
export default Dashboard;
