import React, { Component } from "react";
import { AddServerEntry } from "Api/serverDashboardApi";
import {
  Button,
  Form,
  Input,
  ButtonGroup,
  Col,
  Row,
  FormGroup,
  Modal,
  ModalHeader,
  ModalBody,
  Label,
} from "reactstrap";
 
type Props = {
  translatedFields?: any;
  closeModalfn?: Function;
};
 
class AddServerModal extends Component<Props, any> {
  static defaultProps: Partial<Props> = {
    translatedFields: {},
    closeModalfn: Function.prototype,
  };
 
  state = {
    fields: { ServerName: null, UserName: null, Password: false },
    errors: {},
    validationGood: false,
  };
 
  toggle = () => {
    this.props.closeModalfn();
  };
 
  validate = () => {
    const fields: any = this.state.fields;
    const errors: any = {};
    let formIsValid = true;
 
    if (!fields.ServerName) {
      formIsValid = false;
      errors.ServerName = this.props.translatedFields.ServerName + " " + this.props.translatedFields.IsRequired;
    }
 
    if (!fields.UserName) {
      formIsValid = false;
      errors.UserName = this.props.translatedFields.UserName + " " + this.props.translatedFields.IsRequired;
    }
 
    if (!fields.Password) {
      formIsValid = false;
      errors.Password = this.props.translatedFields.Password + " " + this.props.translatedFields.IsRequired;
    }
 
    this.setState({ validationGood: formIsValid });
  };
 
  handleCreateServerDashboard = async event => {
    event.preventDefault();
    const fields: any = this.state.fields;
    try {
      if (this.state.validationGood) {
        await AddServerEntry({ servername: fields.ServerName, username: fields.UserName, password: fields.Password });
        this.toggle();
      } else {
        this.setState({ errors: false });
      }
    } catch (err) {
      const errors: any = {};
      errors.LocatorSerialNumber = err.response.data.ExceptionMessage;
      this.setState({ errors: errors });
      return errors;
    }
  };
 
  render() {
    const fields: any = this.state.fields;
    const errors: any = this.state.errors;
    return (
      <div>
        <Modal isOpen={false}>
          <ModalHeader>{this.props.translatedFields.LocatorLookupTitle}</ModalHeader>
          <ModalBody>
            <Form onSubmit={this.handleCreateServerDashboard}>
              <FormGroup>
                <Label>{this.props.translatedFields.ServerName}</Label>
                <Input type="text" name="ServerName" value={fields.ServerName} />
              </FormGroup>
              <FormGroup>
                <Label>{this.props.translatedFields.UserName}</Label>
                <Input type="text" name="UserName" value={fields.UserName} />
              </FormGroup>
              <FormGroup>
                <Label>{this.props.translatedFields.Password}</Label>
                <Input type="password" name="Password" value={fields.Password} />
              </FormGroup>
              <Row>
                <Col>
                  <ButtonGroup>
                    <Button type="submit" name="Submit" color="primary">
                      {this.props.translatedFields.SubmitButton}
                    </Button>
                    <Button color="warning" onClick={this.toggle}>
                      {this.props.translatedFields.CancelText}
                    </Button>
                  </ButtonGroup>
                </Col>
              </Row>
            </Form>
          </ModalBody>
        </Modal>
      </div>
    );
  }
}
 
export default AddServerModal;
