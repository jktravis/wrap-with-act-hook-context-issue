import React, { Component, createRef } from "react";
import { GetServerDetails, RemoveServerEntry } from "Api/serverDashboardApi";
import { Card, Row, Col, CardBody, CardHeader, Button, CardTitle, CardFooter } from "reactstrap";
import { FaTrashAlt } from "react-icons/fa";
import LoadSpinner from "../../../Tools/Map/Components/LoadSpinner";
 
interface Props {
  server: IServerInfo;
}
interface IServerInfo {
  ServerDashboardLoginInfoGUID: string;
  ServerName: string;
}
 
class ServerCard extends Component<Props, any> {
  state = {
    serverDetails: undefined,
  };
  componentDidMount() {
    this.loadServerDetails();
  }
 
  loadServerDetails = async () => {
    const severDet = await GetServerDetails({
      serverDashboardLoginInfoGUID: this.props.server.ServerDashboardLoginInfoGUID,
    });
    this.setState(() => {
      if (severDet) {
        return { serverDetails: severDet };
      } else {
        return {
          serverDetails: undefined,
        };
      }
    });
  };
 
  removeServerEntry = async () => {
    RemoveServerEntry({ serverDashboardLoginInfoGUID: this.props.server.ServerDashboardLoginInfoGUID });
  };
 
  render() {
    const { server } = this.props;
    const { serverDetails } = this.state;
    if (!server) {
      return null;
    } else if (!serverDetails) {
      return (
        <div>
          <Card>
            <CardHeader>
              <label>{server.ServerName}</label>
            </CardHeader>
            <CardBody>
              <LoadSpinner />
            </CardBody>
          </Card>
        </div>
      );
    } else {
      return (
        <Card>
          <CardHeader>
            <Row>
              <Col xs="auto">
                <a
                  href={`${server.ServerName}`}
                  className="btn btn-link"
                  target="_blank"
                  rel="noopener noreferrer"
                  title="Click here to visit server"
                >
                  {serverDetails.ServerDisplay}
                </a>
              </Col>
              <Col />
              <Col xs="auto">
                <Button color="danger" onClick={this.removeServerEntry}>
                  <FaTrashAlt />
                </Button>
              </Col>
            </Row>
          </CardHeader>
          <CardBody className="medCardContent">
            {serverDetails.Orgs &&
              serverDetails.Orgs.map(org => (
                <dl key={`${server.ServerName + org.OrganizationName}`}>
                  <dt className="badge badge-info">{org.OrganizationName}</dt>
                  <dd>
                    <label>Incident Count:</label>
                    {org.IncidentCount}
                  </dd>
                  <dd>
                    <a
                      title="Click here to navigate to Incident"
                      className="btn btn-link"
                      target="_blank"
                      rel="noopener noreferrer"
                      href={`${server.ServerName +
                        "/Incident/IncidentHome/Index/" +
                        org.MostRecentIncident.ProjectGUID}`}
                    >
                      <label>Most Recent Incident:</label>
                    </a>
                    <time data-format="lll" dateTime="{org.UnixMostRecentIncident}">
                      {org.MostRecentIncident.DateCreated}
                    </time>
                  </dd>
                </dl>
              ))}
          </CardBody>
          {serverDetails.MostRecentServerException && (
            <Card color="dark" className="text-white m-1 p-1">
              <CardBody>
                <CardTitle>Exceptions</CardTitle>
                <dl>
                  <dt>
                    <a
                      title="Click here to view exceptions"
                      className={`${serverDetails.ServerExceptionLinkClass}`}
                      target="_blank"
                      rel="noopener noreferrer"
                      href={`${server.ServerName + "/Audit/ServerException"}`}
                    >
                      <label>Sever Exceptions</label>
                    </a>
                  </dt>
                  <dd>
                    <label># of Exceptions</label>
                    {serverDetails.ServerExceptions.length}
                  </dd>
                  <dd>
                    <label>Most Recent Exception</label>{" "}
                    <time data-format="lll" dateTime="{serverDetails.UnixMostRecentServerExceptionTime}">
                      {serverDetails.MostRecentServerException.DateCreated}
                    </time>
                  </dd>
                </dl>
              </CardBody>
            </Card>
          )}
          <CardFooter>
            {serverDetails.ServerMetaData && (
              <em className="text-right">Server Reports Time as {serverDetails.ServerMetaData.ServerTime}</em>
            )}
          </CardFooter>
        </Card>
      );
    }
  }
}
export default ServerCard;
