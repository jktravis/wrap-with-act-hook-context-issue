namespace dataMaint {
  interface IGlobalPrompt {
    OrganizationTitle: string;
    UserTitle: string;
    SubOrgTitle: string;
    FacilityTitle: string;
  }
  let _globalPrompts: IGlobalPrompt;
  export function init(globalPrompts) {
    OrgCheckBox();
    UserCheckBox();
    FacilityCheckBox();
    SubOrgCheckBox();
    _globalPrompts = globalPrompts;
  }
 
  export function emailSelectedDiv() {
    const selected = [];
    const divName = getGridName();
    $("#" + divName + " input:checked").each(function() {
      const tmpName = $(this).attr("data-key");
      const tmpType = $(this).attr("data-type");
      const tmpEmail = $(this).attr("data-email");
      selected.push({
        PrimaryKey: tmpName,
        InformationType: tmpType,
        EmailAddress: tmpEmail,
      });
    });
    const reqs = JSON.stringify(selected);
    //now send these IDs to be emailed
    $Ajax.ajaxAntiForgery({
      url: currentUrlPrefix() + "Audit/DataMaintenance/EmailSelected/", //i chose account here, will probably be something else.  this method doesn't exist
      data: {
        emailRequests: reqs,
      },
      type: "POST",
      success: function(response) {
        if (response.success === false) {
          $.notify({ message: response.Message }, { type: "danger" });
        } else {
          //refresh the current grid?  or just send a message to the user that it was done?
          $.notify({ message: "Messages Sent" }, { type: "info" });
        }
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }
 
  function getGridName() {
    const tabStrip = $("#globalTab").data("kendoTabStrip");
    if (tabStrip != null) {
      const selectedTab = tabStrip.select()[0];
      let divName = "";
      switch (
        (<any>selectedTab).innerText //this is a good idea, but it will break if we localize these headings.  can you think of another solution?
      ) {
        case _globalPrompts.OrganizationTitle:
          divName = "orgGrid";
          break;
        case _globalPrompts.UserTitle:
          divName = "usrGrid";
          break;
        case _globalPrompts.FacilityTitle:
          divName = "facGrid";
          break;
        case _globalPrompts.SubOrgTitle:
          divName = "subOrgGrid";
          break;
        default:
          console.log((<any>selectedTab).innerText);
          break;
      }
      return divName;
    }
  }
 
  function OrgCheckBox() {
    $("#header-Orgchb").change(selectAllCheckBoxes);
  }
 
  function UserCheckBox() {
    $("#header-Userchb").change(selectAllCheckBoxes);
  }
 
  function FacilityCheckBox() {
    $("#header-facChb").change(selectAllCheckBoxes);
  }
 
  function SubOrgCheckBox() {
    $("#header-subOrgChb").change(selectAllCheckBoxes);
  }
 
  function selectAllCheckBoxes(ev) {
    const foundGridName = getGridName();
    const checked = (<any>ev.target).checked;
    //selectedTab
    $("#" + foundGridName + " .checkbox").each(function(idx, item) {
      $(item).prop("checked", checked);
    });
  }
