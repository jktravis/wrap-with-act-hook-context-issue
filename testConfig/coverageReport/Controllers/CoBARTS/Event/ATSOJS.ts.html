namespace Dashboard_ATSO {
  interface IATSOScope extends ng.IScope {
    atsoDataSourceL: kendo.data.DataSource;
    atsoDataSourceM: kendo.data.DataSource;
  }
  export function saveATSO(baseurl) {
    const title = $("#Title").val();
    const sev = $("#Severity").val();
    const stat = $("#Status").val();
    const det = $("#Details").val();
    let latval;
    if ($("#Lat")) {
      latval = $("#Lat").val();
    }
    let lonval;
    if ($("#Lon")) {
      lonval = $("#Lon").val();
    }
    if (title !== null && title.length > 0) {
      const url = baseurl; //+ "?Title=" + title + "&Severity=" + sev + "&Status=" + stat + "&Details=" + det;
 
      $Ajax.ajaxAntiForgery({
        url: url,
        type: "POST",
        data: {
          Lat: latval,
          Lon: lonval,
          Title: title,
          Severity: sev,
          Status: stat,
          Details: det,
        },
        success: function(response) {
          if (response !== null) {
            Dashboard.jqueryclosemodal();
            //add refresh to RR window here if you like
            //refreshAllATSO(true);
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }
  function refreshAllATSO() {
    if (angular.element("#atsocontrollerID").length) {
      (<IATSOScope>angular.element("#atsocontrollerID").scope()).atsoDataSourceL.read();
    }
    //would like to figure out a way to make this repository reside in the dashboard controller somehow, so that i
    //only need to refresh this once.
    if (angular.element("#atsocontrollerIDM").length) {
      (<IATSOScope>angular.element("#atsocontrollerIDM").scope()).atsoDataSourceM.read();
    }
  }
 
  angular.module("cobraapp").factory("atsoRepository", function() {
    return {
      getDataSourceM: function(hostURL, projectGUID) {
        return new kendo.data.DataSource({
          transport: {
            read: {
              url: hostURL + "/COBARTS/Event/_GetEvListJSON/" + projectGUID,
            },
          },
          schema: {
            model: {
              id: "LogEntryGUID",
              fields: {
                LogEntryGUID: { editable: false },
                StatusValue: { editable: false },
                COBARTSType: { editable: false },
                StatusClass: { editable: false },
                COBARTSImage: { editable: false },
                Description: { editable: false },
                FormattedDateCreated: { type: "date", editable: false },
                Comments: { editable: false },
                Coordinates: { editable: false },
                ZoneName: { editable: false },
              },
            },
          },
        });
      },
      getDataSourceL: function(hostURL, projectGUID) {
        return new kendo.data.DataSource({
          transport: {
            read: {
              url: hostURL + "/COBARTS/Event/_GetEvListJSON/" + projectGUID,
            },
          },
          pageSize: 7,
          schema: {
            model: {
              id: "LogEntryGUID",
              fields: {
                LogEntryGUID: { editable: false },
                StatusValue: { editable: false },
                StatusClass: { editable: false },
                COBARTSImage: { editable: false },
                COBARTSType: { editable: false },
                Description: { editable: false },
                FormattedDateCreated: { type: "date", editable: false },
                Comments: { editable: false },
                Coordinates: { editable: false },
                ZoneName: { editable: false },
              },
            },
          },
        });
      },
    };
  });
 
  angular.module("cobraapp").controller("atsocontroller", function($scope, atsoRepository) {

        toolname === "OtherTool" ||
        toolname === "ContamTool" ||
