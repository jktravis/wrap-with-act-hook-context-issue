namespace zoneFunctions {
  let typeChangeURL = null;
 
  export function init(typeURL: string) {
    typeChangeURL = typeURL;
    $("#ZoneForm").submit(function(e) {
      //validation check
      if (!Validate()) {
        e.preventDefault();
        return false;
      } else {
        Main.blockit($("#ZoneForm"));
      }
    });
  }
 
  function Validate() {
    let errResponse = {
      Messages: [],
    };
    const guid = $("#ZoneTypeGUID").val();
    if (guid === Guid.emptyGuidString || guid === "" || guid === undefined) {
      Main.markObjectAsError($("#ZoneTypeGUID"));
      errResponse.Messages.push("Must Select a Zone Type");
    }
 
    if (errResponse.Messages.length > 0) {
      Main.handleValErr(null, errResponse);
      return false;
    } else {
      return true;
    }
  }
 
  export function typechanged() {
    const inctypID = $("#ZoneTypeGUID").val();
    const urlToPost = typeChangeURL;
 
    $.ajax({
      url: urlToPost,
      data: {
        SelectedIncType: inctypID,
      },
      type: "GET",
      success: function(response) {
        $("#psIncTypes").html(response);
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }
}
