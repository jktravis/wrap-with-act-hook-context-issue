namespace Dashboard_ESF {
  interface IESFScope extends ng.IScope {
    esfDataSourceL: kendo.data.DataSource;
    esfDataSourceM: kendo.data.DataSource;
  }
  export function saveESF(baseurl) {
    var esftype = $("#ESFType").val();
    var objname = $("#ObjectName").val();
    var statval = $("#StatusValue").val();
    var znguid = $("#ZoneGuid").val();
    var comm = $("#Comments").val();
    var latval;
    if ($("#Lat")) {
      latval = $("#Lat").val();
    }
    var lonval;
    if ($("#Lon")) {
      lonval = $("#Lon").val();
    }
    var logguid = $("#LogEntryGUID").val();
    var statguid = $("#StaticGUID").val();
    if (objname !== null && objname.length > 0) {
      var url = baseurl;
 
      $Ajax.ajaxAntiForgery({
        url: url,
        type: "POST",
        data: {
          Lat: latval,
          Lon: lonval,
          ESFType: esftype,
          ObjectName: objname,
          StatusValue: statval,
          ZoneGUID: znguid,
          Comments: comm,
          LogEntryGUID: logguid,
          StaticGUID: statguid,
        },
        success: function(response) {
          if (response !== null) {
            Dashboard.jqueryclosemodal();
            //add refresh to RR window here if you like
            //refreshAllESF(true);
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }
  function refreshAllESF() {
    if (angular.element("#esfControllerID").length) {
      (<IESFScope>angular.element("#esfControllerID").scope()).esfDataSourceL.read();
    }
    //would like to figure out a way to make this repository reside in the dashboard controller somehow, so that i
    //only need to refresh this once.
    if (angular.element("#esfControllerIDM").length) {
      (<IESFScope>angular.element("#esfControllerIDM").scope()).esfDataSourceM.read();
    }
  }
 
  angular.module("cobraapp").factory("esfRepository", function() {
    return {
      getDataSourceM: function(hostURL, projectGUID) {
        return new kendo.data.DataSource({
          transport: {
            read: {
              url: hostURL + "/ESF/ESF/_GetEvListJSON/" + projectGUID,
            },
          },
          schema: {
            model: {
              id: "LogEntryGUID",
              fields: {
                LogEntryGUID: { editable: false },
                ObjectName: { editable: false },
                StatusValue: { editable: false },
                ESFTypeString: { editable: false },
                StatusClass: { editable: false },
                Description: { editable: false },
                FormattedDateCreated: { type: "date", editable: false },
                Comments: { editable: false },
                Coordinates: { editable: false },
                ZoneName: { editable: false },
              },
            },
          },
        });
      },
      getDataSourceL: function(hostURL, projectGUID) {
        return new kendo.data.DataSource({
          transport: {
            read: {
              url: hostURL + "/ESF/ESF/_GetEvListJSON/" + projectGUID,
            },
          },
          pageSize: 7,
          schema: {
            model: {
              id: "LogEntryGUID",
              fields: {
                LogEntryGUID: { editable: false },
                ObjectName: { editable: false },
                StatusValue: { editable: false },
                StatusClass: { editable: false },
                ESFTypeString: { editable: false },
                Description: { editable: false },
                FormattedDateCreated: { type: "date", editable: false },
                Comments: { editable: false },
                Coordinates: { editable: false },
                ZoneName: { editable: false },
              },
            },
          },
        });
      },
    };
  });
