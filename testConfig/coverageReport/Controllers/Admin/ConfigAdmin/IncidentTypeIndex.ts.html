  function clearValues() {
    $("#txtSelectedIncidentType").val("");

    $("#txtde").val("");
    $("#txtnl").val("");
    $("#txtfr").val("");
    $("#txtValSummIncidentType").text("");
  }
 
  export function addIncidentType() {
    clearValues();
    $("#IncidentTypeModal").modal("show");
  }
  function updateIncidentType(key, envalue, devalue, frvalue, nlvalue) {
    $("#txtSelectedIncidentType").val(key);
    $("#txten").val(envalue);
    $("#txtde").val(devalue);
    $("#txtfr").val(frvalue);
    $("#txtnl").val(nlvalue);
    $("#IncidentTypeModal").modal("show");
  }
  export function saveIncidentType(saveURL) {
    const txten = $("#txten").val();
    const txtfr = $("#txtfr").val();
    const txtde = $("#txtde").val();
    const txtnl = $("#txtnl").val();
    const txtSelCat = $("#txtSelectedIncidentType").val();
    if (txten.length > 0) {
      $Ajax.ajaxAntiForgery({
        url: saveURL,
        type: "POST",
        data: {
          enText: txten,
          deText: txtde,
          nlText: txtnl,
          frText: txtfr,
          SelectedIncidentType: txtSelCat,
        },
        success: function(response) {
          if (response.success === false) {
            $("#txtValSummIncidentType").text(response.Message);
          } else {
            reloadIncidentTypeDD();
            clearValues();
            $("#IncidentTypeModal").modal("hide");
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }
