namespace Hospital_Validate {
  //for this method, you just need to pass the form and the warning
  interface IGlobalPrompt {
    GenericBedsCannotBeZero: string;
  }
  let _globalPrompts: IGlobalPrompt;
 
  export function init(globalPrompts) {
    _globalPrompts = globalPrompts;
 
    $("form").on("submit", function(e) {
      e.preventDefault();
      //populate hidden fields
      const primContacts = $("#BedReservationContactsDiv").data("userlist");
      $("#BedReservationContactsList").val(JSON.stringify(primContacts));
 
      if (
        $("#NumberOfGeneralBeds")
          .data("kendoNumericTextBox")
          .value() <= 0
      ) {
        $.notify({ message: _globalPrompts.GenericBedsCannotBeZero }, { type: "danger" });
        return false;
      }
 
      this.submit();
    });
  }
}
