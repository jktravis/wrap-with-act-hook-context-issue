  function validateForm(e): Boolean {
    const errResponse = {
      Messages: [],
    };
    let latVal, lonVal;
    const kendoLat = $("#Lat").data("kendoNumericTextBox");
    const kendoLon = $("#Lon").data("kendoNumericTextBox");
 
    if (kendoLat !== null && kendoLat !== undefined) {
      latVal = kendoLat.value(); // to get the non-localized value
    }
    if (kendoLon !== null && kendoLon !== undefined) {
      lonVal = kendoLon.value(); // to get the non-localized value
    }
    if (latVal !== null && latVal !== "" && lonVal !== null && lonVal !== "") {
      Main.clearObjectErrorStatus($("#Lat"));
      Main.clearObjectErrorStatus($("#Lon"));
    } else {
      Main.markObjectAsError($("#Lat"));
      Main.markObjectAsError($("#Lon"));
      errResponse.Messages.push(_globalPrompts.MissingLatLonMessage);
 
      if ($("#OrganizationName").length > 0) {
        Main.clearObjectErrorStatus($("#OrganizationName"));
      } else {
        Main.markObjectAsError($("#OrganizationName"));
        errResponse.Messages.push(_globalPrompts.MissingOrgNameMessage);
      }
    }
 
    const CrisisHotline = $("#CrisisHotline");
    const NonEmergencyPhone = $("#NonEmergencyPhone");
    const FaxNumber = $("#FaxNumber");
 
    var phoneNumberValue = (CrisisHotline[0] as HTMLInputElement).value;
    if (phoneNumberValue.length > 0 && !Main.validatePhoneNumber(phoneNumberValue)) {
      Main.markObjectAsError(CrisisHotline);
      errResponse.Messages.push(_globalPrompts.CrisisHotlineMessage);
    } else {
      Main.clearObjectErrorStatus(CrisisHotline);
    }
 
    phoneNumberValue = (NonEmergencyPhone[0] as HTMLInputElement).value;
    if (phoneNumberValue.length > 0 && !Main.validatePhoneNumber(phoneNumberValue)) {
      Main.markObjectAsError(NonEmergencyPhone);
      errResponse.Messages.push(_globalPrompts.NonEmergencyNumberMessage);
    } else {
      Main.clearObjectErrorStatus(NonEmergencyPhone);
    }
 
    phoneNumberValue = (FaxNumber[0] as HTMLInputElement).value;
    if (phoneNumberValue.length > 0 && !Main.validatePhoneNumber(phoneNumberValue)) {
      Main.markObjectAsError(FaxNumber);
      errResponse.Messages.push(_globalPrompts.FaxNumberMessage);
    } else {
      Main.clearObjectErrorStatus(FaxNumber);
    }
 
    if (errResponse.Messages.length > 0) {
      Main.silentHandleValErr(null, errResponse);
      return false;
    } else {
      return true;
    }
  }
 
  let exportFlag = false;
  export function OrgExcelExport(e) {
    if (!exportFlag) {
      //Show the column
      e.sender.showColumn(8);
      //Prevent export, as it initially fired with the column hidden
      e.preventDefault();
      //Update for the new call we re about to make
      exportFlag = true;
      //Call the save again, which will hit this function, fail the if statement and cascade down to hide the column and finish the export.
      setTimeout(function() {
        e.sender.saveAsExcel();
      });
    } else {
      e.sender.hideColumn(8);
      exportFlag = false;
    }
  }

  function HandleItemRemoval(item, currentlySelected) {
    if (item.hasChildren) {
      for (let i = 0; i < item.items.length; i++) {
        currentlySelected = HandleItemRemoval(item.items[i], currentlySelected);
      }
    }
    currentlySelected = currentlySelected.filter(function(value, index, arr) {
      return value !== item.value;
    });
    return currentlySelected;
  }
 
  function HandleItemAddition(item, currentlySelected) {
    if (item.hasChildren) {
      for (let i = 0; i < item.items.length; i++) {
        currentlySelected = HandleItemAddition(item.items[i], currentlySelected);
      }
    }
    if (!currentlySelected.includes(item.value)) {
      currentlySelected.push(item.value);
    }
    return currentlySelected;
  }
 
  export function buildBelgiumAreaDropDownTree(inputID, url) {
    $Ajax.ajaxAntiForgery({
      url: url,
      type: "GET",
      success: function(response) {
        let onSelect = function(e) {
          //Prevent the addition of this item
          e.preventDefault();
          //Get the dataItem instead of just the html element.
          let val = e.sender.dataItem(e.node);
          //Get the list of currently active items in the tree
          let currentValues = $("#" + inputID)
            .data("kendoDropDownTree")
            .value();
          //If the Item to be added is already in the list, then that means it must
          //need to be removed
          if (currentValues.includes(val.value)) {
            currentValues = HandleItemRemoval(val, currentValues);
          } else {
            currentValues = HandleItemAddition(val, currentValues);
          }
          //Updates the selected items in the Kendo DropDownTree and the hidden MVC for.
          $("#" + inputID)
            .data("kendoDropDownTree")
            .value(currentValues);
          $("#AreasOfResponsibilityAsString").val(currentValues);
        };
 
        let onChange = function(e) {
          let currentValues = $("#" + inputID)
            .data("kendoDropDownTree")
            .value();
          $("#AreasOfResponsibilityAsString").val(currentValues);
        };
        if (response && response.length > 0) {
          $("#" + inputID).kendoDropDownTree({
            placeholder: "Select...",
            checkboxes: true,
            checkAll: true,
            autoClose: false,
            filter: "startswith",
            dataSource: response,
            select: onSelect,
            change: onChange,
          });
 
          $(".k-checkbox-wrapper").hide();
        }
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }
