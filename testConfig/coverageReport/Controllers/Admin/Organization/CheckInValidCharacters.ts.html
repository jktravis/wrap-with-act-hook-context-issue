//import { Main } from '../../../basicCoBRAScript';
namespace CheckInValidCharacters {
  export function init() {
    $("#OrganizationName").change(function() {
      checkForInvalidCharacters();
    });
    $("#warningSpan").hide();
    $("#OrganizationName").focusin();
  }
  function checkForInvalidCharacters() {
    const orgInput = $("#OrganizationName");
    let isValid = true;
 
    isValid = orgInput.val().match(/[:*&+]/) ? false : true;
 
    //if (orgInput.val().includes(':')) {
    //    isValid = false;
    //}
    //if (orgInput.val().includes('*')) {
    //    isValid = false;
    //}
    //if (orgInput.val().includes('&')) {
    //    isValid = false;
    //}
    //if (orgInput.val().includes('+')) {
    //    isValid = false;
    //}
    if (isValid === false) {
      $("#saveButton").prop("disabled", true);
      $("#warningSpan").show();
    }
    if (isValid === true) {
      $("#saveButton").prop("disabled", false);
      $("#warningSpan").hide();
    }
  }
}
