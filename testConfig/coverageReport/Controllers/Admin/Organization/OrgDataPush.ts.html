namespace orgDataPush {
  declare function pushToSelectedOrgs(tempateOrg: string, destOrgs: any);
  export const cbraModule = angular.module(
    "cobraapp",
    ["ngRoute", "ngAnimate", "ngSanitize", "kendo.directives"],
    function($httpProvider) {
      //from http://victorblog.com/2012/12/20/make-angularjs-http-service-behave-like-jquery-ajax/
      // Use x-www-form-urlencoded Content-Type
      $httpProvider.defaults.headers.post["Content-Type"] = "application/x-www-form-urlencoded;charset=utf-8";
 
      // Override $http service's default transformRequest
      $httpProvider.defaults.transformRequest = [
        function(data) {
          /**
           * The workhorse; converts an object to x-www-form-urlencoded serialization.
           * @param {Object} obj
           * @return {String}
           */
          const param = function(obj) {
            let query = "";
            let name, value, fullSubName, subName, subValue, innerObj, i;
 
            for (name in obj) {
              if (obj.hasOwnProperty(name)) {
                value = obj[name];
 
                if (value instanceof Array) {
                  for (i = 0; i < value.length; ++i) {
                    subValue = value[i];
                    fullSubName = name + "[" + i + "]";
                    innerObj = {};
                    innerObj[fullSubName] = subValue;
                    query += param(innerObj) + "&";
                  }
                } else if (value instanceof Object) {
                  for (subName in value) {
                    if (value.hasOwnProperty(subName)) {
                      subValue = value[subName];
                      fullSubName = name + "[" + subName + "]";
                      innerObj = {};
                      innerObj[fullSubName] = subValue;
                      query += param(innerObj) + "&";
                    }
                  }
                } else if (value !== undefined && value !== null) {
                  query += encodeURIComponent(name) + "=" + encodeURIComponent(value) + "&";
                }
              }
            }
            return query.length ? query.substr(0, query.length - 1) : query;
          };
 
          return angular.isObject(data) && String(data) !== "[object File]" ? param(data) : data;
        },
      ];
    },
  );

    $scope.tmpLstDataSource = orgDataPushRepository.getOrganizations();

      dataSource: $scope.tmpLstDataSource,
      template:
        "<div class='listviewtmp'><dl><dt><time data-format='lll' datetime='{{dataItem.UnixDateCreated}}'>{{angDateFormatter(dataItem.FormattedDateCreated)}}</time>:<span></dt><dd>{{dataItem.CurrentCultureTranslation}}</dd></dl></div>",
      altTemplate:
        "<div class='listviewalttmp'><dl><dt><time data-format='lll' datetime='{{dataItem.UnixDateCreated}}'>{{angDateFormatter(dataItem.FormattedDateCreated)}}</time>:<span></dt><dd>{{dataItem.CurrentCultureTranslation}}</dd></dl></div>",
      dataBinding: function() {
        //do nothing
      },
      change: function(e) {
        //const sel = $scope.grplsto.select();
        const index = this.select().index(),
          dataItem = this.dataSource.view()[index];
        $scope.TemplateOrg = dataItem.OrganizationName;
      },
      dataBound: function() {
        //Main.qtipSection('grplst');
      },
    };
 
    $scope.destLstDataSource = orgDataPushRepository.getOrganizations();
    $scope.destGridOptions = {
      selectable: false,
      pageable: false,
      dataSource: $scope.destLstDataSource,
      template:
        "<div class='listviewtmp'><div class='form-inline'><input type='checkbox' class='checkbox' id='{{dataItem.OrganizationName}}' value='{{dataItem.OrganizationName}}' ng-checked='selection.indexOf(dataItem.OrganizationName) > -1' ng-click='toggleSelection(dataItem.OrganizationName)'> {{dataItem.CurrentCultureTranslation}}</div></div>",
      altTemplate:
        "<div class='listviewalttmp'><div class='form-inline'><input type='checkbox' class='checkbox' id='{{dataItem.OrganizationName}}' value='{{dataItem.OrganizationName}}' ng-checked='selection.indexOf(dataItem.OrganizationName) > -1' ng-click='toggleSelection(dataItem.OrganizationName)'> {{dataItem.CurrentCultureTranslation}}</div></div>",
      dataBinding: function() {
        //do nothing
      },
      change: function(e) {
        //const sel = $scope.grplsto.select();
      },
      dataBound: function() {
        //Main.qtipSection('grplst');
      },
    };
    $scope.selection = [];
    $scope.toggleSelection = function(orgName: string) {
      const idx = $scope.selection.indexOf(orgName);
      // is currently selected
      if (idx > -1) {
        $scope.selection.splice(idx, 1);
      } else {
        // is newly selected
        $scope.selection.push(orgName);
      }
    };
    $scope.selectAll = function() {
      if ($scope.selectAllChecked === true) {
        const datasourcedata = $scope.tmpLstDataSource.data();
 
        for (let i = 0; i < datasourcedata.length; i++) {
          const dataitem = datasourcedata[i].OrganizationName;
          const idx = $scope.selection.indexOf(dataitem);
          if (idx > -1) {
            //do nothing
          } else {
            $scope.selection.push(dataitem);
          }
        }
      } else {
        $scope.selection = [];
      }
    };
    $scope.angDateFormatter = function(datetouse) {
      return kendo.toString(datetouse, "G");
    };
    $scope.saveOrgChanges = function() {
      pushToSelectedOrgs($scope.TemplateOrg, $scope.selection);
    };
  });
