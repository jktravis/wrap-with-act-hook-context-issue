import { forEach } from "ramda";
/* global
Main,
$Ajax,
*/
 
interface IGlobalPrompt {
  firstNameValidate: string;
  lastNameValidate: string;
  emailAddressValidate: string;
  PhoneNumberMessage: string;
  MobileNumberMessage: string;
  contactUrl: string;
  returnURL: string;
  markAsIncorrectUrl: string;
}
export class Phonebook {
  private _globalPrompts: IGlobalPrompt;
  constructor(prompts: IGlobalPrompt) {
    this._globalPrompts = prompts;
  }
 
  private checkField;
 
  private validateForm(): boolean {
    const errResponse = {
      Messages: [],
    };
 
    const fieldsToCheck = [
      {
        field: $("#FirstName"),
        validation: this._globalPrompts.firstNameValidate,
      },
      {
        field: $("#LastName"),
        validation: this._globalPrompts.lastNameValidate,
      },
      {
        field: $("#EmailAddress"),
        validation: this._globalPrompts.emailAddressValidate,
      },
    ];
 
    forEach(inputData => {
      const currentFieldCheck = inputData.field.val();
      if (!currentFieldCheck || !currentFieldCheck.length) {
        Main.markObjectAsError(inputData.field);
        errResponse.Messages.push(inputData.validation);
      } else {
        Main.clearObjectErrorStatus(inputData.field);
      }
    }, fieldsToCheck);
 
    const phonesToCheck = [
      {
        field: $("#PhoneNum"),
        validation: this._globalPrompts.PhoneNumberMessage,
      },
      //{
      //  field: $("#MobileNum"),
      //  validation: this._globalPrompts.MobileNumberMessage,
      //},
    ];
 
    forEach(inputData => {
      const currentFieldCheck = inputData.field.val();
      if (currentFieldCheck && currentFieldCheck.length > 0 && Main.validatePhoneNumber(currentFieldCheck)) {
        Main.clearObjectErrorStatus(inputData.field);
      } else {
        Main.markObjectAsError(inputData.field);
        errResponse.Messages.push(inputData.validation);
      }
    }, phonesToCheck);
 
    if (errResponse.Messages.length > 0) {
      Main.silentHandleValErr(null, errResponse);
      return false;
    } else {
      return true;
    }
  }
 
  public handleOrgReturn(orgObj, sourceButton) {
    let val = "";
    if (orgObj !== null) {
      val = orgObj.OrganizationName;
    }
    $("#OrganizationName").val(val);
  }
 
  public handleUserReturn(userObj, sourceButton) {
    let val = "";
    if (userObj !== null) {
      if (userObj.UserGUID) {
        val = userObj.UserGUID;
      } else {
        val = userObj.CoBRAUserGUID;
      }
    }
    $("#DataOwner2GUID").val(val);
  }
 
  public handleAddressReturn(selectedAddress) {
    let streetText = selectedAddress.HouseNumber + " " + selectedAddress.StreetName;
    let country = "";
    if (Main.serverBuildType === "Belgium") {
      streetText = selectedAddress.StreetName + " " + selectedAddress.HouseNumber;
      country = "Belgium";
    }
 
    $("#WorkAddress_Street1").val($.trim(streetText));
    $("#WorkAddress_City").val($.trim(selectedAddress.City));
    $("#WorkAddress_Zip_PostalCode").val($.trim(selectedAddress.ZipCode));
 
    $("#WorkAddress_Country")
      .data("kendoDropDownList")
      .text($.trim(country));
    $("#WorkAddress_Country")
      .data("kendoDropDownList")
      .trigger("change");
 
    $("#WorkAddress_State_Province")
      .data("kendoComboBox")
      .text($.trim(selectedAddress.State));
    $("#WorkAddress_State_Province")
      .data("kendoComboBox")
      .trigger("change");
  }
 
  public filterStateListWork() {
    return {
      CountryCode: $("WorkAddress_Country").val(),
    };
  }
 
  public SaveContact() {
    $(this)
      .find("input[type=submit]")
      .prop("disabled", true);
    if (this.validateForm()) {
      //good, send off
      const contact = {
        FirstName: $("#FirstName").val(),
        LastName: $("#LastName").val(),
        EmailAddress: $("#EmailAddress").val(),
        PhoneNum: $("#PhoneNum").val(),
        MobileNum: $("#MobileNum").val(),
        OrganizationName: $("#OrganizationName").val(),
        DataOwner2GUID: $("#DataOwner2GUID").val(),
        PhonebookContactGUID: $("#PhonebookContactGUID").val(),
        CreatedByGUID: $("#CreatedByGUID").val(), //this is probably a bad idea honestly.
        WorkAddress: {
          Street1: $("#WorkAddress_Street1").val(),
          Street2: $("#WorkAddress_Street2").val(),
          City: $("#WorkAddress_City").val(),
          State_Province: $("#WorkAddress_State_Province") // eslint-disable-line @typescript-eslint/camelcase
            .data("kendoComboBox")
            .value(),
          Zip_PostalCode: $("#WorkAddress_Zip_PostalCode").val(), // eslint-disable-line  @typescript-eslint/camelcase
          Country: $("#WorkAddress_Country")
            .data("kendoDropDownList")
            .value(),
          County_Region: $("#WorkAddress_County_Region").val(), // eslint-disable-line  @typescript-eslint/camelcase
          Lat: "",
          Lon: "",
          What3Words: "",
        },
      };
      const returnUrl = this._globalPrompts.returnURL;
      $Ajax.ajaxAntiForgery({
        url: this._globalPrompts.contactUrl,
        data: JSON.stringify(contact),
        contentType: "application/json",
        type: "POST",
        success: function() {
          window.location.href = returnUrl;
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    } else {
      $(this)
        .find("input[type=submit]")
        .prop("disabled", false);
    }
  }
 
  public RemoveContact(contactGUID) {
    if (contactGUID) {
      $Ajax.ajaxAntiForgery({
        url: this._globalPrompts.contactUrl + "?PhonebookContactGUID=" + contactGUID,
        type: "DELETE",
        success: function() {
          $("#phnlistgrid")
            .data("kendoGrid")
            .dataSource.read();
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }
  public MarkAsIncorrect(contactGUID) {
    if (contactGUID) {
      const returnUrl = this._globalPrompts.returnURL;
      $Ajax.ajaxAntiForgery({
        url: this._globalPrompts.markAsIncorrectUrl + "?PhonebookContactGUID=" + contactGUID,
        type: "POST",
        success: function() {
          window.location.href = returnUrl;
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }
}
export default Phonebook;
