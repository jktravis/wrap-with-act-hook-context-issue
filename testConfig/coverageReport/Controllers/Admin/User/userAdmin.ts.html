  export function userDelete(userGUID, deleteConfirm, urlToPost) {
    const result = confirm(deleteConfirm);
    if (result) {
      $Ajax.ajaxAntiForgery({
        url: urlToPost,
        type: "POST",
        data: {
          CoBRAUserGUID: userGUID,
        },
        success: function(response) {
          if (response.success === true) {
            if ($("#usrlistgrid").length > 0) {
              $("#usrlistgrid")
                .data("kendoGrid")
                .dataSource.read();
            }
            if ($("#usrgrid").length > 0) {
              $("#usrgrid")
                .data("kendoGrid")
                .dataSource.read();
            }
          } else {
            $.notify({ message: response.Message }, { type: "danger" });
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }
