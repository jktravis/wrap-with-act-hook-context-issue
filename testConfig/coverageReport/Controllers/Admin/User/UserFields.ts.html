  export function addfunctionalAdherence() {
    //Set this field to 0, because it is not an update.
    //0 == Creating new functional adherence
    //1 == updating a functional adherence.
    //This lets us distinguish between the two since they both call saveFunctionalAdherence function
    $("#txtIsUpdateAdherence").val(0);
    clearfunctionalAdherenceValues();
    $("#functionalAdherenceModal").modal("show");
  }
 
  export function updatefunctionalAdherence() {
    const dropdownlist = $("#drpFunctionalAdherence").data("kendoDropDownList");
    const temp = dropdownlist.dataItem();
 
    if (temp.CurrentCultureTranslation !== "") {
      $("#txtenfunctionalAdherence").val(temp.enValue);
      $("#txtdefunctionalAdherence").val(temp.deValue);
      $("#txtfrfunctionalAdherence").val(temp.frValue);
      $("#txtnlfunctionalAdherence").val(temp.nlValue);
      $("#txtIsUpdateAdherence").val(1);
      $("#functionalAdherenceModal").modal("show");
    }
  }
  export function savefunctionalAdherence(saveURL) {
    $("#AdherenceModalSaveBtn").attr("disabled", "disabled");
    const txten = $("#txtenfunctionalAdherence").val();
    const txtfr = $("#txtfrfunctionalAdherence").val();
    const txtde = $("#txtdefunctionalAdherence").val();
    const txtnl = $("#txtnlfunctionalAdherence").val();
    //const txtSel = $("#txtSelectedfunctionalAdherence").val();
    let txtSel;
    //If this value is 1, that means we are updating a functional adherence and need to overwrite its information
    //using its guid.
    if (Number($("#txtIsUpdateAdherence").val()) === 1) {
      txtSel = $("#drpFunctionalAdherence")
        .data("kendoDropDownList")
        .dataItem().MultiLanguageCategoryGUID;
    } else {
      //If the value is 0, that means we are creating a new functional adherence and need to
      //"Overwrite" the non-existent functional adherence with its empty guid.
      txtSel = Guid.empty();
    }
 
    if (txten.length > 0) {
      $Ajax.ajaxAntiForgery({
        url: saveURL,
        type: "POST",
        data: {
          enText: txten,
          deText: txtde,
          nlText: txtnl,
          frText: txtfr,
          SelectedfunctionalAdherence: txtSel,
        },
        success: function(response) {
          if (response.success === false) {
            $("#txtValSummfunctionalAdherence").text(response.Message);
          } else {
            reloadfunctionalAdherenceDD();
            clearfunctionalAdherenceValues();
            $("#functionalAdherenceModal").modal("hide");
          }
          $("#AdherenceModalSaveBtn").removeAttr("disabled");
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }
  export function deletefunctionalAdherence(delURL) {
    const txtSel = $("#drpFunctionalAdherence")
      .data("kendoDropDownList")
      .value();
    $Ajax.ajaxAntiForgery({
      url: delURL,
      type: "POST",
      data: {
        SelectedGUID: txtSel,
      },
      success: function(response) {
        if (response.success === false) {
          $.notify({ message: response.Message }, { type: "danger" });
        } else {
          reloadfunctionalAdherenceDD();
        }
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }
