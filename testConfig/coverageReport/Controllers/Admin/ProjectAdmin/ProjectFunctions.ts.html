  export function changeChecksByIncType() {
    if (Main.serverBuildType === "Belgium") {
      let prjTypeText = $("#IncidentTypeGUID")
        .data("kendoDropDownList")
        .text();
      $("#SendNotificationOfNewIncident").removeAttr("disabled");
      if (prjTypeText.length > 0) {
        prjTypeText = prjTypeText.toUpperCase();
        if (prjTypeText === "EXER" || prjTypeText === "EVENT" || prjTypeText === "TEST" || prjTypeText === "SCEN") {
          $("#SendNotificationOfNewIncident").prop("checked", false);
          if (prjTypeText === "TEST" || prjTypeText === "SCEN") {
            $("#SendNotificationOfNewIncident").attr("disabled", "disabled");
          }
        } else {
          $("#SendNotificationOfNewIncident").prop("checked", true);
        }
      }
    }
  }
 
  export function validate(): boolean {
    //check for a project name, lat, and lon
    //if any are missing, show popup and return false
    let errResponse = {
      Messages: [],
    };
    const projName = $("#ProjectName").val();
    if (!(projName.length > 0)) {
      Main.markObjectAsError($("#ProjectName"));
      errResponse.Messages.push(_globalVars.MissingIncidentMessage);
    } else {
      Main.clearObjectErrorStatus($("#ProjectName"));
    }
    if (Main.serverBuildType === "Belgium") {
      const latVal = MiniMap.getLatObj().value();
      if (latVal === null || latVal === undefined) {
        Main.markObjectAsError($("#Lat"));
        errResponse.Messages.push(_globalVars.MissingLatitudeMessage);
      } else {
        Main.clearObjectErrorStatus($("#Lat"));
      }
      const lonVal = MiniMap.getLonObj().value();
      if (lonVal === null || lonVal === undefined) {
        Main.markObjectAsError($("#Lon"));
        errResponse.Messages.push(_globalVars.MissingLongitudeMessage);
      } else {
        Main.clearObjectErrorStatus($("#Lon"));
      }
    }
    const prjType = $("#IncidentTypeGUID")
      .data("kendoDropDownList")
      .value();
    if (!(prjType.length > 0)) {
      Main.markObjectAsError($("#IncidentTypeGUID"));
      errResponse.Messages.push(_globalVars.MissingProjectTypeMessage);
    } else {
      Main.clearObjectErrorStatus($("#IncidentTypeGUID"));
    }
 
    if (errResponse.Messages.length > 0) {
      Main.handleValErr(null, errResponse);
      return false;
    } else {
      return true;
    }
  }

  function shareIncidentButtonClick(e) {
    e.preventDefault();
    let ul = $("#checkedOrganizationList");
    ul.empty(); //clear out earlier organization list
    //build new org list (checking previously selected orgs?)
    $.each(_matchingOrgList, function(index, item) {
      let checkbox = $("<input type='checkbox'>");
      let index2 = $.inArray(item.OrganizationName, _selectedOrgNames);
      if (index2 > -1) {

      }
      let label = $("<label >" + "</label>");
      label.append(checkbox);
      label.append(item.OrganizationName);
      let li = $("<li class='list-box-item checkbox'></li>");
      li.append(label);
      //const li = $("<li><label class="checkbox"><input type="checkbox">" + item.OrganizationName + "</label></li>");
