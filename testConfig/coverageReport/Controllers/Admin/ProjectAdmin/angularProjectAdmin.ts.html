    // from http://victorblog.com/2012/12/20/make-angularjs-http-service-behave-like-jquery-ajax/
    // use x-www-form-urlencoded Content-Type
    $httpProvider.defaults.headers.post["Content-Type"] = "application/x-www-form-urlencoded;charset=utf-8";
 
    // override $http service's default transformRequest
    $httpProvider.defaults.transformRequest = [
      function(data) {
        const param = function(obj) {
          let query = "";
          let name, value, fullSubName, subName, subValue, innerObj, i;
 
          for (name in obj) {
            if (obj.hasOwnProperty(name)) {
              value = obj[name];
              if (value instanceof Array) {
                for (i = 0; i < value.length; ++i) {
                  subValue = value[i];
                  fullSubName = name + "[" + i + "]";
                  innerObj = {};
                  innerObj[fullSubName] = subValue;
                  query += param(innerObj) + "&";
                }
              } else if (value instanceof Object) {
                for (subName in value) {
                  if (value.hasOwnProperty(subName)) {
                    subValue = value[subName];
                    fullSubName = name + "[" + subName + "]";
                    innerObj = {};
                    innerObj[fullSubName] = subValue;
                    query += param(innerObj) + "&";
                  }
                }
              } else if (value !== undefined && value !== null) {
                query += encodeURIComponent(name) + "=" + encodeURIComponent(value) + "&";
              }
            }
          }
 
          return query.length ? query.substr(0, query.length - 1) : query;
        };
 
        return angular.isObject(data) && String(data) !== "[object File]" ? param(data) : data;
      },
    ];
  });

          break;

    $scope.searchOrgs = function() {
      $scope.setIsOrgSearching(true);

      setTimeout(() => {

    $scope.FilterOrgSearchByMapVisibility = function(foundOrgNames: string[]) {
      //need a way to filter the orgsearchlist datasource
      $scope.boundFromSearch = false;
      $scope.setIsOrgSearching(true);
      $scope.orgSrchlst.dataSource.filter({
        field: "",
        operator: function(item, filter) {
          if (foundOrgNames.length === 0) {
            return false;
          }
 
          if (item.LocationInfo === "No Organization Location Defined" || item.LocationInfo.startsWith("Requesting")) {
            // orgs without location should be returned
            return true;
          }
          const found = $.inArray(item.OrganizationName, foundOrgNames);
          return found > -1;
        },
        value: "",
      });
    };
 
    $scope.FilterJurisdictionSearchByMapVisibility = function(foundOrgNames: string[]) {
      //need a way to filter the jursSrchlstJ datasource
      $scope.boundFromSearch = false;
      $scope.setIsJursSearching(true);
      $scope.jursSrchlst.dataSource.filter({
        field: "",
        operator: function(item, filter) {
          if (foundOrgNames.length === 0) {
            return false;
          }
          let rtnVal = false;
          if (item.AssociatedOrganizations !== null && item.AssociatedOrganizations !== undefined) {
            item.AssociatedOrganizations.forEach(o => {
              const found = $.inArray(o, foundOrgNames);
              if (found > -1) {
                rtnVal = true;
                return;
              }
            });
          }
          return rtnVal;
        },
        value: "",
      });
    };

    $scope.getIsCollapsed = function(jurisdictionGUID: string): boolean {
      let ret = false;
      for (let index = 0; index < $scope.collapsedOrgs.length; ++index) {
        const entry = $scope.collapsedOrgs[index];
        if (entry === jurisdictionGUID) {
          ret = true;
        }
      }
      return ret;
    };
 
    $scope.flipIsCollapsed = function(jurisdictionGUID: string) {
      if ($scope.getIsCollapsed(jurisdictionGUID)) {
        for (let index = 0; index < $scope.collapsedOrgs.length; ++index) {
          const entry = $scope.collapsedOrgs[index];
          if (entry === jurisdictionGUID) {
            $scope.collapsedOrgs.splice(index, 1);
          }
        }
      } else {
        $scope.collapsedOrgs.push(jurisdictionGUID);
      }
    };

    $scope.saveScratch = function() {
      const url = $scope.hostURL + "Admin/ProjectAdmin/AddShare/" + $scope.projectGUID;
      const scratchAsJSON = JSON.stringify($scope.scratchData);
      $scope.isShareSaving = true;
      $Ajax.ajaxAntiForgery({
        url: url,
        type: "POST",
        data: {
          scratchPadAsJSON: scratchAsJSON,
          sendNotification: $scope.sendNotification,
