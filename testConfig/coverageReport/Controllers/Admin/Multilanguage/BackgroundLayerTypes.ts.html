namespace BackgroundLayerTypes {
  function reloadBackgroundLayerTypeDD() {
    const dropdownlist = $("#drpBackgroundLayerType").data("kendoDropDownList");
    dropdownlist.dataSource.read();
  }
  function clearBackgroundLayerTypeValues() {
    $("#txtSelectedBackgroundLayerType").val("");
    $("#txtenBackgroundLayerType").val("");
    $("#txtdeBackgroundLayerType").val("");
    $("#txtnlBackgroundLayerType").val("");
    $("#txtfrBackgroundLayerType").val("");
    $("#txtValSummBackgroundLayerType").text("");
  }
 
  export function addBackgroundLayerType() {
    clearBackgroundLayerTypeValues();
    $("#BackgroundLayerTypeModal").modal("show");
  }

  export function saveBackgroundLayerType(url) {
    const txten = $("#txtenBackgroundLayerType").val();
    const txtfr = $("#txtfrBackgroundLayerType").val();
    const txtde = $("#txtdeBackgroundLayerType").val();
    const txtnl = $("#txtnlBackgroundLayerType").val();
    const txtSel = $("#drpBackgroundLayerType")
      .data("kendoDropDownList")
      .value();
    if (txten.length > 0) {
      $Ajax.ajaxAntiForgery({
        url: url,
        type: "POST",
        data: {
          enText: txten,
          deText: txtde,
          nlText: txtnl,
          frText: txtfr,
          SelectedBackgroundLayerType: txtSel,
        },
        success: function(response) {
          if (response.success === false) {
            $("#txtValSummBackgroundLayerType").text(response.Message);
          } else {
            reloadBackgroundLayerTypeDD();
            clearBackgroundLayerTypeValues();
            $("#BackgroundLayerTypeModal").modal("hide");
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }
  export function deleteBackgroundLayerType(url) {
    const txtSel = $("#drpBackgroundLayerType")
      .data("kendoDropDownList")
      .value();
    $Ajax.ajaxAntiForgery({
      url: url,
      type: "POST",
      data: {
        SelectedGUID: txtSel,
      },
      success: function(response) {
        if (response.success === false) {
          $.notify({ message: response.Message }, { type: "danger" });
        } else {
          reloadBackgroundLayerTypeDD();
        }
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }
 
  export function BackgroundLayerTypeChanged() {
    if (IsBackgroundLayerTypeTextboxValid()) {
      EnableBackgroundLayerTypeButtons();
    } else {
      DisableBackgroundLayerTypeButtons();
    }
  }
  function IsBackgroundLayerTypeTextboxValid() {
    return $("#drpBackgroundLayerType").val().length > 0;
  }
  function EnableBackgroundLayerTypeButtons() {
    $(".BackgroundLayerTypechng").removeAttr("disabled");
  }
  function DisableBackgroundLayerTypeButtons() {
    $(".BackgroundLayerTypechng").attr("disabled", "disabled");
  }
}
