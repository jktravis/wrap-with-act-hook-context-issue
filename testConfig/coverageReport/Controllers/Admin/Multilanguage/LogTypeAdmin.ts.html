namespace LogTypeAdmin {
  function reloadLogTypeDD() {
    $("#lgTypGrd")
      .data("kendoGrid")
      .dataSource.read();
  }
  function clearValues() {
    $("#txtSelectedLogType").val("");
    $("#txten").val("");
    $("#txtde").val("");
    $("#txtnl").val("");
    $("#txtfr").val("");
    //$("#drpImgUrl").data('kendoDropDownList').selectedIndex = -1;
    $("#txtValSummLogType").text("");
  }
 
  export function addLogType() {
    clearValues();
    $("#LogTypeModal").modal("show");
  }
  export function updateLogType(key, envalue, devalue, frvalue, nlvalue, image) {
    $("#txtSelectedLogType").val(key);
    $("#txten").val(envalue);
    $("#txtde").val(devalue);
    $("#txtfr").val(frvalue);
    $("#txtnl").val(nlvalue);
    $("#drpImgUrl")
      .data("kendoDropDownList")
      .value(image);
    $("#LogTypeModal").modal("show");
  }
  export function saveLogType(url) {
    const txten = $("#txten").val();
    const txtfr = $("#txtfr").val();
    const txtde = $("#txtde").val();
    const txtnl = $("#txtnl").val();
    const txtSelCat = $("#txtSelectedLogType").val();
    const img = $("#drpImgUrl")
      .data("kendoDropDownList")
      .value();
    if (txten.length > 0) {
      $Ajax.ajaxAntiForgery({
        url: url,
        type: "POST",
        data: {
          enText: txten,
          deText: txtde,
          nlText: txtnl,
          frText: txtfr,
          SelectedLogType: txtSelCat,
          ImageUrl: img,
        },
        success: function(response) {
          if (response.success === false) {
            $("#txtValSummLogType").text(response.Message);
          } else {
            reloadLogTypeDD();
            clearValues();
            $("#LogTypeModal").modal("hide");
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }
}
