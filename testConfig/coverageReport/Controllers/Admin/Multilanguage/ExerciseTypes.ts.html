namespace ExerciseTypes {
  function reloadExerciseTypeDD() {
    $("#lgTypGrd")
      .data("kendoGrid")
      .dataSource.read();
  }
  function clearValues() {
    $("#txtSelectedExerciseType").val("");
    $("#txten").val("");
    $("#txtde").val("");
    $("#txtnl").val("");
    $("#txtfr").val("");
    $("#txtValSummExerciseType").text("");
  }
 
  export function addExerciseType() {
    clearValues();
    $("#ExerciseTypeModal").modal("show");
  }
  export function updateExerciseType(key, envalue, devalue, frvalue, nlvalue, image) {
    $("#txtSelectedExerciseType").val(key);
    $("#txten").val(envalue);
    $("#txtde").val(devalue);
    $("#txtfr").val(frvalue);
    $("#txtnl").val(nlvalue);
    $("#ExerciseTypeModal").modal("show");
  }
  export function saveExerciseType(url) {
    const txten = $("#txten").val();
    const txtfr = $("#txtfr").val();
    const txtde = $("#txtde").val();
    const txtnl = $("#txtnl").val();
    const txtSelCat = $("#txtSelectedExerciseType").val();
    if (txten.length > 0) {
      $Ajax.ajaxAntiForgery({
        url: url,
        type: "POST",
        data: {
          enText: txten,
          deText: txtde,
          nlText: txtnl,
          frText: txtfr,
          SelectedExerciseType: txtSelCat,
        },
        success: function(response) {
          if (response.success === false) {
            $("#txtValSummExerciseType").text(response.Message);
          } else {
            reloadExerciseTypeDD();
            clearValues();
            $("#ExerciseTypeModal").modal("hide");
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }
}
