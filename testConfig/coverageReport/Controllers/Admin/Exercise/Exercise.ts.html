  function init(globalPrompts) {
    _globalPrompts = globalPrompts;
    $("form").on("submit", function(e) {
      e.preventDefault();
      $(this)
        .find("input[type=submit]")
        .prop("disabled", true);
      if (validateForm()) {
        //populate hidden fields
        const exercisePOCs = $("#exercisePOCsDiv").data("userlist");
        $("#exercisePOCs").val(JSON.stringify(exercisePOCs));
 
        if ($("#exercisePOCSupportsDiv").length > 0) {
          const supportPOCs = $("#exercisePOCSupportsDiv").data("userlist");
          $("#exercisePOCSupports").val(JSON.stringify(supportPOCs));
        }
 
        if ($("#sponsorOrgsDiv").length > 0) {
          const sponsorOrgs = $("#sponsorOrgsDiv").data("orglist");
          $("#sponsorOrgs").val(JSON.stringify(sponsorOrgs));
        } else {
          $("#sponsorOrgs").val("");
        }
 
        if ($("#participantOrgsDiv").length > 0) {
          const participantOrgs = $("#participantOrgsDiv").data("orglist");
          $("#participantOrgs").val(JSON.stringify(participantOrgs));
        } else {
          $("#participantOrgs").val("");
        }
 
        if ($("#entityFacsDiv").length > 0) {
          const entityFacs = $("#entityFacsDiv").data("faclist");
          $("#entityFacs").val(JSON.stringify(entityFacs));
        } else {
          $("#entityFacs").val("");
        }
 
        if ($("#MeetingsAsJSON").length > 0) {
          $("#MeetingsAsJSON").val(JSON.stringify(customDates));
        }
        this.submit();
      } else {
        $(this)
          .find("input[type=submit]")
          .prop("disabled", false);
      }
    });
  }

  function validateForm(): Boolean {
    const errResponse = {
      Messages: [],
    };
    const nm = $("#ExerciseName").val();
    if (!(nm.length > 0)) {
      Main.markObjectAsError($("#ExerciseName"));
      errResponse.Messages.push(_globalPrompts.MissingNameMessage);
    } else {
      Main.clearObjectErrorStatus($("#ExerciseName"));
    }
 
    const typ = $("#ExerciseTypeGUID")
      .data("kendoDropDownList")
      .value();
    if (!(typ.length > 0) || typ === Guid.emptyGuidString) {
      Main.markObjectAsError($("#ExerciseTypeGUID"));
      errResponse.Messages.push(_globalPrompts.MissingExerciseTypeMessage);
    } else {
      Main.clearObjectErrorStatus($("#ExerciseTypeGUID"));
    }
 
    if (Main.serverBuildType === "Belgium") {
      const sDate = $("#StartDate")
        .data("kendoDatePicker")
        .value();
      if (sDate === null || !(sDate.toString().length > 0)) {
        Main.markObjectAsError($("#StartDate"));
        errResponse.Messages.push(_globalPrompts.MissingStartDateMessage);
      } else {
        Main.clearObjectErrorStatus($("#StartDate"));
      }
      const eDate = $("#EndDate")
        .data("kendoDatePicker")
        .value();
      if (eDate === null || !(eDate.toString().length > 0)) {
        Main.markObjectAsError($("#EndDate"));
        errResponse.Messages.push(_globalPrompts.MissingEndDateMessage);
      } else {
        Main.clearObjectErrorStatus($("#EndDate"));
      }
      const exercisePOCs = $("#exercisePOCsDiv").data("userlist");
      if (exercisePOCs.length > 0 && exercisePOCs[0] !== Guid.emptyGuidString) {
        Main.clearObjectErrorStatus($("#exercisePOCsDiv"));
      } else {
        Main.markObjectAsError($("#exercisePOCsDiv"));
        //errResponse.Message = _globalPrompts.MissingSubmittedByMessage + '<br/>';
        errResponse.Messages.push(_globalPrompts.MissingPOCMessage);
      }
    }
    if (errResponse.Messages.length > 0) {
      Main.handleValErr(null, errResponse);
      return false;
    } else {
      return true;
    }
  }

  function loadCustomDateGrid() {
    const customDatesAsJSON = $("#MeetingsAsJSON").val();
    if (customDatesAsJSON.length > 0) {
      customDates = JSON.parse(customDatesAsJSON);
    }
    const dataSource = new kendo.data.DataSource({
      transport: {
        read: function(e) {
          e.success(customDates);
        },
      },
      batch: false,
      schema: {
        model: {
          id: "ID",
          fields: {
            ID: { type: "number", editable: false },
            CustomDateTitle: { type: "string", editable: false },
            CustomDate: { type: "date", editable: false },
          },
        },
      },
    });
    const grid = $("#customDateGrid").data("kendoGrid");
    grid.setDataSource(dataSource);
    reloadCustomDateGrid();
  }

  export function saveDate() {
    const customDateTitle = $("#txtCustomDateTitle").val();
    const customDate = $("#dteCustomDate")
      .data("kendoDateTimePicker")
      .value();
    const customEndDate = $("#dteCustomEndDate")
      .data("kendoDateTimePicker")
      .value();
    if (customDateTitle.length > 0 && customDate !== null) {
      //create object
      const obj: ExerciseCustomDate = new ExerciseCustomDate();
      obj.MeetingTitle = customDateTitle;
      obj.MeetingDate = customDate;
      obj.MeetingEndDate = customEndDate;
      obj.IsVirtual = IsVirtualIsChecked();
      obj.FederalNum = $("#txtFederalNum")
        .data("kendoNumericTextBox")
        .value();
      obj.OtherNum = $("#txtOtherNum")
        .data("kendoNumericTextBox")
        .value();
      obj.StateNum = $("#txtStateNum")
        .data("kendoNumericTextBox")
        .value();
      obj.LocalNum = $("#txtLocalNum")
        .data("kendoNumericTextBox")
        .value();
      obj.PrivateNum = $("#txtPrivateNum")
        .data("kendoNumericTextBox")
        .value();
      obj.OrgNum = $("#txtOrgNum")
        .data("kendoNumericTextBox")
        .value();
      obj.TotalNum = $("#txtTotalNum")
        .data("kendoNumericTextBox")
        .value();
      obj.Location = $("#txtLocation").val();
      obj.MeetingType = $("#meetingDropDown")
        .data("kendoDropDownList")
        .value();
 
      let idVal = $("#idVal").val();
      //console.log("id to edit " + idVal);
      if (idVal.length > 0) {
        let idValAsNum: number = parseInt(idVal, 10);
        obj.ID = idValAsNum;
        //console.log("index to edit " + getIndexById(idValAsNum));
        customDates.splice(getIndexById(idValAsNum), 1);
      } else {
        obj.ID = customDates.length + 1;
      }
 
      customDates.push(obj);
      reloadCustomDateGrid();
      $("#customDateModal").modal("hide");
    }
  }

  export function deleteAttachment(attGUID, theUrl) {
    $Ajax.ajaxAntiForgery({
      url: theUrl,
      type: "POST",
      data: {
        PrimaryKey: attGUID,
      },
      success: function(response) {
        if (response.success === false) {
          $.notify({ message: response.Message }, { type: "danger" });
        } else {
          $("#lstPreAtt")
            .data("kendoListView")
            .dataSource.read();
          if ($("#lstPostAtt").length > 0) {
            $("#lstPostAtt")
              .data("kendoListView")
              .dataSource.read();
          }
        }
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }
