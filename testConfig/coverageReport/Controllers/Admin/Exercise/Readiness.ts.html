  export function BuildWordCloud(div) {
    const insertDiv = document.getElementById(div);
    const mainSVG = initSVG();
    let total;
    let largestCount;
    const countDisplayBuffer = 15;
 
    //Create SVG
    insertDiv.appendChild(mainSVG);
 
    const svgWidth = mainSVG.getBoundingClientRect().width;
    const svgHeight = mainSVG.getBoundingClientRect().height;
    const maxRadius = svgWidth > svgHeight ? svgHeight / 3 : svgWidth / 3;
 
    //Get Information
    $.ajax({
      url: WordCloudSectorUrl,
      type: "GET",
      success: function(response) {
        if (response && !jQuery.isEmptyObject(response)) {
          let largestFound = false;
          let largestArea;
          const existingCircles = [];
 
          response.forEach(function(item, index) {
            if (!largestFound) {
              if (item.Key === totalString) {
                total = item.Value;
              } else {
                existingCircles.push({
                  x: svgWidth / 2,
                  y: svgHeight / 2,
                  r: (item.Value / total) * maxRadius,
                  text: item.Key,
                  value: item.Value,
                });
                largestCount = item.Value;
                largestArea = Math.PI * Math.pow((item.Value / total) * maxRadius, 2);
                largestFound = true;
              }
            } else {
              //Generate x, y and radius for new circles
              let newX;
              let newY;
              let newRad;
              do {
                //Good for new radius. Nice large circles
                newRad = Math.floor(Math.sqrt(((item.Value / largestCount) * largestArea * 2) / Math.PI));
                newX = Math.floor(Math.random() * (svgWidth - newRad * 2) + newRad);
                newY = Math.floor(Math.random() * (svgHeight - newRad * 2 - countDisplayBuffer) + newRad);
              } while (DoesItCollide(existingCircles, newX, newY, newRad));
 
              existingCircles.push({ x: newX, y: newY, r: newRad, text: item.Key, value: item.Value });
            }
          });
 
          const defs = document.createElementNS("http://www.w3.org/2000/svg", "defs");
          mainSVG.appendChild(defs);
 
          //Draw stuff.
          existingCircles.forEach(function(item, index) {
            const pattern = document.createElementNS("http://www.w3.org/2000/svg", "pattern");
            pattern.setAttributeNS(null, "id", "Sector_" + index + "_Pattern");
            pattern.setAttributeNS(null, "width", "1");
            pattern.setAttributeNS(null, "height", "1");
 
            //If file exists, make the image
            let flag = false;
            $.ajax({
              url: `../../Content/images/Exercise/Sector_${item.text}.jpg`,
              async: false,
              type: "HEAD",
              success: function() {
                const image = document.createElementNS("http://www.w3.org/2000/svg", "image");
                image.setAttributeNS(
                  "http://www.w3.org/1999/xlink",
                  "href",
                  `../../Content/images/Exercise/Sector_${item.text}.jpg`,
                );
                image.setAttributeNS(null, "width", String(item.r * 2));
                image.setAttributeNS(null, "height", String(item.r * 2));
                image.setAttributeNS(null, "x", String(0));
                image.setAttributeNS(null, "y", String(0));
 
                pattern.appendChild(image);
                flag = true;
              },
            });
 
            //Build circles here. Add text/groups/defs.
 
            //HIGHER RES IMAGES AND FIGURE OUT WHAT TO DO WHEN IMAGE DOESNT EXIST
            //ALSO TEXT
            const group = document.createElementNS("http://www.w3.org/2000/svg", "g");
 
            const circle = document.createElementNS("http://www.w3.org/2000/svg", "circle");
            circle.setAttributeNS(null, "cx", item.x);
            circle.setAttributeNS(null, "cy", item.y);
            circle.setAttributeNS(null, "r", item.r);
            circle.setAttributeNS(null, "opacity", "0.6");
            circle.setAttributeNS(null, "style", "fill: #f49b42");
 
            const titleText = document.createElementNS("http://www.w3.org/2000/svg", "text");
            if (flag) {
              circle.setAttributeNS(null, "style", 'fill: url("#Sector_' + index + '_Pattern");');
            } else {
              titleText.setAttributeNS(null, "x", item.x);
              titleText.setAttributeNS(null, "y", item.y);
              titleText.setAttributeNS(null, "text-anchor", "middle");
              titleText.textContent = addNewLines(item.text);
            }
 
            const numberValueText = document.createElementNS("http://www.w3.org/2000/svg", "text");
            numberValueText.setAttributeNS(null, "x", item.x);
            numberValueText.setAttributeNS(null, "y", item.y + item.r + countDisplayBuffer);
            numberValueText.setAttributeNS(null, "text-anchor", "middle");
            numberValueText.textContent = item.value;
 
            group.appendChild(circle);
            group.appendChild(numberValueText);
            group.appendChild(titleText);
 
            defs.appendChild(pattern);
            mainSVG.appendChild(group);
          });
        }
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }
 
  function initSVG() {
    const element = document.createElementNS("http://www.w3.org/2000/svg", "svg");
    element.setAttributeNS(null, "style", "height: 500px; width: 100%; overflow: hidden; left: 0px; top: -0.4px;");
    element.setAttributeNS(null, "version", "1.1");
 
    return element;
  }
 
  function DoesItCollide(circleList, currentX, currentY, currentRadius) {
    let flag = false;
 
    if (circleList) {
      circleList.forEach(function(item, index) {
        if (
          (currentX - currentRadius <= item.x + item.r && currentX - currentRadius >= item.x - item.r) ||
          (currentX + currentRadius <= item.x + item.r && currentX + currentRadius >= item.x - item.r)
        ) {
          if (
            (currentY - currentRadius <= item.y + item.r && currentY - currentRadius >= item.y - item.r) ||
            (currentY + currentRadius <= item.y + item.r && currentY + currentRadius >= item.y - item.r)
          ) {
            flag = true;
          }
        }
      });
    }
 
    return flag;
  }
 
  export function addNewLines(e) {
    e = e.split(",").join("\n");
    if (e.split(" ").length <= 3) {
      e = e.split(" ").join("\n");
    }
    e = e.split("and").join("\n");
    return e;
  }
