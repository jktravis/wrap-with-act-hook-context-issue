namespace AdminSEVESO {
  function reloadstructureCatDD() {
    const dropdownlist = $("#Primary").data("kendoDropDownList");
    dropdownlist.dataSource.read();
  }
  function clearValues() {
    $("#txtSelectedPrimary").val("");
    $("#txtPrien").val("");
    $("#txtPride").val("");
    $("#txtPrinl").val("");
    $("#txtPrifr").val("");
    $("#txtValSummPri").text("");
  }
 
  export function addPrimary() {
    clearValues();
    $("#primaryModal").modal("show");
  }
  export function updatePrimary() {
    const selCat = $("#Primary").data("kendoDropDownList");
    const object = selCat.dataItem(selCat.select());
    $("#txtSelectedPrimary").val(selCat.value());
    $("#txtPrien").val(object.enValue);
    $("#txtPride").val(object.deValue);
    $("#txtPrinl").val(object.nlValue);
    $("#txtPrifr").val(object.frValue);
    $("#primaryModal").modal("show");
  }
  export function savePrimary(saveURL) {
    const txten = $("#txtPrien").val();
    const txtfr = $("#txtPrifr").val();
    const txtde = $("#txtPride").val();
    const txtnl = $("#txtPrinl").val();
    const txtSelCat = $("#txtSelectedPrimary").val();
    if (txten.length > 0) {
      $Ajax.ajaxAntiForgery({
        url: saveURL,
        type: "POST",
        data: {
          enText: txten,
          deText: txtde,
          nlText: txtnl,
          frText: txtfr,
          SelectedPrimary: txtSelCat,
        },
        success: function(response) {
          if (response.success === false) {
            $("#txtValSummPri").text(response.Message);
          } else {
            reloadstructureCatDD();
            clearValues();
            $("#primaryModal").modal("hide");
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }
  export function deletePrimary(delURL) {
    const sel = $("#Primary").val();
    $Ajax.ajaxAntiForgery({
      url: delURL,
      type: "POST",
      data: {
        SelectedPrimary: sel,
      },
      success: function(response) {
        if (response.success === false) {
          $.notify({ message: response.Message }, { type: "danger" });
        } else {
          reloadstructureCatDD();
        }
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }
  function catChanged() {
    if (IsCatTextboxValid()) {
      EnableCatButtons();
    } else {
      DisableCatButtons();
    }
  }
  function IsCatTextboxValid() {
    return $("#Primary").val().length > 0;
  }
  function EnableCatButtons() {
    $(".catchng").removeAttr("disabled");
  }
  function DisableCatButtons() {
    $(".catchng").attr("disabled", "disabled");
  }
}
