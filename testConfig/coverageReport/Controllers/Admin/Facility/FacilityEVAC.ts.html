namespace Facility_EVAC {
  interface IGlobalPrompt {
    MissingNormalBuildingMessage: string;
    MissingSomeoneIsPresentMessage: string;
    MissingPrimaryContactMessage: string;
  }
  let _globalPrompts: IGlobalPrompt;
  let shelterChar = [];
  export function Init(globalPrompts) {
    _globalPrompts = globalPrompts;
    $("form").on("submit", function(e) {
      e.preventDefault();
      $(this)
        .find("input[type=submit]")
        .prop("disabled", true);
      if (validateForm()) {
        //populate hidden fields
        const primContacts = $("#primaryContactsDiv").data("userlist");
        $("#POCContactsList").val(JSON.stringify(primContacts));
 
        const supContacts = $("#supContactsDiv").data("userlist");
        $("#SupContactsList").val(JSON.stringify(supContacts));
 
        $("#ShelterRoomDetailsAsJSON").val(JSON.stringify(shelterChar));
 
        this.submit();
      } else {
        $(this)
          .find("input[type=submit]")
          .prop("disabled", false);
      }
    });
    $("#ShelCharModal").on("shown.bs.modal", function() {
      $("#txtIdentification").focus();
    });
    loadShelterGrid();
  }
  export function validateForm(): Boolean {
    const errResponse = {
      Messages: [],
    };
    const buildUsage = $("#NormalBuildingUsage").val();
    if (!(buildUsage.length > 0)) {
      Main.markObjectAsError($("#NormalBuildingUsage"));
      errResponse.Messages.push(_globalPrompts.MissingNormalBuildingMessage);
    } else {
      Main.clearObjectErrorStatus($("#NormalBuildingUsage"));
    }
    const contacts = $("#primaryContactsDiv").data("userlist");
    const contactsString = JSON.stringify(contacts);
    if (contacts.length > 0 && contacts[0] !== Guid.emptyGuidString) {
      Main.clearObjectErrorStatus($("#primaryContactsDiv"));
    } else {
      Main.markObjectAsError($("#primaryContactsDiv"));
      errResponse.Messages.push(_globalPrompts.MissingPrimaryContactMessage);
    }
    const someone = $("#SomeoneIsPresent")
      .data("kendoDropDownList")
      .value();
    if (!(someone.length > 0) || someone === Guid.emptyGuidString) {
      Main.markObjectAsError($("#SomeoneIsPresent"));
      errResponse.Messages.push(_globalPrompts.MissingSomeoneIsPresentMessage);
    } else {
      Main.clearObjectErrorStatus($("#SomeoneIsPresent"));
    }
 
    if (errResponse.Messages.length > 0) {
      Main.handleValErr(null, errResponse);
      return false;
    } else {
      return true;
    }
  }
 
  function loadShelterGrid() {
    const shelterAsJSON = $("#ShelterRoomDetailsAsJSON").val();
    if (shelterAsJSON.length > 0) {
      shelterChar = JSON.parse(shelterAsJSON);
    }
    const dataSource = new kendo.data.DataSource({
      transport: {
        read: function(e) {
          e.success(shelterChar);
        },
      },
      batch: false,
      schema: {
        model: {
          id: "ID",
          fields: {
            ID: { type: "number", editable: false },
            Identification: { type: "string", editable: true },
            Description: { type: "string", editable: true },
            NumericalCapacityDay: { type: "number", defaultValue: null, editable: true },
            EquipmentForDayShelter: { type: "string", editable: true },
            NumericalCapacityNightDorm: { type: "number", defaultValue: null, editable: true },
            NumericalCapacityRoom: { type: "number", defaultValue: null, editable: true },
            EquipmentForNightShelter: { type: "string", editable: true },
            TimeToOPsStatus: { type: "number", defaultValue: null, editable: true },
            PublicInfoEquipment: { type: "string", editable: true },
            Notes: { type: "string", editable: true },
          },
        },
      },
    });
    const grid = $("#shelGrid").data("kendoGrid");
    grid.setDataSource(dataSource);
    reloadShelChar();
  }
  function reloadShelChar() {

  function getIndexById(id) {
    let idx,
      l = shelterChar.length;
 
    for (let j = 0; j < l; j++) {
      if (shelterChar[j].ID === id) {
        return j;
      }
    }
    return null;
  }
  export function deleteShelterRow(idVal, deleteText) {
    if (confirm(deleteText)) {
      shelterChar.splice(getIndexById(idVal), 1);
      reloadShelChar();
    }
  }
  class ShelterDetails {
    Identification: string;
    Description: string;
    NumericalCapacityDay: string;
    EquipmentForDayShelter: string;
    NumericalCapacityNightDorm: string;
    NumericalCapacityRoom: string;
    EquipmentForNightShelter: string;
    TimeToOPsStatus: string;
    PublicInfoEquipment: string;
    Notes: string;
    ID: number;
  }
  export function saveShelChar() {
    const txtIdentification = $("#txtIdentification").val();
    if (txtIdentification.length > 0) {
      //create object
      const obj: ShelterDetails = new ShelterDetails();
      obj.Identification = txtIdentification;
      obj.Description = $("#txtDescription").val();
      obj.NumericalCapacityDay = $("#txtNumericalCapacityDay").val();
      obj.EquipmentForDayShelter = $("#txtEquipmentForDayShelter").val();
      obj.NumericalCapacityNightDorm = $("#txtNumericalCapacityNightDorm").val();
      obj.NumericalCapacityRoom = $("#txtNumericalCapacityRoom").val();
      obj.EquipmentForNightShelter = $("#txtEquipmentForNightShelter").val();
      obj.TimeToOPsStatus = $("#txtTimeToOPsStatus").val();
      obj.PublicInfoEquipment = $("#txtPublicInfoEquipment").val();
      obj.Notes = $("#txtNotes").val();
      //add to shelterChar array
      obj.ID = shelterChar.length + 1;
      shelterChar.push(obj);
      reloadShelChar();
      $("#ShelCharModal").modal("hide");
    }
  }
  export function deleteAttachment(attGUID, theUrl) {
    $Ajax.ajaxAntiForgery({
      url: theUrl,
      type: "POST",
      data: {
        PrimaryKey: attGUID,
      },
      success: function(response) {
        if (response.success === false) {
          $.notify({ message: response.Message }, { type: "danger" });
        } else {
          $("#lstattAtt")
            .data("kendoListView")
            .dataSource.read();
          $("#lstothAtt")
            .data("kendoListView")
            .dataSource.read();
        }
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }
