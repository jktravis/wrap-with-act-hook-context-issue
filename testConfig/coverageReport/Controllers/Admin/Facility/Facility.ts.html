  interface IGlobalPrompt {
    MissingPOCMessage: string;
    MissingLatLonMessage: string;

  export function Init(globalPrompts) {
    _globalPrompts = globalPrompts;
 
    $("input").on("blur", function(e) {
      validateForm(e);
    });
 
    $("form").on("submit", function(e) {
      e.preventDefault();
      $(this)
        .find("input[type=submit]")
        .prop("disabled", true);
      //populate hidden fields
      if (validateForm(e)) {
        if ($("#PrimaryPOCDiv").length > 0) {
          const contacts = $("#PrimaryPOCDiv").data("userlist");
          $("#PrimaryPOC").val(contacts[0]);
        }
        this.submit();
      } else {
        $(this)
          .find("input[type=submit]")
          .prop("disabled", false);
      }
    });
  }
  export function validateForm(e): boolean {
    const errResponse = {
      Messages: [],
    };
    if (Main.serverBuildType === "Belgium") {
      let latVal, lonVal;
      const kendoLat = $("#Lat").data("kendoNumericTextBox");
      const kendoLon = $("#Lon").data("kendoNumericTextBox");
      if (kendoLat !== null && kendoLat !== undefined) {
        latVal = kendoLat.value(); // to get the non-localized value
      }
      if (kendoLon !== null && kendoLon !== undefined) {
        lonVal = kendoLon.value(); // to get the non-localized value
      }
      if (latVal !== null && latVal !== "" && lonVal !== null && lonVal !== "") {
        Main.clearObjectErrorStatus($("#Lat"));
        Main.clearObjectErrorStatus($("#Lon"));
      } else {
        Main.markObjectAsError($("#Lat"));
        Main.markObjectAsError($("#Lon"));
        errResponse.Messages.push(_globalPrompts.MissingLatLonMessage);
      }
 
      if ($("#PrimaryPOCDiv").length > 0) {
        const contacts = $("#PrimaryPOCDiv").data("userlist");
        const contactsString = JSON.stringify(contacts);
        if (contacts.length > 0 && contacts[0] !== Guid.emptyGuidString) {
          Main.clearObjectErrorStatus($("#PrimaryPOCDiv"));
        } else {
          Main.markObjectAsError($("#PrimaryPOCDiv"));
          errResponse.Messages.push(_globalPrompts.MissingPOCMessage);
        }
      }
    }
 
    const CrisisHotline = $("#CrisisHotline");
    const FaxNumber = $("#FaxNumber");
 
    var phoneNumberValue = (CrisisHotline[0] as HTMLInputElement).value;
    if (phoneNumberValue.length > 0 && !Main.validatePhoneNumber(phoneNumberValue)) {
      Main.markObjectAsError(CrisisHotline);
      errResponse.Messages.push(_globalPrompts.CrisisHotlineMessage);
    } else {
      Main.clearObjectErrorStatus(CrisisHotline);
    }
 
    phoneNumberValue = (FaxNumber[0] as HTMLInputElement).value;
    if (phoneNumberValue.length > 0 && !Main.validatePhoneNumber(phoneNumberValue)) {
      Main.markObjectAsError(FaxNumber);
      errResponse.Messages.push(_globalPrompts.FaxNumberMessage);
    } else {
      Main.clearObjectErrorStatus(FaxNumber);
    }
 
    if (errResponse.Messages.length > 0) {
      Main.silentHandleValErr(null, errResponse);
      return false;
    } else {
      return true;
    }
  }

  export function deleteAttachment(attGUID, theUrl) {
    $Ajax.ajaxAntiForgery({
      url: theUrl,
      type: "POST",
      data: {
        PrimaryKey: attGUID,
      },
      success: function(response) {
        if (response.success === false) {
          $.notify({ message: response.Message }, { type: "danger" });
        } else {
          $("#lstAtt")
            .data("kendoListView")
            .dataSource.read();
        }
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }

  let exportFlag = false;
  export function FacilityListExcelExport(e) {
    if (!exportFlag) {
      //Show the columns
      e.sender.showColumn(10);
      e.sender.showColumn(12);
      if (e.sender.columns.length >= 14) {
        //Hide facility guid col
        e.sender.hideColumn(14);
      }
      //Prevent export, as it initially fired with the column hidden
      e.preventDefault();
      //Update for the new call we re about to make
      exportFlag = true;
      //Call the save again, which will hit this function, fail the if statement and cascade down to hide the column and finish the export.
      setTimeout(function() {
        e.sender.saveAsExcel();
      });
    } else {
      e.sender.hideColumn(10);
      e.sender.hideColumn(12);
      if (e.sender.columns.length >= 14) {
        //Hide facility guid col
        e.sender.showColumn(14);
      }
 
      for (let i = 1; i < e.workbook.sheets[0].rows.length; i++) {
        e.workbook.sheets[0].rows[i].cells[2].value = Main.htmlUnescape(e.workbook.sheets[0].rows[i].cells[2].value);
      }
 
      exportFlag = false;
    }
  }
