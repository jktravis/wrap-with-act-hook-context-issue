  export function saveCategory(saveURL) {
    const txten = $("#txtCaten").val();
    const txtfr = $("#txtCatfr").val();
    const txtde = $("#txtCatde").val();
    const txtnl = $("#txtCatnl").val();
    const txtSelCat = $("#txtSelectedCategory").val();
    if (txten.length > 0) {
      $Ajax.ajaxAntiForgery({
        url: saveURL,
        type: "POST",
        data: {
          enText: txten,
          deText: txtde,
          nlText: txtnl,
          frText: txtfr,
          SelectedCategory: txtSelCat,
        },
        success: function(response) {
          if (response.success === false) {
            $("#txtValSummCat").text(response.Message);
          } else {
            reloadstructureCatDD();
            clearValues();
            $("#categoryModal").modal("hide");
          }
        },
        error: function(xhr, status, error) {
          Main.handlejqerr(xhr, status, error);
        },
      });
    }
  }
