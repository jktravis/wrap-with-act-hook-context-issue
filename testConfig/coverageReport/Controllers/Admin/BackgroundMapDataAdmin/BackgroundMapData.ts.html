  function validateSymbology() {
    let rtnVal = true;
    const errorMessages = [];
    $("#symbolizationErrorsDiv").hide();
 
    const valid = backgroundLayerBuilder.UsingPreDefinedSymbology(_Analysis);
    if (valid !== null && valid.isSet) {
      if (valid.isValid) {
        return true;
      } else {
        errorMessages.push(valid.Message);
        rtnVal = false;
      }
    } else {
      //var kmlValid = backgroundLayerBuilder.UsingKMLDefinedSymbology(analysis);
      //if (kmlValid !== null && kmlValid.isSet) {
      //    if (kmlValid.isValid) {
      //        return true;
      //    } else {
      //        errorMessages.push(kmlValid.Message);
      //        rtnVal = false;
      //    }
      //}
      rtnVal = false;
      //nothing valid found yet
      if (rtnVal === false) {
        rtnVal = true; //reset return value
        if (_Analysis.NumberOfPoints > 0) {
          //check to see if a point symbol has been selected
          rtnVal = backgroundLayerBuilder.HasPointSymbolDefined();
          if (!rtnVal) {
            errorMessages.push(_localization.BGLayerPointSymbolError);
          } else {
            //clear error messages
            const index = $.inArray(_localization.BGLayerPointSymbolError, errorMessages);
            if (index > -1) {
              errorMessages.splice(index, 1); //remove message
            }
          }
        }
        if (_Analysis.NumberOfLines > 0) {
          //check to see if a line symbology has been created
          const createdLineSymbology: boolean = CreateLineSymbology();
 
          if (!createdLineSymbology) {
            errorMessages.push(_localization.BGLayerLineStyleError);
          } else {
            //clear error messages
            const index3 = $.inArray(_localization.BGLayerLineStyleError, errorMessages);
            if (index3 > -1) {
              errorMessages.splice(index3, 1); //remove message
            }
          }
          rtnVal = rtnVal && createdLineSymbology;
        }
        if (_Analysis.NumberOfPolygons > 0) {
          //check to see if a polygon symbol has been selected
          const createdArea: boolean = CreateAreaSymbology();
          if (!createdArea) {
            errorMessages.push(_localization.BGLayerPolygonStyleError);
          } else {
            //clear error messages
            const index2 = $.inArray(_localization.BGLayerPolygonStyleError, errorMessages);
            if (index2 > -1) {
              errorMessages.splice(index2, 1); //remove message
            }
          }
          rtnVal = rtnVal && createdArea;
        }
      }
    }
    if (errorMessages.length > 0) {
      const msg = errorMessages.join("\n");
      $("#symbolizationErrorsDiv").show();
      $("#symbolizationErrors").text(msg);
    }
    return rtnVal;
  }

  export function SubmitBackgroundLayer(nextFunc, errorFunc) {
    //wizard should have validated the inputs except for the jurisdiction selection.
    //take all inputs and build a submit object (what about the image upload?)
    const data = new FormData();
 
    //get shared jurisdictions
    if (_submitLayerObject.sharingType === 2) {
      //get all selected jurisdiction guids
      const multiselect = $("#jurisdictionsSelector").data("kendoMultiSelect");
      const items = multiselect.dataItems();
      $.each(items, function() {
        _submitLayerObject.sharedJurisdictions.push(this.JurisdictionGUID);
      });
    }
 
    if (_submitLayerObject.pointStyleSource === "uploaded" && _imageFile !== null) {
      data.append("file", _imageFile); //get file $('#newPointSymbol').get(0).files[0]
    }
    _submitLayerObject.minZoomLevel = $("#minZoomLevel").val();
    _submitLayerObject.maxZoomLevel = $("#maxZoomLevel").val();
 
    //get selected attribute data fields
    const checkedBoxes = $("input:checkbox[class=attributeDataCheckItem]:checked");
    if (checkedBoxes.length > 0) {
      $.each(checkedBoxes, function() {
        _submitLayerObject.includedDataFields.push($(this).val());
      });
    } else {
      _submitLayerObject.includedDataFields = [];
    }
 
    const dropdownlist = $("#namingFieldDropDown").data("kendoDropDownList");
    _submitLayerObject.nameDataField = dropdownlist.value();
 
    const categorySelect = $("#categoryMultiSelect").data("kendoMultiSelect");
    _submitLayerObject.categories = categorySelect.value();
 
    data.append("layerData", JSON.stringify(_submitLayerObject));
    //manually inject the request verification token!
    data.append("__RequestVerificationToken", $("input[name='__RequestVerificationToken']").val());
    //needs to be straight ajax call because of the combination file upload with json form data.
    $.ajax({
      type: "POST",
      url: _endPoints.submit,
      data: data,
      contentType: false,
      processData: false,
      dataType: "json", //datatype we expect back from server
      success: function(response) {
        if (response.success === false) {
          $.notify({ message: response.Message + $("#symbolizationErrors").text() }, { type: "danger", z_index: 3000 });
          $(".submissionProcessing").hide();
        } else if (nextFunc && $.isFunction(nextFunc)) {
          nextFunc(response);
        }
        //alert('success!!');
      },
      error: function(error) {
        if (errorFunc && $.isFunction(errorFunc)) {
          errorFunc(error);
        } else {
          /* eslint-disable-next-line no-console*/
          console.log(error);
        }
      },
    });
  }

  export function PopulateUIFromExistingSymbologySelection(symbologyObj) {
    $Ajax.ajaxAntiForgery({
      url: _endPoints.retrieve,
      type: "GET",
      data: {
        symbologyID: symbologyObj.SymbologyGUID,
      },
      datatype: "json",
      success: function(response) {
        if (response.success) {
          _submitLayerObject.existingSymbology = symbologyObj;
          _submitLayerObject.pointStyleId = "";
          _selectedSymbology = response.Symbology;
          ClearExistingSymbologyDivs(response.Symbology);
        } else {
          //alert the user on error? yes
          $.notify({ message: response.Message }, { type: "danger" });
        }
      },
      error: function(response) {
        //alert the user on error?
        /* eslint-disable-next-line no-console*/
        console.log(response);
      },
    });
  }
 
  export function ClearExistingSymbologyDivs(symbology) {
    $("#symbolizationErrorsDiv").hide();

  function updateDataFieldsPane(analysis) {
    let data = [];
    let nameIndex = 0;
    if (analysis === null || analysis === undefined) {
      //clear out data and hide list
      $("#dataFieldsList").hide();
    } else {
      $("#dataFieldsList").show();
      data = analysis.FoundDataFields;
      if (data.length > 0) {
        //try to find index where there is a name field
        const indexes = $.map(data, function(obj, index) {
          if (obj.Key.toLowerCase().indexOf("nam") >= 0) {
            //want to default to a name *type* field
            return index;
          }
        });
        nameIndex = indexes[0];
      }
    }
    const ds = new kendo.data.DataSource({ data: data });
 
    $("#dataFieldsList").kendoListView({
      dataSource: ds,
      template: kendo.template($("#attributeDataTemplate").html()),
      dataBound: function(e) {
        //checkboxEventBinding();
      },
    });
 
    $("#namingFieldDropDown").kendoDropDownList({
      dataSource: ds,
      dataTextField: "Key",
      dataValueField: "Key",
      index: nameIndex,
    });
  }
