  export function DeleteServiceLayer(element) {
    const theUrl = _endPoints.delete;
    const tr = $(element).closest("tr");
    const grid: kendo.ui.Grid = $("#servicesListGrid").data("kendoGrid");
    const rowDataItem = <IFeature>grid.dataItem(tr);
    $Ajax.ajaxAntiForgery({
      url: theUrl,
      type: "POST",
      data: {
        layerID: rowDataItem.WMSLayerGUID,
      },
      success: function(response) {
        if (response.success === false) {
          $.notify({ message: response.Message }, { type: "danger" });
        } else {
          //update the grid
          grid.removeRow(tr);
        }
      },
      error: function(xhr, status, error) {
        Main.handlejqerr(xhr, status, error);
      },
    });
  }

    //get shared jurisdictions
    if (_submitLayerObject.sharingType === 2) {

  function updateServiceDetailsPane(details: any, isWMS: boolean) {
    if (details === null) {
      $("#serviceLayerDetailsDiv").hide();
      $("#serviceLayerDefinedName").val("");
      $("#serviceDescription").val("");
      $("#serviceOwnerInfo").val("");
      updateNextButtonState(false);
    } else {
      updateNextButtonState(true);
      if (details.DefinedTitle !== null && details.DefinedTitle !== undefined) {
        $("#serviceLayerDefinedName").val(details.DefinedTitle);
      }
      if (details.Description !== null && details.Description !== undefined) {
        $("#serviceDescription").val(details.Description);
      }
      if (details.ContactInfo !== null && details.ContactInfo !== undefined) {
        let contactStr: string = "";
        //need to change logic if it is a WFS
        if (isWMS) {
          if (details.ContactInfo.ContactPersonPrimary !== undefined) {
            contactStr += details.ContactInfo.ContactPersonPrimary.ContactPerson + "\n";
          }
          if (details.ContactInfo.ContactPersonPrimary.ContactOrganization !== undefined) {
            contactStr += details.ContactInfo.ContactPersonPrimary.ContactOrganization + "\n";
          }
          if (details.ContactInfo.ContactVoiceTelephone !== undefined) {
            contactStr += details.ContactInfo.ContactVoiceTelephone + "\n";
          }
          if (details.ContactInfo.ContactElectronicMailAddress !== undefined) {
            contactStr += details.ContactInfo.ContactElectronicMailAddress;
          }
        } else {
          if (details.ContactInfo.Phone !== undefined) {
            contactStr += details.ContactInfo.Phone.Voice + "\n";
            if (details.ContactInfo.Phone.Facsimile !== "" && details.ContactInfo.Phone.Facsimile !== null) {
              contactStr += details.ContactInfo.Phone.Facsimile + "\n";
            }
          }
          if (details.ContactInfo.Address !== undefined) {
            contactStr += details.ContactInfo.Address.City + "\n";
          }
        }
        $("#serviceOwnerInfo").val(contactStr);
      }
      $("#serviceLayerDetailsDiv").show();
    }
  }
