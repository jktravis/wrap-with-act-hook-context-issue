namespace IncidentSummary {
  function getSelectedURL(asNetworkLink) {
    const dwnType = $("#DownloadType").val();
    let url = currentUrlPrefix() + "File/ShowKML?ProjectGUID=" + Main.currentProjectGUID;
    if (dwnType === "Incident") {
      if (asNetworkLink === true) {
        url = url + "&UseExplicitKML=false&IncludeOrgData=false";
      } else {
        url = url + "&UseExplicitKML=true&IncludeOrgData=false";
      }
    } else if (dwnType === "Incident and Org") {
      if (asNetworkLink === true) {
        url = url + "&UseExplicitKML=false&IncludeOrgData=true";
      } else {
        url = url + "&UseExplicitKML=true&IncludeOrgData=true";
      }
    }
    return url;
  }
  export function GetKML(asNetworkLink) {
    const url = getSelectedURL(asNetworkLink);
    window.location.href = url;
  }
}
