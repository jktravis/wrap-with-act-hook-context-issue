/*eslint-disable
@typescript-eslint/no-namespace
 */
interface ITranslations{
  contactEntity: string;
  facility: string;
  subOrg: string;
}
 
let _translations: ITranslations;
 
namespace Search {
  export function initTranslations(translations){
    _translations = translations;
  }
 
  $(function() {
    $("#results").hide();
    $("#SearchText").keydown(function(event) {
      if (event.keyCode && event.keyCode === 13) {
        $("#srchButton").click();
        return false;
      } else {
        return true;
      }
    });
    if ($("#LocationText").length > 0) {
      $("#LocationText").keydown(function(event) {
        if (event.keyCode && event.keyCode === 13) {
          $("#srchButton").click();
          return false;
        } else {
          return true;
        }
      });
    }
  });
  export function showNorm() {
    $("#divCatSearch").hide("slow");
    $("#divNormSearch").show("slow");
  }
  export function showCat() {
    $("#divNormSearch").hide("slow");
    $("#divCatSearch").show("slow");
  }
  export function searchFilter() {
    //Will return true if uncollapsed
    const catVisible = $("#divCatSearch").is(":visible");
    const srchType = $("input[type='radio'][name='SearchType']:checked").val();
    if (!catVisible) {
      let locText = "";
      if ($("#LocationText").length > 0) {
        locText = $("#LocationText").val();
      }
 
      return { SearchText: $("#SearchText").val(), LocationText: locText, SearchType: srchType };
    } else {
      const catGUID = $("#FacilityCategoryGUID")
        .data("kendoDropDownList")
        .value();
      const subCatGUID = $("#FacilitySubCategoryGUID")
        .data("kendoDropDownList")
        .value();
      return {
        SearchText: "",
        LocationText: $("#CatLocationText").val(),
        SearchType: srchType,
        CategoryGUID: catGUID,
        SubCategoryGUID: subCatGUID,
      };
    }
  }
  export function filterSub() {
    return {
      FacilityCategoryGUID: $("#FacilityCategoryGUID").val(),
    };
  }
  export function searchAll() {
    var phonebookGridSource = $("#phonebookGrid")
                                .data("kendoGrid")
                                .dataSource;
    var facGridSource = $("#facGrid")
                          .data("kendoGrid")
                          .dataSource;
    var subOrgGridSource = $("#subOrgGrid")
                            .data("kendoGrid")
                            .dataSource;
    $("#results").show();
 
    var tabs = document.getElementsByClassName("k-link");
 
    phonebookGridSource.read().then(() => { tabs[0].innerHTML = _translations.contactEntity + " (" + phonebookGridSource.total() + ")"; });
    facGridSource.read().then(() => { tabs[1].innerHTML = _translations.facility + " (" + facGridSource.total() + ")"; });
    subOrgGridSource.read().then(() => { tabs[2].innerHTML = _translations.subOrg + " (" + subOrgGridSource.total() + ")"; });
  }
}
